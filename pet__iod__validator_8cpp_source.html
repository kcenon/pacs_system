89validation_result pet_iod_validator::validate(const dicom_dataset& dataset) const {
90 validation_result result;
91 result.is_valid = true;
92
93 // Validate mandatory modules
94 if (options_.check_type1 || options_.check_type2) {
95 validate_patient_module(dataset, result.findings);
96 validate_general_study_module(dataset, result.findings);
97 validate_general_series_module(dataset, result.findings);
98 validate_pet_series_module(dataset, result.findings);
99 validate_frame_of_reference_module(dataset, result.findings);
100 validate_pet_image_module(dataset, result.findings);
101 validate_sop_common_module(dataset, result.findings);
102 }
103
104 if (options_.validate_pixel_data) {
105 validate_image_pixel_module(dataset, result.findings);
106 validate_image_plane_module(dataset, result.findings);
107 }
108
109 // PET-specific validation
110 if (options_.validate_pet_specific) {
111 if (options_.validate_radiopharmaceutical) {
112 validate_radiopharmaceutical_info(dataset, result.findings);
113 }
114 validate_suv_parameters(dataset, result.findings);
115 validate_reconstruction_info(dataset, result.findings);
116 if (options_.validate_corrections) {
117 validate_corrections(dataset, result.findings);
118 }
119 }
120
121 // Check for errors
122 for (const auto& finding : result.findings) {
123 if (finding.severity == validation_severity::error) {
124 result.is_valid = false;
125 break;
126 }
127 if (options_.strict_mode && finding.severity == validation_severity::warning) {
128 result.is_valid = false;
129 break;
130 }
131 }
132
133 return result;
134}

136bool pet_iod_validator::quick_check(const dicom_dataset& dataset) const {
137 // Check only Type 1 required attributes for quick validation
138
139 // General Study Module Type 1
140 if (!dataset.contains(tags::study_instance_uid)) return false;
141
142 // General Series Module Type 1
143 if (!dataset.contains(tags::modality)) return false;
144 if (!dataset.contains(tags::series_instance_uid)) return false;
145
146 // Check modality is PT
147 auto modality = dataset.get_string(tags::modality);
148 if (modality != "PT") return false;
149
150 // Frame of Reference Module Type 1
151 if (!dataset.contains(pet_tags::frame_of_reference_uid)) return false;
152
153 // Image Pixel Module Type 1
154 if (!dataset.contains(tags::samples_per_pixel)) return false;
155 if (!dataset.contains(tags::photometric_interpretation)) return false;
156 if (!dataset.contains(tags::rows)) return false;
157 if (!dataset.contains(tags::columns)) return false;
158 if (!dataset.contains(tags::bits_allocated)) return false;
159 if (!dataset.contains(tags::bits_stored)) return false;
160 if (!dataset.contains(tags::high_bit)) return false;
161 if (!dataset.contains(tags::pixel_representation)) return false;
162 if (!dataset.contains(tags::pixel_data)) return false;
163
164 // SOP Common Module Type 1
165 if (!dataset.contains(tags::sop_class_uid)) return false;
166 if (!dataset.contains(tags::sop_instance_uid)) return false;
167
168 return true;
169}

234void pet_iod_validator::validate_pet_series_module(
235 const dicom_dataset& dataset,
236 std::vector<validation_finding>& findings) const {
237
238 // PET Series Module Type 1 attributes
239 if (options_.check_type1) {
240 check_type1_attribute(dataset, pet_tags::series_type, "SeriesType", findings);
241 check_type1_attribute(dataset, pet_tags::units, "Units", findings);
242 check_type1_attribute(dataset, pet_tags::counts_source, "CountsSource", findings);
243 }
244
245 // Validate Units value
246 if (dataset.contains(pet_tags::units)) {
247 auto units = dataset.get_string(pet_tags::units);
248 if (units != "CNTS" && units != "BQML" && units != "GML" &&
249 units != "PROPCNTS" && units != "NONE") {
250 findings.push_back({
251 validation_severity::warning,
252 pet_tags::units,
253 "Unusual Units value for PET: " + units +
254 " (expected CNTS, BQML, or GML)",
255 "PT-WARN-001"
256 });
257 }
258 }
259
260 // Validate SeriesType
261 if (dataset.contains(pet_tags::series_type)) {
262 auto series_type = dataset.get_string(pet_tags::series_type);
263 // SeriesType should be like "STATIC\IMAGE" or "WHOLE BODY\IMAGE"
264 if (series_type.find('\\') == std::string::npos) {
265 findings.push_back({
266 validation_severity::info,
267 pet_tags::series_type,
268 "SeriesType should contain backslash separator (e.g., STATIC\\IMAGE)",
269 "PT-INFO-001"
270 });
271 }
272 }

292void pet_iod_validator::validate_pet_image_module(
293 const dicom_dataset& dataset,
294 std::vector<validation_finding>& findings) const {
295
296 // Type 1
297 if (options_.check_type1) {
298 check_type1_attribute(dataset, pet_tags::image_type, "ImageType", findings);
299 check_type1_attribute(dataset, pet_tags::frame_reference_time,
300 "FrameReferenceTime", findings);
301 }
302
303 // Type 1C - Decay Correction (required if Units = BQML)
304 if (options_.check_conditional) {
305 if (dataset.contains(pet_tags::units)) {
306 auto units = dataset.get_string(pet_tags::units);
307 if (units == "BQML") {
308 if (!dataset.contains(pet_tags::decay_correction)) {
309 findings.push_back({
310 validation_severity::error,
311 pet_tags::decay_correction,
312 "DecayCorrection is required when Units = BQML",
313 "PT-ERR-001"
314 });
315 }
316 }
317 }
318 }
319
320 // Validate DecayCorrection value
321 if (dataset.contains(pet_tags::decay_correction)) {
322 auto decay = dataset.get_string(pet_tags::decay_correction);
323 if (decay != "NONE" && decay != "START" && decay != "ADMIN") {
324 findings.push_back({
325 validation_severity::warning,
326 pet_tags::decay_correction,
327 "Invalid DecayCorrection value: " + decay +
328 " (expected NONE, START, or ADMIN)",
329 "PT-WARN-002"
330 });
331 }
332 }

406void pet_iod_validator::validate_radiopharmaceutical_info(
407 const dicom_dataset& dataset,
408 std::vector<validation_finding>& findings) const {
409
410 // Radiopharmaceutical Information Sequence is Type 2
411 if (options_.check_type2) {
412 check_type2_attribute(dataset, pet_tags::radiopharmaceutical_info_sequence,
413 "RadiopharmaceuticalInformationSequence", findings);
414 }
415
416 // If sequence is present, validate its contents
417 if (dataset.contains(pet_tags::radiopharmaceutical_info_sequence)) {
418 // For quantitative PET (BQML), dose and half-life are critical
419 if (dataset.contains(pet_tags::units)) {
420 auto units = dataset.get_string(pet_tags::units);
421 if (units == "BQML" || units == "GML") {
422 // Check for required dose information
423 if (!dataset.contains(pet_tags::radionuclide_total_dose)) {
424 findings.push_back({
425 validation_severity::warning,
426 pet_tags::radionuclide_total_dose,
427 "RadionuclideTotalDose recommended for quantitative PET",
428 "PT-WARN-003"
429 });
430 }
431 if (!dataset.contains(pet_tags::radionuclide_half_life)) {
432 findings.push_back({
433 validation_severity::warning,
434 pet_tags::radionuclide_half_life,
435 "RadionuclideHalfLife recommended for quantitative PET",
436 "PT-WARN-004"
437 });
438 }
439 }
440 }

594void pet_iod_validator::check_pixel_data_consistency(
595 const dicom_dataset& dataset,
596 std::vector<validation_finding>& findings) const {
597
598 // Check BitsStored <= BitsAllocated
599 auto bits_allocated = dataset.get_numeric<uint16_t>(tags::bits_allocated);
600 auto bits_stored = dataset.get_numeric<uint16_t>(tags::bits_stored);
601 auto high_bit = dataset.get_numeric<uint16_t>(tags::high_bit);
602
603 if (bits_allocated && bits_stored) {
604 if (*bits_stored > *bits_allocated) {
605 findings.push_back({
606 validation_severity::error,
607 tags::bits_stored,
608 "BitsStored (" + std::to_string(*bits_stored) +
609 ") exceeds BitsAllocated (" + std::to_string(*bits_allocated) + ")",
610 "PT-ERR-004"
611 });
612 }
613 }
614
615 // Check HighBit == BitsStored - 1
616 if (bits_stored && high_bit) {
617 if (*high_bit != *bits_stored - 1) {
618 findings.push_back({
619 validation_severity::warning,
620 tags::high_bit,
621 "HighBit (" + std::to_string(*high_bit) +
622 ") should typically be BitsStored - 1 (" +
623 std::to_string(*bits_stored - 1) + ")",
624 "PT-WARN-005"
625 });
626 }
627 }
628
629 // Check photometric interpretation is valid for PET
630 if (dataset.contains(tags::photometric_interpretation)) {
631 auto photometric = dataset.get_string(tags::photometric_interpretation);
632 if (!sop_classes::is_valid_pet_photometric(photometric)) {
633 findings.push_back({
634 validation_severity::warning,
635 tags::photometric_interpretation,
636 "Unusual photometric interpretation for PET: " + photometric +
637 " (expected MONOCHROME2)",
638 "PT-WARN-006"
639 });
640 }
641 }
642
643 // PET images should be grayscale (SamplesPerPixel = 1)
644 auto samples = dataset.get_numeric<uint16_t>(tags::samples_per_pixel);
645 if (samples && *samples != 1) {
646 findings.push_back({
647 validation_severity::warning,
648 tags::samples_per_pixel,
649 "PET images should have SamplesPerPixel = 1, found: " +
650 std::to_string(*samples),
651 "PT-WARN-007"
652 });
653 }
654}
