72struct print_result {

89 [[nodiscard]] bool is_success() const noexcept {
90 return status == 0x0000;
91 }
92
94 [[nodiscard]] bool is_warning() const noexcept {
95 return (status & 0xF000) == 0xB000;
96 }
97
99 [[nodiscard]] bool is_error() const noexcept {
100 return !is_success() && !is_warning();
101 }
102};
