71 static user_context system_context() {
72 User system_user;
73 system_user.id = "system";
74 system_user.username = "system";
75 system_user.roles = {Role::System};
76 system_user.active = true;
77 return user_context(std::move(system_user), "system-internal");

83 static user_context anonymous_context(const std::string &session_id) {
