24std::pair<ResourceType, std::uint32_t>
25access_control_manager::map_dicom_operation(DicomOperation op) {
26 switch (op) {
27 case DicomOperation::CStore:
28 return {ResourceType::Study, Action::Write};
29 case DicomOperation::CFind:
30 return {ResourceType::Metadata, Action::Read};
31 case DicomOperation::CMove:
32 return {ResourceType::Study, Action::Export};
33 case DicomOperation::CGet:
34 return {ResourceType::Study, Action::Read | Action::Export};
35 case DicomOperation::CEcho:
36 return {ResourceType::System, Action::Execute};
37 case DicomOperation::NCreate:
38 return {ResourceType::Study, Action::Write};
39 case DicomOperation::NSet:
40 return {ResourceType::Study, Action::Write};
41 case DicomOperation::NGet:
42 return {ResourceType::Study, Action::Read};
43 case DicomOperation::NDelete:
44 return {ResourceType::Study, Action::Delete};
45 case DicomOperation::NAction:
46 return {ResourceType::Study, Action::Execute};
47 case DicomOperation::NEventReport:
48 return {ResourceType::Study, Action::Read};
49 }
50 return {ResourceType::System, Action::None};
51}

53void access_control_manager::initialize_default_permissions() {
54 // Viewer: Read-only access to Studies
55 role_permissions_[Role::Viewer] = {{ResourceType::Study, Action::Read},
56 {ResourceType::Metadata, Action::Read}};
57
58 // Technologist: Read access, Create/Update studies (but not delete)
59 role_permissions_[Role::Technologist] = {
60 {ResourceType::Study, Action::Read | Action::Write},
61 {ResourceType::Metadata, Action::Read | Action::Write}};
62
63 // Radiologist: Same as Technologist + Verification (represented as Write in
64 // this simple model or could be separate) Giving Full study access for now
65 // (Delete/Export)
66 role_permissions_[Role::Radiologist] = {
67 {ResourceType::Study, Action::Full},
68 {ResourceType::Metadata, Action::Read | Action::Write}};
69
70 // Administrator: User management, System config
71 role_permissions_[Role::Administrator] = {
72 {ResourceType::System, Action::Full},
73 {ResourceType::Role, Action::Full},
74 {ResourceType::Audit, Action::Full},
75 {ResourceType::Study, Action::Full},
76 {ResourceType::Series, Action::Full},
77 {ResourceType::Image, Action::Full}};
78
79 // System: Internal superuser
80 role_permissions_[Role::System] = {{ResourceType::Study, Action::Full},
81 {ResourceType::Metadata, Action::Full},
82 {ResourceType::System, Action::Full},
83 {ResourceType::User, Action::Full},
84 {ResourceType::Audit, Action::Full}};
85}

108auto access_control_manager::assign_role(std::string_view user_id, Role role)
109 -> kcenon::common::VoidResult {
110 if (!storage_)
111 return kcenon::common::make_error<std::monostate>(
112 1, "Storage not configured", "access_control");
113
114 // Get user, add role, update
115 auto user_res = storage_->get_user(user_id);
116 if (user_res.is_err())
117 return kcenon::common::make_error<std::monostate>(2, "User not found",
118 "access_control");
119
120 auto user = user_res.unwrap();
121 if (!user.has_role(role)) {
122 user.roles.push_back(role);
123 return storage_->update_user(user);
124 }
125 return kcenon::common::ok();
126}

128bool access_control_manager::check_permission(const User &user,
129 ResourceType resource,
130 std::uint32_t action_mask) const {
131 if (!user.active)
132 return false;
133
134 for (const auto &role : user.roles) {
135 auto it = role_permissions_.find(role);
136 if (it != role_permissions_.end()) {
137 for (const auto &perm : it->second) {
138 if (perm.resource == resource && perm.has_action(action_mask)) {
139 return true;
140 }
141 }
142 }
143 }
144 return false;
145}

170auto access_control_manager::validate_access(const user_context &ctx,
171 ResourceType resource,
172 std::uint32_t action_mask)
173 -> kcenon::common::VoidResult {
174 if (!ctx.is_valid()) {
175 return kcenon::common::make_error<std::monostate>(
176 1, "User context is not valid (inactive user)", "access_control");
177 }
178
179 if (!check_permission(ctx.user(), resource, action_mask)) {
180 return kcenon::common::make_error<std::monostate>(
181 2, "Access denied: insufficient permissions", "access_control");
182 }
183
184 return kcenon::common::ok();
185}

187AccessCheckResult
188access_control_manager::check_dicom_operation(const user_context &ctx,
189 DicomOperation op) const {
190 if (!ctx.is_valid()) {
191 auto result = AccessCheckResult::deny("User context is not valid");
192 if (audit_callback_) {
193 audit_callback_(ctx, op, result);
194 }
195 return result;
196 }
197
198 auto [resource, action] = map_dicom_operation(op);
199 bool allowed = check_permission(ctx.user(), resource, action);
200
201 AccessCheckResult result =
202 allowed ? AccessCheckResult::allow()
203 : AccessCheckResult::deny("Insufficient permissions for operation");
204
205 if (audit_callback_) {
206 audit_callback_(ctx, op, result);
207 }
208
209 return result;
210}

212std::optional<user_context> access_control_manager::get_context_for_ae(
213 std::string_view ae_title, const std::string &session_id) const {
214 std::lock_guard<std::mutex> lock(mutex_);
215
216 // Look up AE Title to user mapping
217 auto it = ae_to_user_id_.find(std::string(ae_title));
218 if (it == ae_to_user_id_.end()) {
219 // Reject unregistered AE titles
220 return std::nullopt;
221 }
222
223 // Get user from storage
224 if (!storage_) {
225 return std::nullopt;
226 }
227
228 auto user_result = storage_->get_user(it->second);
229 if (user_result.is_err()) {
230 return std::nullopt;
231 }
232
233 auto ctx = user_context(user_result.unwrap(), session_id);
234 ctx.set_source_ae_title(std::string(ae_title));
235 return ctx;
236}
