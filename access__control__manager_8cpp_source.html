49std::pair<ResourceType, std::uint32_t>
50access_control_manager::map_dicom_operation(DicomOperation op) {
51 switch (op) {
52 case DicomOperation::CStore:
53 return {ResourceType::Study, Action::Write};
54 case DicomOperation::CFind:
55 return {ResourceType::Metadata, Action::Read};
56 case DicomOperation::CMove:
57 return {ResourceType::Study, Action::Export};
58 case DicomOperation::CGet:
59 return {ResourceType::Study, Action::Read | Action::Export};
60 case DicomOperation::CEcho:
61 return {ResourceType::System, Action::Execute};
62 case DicomOperation::NCreate:
63 return {ResourceType::Study, Action::Write};
64 case DicomOperation::NSet:
65 return {ResourceType::Study, Action::Write};
66 case DicomOperation::NGet:
67 return {ResourceType::Study, Action::Read};
68 case DicomOperation::NDelete:
69 return {ResourceType::Study, Action::Delete};
70 case DicomOperation::NAction:
71 return {ResourceType::Study, Action::Execute};
72 case DicomOperation::NEventReport:
73 return {ResourceType::Study, Action::Read};
74 }
75 return {ResourceType::System, Action::None};
76}

78void access_control_manager::initialize_default_permissions() {
79 // Viewer: Read-only access to Studies
80 role_permissions_[Role::Viewer] = {{ResourceType::Study, Action::Read},
81 {ResourceType::Metadata, Action::Read}};
82
83 // Technologist: Read access, Create/Update studies (but not delete)
84 role_permissions_[Role::Technologist] = {
85 {ResourceType::Study, Action::Read | Action::Write},
86 {ResourceType::Metadata, Action::Read | Action::Write}};
87
88 // Radiologist: Same as Technologist + Verification (represented as Write in
89 // this simple model or could be separate) Giving Full study access for now
90 // (Delete/Export)
91 role_permissions_[Role::Radiologist] = {
92 {ResourceType::Study, Action::Full},
93 {ResourceType::Metadata, Action::Read | Action::Write}};
94
95 // Administrator: User management, System config
96 role_permissions_[Role::Administrator] = {
97 {ResourceType::System, Action::Full},
98 {ResourceType::Role, Action::Full},
99 {ResourceType::Audit, Action::Full},
100 {ResourceType::Study, Action::Full},
101 {ResourceType::Series, Action::Full},
102 {ResourceType::Image, Action::Full}};
103
104 // System: Internal superuser
105 role_permissions_[Role::System] = {{ResourceType::Study, Action::Full},
106 {ResourceType::Metadata, Action::Full},
107 {ResourceType::System, Action::Full},
108 {ResourceType::User, Action::Full},
109 {ResourceType::Audit, Action::Full}};
110}

133auto access_control_manager::assign_role(std::string_view user_id, Role role)
134 -> kcenon::common::VoidResult {
135 if (!storage_)
136 return kcenon::common::make_error<std::monostate>(
137 1, "Storage not configured", "access_control");
138
139 // Get user, add role, update
140 auto user_res = storage_->get_user(user_id);
141 if (user_res.is_err())
142 return kcenon::common::make_error<std::monostate>(2, "User not found",
143 "access_control");
144
145 auto user = user_res.unwrap();
146 if (!user.has_role(role)) {
147 user.roles.push_back(role);
148 return storage_->update_user(user);
149 }
150 return kcenon::common::ok();
151}

153bool access_control_manager::check_permission(const User &user,
154 ResourceType resource,
155 std::uint32_t action_mask) const {
156 if (!user.active)
157 return false;
158
159 for (const auto &role : user.roles) {
160 auto it = role_permissions_.find(role);
161 if (it != role_permissions_.end()) {
162 for (const auto &perm : it->second) {
163 if (perm.resource == resource && perm.has_action(action_mask)) {
164 return true;
165 }
166 }
167 }
168 }
169 return false;
170}

195auto access_control_manager::validate_access(const user_context &ctx,
196 ResourceType resource,
197 std::uint32_t action_mask)
198 -> kcenon::common::VoidResult {
199 if (!ctx.is_valid()) {
200 return kcenon::common::make_error<std::monostate>(
201 1, "User context is not valid (inactive user)", "access_control");
202 }
203
204 if (!check_permission(ctx.user(), resource, action_mask)) {
205 return kcenon::common::make_error<std::monostate>(
206 2, "Access denied: insufficient permissions", "access_control");
207 }
208
209 return kcenon::common::ok();
210}

212AccessCheckResult
213access_control_manager::check_dicom_operation(const user_context &ctx,
214 DicomOperation op) const {
215 if (!ctx.is_valid()) {
216 auto result = AccessCheckResult::deny("User context is not valid");
217 if (audit_callback_) {
218 audit_callback_(ctx, op, result);
219 }
220 return result;
221 }
222
223 auto [resource, action] = map_dicom_operation(op);
224 bool allowed = check_permission(ctx.user(), resource, action);
225
226 AccessCheckResult result =
227 allowed ? AccessCheckResult::allow()
228 : AccessCheckResult::deny("Insufficient permissions for operation");
229
230 if (audit_callback_) {
231 audit_callback_(ctx, op, result);
232 }
233
234 return result;
235}

237std::optional<user_context> access_control_manager::get_context_for_ae(
238 std::string_view ae_title, const std::string &session_id) const {
239 std::lock_guard<std::mutex> lock(mutex_);
240
241 // Look up AE Title to user mapping
242 auto it = ae_to_user_id_.find(std::string(ae_title));
243 if (it == ae_to_user_id_.end()) {
244 // Reject unregistered AE titles
245 return std::nullopt;
246 }
247
248 // Get user from storage
249 if (!storage_) {
250 return std::nullopt;
251 }
252
253 auto user_result = storage_->get_user(it->second);
254 if (user_result.is_err()) {
255 return std::nullopt;
256 }
257
258 auto ctx = user_context(user_result.unwrap(), session_id);
259 ctx.set_source_ae_title(std::string(ae_title));
260 return ctx;
261}
