96auto get_all_identifier_tags() -> std::vector<dicom_tag> {
97 std::vector<dicom_tag> all_tags;
98
99 auto names = get_name_tags();
100 auto geo = get_geographic_tags();
101 auto dates = get_date_tags();
102 auto comm = get_communication_tags();
103 auto ids = get_unique_id_tags();
104
105 all_tags.reserve(names.size() + geo.size() + dates.size() +
106 comm.size() + ids.size());
107
108 all_tags.insert(all_tags.end(), names.begin(), names.end());
109 all_tags.insert(all_tags.end(), geo.begin(), geo.end());
110 all_tags.insert(all_tags.end(), dates.begin(), dates.end());
111 all_tags.insert(all_tags.end(), comm.begin(), comm.end());
112 all_tags.insert(all_tags.end(), ids.begin(), ids.end());
113
114 return all_tags;
115}
