82auto hsm_storage::store(const core::dicom_dataset& dataset) -> VoidResult {
83 // Extract SOP Instance UID for metadata tracking
84 auto sop_uid = dataset.get_string(core::tags::sop_instance_uid);
85 if (sop_uid.empty()) {
86 return make_error<std::monostate>(
87 kInvalidConfiguration, "Missing SOP Instance UID", "hsm_storage");
88 }
89
90 // Store in hot tier
91 auto result = hot_tier_->store(dataset);
92 if (!result.is_ok()) {
93 return result;
94 }
95
96 // Update metadata
97 std::unique_lock lock(mutex_);
98 update_metadata(sop_uid, storage_tier::hot, dataset);
99
100 return ok();
101}

103auto hsm_storage::retrieve(std::string_view sop_instance_uid)
104 -> Result<core::dicom_dataset> {
105 // Find which tier contains the instance
106 auto tier = find_tier(sop_instance_uid);
107 if (!tier.has_value()) {
108 return make_error<core::dicom_dataset>(
109 kInstanceNotFound,
110 "Instance not found: " + std::string(sop_instance_uid),
111 "hsm_storage");
112 }
113
114 // Retrieve from the tier
115 auto* storage = get_storage(*tier);
116 if (storage == nullptr) {
117 return make_error<core::dicom_dataset>(
118 kTierNotAvailable, "Tier storage not available", "hsm_storage");
119 }
120
121 auto result = storage->retrieve(sop_instance_uid);
122 if (!result.is_ok()) {
123 return result;
124 }
125
126 // Update access time if tracking is enabled
127 if (config_.track_access_time) {
128 std::unique_lock lock(mutex_);
129 update_access_time(sop_instance_uid);
130 }
131
132 return result;
133}

135auto hsm_storage::remove(std::string_view sop_instance_uid) -> VoidResult {
136 // Find which tier contains the instance
137 auto tier = find_tier(sop_instance_uid);
138 if (!tier.has_value()) {
139 // Not found is not an error for remove
140 return ok();
141 }
142
143 // Remove from the tier
144 auto* storage = get_storage(*tier);
145 if (storage == nullptr) {
146 return make_error<std::monostate>(
147 kTierNotAvailable, "Tier storage not available", "hsm_storage");
148 }
149
150 auto result = storage->remove(sop_instance_uid);
151 if (!result.is_ok()) {
152 return result;
153 }
154
155 // Remove metadata
156 std::unique_lock lock(mutex_);
157 remove_metadata(sop_instance_uid);
158
159 return ok();
160}

167auto hsm_storage::find(const core::dicom_dataset& query)
168 -> Result<std::vector<core::dicom_dataset>> {
169 std::vector<core::dicom_dataset> combined_results;
170
171 // Search all tiers and combine results
172 std::vector<storage_interface*> tiers = {hot_tier_.get(), warm_tier_.get(),
173 cold_tier_.get()};
174
175 for (auto* tier : tiers) {
176 if (tier == nullptr) {
177 continue;
178 }
179
180 auto result = tier->find(query);
181 if (result.is_ok()) {
182 auto& datasets = result.value();
183 combined_results.insert(combined_results.end(), datasets.begin(),
184 datasets.end());
185 }
186 }
187
188 return combined_results;
189}

191auto hsm_storage::get_statistics() const -> storage_statistics {
192 storage_statistics stats;
193
194 std::shared_lock lock(mutex_);
195
196 // Aggregate from all tiers
197 std::vector<storage_interface*> tiers = {hot_tier_.get(), warm_tier_.get(),
198 cold_tier_.get()};
199
200 std::set<std::string> studies;
201 std::set<std::string> series;
202 std::set<std::string> patients;
203
204 for (auto* tier : tiers) {
205 if (tier == nullptr) {
206 continue;
207 }
208
209 auto tier_stats = tier->get_statistics();
210 stats.total_instances += tier_stats.total_instances;
211 stats.total_bytes += tier_stats.total_bytes;
212 }
213
214 // Count unique studies/series from metadata
215 for (const auto& [uid, meta] : metadata_index_) {
216 if (!meta.study_instance_uid.empty()) {
217 studies.insert(meta.study_instance_uid);
218 }
219 if (!meta.series_instance_uid.empty()) {
220 series.insert(meta.series_instance_uid);
221 }
222 }
223
224 stats.studies_count = studies.size();
225 stats.series_count = series.size();
226
227 return stats;
228}

230auto hsm_storage::verify_integrity() -> VoidResult {
231 // Verify each tier
232 std::vector<std::pair<storage_tier, storage_interface*>> tiers = {
233 {storage_tier::hot, hot_tier_.get()},
234 {storage_tier::warm, warm_tier_.get()},
235 {storage_tier::cold, cold_tier_.get()}};
236
237 for (const auto& [tier, storage] : tiers) {
238 if (storage == nullptr) {
239 continue;
240 }
241
242 auto result = storage->verify_integrity();
243 if (!result.is_ok()) {
244 return make_error<std::monostate>(
245 kIntegrityError,
246 "Integrity check failed for " + std::string(to_string(tier)) +
247 " tier: " + std::string(result.error().message),
248 "hsm_storage");
249 }
250 }
251
252 // Verify metadata consistency
253 std::shared_lock lock(mutex_);
254 for (const auto& [uid, meta] : metadata_index_) {
255 auto* storage = get_storage(meta.current_tier);
256 if (storage == nullptr) {
257 continue;
258 }
259
260 if (!storage->exists(uid)) {
261 return make_error<std::monostate>(
262 kIntegrityError,
263 "Metadata references non-existent instance: " + uid,
264 "hsm_storage");
265 }
266 }
267
268 return ok();
269}

295auto hsm_storage::migrate(std::string_view sop_instance_uid,
296 storage_tier target_tier) -> VoidResult {
297 // Find current tier
298 auto current_tier_opt = find_tier(sop_instance_uid);
299 if (!current_tier_opt.has_value()) {
300 return make_error<std::monostate>(
301 kInstanceNotFound,
302 "Instance not found: " + std::string(sop_instance_uid),
303 "hsm_storage");
304 }
305
306 auto current_tier = *current_tier_opt;
307 if (current_tier == target_tier) {
308 // Already in target tier
309 return ok();
310 }
311
312 return migrate_instance(sop_instance_uid, current_tier, target_tier);
313}

315auto hsm_storage::get_migration_candidates(storage_tier from_tier,
316 storage_tier to_tier) const
317 -> std::vector<tier_metadata> {
318 std::vector<tier_metadata> candidates;
319
320 std::shared_lock lock(mutex_);
321 for (const auto& [uid, meta] : metadata_index_) {
322 if (meta.current_tier == from_tier &&
323 meta.should_migrate(config_.policy, to_tier)) {
324 candidates.push_back(meta);
325 }
326 }
327
328 // Sort by age (oldest first)
329 std::sort(candidates.begin(), candidates.end(),
330 [](const tier_metadata& a, const tier_metadata& b) {
331 return a.stored_at < b.stored_at;
332 });
333
334 return candidates;
335}

337auto hsm_storage::run_migration_cycle() -> migration_result {
338 migration_result result;
339 auto start_time = std::chrono::steady_clock::now();
340
341 // Hot to warm migration
342 if (warm_tier_) {
343 auto candidates =
344 get_migration_candidates(storage_tier::hot, storage_tier::warm);
345
346 for (const auto& meta : candidates) {
347 if (result.instances_migrated >=
348 config_.policy.max_instances_per_cycle) {
349 break;
350 }
351 if (result.bytes_migrated >= config_.policy.max_bytes_per_cycle) {
352 break;
353 }
354
355 auto migrate_result = migrate_instance(
356 meta.sop_instance_uid, storage_tier::hot, storage_tier::warm);
357
358 if (migrate_result.is_ok()) {
359 result.instances_migrated++;
360 result.bytes_migrated += meta.size_bytes;
361 } else {
362 result.failed_uids.push_back(meta.sop_instance_uid);
363 }
364 }
365 }
366
367 // Warm to cold migration
368 if (cold_tier_) {
369 auto candidates =
370 get_migration_candidates(storage_tier::warm, storage_tier::cold);
371
372 for (const auto& meta : candidates) {
373 if (result.instances_migrated >=
374 config_.policy.max_instances_per_cycle) {
375 break;
376 }
377 if (result.bytes_migrated >= config_.policy.max_bytes_per_cycle) {
378 break;
379 }
380
381 auto migrate_result = migrate_instance(
382 meta.sop_instance_uid, storage_tier::warm, storage_tier::cold);
383
384 if (migrate_result.is_ok()) {
385 result.instances_migrated++;
386 result.bytes_migrated += meta.size_bytes;
387 } else {
388 result.failed_uids.push_back(meta.sop_instance_uid);
389 }
390 }
391 }
392
393 // Also check hot to cold (if warm tier is skipped)
394 if (!warm_tier_ && cold_tier_) {
395 auto candidates =
396 get_migration_candidates(storage_tier::hot, storage_tier::cold);
397
398 for (const auto& meta : candidates) {
399 if (result.instances_migrated >=
400 config_.policy.max_instances_per_cycle) {
401 break;
402 }
403 if (result.bytes_migrated >= config_.policy.max_bytes_per_cycle) {
404 break;
405 }
406
407 auto migrate_result = migrate_instance(
408 meta.sop_instance_uid, storage_tier::hot, storage_tier::cold);
409
410 if (migrate_result.is_ok()) {
411 result.instances_migrated++;
412 result.bytes_migrated += meta.size_bytes;
413 } else {
414 result.failed_uids.push_back(meta.sop_instance_uid);
415 }
416 }
417 }
418
419 auto end_time = std::chrono::steady_clock::now();
420 result.duration = std::chrono::duration_cast<std::chrono::milliseconds>(
421 end_time - start_time);
422
423 return result;
424}

436auto hsm_storage::get_hsm_statistics() const -> hsm_statistics {
437 hsm_statistics stats;
438
439 std::shared_lock lock(mutex_);
440
441 std::set<std::string> hot_studies, hot_series;
442 std::set<std::string> warm_studies, warm_series;
443 std::set<std::string> cold_studies, cold_series;
444
445 for (const auto& [uid, meta] : metadata_index_) {
446 switch (meta.current_tier) {
447 case storage_tier::hot:
448 stats.hot.instance_count++;
449 stats.hot.total_bytes += meta.size_bytes;
450 if (!meta.study_instance_uid.empty()) {
451 hot_studies.insert(meta.study_instance_uid);
452 }
453 if (!meta.series_instance_uid.empty()) {
454 hot_series.insert(meta.series_instance_uid);
455 }
456 break;
457 case storage_tier::warm:
458 stats.warm.instance_count++;
459 stats.warm.total_bytes += meta.size_bytes;
460 if (!meta.study_instance_uid.empty()) {
461 warm_studies.insert(meta.study_instance_uid);
462 }
463 if (!meta.series_instance_uid.empty()) {
464 warm_series.insert(meta.series_instance_uid);
465 }
466 break;
467 case storage_tier::cold:
468 stats.cold.instance_count++;
469 stats.cold.total_bytes += meta.size_bytes;
470 if (!meta.study_instance_uid.empty()) {
471 cold_studies.insert(meta.study_instance_uid);
472 }
473 if (!meta.series_instance_uid.empty()) {
474 cold_series.insert(meta.series_instance_uid);
475 }
476 break;
477 }
478 }
479
480 stats.hot.study_count = hot_studies.size();
481 stats.hot.series_count = hot_series.size();
482 stats.warm.study_count = warm_studies.size();
483 stats.warm.series_count = warm_series.size();
484 stats.cold.study_count = cold_studies.size();
485 stats.cold.series_count = cold_series.size();
486
487 return stats;
488}

499auto hsm_storage::find_tier(std::string_view sop_instance_uid) const
500 -> std::optional<storage_tier> {
501 std::shared_lock lock(mutex_);
502 auto it = metadata_index_.find(std::string(sop_instance_uid));
503 if (it != metadata_index_.end()) {
504 return it->second.current_tier;
505 }
506
507 // Fallback: search each tier directly
508 lock.unlock();
509
510 if (hot_tier_ && hot_tier_->exists(sop_instance_uid)) {
511 return storage_tier::hot;
512 }
513 if (warm_tier_ && warm_tier_->exists(sop_instance_uid)) {
514 return storage_tier::warm;
515 }
516 if (cold_tier_ && cold_tier_->exists(sop_instance_uid)) {
517 return storage_tier::cold;
518 }
519
520 return std::nullopt;
521}

567auto hsm_storage::migrate_instance(std::string_view uid, storage_tier from_tier,
568 storage_tier to_tier) -> VoidResult {
569 auto* source = get_storage(from_tier);
570 auto* target = get_storage(to_tier);
571
572 if (source == nullptr) {
573 return make_error<std::monostate>(
574 kTierNotAvailable,
575 "Source tier not available: " + std::string(to_string(from_tier)),
576 "hsm_storage");
577 }
578 if (target == nullptr) {
579 return make_error<std::monostate>(
580 kTierNotAvailable,
581 "Target tier not available: " + std::string(to_string(to_tier)),
582 "hsm_storage");
583 }
584
585 // Retrieve from source
586 auto retrieve_result = source->retrieve(uid);
587 if (!retrieve_result.is_ok()) {
588 return make_error<std::monostate>(
589 kMigrationFailed,
590 "Failed to retrieve from source: " +
591 std::string(retrieve_result.error().message),
592 "hsm_storage");
593 }
594
595 auto& dataset = retrieve_result.value();
596
597 // Store to target
598 auto store_result = target->store(dataset);
599 if (!store_result.is_ok()) {
600 return make_error<std::monostate>(
601 kMigrationFailed,
602 "Failed to store to target: " +
603 std::string(store_result.error().message),
604 "hsm_storage");
605 }
606
607 // Verify if configured
608 if (config_.verify_after_migration) {
609 if (!target->exists(uid)) {
610 return make_error<std::monostate>(
611 kMigrationFailed,
612 "Verification failed: instance not found in target tier",
613 "hsm_storage");
614 }
615 }
616
617 // Remove from source if configured
618 if (config_.delete_after_migration) {
619 auto remove_result = source->remove(uid);
620 // Ignore remove failure - the instance is already in target tier
621 (void)remove_result;
622 }
623
624 // Update metadata
625 {
626 std::unique_lock lock(mutex_);
627 auto it = metadata_index_.find(std::string(uid));
628 if (it != metadata_index_.end()) {
629 it->second.current_tier = to_tier;
630 }
631 }
632
633 return ok();
634}
