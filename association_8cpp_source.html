59void rejection_info::build_description() {
60 std::ostringstream oss;
61 oss << "Association rejected: ";
62
63 if (result == reject_result::rejected_permanent) {
64 oss << "permanent, ";
65 } else {
66 oss << "transient, ";
67 }
68
69 switch (static_cast<reject_source>(source)) {
70 case reject_source::service_user:
71 oss << "source=service-user, ";
72 switch (static_cast<reject_reason_user>(reason)) {
73 case reject_reason_user::no_reason:
74 oss << "reason=no reason given";
75 break;
76 case reject_reason_user::application_context_not_supported:
77 oss << "reason=application context not supported";
78 break;
79 case reject_reason_user::calling_ae_not_recognized:
80 oss << "reason=calling AE title not recognized";
81 break;
82 case reject_reason_user::called_ae_not_recognized:
83 oss << "reason=called AE title not recognized";
84 break;
85 default:
86 oss << "reason=unknown (" << static_cast<int>(reason) << ")";
87 }
88 break;
89 case reject_source::service_provider_acse:
90 oss << "source=service-provider (ACSE), ";
91 switch (static_cast<reject_reason_provider_acse>(reason)) {
92 case reject_reason_provider_acse::no_reason:
93 oss << "reason=no reason given";
94 break;
95 case reject_reason_provider_acse::protocol_version_not_supported:
96 oss << "reason=protocol version not supported";
97 break;
98 default:
99 oss << "reason=unknown (" << static_cast<int>(reason) << ")";
100 }
101 break;
102 case reject_source::service_provider_presentation:
103 oss << "source=service-provider (Presentation), ";
104 switch (static_cast<reject_reason_provider_presentation>(reason)) {
105 case reject_reason_provider_presentation::temporary_congestion:
106 oss << "reason=temporary congestion";
107 break;
108 case reject_reason_provider_presentation::local_limit_exceeded:
109 oss << "reason=local limit exceeded";
110 break;
111 default:
112 oss << "reason=unknown (" << static_cast<int>(reason) << ")";
113 }
114 break;
115 default:
116 oss << "source=unknown (" << static_cast<int>(source) << ")";
117 }
118
119 description = oss.str();
120}

205Result<association> association::connect(
206 const std::string& host,
207 uint16_t port,
208 const association_config& config,
209 duration /*timeout*/) {
210
211 association assoc;
212 assoc.is_scu_ = true;
213 assoc.calling_ae_ = config.calling_ae_title;
214 assoc.called_ae_ = config.called_ae_title;
215 assoc.our_ae_ = config.calling_ae_title;
216 assoc.max_pdu_size_ = config.max_pdu_length;
217 assoc.our_implementation_class_ = config.implementation_class_uid;
218 assoc.our_implementation_version_ = config.implementation_version_name;
219
220 // Copy proposed contexts
221 for (const auto& pc : config.proposed_contexts) {
222 assoc.proposed_contexts_.push_back(pc);
223 }
224
225 // Transition to awaiting state
226 assoc.transition(association_state::awaiting_associate_ac);
227
228 // Check for in-memory server (Test Hook)
229 auto* server = dicom_server::get_server_on_port(port);
230 if (server) {
231 auto rq = assoc.build_associate_rq();
232 auto result = server->simulate_association_request(rq, &assoc);
233 if (result.is_ok()) {
234 if (assoc.process_associate_ac(result.value())) {
235 return assoc;
236 } else {
237 return error_info{no_acceptable_context, "Association negotiation failed", "network"};
238 }
239 } else {
240 return result.error();
241 }
242 }
243
244 // Note: Actual network connection should be handled by network_system
245 // This implementation prepares the association for PDU exchange
246 (void)host; // Will be used with network_system integration
247 (void)port;
248
249 return assoc;
250}

252association association::accept(
253 const associate_rq& rq,
254 const scp_config& config) {
255
256 association assoc;
257 assoc.is_scu_ = false;
258 assoc.calling_ae_ = rq.calling_ae_title;
259 assoc.called_ae_ = rq.called_ae_title;
260 assoc.our_ae_ = config.ae_title;
261 assoc.max_pdu_size_ = (std::min)(config.max_pdu_length, rq.user_info.max_pdu_length);
262 assoc.our_implementation_class_ = config.implementation_class_uid;
263 assoc.our_implementation_version_ = config.implementation_version_name;
264 assoc.remote_implementation_class_ = rq.user_info.implementation_class_uid;
265 assoc.remote_implementation_version_ = rq.user_info.implementation_version_name;
266
267 // Negotiate presentation contexts
268 assoc.negotiate_contexts(rq, config);
269
270 // Build lookup maps
271 assoc.build_context_map();
272
273 // Establish association if at least one context was accepted
274 // Note: For SCP, TCP connection is already established when accept() is called,
275 // so we directly set state to established (bypassing state machine)
276 bool has_accepted = std::any_of(
277 assoc.accepted_contexts_.begin(),
278 assoc.accepted_contexts_.end(),
279 [](const auto& ctx) { return ctx.is_accepted(); });
280
281 if (has_accepted) {
282 // SCP: A-ASSOCIATE-RQ received and accepted, go directly to established
283 assoc.state_ = association_state::established;
284 } else {
285 // No acceptable contexts - will need to reject
286 assoc.state_ = association_state::idle;
287 }
288
289 return assoc;
290}

394Result<std::monostate> association::send_dimse(
395 uint8_t context_id,
396 const dimse::dimse_message& msg) {
397
398 std::lock_guard<std::mutex> lock(mutex_);
399
400 if (state_ != association_state::established) {
401 return error_info{invalid_association_state, "Cannot send DIMSE: association not established", "network"};
402 }
403
404 // Verify context exists
405 if (context_to_transfer_syntax_.find(context_id) ==
406 context_to_transfer_syntax_.end()) {
407 return error_info{kcenon::pacs::error_codes::dimse_error, "Invalid presentation context ID", "network"};
408 }
409
410 // Message encoding and network send would be handled by network_system
411 // For now, validate the message
412 if (!msg.is_valid()) {
413 return error_info{kcenon::pacs::error_codes::dimse_error, "Invalid DIMSE message", "network"};
414 }
415
416 // Placeholder for actual send implementation
417 (void)msg;
418
419 if (peer_) {
420 peer_->enqueue_message(context_id, msg);
421 return std::monostate{};
422 }
423
424 return std::monostate{};
425}

427Result<std::pair<uint8_t, dimse::dimse_message>> association::receive_dimse(
428 duration timeout) {
429
430 {
431 std::lock_guard<std::mutex> lock(mutex_);
432 if (state_ != association_state::established) {
433 return error_info{invalid_association_state, "Cannot receive DIMSE: association not established", "network"};
434 }
435 }
436
437 // Placeholder - actual receive would be implemented with network_system
438 // return error_info("Not implemented: requires network_system integration");
439
440 // Use lock-free queue with blocking wait
441 if (auto item = incoming_queue_->wait_dequeue(timeout)) {
442 return std::move(*item);
443 }
444
445 // Check if we were aborted/released while waiting
446 {
447 std::lock_guard<std::mutex> lock(mutex_);
448 if (state_ != association_state::established) {
449 return error_info{association_aborted, "Association aborted or released", "network"};
450 }
451 }
452
453 return error_info{receive_timeout, "Receive timeout", "network"};
454}

460associate_rq association::build_associate_rq() const {
461 std::lock_guard<std::mutex> lock(mutex_);
462
463 associate_rq rq;
464 rq.calling_ae_title = calling_ae_;
465 rq.called_ae_title = called_ae_;
466 rq.application_context = DICOM_APPLICATION_CONTEXT;
467
468 // Convert proposed contexts
469 for (const auto& pc : proposed_contexts_) {
470 rq.presentation_contexts.push_back(
471 presentation_context_rq{pc.id, pc.abstract_syntax, pc.transfer_syntaxes});
472 }
473
474 // User information
475 rq.user_info.max_pdu_length = max_pdu_size_;
476 rq.user_info.implementation_class_uid = our_implementation_class_;
477 rq.user_info.implementation_version_name = our_implementation_version_;
478
479 return rq;
480}

482associate_ac association::build_associate_ac() const {
483 std::lock_guard<std::mutex> lock(mutex_);
484
485 associate_ac ac;
486 ac.calling_ae_title = calling_ae_;
487 ac.called_ae_title = called_ae_;
488 ac.application_context = DICOM_APPLICATION_CONTEXT;

491 for (const auto& ctx : accepted_contexts_) {
492 ac.presentation_contexts.push_back(
493 presentation_context_ac{ctx.id, ctx.result, ctx.transfer_syntax});
494 }

496 // User information
497 ac.user_info.max_pdu_length = max_pdu_size_;
498 ac.user_info.implementation_class_uid = our_implementation_class_;
499 ac.user_info.implementation_version_name = our_implementation_version_;
500

504bool association::process_associate_ac(const associate_ac& ac) {
505 std::lock_guard<std::mutex> lock(mutex_);
506
507 if (state_ != association_state::awaiting_associate_ac) {
508 return false;
509 }
510
511 // Store remote implementation info
512 remote_implementation_class_ = ac.user_info.implementation_class_uid;
513 remote_implementation_version_ = ac.user_info.implementation_version_name;
514
515 // Update max PDU size to negotiated value
516 max_pdu_size_ = (std::min)(max_pdu_size_, ac.user_info.max_pdu_length);
517
518 // Process accepted presentation contexts
519 accepted_contexts_.clear();
520 for (const auto& pc_ac : ac.presentation_contexts) {
521 // Find the corresponding proposed context
522 for (const auto& pc_rq : proposed_contexts_) {
523 if (pc_rq.id == pc_ac.id) {
524 accepted_contexts_.push_back(accepted_presentation_context{
525 pc_ac.id,
526 pc_rq.abstract_syntax,
527 pc_ac.transfer_syntax,
528 pc_ac.result
529 });
530 break;
531 }
532 }
533 }
534
535 // Build lookup maps
536 build_context_map();
537
538 // Check if at least one context was accepted
539 bool has_accepted = std::any_of(
540 accepted_contexts_.begin(),
541 accepted_contexts_.end(),
542 [](const auto& ctx) { return ctx.is_accepted(); });
543
544 if (has_accepted) {
545 transition(association_state::established);
546 return true;
547 }
548
549 return false;
550}

568Result<std::monostate> association::release(duration /*timeout*/) {
569 std::lock_guard<std::mutex> lock(mutex_);
570
571 if (state_ == association_state::released) {
572 return error_info{already_released, "Association already released", "network"};
573 }
574
575 if (state_ != association_state::established) {
576 return error_info{invalid_association_state, "Cannot release: association not established", "network"};
577 }
578
579 // Transition to awaiting release response
580 state_ = association_state::awaiting_release_rp;
581
582 // Note: Actual A-RELEASE-RQ sending and A-RELEASE-RP receiving
583 // would be handled by network_system integration
584
585 // For now, simulate successful release
586 state_ = association_state::released;
587
588 return std::monostate{};
589}

655bool association::can_transition_to(association_state new_state) const {
656 // Define valid state transitions per PS3.8
657 switch (state_) {
658 case association_state::idle:
659 return new_state == association_state::awaiting_associate_ac ||
660 new_state == association_state::awaiting_associate_rq;
661
662 case association_state::awaiting_associate_ac:
663 return new_state == association_state::established ||
664 new_state == association_state::idle ||
665 new_state == association_state::aborted;
666
667 case association_state::awaiting_associate_rq:
668 return new_state == association_state::established ||
669 new_state == association_state::idle ||
670 new_state == association_state::aborted;
671
672 case association_state::established:
673 return new_state == association_state::awaiting_release_rp ||
674 new_state == association_state::awaiting_release_rq ||
675 new_state == association_state::aborted;
676
677 case association_state::awaiting_release_rp:
678 return new_state == association_state::released ||
679 new_state == association_state::aborted;
680
681 case association_state::awaiting_release_rq:
682 return new_state == association_state::released ||
683 new_state == association_state::aborted;
684
685 case association_state::released:
686 case association_state::aborted:
687 return false; // Terminal states
688
689 default:
690 return false;
691 }
692}
693
694void association::transition(association_state new_state) {
695 if (can_transition_to(new_state)) {
696 state_ = new_state;
697 }
698}

700void association::build_context_map() {
701 abstract_syntax_to_context_.clear();
702 context_to_transfer_syntax_.clear();
703
704 for (const auto& ctx : accepted_contexts_) {
705 if (ctx.is_accepted()) {
706 abstract_syntax_to_context_[ctx.abstract_syntax] = ctx.id;
707
708 encoding::transfer_syntax ts{ctx.transfer_syntax};
709 if (ts.is_valid()) {
710 context_to_transfer_syntax_.emplace(ctx.id, std::move(ts));
711 }
712 }
713 }
714}
715
716void association::negotiate_contexts(
717 const associate_rq& rq,
718 const scp_config& config) {
719
720 accepted_contexts_.clear();
721
722 for (const auto& pc_rq : rq.presentation_contexts) {
723 // Check if we support the abstract syntax
724 bool supports_abstract = config.supported_abstract_syntaxes.empty() ||
725 std::find(config.supported_abstract_syntaxes.begin(),

728

731 pc_rq.id,
732 pc_rq.abstract_syntax,
733 "",
734 presentation_context_result::abstract_syntax_not_supported
735 });
736 continue;
737 }
738
739 // Find first supported transfer syntax
740 std::string accepted_ts;
741 for (const auto& proposed_ts : pc_rq.transfer_syntaxes) {
742 bool supports_ts = config.supported_transfer_syntaxes.empty() ||
743 std::find(config.supported_transfer_syntaxes.begin(),
744 config.supported_transfer_syntaxes.end(),
745 proposed_ts) != config.supported_transfer_syntaxes.end();
746
747 if (supports_ts) {
748 accepted_ts = proposed_ts;
749 break;
750 }
751 }
752
753 if (accepted_ts.empty()) {
754 accepted_contexts_.push_back(accepted_presentation_context{
755 pc_rq.id,
756 pc_rq.abstract_syntax,
757 "",
758 presentation_context_result::transfer_syntaxes_not_supported
759 });
760 } else {
761 accepted_contexts_.push_back(accepted_presentation_context{
762 pc_rq.id,
763 pc_rq.abstract_syntax,
764 accepted_ts,
765 presentation_context_result::acceptance
766 });
767 }
