84auto thread_pool_adapter::start() -> bool {
85 std::lock_guard<std::mutex> lock(mutex_);
86
87 if (initialized_ && pool_ && pool_->is_running()) {
88 return true; // Already running
89 }
90
91 // Create pool if not exists
92 if (!pool_) {
93 kcenon::thread::thread_context context;
94 pool_ = std::make_shared<kcenon::thread::thread_pool>(config_.pool_name, context);
95 }
96
97 // Create worker threads
98 std::vector<std::unique_ptr<kcenon::thread::thread_worker>> workers;
99 workers.reserve(config_.min_threads);
100
101 kcenon::thread::thread_context context;
102 for (std::size_t i = 0; i < config_.min_threads; ++i) {
103 workers.push_back(std::make_unique<kcenon::thread::thread_worker>(false, context));
104 }
105
106 // Enqueue workers
107 auto enqueue_result = pool_->enqueue_batch(std::move(workers));
108 if (enqueue_result.is_err()) {
109 return false;
110 }
111
112 // Start the pool
113 auto start_result = pool_->start();
114 if (start_result.is_err()) {
115 return false;
116 }
117
118 initialized_ = true;
119 return true;
120}

141auto thread_pool_adapter::submit(std::function<void()> task)
142 -> std::future<void> {
143 auto promise = std::make_shared<std::promise<void>>();
144 auto future = promise->get_future();
145
146 submit_internal(
147 [task = std::move(task), promise]() mutable {
148 try {
149 task();

184void thread_pool_adapter::submit_internal(
185 std::function<void()> task,
186 job_priority /*priority*/) {
187 // Ensure pool is started
188 if (!is_running()) {
189 if (!start()) {
190 throw std::runtime_error("Failed to start thread pool");
191 }
192 }
193
194 std::lock_guard<std::mutex> lock(mutex_);
195 if (pool_) {
196 // Use job_builder for modern API (replaces deprecated future_job)
197 // Note: Priority is not currently used as the base thread_pool
198 // doesn't support it. For future implementation, consider using
199 // typed_thread_pool or a priority queue wrapper.
200 try {
201 auto job = kcenon::thread::job_builder()
202 .name("pacs_pool_task")
203 .work([task = std::move(task)]() -> kcenon::common::VoidResult {
204 task();
205 return kcenon::common::ok();
206 })
207 .build();
208
209 auto result = pool_->enqueue(std::move(job));
210 if (result.is_err()) {
211 throw std::runtime_error("Failed to enqueue task to thread pool");
212 }
213 } catch (const std::exception&) {
214 throw std::runtime_error("Failed to enqueue task to thread pool");
215 }
216 }
217}
