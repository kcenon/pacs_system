64auto generate_temp_filename(const std::filesystem::path& base)
65 -> std::filesystem::path {
66 static std::random_device rd;
67 static std::mt19937 gen(rd());
68 static std::uniform_int_distribution<uint64_t> dist;
69
70 auto temp_name = base.filename().string() + ".tmp." +
71 std::to_string(dist(gen));
72 return base.parent_path() / temp_name;
73}

100auto file_storage::store(const core::dicom_dataset& dataset) -> VoidResult {
101 // Extract required UIDs
102 auto study_uid = dataset.get_string(core::tags::study_instance_uid);
103 auto series_uid = dataset.get_string(core::tags::series_instance_uid);
104 auto sop_uid = dataset.get_string(core::tags::sop_instance_uid);
105
106 if (study_uid.empty() || series_uid.empty() || sop_uid.empty()) {
107 return make_error<std::monostate>(
108 kMissingRequiredUid,
109 "Missing required UID (Study, Series, or SOP Instance UID)",
110 "file_storage");
111 }
112
113 // Build file path based on naming scheme
114 std::filesystem::path file_path;
115 switch (config_.naming) {
116 case naming_scheme::uid_hierarchical:
117 file_path = build_path(study_uid, series_uid, sop_uid);
118 break;
119 case naming_scheme::date_hierarchical: {
120 auto study_date = dataset.get_string(core::tags::study_date);
121 if (study_date.empty()) {
122 // Use current date if study date not available
123 auto now = std::chrono::system_clock::now();
124 auto time = std::chrono::system_clock::to_time_t(now);
125 std::tm tm_buf{};
126#ifdef _WIN32
127 localtime_s(&tm_buf, &time);
128#else
129 localtime_r(&time, &tm_buf);
130#endif
131 char date_str[9];
132 std::strftime(date_str, sizeof(date_str), "%Y%m%d", &tm_buf);
133 study_date = date_str;
134 }
135 file_path = build_date_path(study_date, study_uid, sop_uid);
136 break;
137 }
138 case naming_scheme::flat:
139 file_path = config_.root_path /
140 (sanitize_uid(sop_uid) + config_.file_extension);
141 break;
142 }
143
144 // Handle duplicate checking
145 {
146 std::shared_lock lock(mutex_);
147 if (index_.contains(sop_uid)) {
148 switch (config_.duplicate) {
149 case duplicate_policy::reject:
150 return make_error<std::monostate>(
151 kDuplicateInstance,
152 "Instance already exists: " + sop_uid,
153 "file_storage");
154 case duplicate_policy::ignore:
155 return ok();
156 case duplicate_policy::replace:
157 // Continue to overwrite
158 break;
159 }
160 }
161 }
162
163 // Create directories if needed
164 if (config_.create_directories) {
165 std::error_code ec;
166 std::filesystem::create_directories(file_path.parent_path(), ec);
167 if (ec) {
168 return make_error<std::monostate>(
169 kDirectoryCreateError,
170 "Failed to create directory: " + ec.message(),
171 "file_storage");
172 }
173 }
174
175 // Create DICOM file and write atomically
176 auto dicom_file = core::dicom_file::create(
177 dataset, encoding::transfer_syntax::explicit_vr_little_endian);
178
179 // Write to temporary file first
180 auto temp_path = generate_temp_filename(file_path);
181 auto save_result = dicom_file.save(temp_path);
182
183 if (save_result.is_err()) {
184 std::filesystem::remove(temp_path);
185 return make_error<std::monostate>(
186 kFileWriteError,
187 "Failed to write DICOM file: " + save_result.error().message,
188 "file_storage");
189 }
190
191 // Atomic rename
192 std::error_code ec;
193 std::filesystem::rename(temp_path, file_path, ec);
194 if (ec) {
195 std::filesystem::remove(temp_path);
196 return make_error<std::monostate>(
197 kFileWriteError,
198 "Failed to rename temp file: " + ec.message(),
199 "file_storage");
200 }
201
202 // Update index
203 {
204 std::unique_lock lock(mutex_);
205 index_[sop_uid] = file_path;
206 }
207
208 return ok();
209}

211auto file_storage::retrieve(std::string_view sop_instance_uid)
212 -> Result<core::dicom_dataset> {
213 std::filesystem::path file_path;
214
215 {
216 std::shared_lock lock(mutex_);
217 auto it = index_.find(std::string{sop_instance_uid});
218 if (it == index_.end()) {
219 return make_error<core::dicom_dataset>(
220 kFileNotFound,
221 "Instance not found: " + std::string{sop_instance_uid},
222 "file_storage");
223 }
224 file_path = it->second;
225 }
226
227 // Read DICOM file
228 auto open_result = core::dicom_file::open(file_path);
229 if (open_result.is_err()) {
230 return make_error<core::dicom_dataset>(
231 kFileReadError,
232 "Failed to read DICOM file: " + open_result.error().message,
233 "file_storage");
234 }
235
236 return open_result.value().dataset();
237}

239auto file_storage::remove(std::string_view sop_instance_uid) -> VoidResult {
240 std::filesystem::path file_path;
241
242 {
243 std::unique_lock lock(mutex_);
244 auto it = index_.find(std::string{sop_instance_uid});
245 if (it == index_.end()) {
246 // Not found is not an error for remove
247 return ok();
248 }
249 file_path = it->second;
250 index_.erase(it);
251 }
252
253 // Delete the file
254 std::error_code ec;
255 std::filesystem::remove(file_path, ec);
256 // Ignore errors - file might have been deleted externally
257
258 // Try to clean up empty parent directories
259 auto parent = file_path.parent_path();
260 while (parent != config_.root_path) {
261 if (std::filesystem::is_empty(parent)) {
262 std::filesystem::remove(parent, ec);
263 parent = parent.parent_path();
264 } else {
265 break;
266 }
267 }
268
269 return ok();
270}

277auto file_storage::find(const core::dicom_dataset& query)
278 -> Result<std::vector<core::dicom_dataset>> {
279 std::vector<core::dicom_dataset> results;
280
281 std::vector<std::filesystem::path> paths_to_check;
282 {
283 std::shared_lock lock(mutex_);
284 paths_to_check.reserve(index_.size());
285 for (const auto& [uid, path] : index_) {
286 paths_to_check.push_back(path);
287 }
288 }
289
290 for (const auto& path : paths_to_check) {
291 auto open_result = core::dicom_file::open(path);
292 if (open_result.is_err()) {
293 continue; // Skip files that can't be read
294 }
295
296 const auto& dataset = open_result.value().dataset();
297 if (matches_query(dataset, query)) {
298 results.push_back(dataset);
299 }
300 }
301
302 return results;
303}

305auto file_storage::get_statistics() const -> storage_statistics {
306 storage_statistics stats;
307
308 std::set<std::string> studies;
309 std::set<std::string> series;
310 std::set<std::string> patients;
311
312 std::vector<std::filesystem::path> paths;
313 {
314 std::shared_lock lock(mutex_);
315 stats.total_instances = index_.size();
316 paths.reserve(index_.size());
317 for (const auto& [uid, path] : index_) {
318 paths.push_back(path);
319 }
320 }
321
322 for (const auto& path : paths) {
323 std::error_code ec;
324 stats.total_bytes += std::filesystem::file_size(path, ec);
325
326 // Read file to get study/series/patient info
327 auto open_result = core::dicom_file::open(path);
328 if (open_result.is_ok()) {
329 const auto& ds = open_result.value().dataset();
330 auto study_uid = ds.get_string(core::tags::study_instance_uid);
331 auto series_uid = ds.get_string(core::tags::series_instance_uid);
332 auto patient_id = ds.get_string(core::tags::patient_id);
333
334 if (!study_uid.empty()) {
335 studies.insert(study_uid);
336 }
337 if (!series_uid.empty()) {
338 series.insert(series_uid);
339 }
340 if (!patient_id.empty()) {
341 patients.insert(patient_id);
342 }
343 }
344 }
345
346 stats.studies_count = studies.size();
347 stats.series_count = series.size();
348 stats.patients_count = patients.size();
349
350 return stats;
351}

353auto file_storage::verify_integrity() -> VoidResult {
354 std::vector<std::pair<std::string, std::filesystem::path>> entries;
355 {
356 std::shared_lock lock(mutex_);
357 entries.reserve(index_.size());
358 for (const auto& [uid, path] : index_) {
359 entries.emplace_back(uid, path);
360 }
361 }
362
363 std::vector<std::string> invalid_entries;
364
365 for (const auto& [uid, path] : entries) {
366 if (!std::filesystem::exists(path)) {
367 invalid_entries.push_back(uid + " (file missing)");
368 continue;
369 }
370
371 auto open_result = core::dicom_file::open(path);
372 if (open_result.is_err()) {
373 invalid_entries.push_back(uid + " (invalid DICOM)");
374 continue;
375 }
376
377 // Verify UID matches
378 auto file_uid =
379 open_result.value().dataset().get_string(core::tags::sop_instance_uid);
380 if (file_uid != uid) {
381 invalid_entries.push_back(uid + " (UID mismatch)");
382 }
383 }
384
385 if (!invalid_entries.empty()) {
386 std::string message = "Integrity check failed for " +
387 std::to_string(invalid_entries.size()) +
388 " entries";
389 return make_error<std::monostate>(kIntegrityError, message,
390 "file_storage");
391 }
392
393 return ok();
394}

410auto file_storage::import_directory(const std::filesystem::path& source)
411 -> VoidResult {
412 if (!std::filesystem::exists(source)) {
413 return make_error<std::monostate>(
414 kFileNotFound,
415 "Source directory does not exist: " + source.string(),
416 "file_storage");
417 }
418
419 std::error_code ec;
420 for (const auto& entry :
421 std::filesystem::recursive_directory_iterator(source, ec)) {
422 if (!entry.is_regular_file()) {
423 continue;
424 }
425
426 // Try to open as DICOM file
427 auto open_result = core::dicom_file::open(entry.path());
428 if (open_result.is_err()) {
429 continue; // Not a DICOM file, skip
430 }
431
432 // Store the dataset
433 auto store_result = store(open_result.value().dataset());
434 if (store_result.is_err()) {
435 // Log error but continue with other files
436 // In production, this should be logged
437 }
438 }
439
440 return ok();
441}

447auto file_storage::rebuild_index() -> VoidResult {
448 std::unique_lock lock(mutex_);
449 index_.clear();
450
451 if (!std::filesystem::exists(config_.root_path)) {
452 return ok();
453 }
454
455 std::error_code ec;
456 for (const auto& entry :
457 std::filesystem::recursive_directory_iterator(config_.root_path, ec)) {
458 if (!entry.is_regular_file()) {
459 continue;
460 }
461
462 // Check file extension
463 if (!config_.file_extension.empty() &&
464 entry.path().extension() != config_.file_extension) {
465 continue;
466 }
467
468 // Try to read as DICOM file
469 auto open_result = core::dicom_file::open(entry.path());
470 if (open_result.is_err()) {
471 continue;
472 }
473
474 auto sop_uid =
475 open_result.value().dataset().get_string(core::tags::sop_instance_uid);
476 if (!sop_uid.empty()) {
477 index_[sop_uid] = entry.path();
478 }
479 }
480
481 return ok();
482}

497auto file_storage::build_date_path(std::string_view study_date,
498 std::string_view study_uid,
499 std::string_view sop_uid) const
500 -> std::filesystem::path {
501 // Parse date YYYYMMDD
502 std::string year = "unknown";
503 std::string month = "01";
504 std::string day = "01";
505
506 if (study_date.length() >= 8) {
507 year = std::string{study_date.substr(0, 4)};
508 month = std::string{study_date.substr(4, 2)};
509 day = std::string{study_date.substr(6, 2)};
510 }
511
512 return config_.root_path / year / month / day / sanitize_uid(study_uid) /
513 (sanitize_uid(sop_uid) + config_.file_extension);

527auto file_storage::matches_query(const core::dicom_dataset& dataset,
528 const core::dicom_dataset& query) -> bool {
529 // If query is empty, match all
530 if (query.empty()) {
531 return true;
532 }
533
534 // Check each query element
535 for (const auto& [tag, element] : query) {
536 auto query_value = element.as_string().unwrap_or("");
537 if (query_value.empty()) {
538 continue; // Empty value acts as wildcard
539 }
540
541 auto dataset_value = dataset.get_string(tag);
542
543 // Support basic wildcard matching (* and ?)
544 if (query_value.find('*') != std::string::npos ||
545 query_value.find('?') != std::string::npos) {
546 // Convert to regex-like matching
547 std::string pattern = query_value;
548 // Escape regex special characters except * and ?
549 std::string escaped;
550 for (char c : pattern) {
551 if (c == '*') {
552 escaped += ".*";
553 } else if (c == '?') {
554 escaped += ".";
555 } else if (c == '.' || c == '[' || c == ']' || c == '(' ||
556 c == ')' || c == '+' || c == '^' || c == '$' ||
557 c == '|' || c == '\\') {
558 escaped += '\\';
559 escaped += c;
560 } else {
561 escaped += c;
562 }
563 }
564
565 // Simple pattern matching (without full regex for performance)
566 // For now, just check if pattern starts with or ends with value
567 if (query_value.front() == '*' && query_value.back() == '*') {
568 // Contains
569 auto inner =
570 query_value.substr(1, query_value.length() - 2);
571 if (dataset_value.find(inner) == std::string::npos) {
572 return false;
573 }
574 } else if (query_value.front() == '*') {
575 // Ends with
576 auto suffix = query_value.substr(1);
577 if (dataset_value.length() < suffix.length() ||
578 dataset_value.substr(dataset_value.length() -
579 suffix.length()) != suffix) {
580 return false;
581 }
582 } else if (query_value.back() == '*') {
583 // Starts with
584 auto prefix = query_value.substr(0, query_value.length() - 1);
585 if (dataset_value.substr(0, prefix.length()) != prefix) {
586 return false;
587 }
588 }
589 } else {
590 // Exact match
591 if (dataset_value != query_value) {
592 return false;
593 }
594 }
595 }
596
597 return true;
598}

600auto file_storage::sanitize_uid(std::string_view uid) -> std::string {
601 std::string result;
602 result.reserve(uid.length());
603
604 for (char c : uid) {
605 // UIDs contain digits and dots, which are safe for filesystems
606 // Replace any other characters with underscore
607 if (std::isalnum(static_cast<unsigned char>(c)) || c == '.') {
608 result += c;
609 } else {
610 result += '_';
611 }
612 }
613
614 return result;
615}
