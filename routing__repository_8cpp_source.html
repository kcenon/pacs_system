65[[nodiscard]] std::string escape_json_string(const std::string& str) {
66 std::ostringstream oss;
67 for (char c : str) {
68 switch (c) {
69 case '"': oss << "\\\""; break;
70 case '\\': oss << "\\\\"; break;
71 case '\b': oss << "\\b"; break;
72 case '\f': oss << "\\f"; break;
73 case '\n': oss << "\\n"; break;
74 case '\r': oss << "\\r"; break;
75 case '\t': oss << "\\t"; break;
76 default: oss << c; break;
77 }
78 }
79 return oss.str();
80}

83[[nodiscard]] std::string unescape_json_string(std::string_view str) {
84 std::string result;
85 result.reserve(str.size());
86 for (size_t i = 0; i < str.size(); ++i) {
87 if (str[i] == '\\' && i + 1 < str.size()) {
88 ++i;
89 switch (str[i]) {
90 case '"': result += '"'; break;
91 case '\\': result += '\\'; break;
92 case 'b': result += '\b'; break;
93 case 'f': result += '\f'; break;
94 case 'n': result += '\n'; break;
95 case 'r': result += '\r'; break;
96 case 't': result += '\t'; break;
97 default: result += str[i]; break;
98 }
99 } else {
100 result += str[i];
101 }
102 }
103 return result;
104}

107[[nodiscard]] std::pair<std::string, size_t> extract_json_string(
108 std::string_view json, size_t pos) {
109 auto start = json.find('"', pos);
110 if (start == std::string_view::npos) return {"", std::string_view::npos};
111
112 size_t end = start + 1;
113 while (end < json.size()) {
114 if (json[end] == '\\' && end + 1 < json.size()) {
115 end += 2;
116 } else if (json[end] == '"') {
117 break;
118 } else {
119 ++end;
120 }
121 }
122
123 if (end >= json.size()) return {"", std::string_view::npos};
124
125 auto value = unescape_json_string(json.substr(start + 1, end - start - 1));
126 return {value, end + 1};
127}

131std::string routing_repository::serialize_conditions(
132 const std::vector<client::routing_condition>& conditions) {
133 if (conditions.empty()) return "[]";
134
135 std::ostringstream oss;
136 oss << "[";
137 for (size_t i = 0; i < conditions.size(); ++i) {
138 if (i > 0) oss << ",";
139 const auto& cond = conditions[i];
140 oss << "{";
141 oss << "\"field\":\"" << client::to_string(cond.match_field) << "\",";
142 oss << "\"pattern\":\"" << escape_json_string(cond.pattern) << "\",";
143 oss << "\"case_sensitive\":" << (cond.case_sensitive ? "true" : "false") << ",";
144 oss << "\"negate\":" << (cond.negate ? "true" : "false");
145 oss << "}";
146 }
147 oss << "]";
148 return oss.str();
149}

151std::vector<client::routing_condition> routing_repository::deserialize_conditions(
152 std::string_view json) {
153 std::vector<client::routing_condition> result;
154 if (json.empty() || json == "[]") return result;
155
156 size_t pos = 0;
157 while (pos < json.size()) {
158 auto obj_start = json.find('{', pos);
159 if (obj_start == std::string_view::npos) break;
160
161 auto obj_end = json.find('}', obj_start);
162 if (obj_end == std::string_view::npos) break;
163
164 auto obj = json.substr(obj_start, obj_end - obj_start + 1);
165
166 client::routing_condition cond;
167
168 // Parse field
169 auto field_pos = obj.find("\"field\"");
170 if (field_pos != std::string_view::npos) {
171 auto [field_value, next] = extract_json_string(obj, field_pos + 7);
172 cond.match_field = client::routing_field_from_string(field_value);
173 }
174
175 // Parse pattern
176 auto pattern_pos = obj.find("\"pattern\"");
177 if (pattern_pos != std::string_view::npos) {
178 auto [pattern_value, next] = extract_json_string(obj, pattern_pos + 9);
179 cond.pattern = pattern_value;
180 }
181
182 // Parse case_sensitive
183 auto case_pos = obj.find("\"case_sensitive\"");
184 if (case_pos != std::string_view::npos) {
185 cond.case_sensitive = (obj.find("true", case_pos) != std::string_view::npos &&
186 obj.find("true", case_pos) < obj.find(',', case_pos));
187 }
188
189 // Parse negate
190 auto negate_pos = obj.find("\"negate\"");
191 if (negate_pos != std::string_view::npos) {
192 cond.negate = (obj.find("true", negate_pos) != std::string_view::npos);
193 }
194
195 result.push_back(std::move(cond));
196 pos = obj_end + 1;
197 }
198
199 return result;
200}

223std::vector<client::routing_action> routing_repository::deserialize_actions(
224 std::string_view json) {
225 std::vector<client::routing_action> result;
226 if (json.empty() || json == "[]") return result;
227
228 size_t pos = 0;
229 while (pos < json.size()) {
230 auto obj_start = json.find('{', pos);
231 if (obj_start == std::string_view::npos) break;
232
233 auto obj_end = json.find('}', obj_start);
234 if (obj_end == std::string_view::npos) break;
235
236 auto obj = json.substr(obj_start, obj_end - obj_start + 1);
237
238 client::routing_action action;
239
240 // Parse destination
241 auto dest_pos = obj.find("\"destination\"");
242 if (dest_pos != std::string_view::npos) {
243 auto [dest_value, next] = extract_json_string(obj, dest_pos + 13);
244 action.destination_node_id = dest_value;
245 }
246
247 // Parse priority
248 auto prio_pos = obj.find("\"priority\"");
249 if (prio_pos != std::string_view::npos) {
250 auto [prio_value, next] = extract_json_string(obj, prio_pos + 10);
251 action.priority = client::job_priority_from_string(prio_value);
252 }
253
254 // Parse delay_minutes
255 auto delay_pos = obj.find("\"delay_minutes\"");
256 if (delay_pos != std::string_view::npos) {
257 auto colon = obj.find(':', delay_pos);
258 if (colon != std::string_view::npos) {
259 int minutes = 0;
260 std::sscanf(obj.data() + colon + 1, "%d", &minutes);
261 action.delay = std::chrono::minutes{minutes};
262 }
263 }
264
265 // Parse delete_after_send
266 auto delete_pos = obj.find("\"delete_after_send\"");
267 if (delete_pos != std::string_view::npos) {
268 action.delete_after_send = (obj.find("true", delete_pos) != std::string_view::npos &&
269 obj.find("true", delete_pos) < obj.find(',', delete_pos + 20));
270 }
271
272 // Parse notify_on_failure
273 auto notify_pos = obj.find("\"notify_on_failure\"");
274 if (notify_pos != std::string_view::npos) {
275 action.notify_on_failure = (obj.find("true", notify_pos) != std::string_view::npos);
276 }
277
278 result.push_back(std::move(action));
279 pos = obj_end + 1;
280 }
281
282 return result;
283}

289auto routing_repository::parse_timestamp(const std::string& str) const
290 -> std::chrono::system_clock::time_point {
291 if (str.empty()) {
292 return {};
293 }
294
295 std::tm tm{};
296 std::istringstream ss(str);
297 ss >> std::get_time(&tm, "%Y-%m-%d %H:%M:%S");
298 if (ss.fail()) {
299 return {};
300 }
301
302#ifdef _WIN32
303 auto time = _mkgmtime(&tm);
304#else
305 auto time = timegm(&tm);
306#endif
307
308 return std::chrono::system_clock::from_time_t(time);
309}
310
311auto routing_repository::format_timestamp(
312 std::chrono::system_clock::time_point tp) const -> std::string {
313 if (tp == std::chrono::system_clock::time_point{}) {
314 return "";
315 }
316
317 auto time = std::chrono::system_clock::to_time_t(tp);
318 std::tm tm{};
319#ifdef _WIN32
320 gmtime_s(&tm, &time);
321#else
322 gmtime_r(&time, &tm);
323#endif
324
325 char buf[32];
326 std::strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M:%S", &tm);
327 return buf;
328}

330auto routing_repository::format_optional_timestamp(
331 const std::optional<std::chrono::system_clock::time_point>& tp) const
332 -> std::string {
333 if (!tp.has_value()) {
334 return "";
335 }
336 return format_timestamp(tp.value());

368auto routing_repository::find_rules(const routing_rule_query_options& options)
369 -> list_result_type {
370 if (!db() || !db()->is_connected()) {
371 return list_result_type(kcenon::common::error_info{
372 -1, "Database not connected", "storage"});
373 }
374
375 auto builder = query_builder();
376 builder.select(select_columns()).from(table_name());
377
378 if (options.enabled_only.has_value()) {
379 builder.where("enabled", "=", options.enabled_only.value() ? 1 : 0);
380 }
381
382 if (options.order_by_priority) {
383 builder.order_by("priority", database::sort_order::desc);
384 builder.order_by("created_at", database::sort_order::asc);
385 } else {
386 builder.order_by("created_at", database::sort_order::desc);
387 }
388
389 builder.limit(options.limit).offset(options.offset);
390
391 auto result = storage_session().select(builder.build());
392 if (result.is_err()) {
393 return list_result_type(result.error());
394 }
395
396 std::vector<client::routing_rule> rules;
397 rules.reserve(result.value().size());
398 for (const auto& row : result.value()) {
399 rules.push_back(map_row_to_entity(row));
400 }
401
402 return list_result_type(std::move(rules));
403}

559auto routing_repository::count_enabled() -> Result<size_t> {
560 if (!db() || !db()->is_connected()) {
561 return Result<size_t>(kcenon::common::error_info{
562 -1, "Database not connected", "storage"});
563 }
564

567 .from(table_name())

571 if (result.is_err()) {
572 return Result<size_t>(result.error());

574

576 return Result<size_t>(static_cast<size_t>(0));
577 }
578
579 const auto& row = result.value()[0];
580 auto it = row.find("COUNT(*)");
581 if (it == row.end()) {
582 it = row.begin();
583 }
584
585 if (it != row.end() && !it->second.empty()) {
586 return Result<size_t>(static_cast<size_t>(std::stoull(it->second)));
587 }
588
589 return Result<size_t>(static_cast<size_t>(0));
590}

596auto routing_repository::map_row_to_entity(const database_row& row) const
597 -> client::routing_rule {
598 client::routing_rule rule;
599
600 // Parse pk if present
601 auto pk_it = row.find("pk");
602 if (pk_it != row.end() && !pk_it->second.empty()) {
603 rule.pk = std::stoll(pk_it->second);
604 }
605
606 rule.rule_id = row.at("rule_id");
607 rule.name = row.at("name");
608
609 auto desc_it = row.find("description");
610 if (desc_it != row.end()) {
611 rule.description = desc_it->second;
612 }
613
614 auto enabled_it = row.find("enabled");
615 if (enabled_it != row.end() && !enabled_it->second.empty()) {
616 rule.enabled = (std::stoi(enabled_it->second) != 0);
617 }
618
619 auto priority_it = row.find("priority");
620 if (priority_it != row.end() && !priority_it->second.empty()) {
621 rule.priority = std::stoi(priority_it->second);
622 }
623
624 auto conditions_it = row.find("conditions_json");
625 if (conditions_it != row.end() && !conditions_it->second.empty()) {
626 rule.conditions = deserialize_conditions(conditions_it->second);
627 }
628
629 auto actions_it = row.find("actions_json");
630 if (actions_it != row.end() && !actions_it->second.empty()) {
631 rule.actions = deserialize_actions(actions_it->second);
632 }
633
634 auto schedule_it = row.find("schedule_cron");
635 if (schedule_it != row.end() && !schedule_it->second.empty()) {
636 rule.schedule_cron = schedule_it->second;
637 }
638
639 auto eff_from_it = row.find("effective_from");
640 if (eff_from_it != row.end() && !eff_from_it->second.empty()) {
641 auto tp = parse_timestamp(eff_from_it->second);
642 if (tp != std::chrono::system_clock::time_point{}) {
643 rule.effective_from = tp;
644 }
645 }
646
647 auto eff_until_it = row.find("effective_until");
648 if (eff_until_it != row.end() && !eff_until_it->second.empty()) {
649 auto tp = parse_timestamp(eff_until_it->second);
650 if (tp != std::chrono::system_clock::time_point{}) {
651 rule.effective_until = tp;
652 }
653 }
654
655 auto triggered_it = row.find("triggered_count");
656 if (triggered_it != row.end() && !triggered_it->second.empty()) {
657 rule.triggered_count = static_cast<size_t>(std::stoll(triggered_it->second));
658 }
659
660 auto success_it = row.find("success_count");
661 if (success_it != row.end() && !success_it->second.empty()) {
662 rule.success_count = static_cast<size_t>(std::stoll(success_it->second));
663 }
664
665 auto failure_it = row.find("failure_count");
666 if (failure_it != row.end() && !failure_it->second.empty()) {
667 rule.failure_count = static_cast<size_t>(std::stoll(failure_it->second));
668 }
669
670 auto last_triggered_it = row.find("last_triggered");
671 if (last_triggered_it != row.end() && !last_triggered_it->second.empty()) {
672 rule.last_triggered = parse_timestamp(last_triggered_it->second);
673 }
674
675 auto created_it = row.find("created_at");
676 if (created_it != row.end() && !created_it->second.empty()) {
677 rule.created_at = parse_timestamp(created_it->second);
678 }
679
680 auto updated_it = row.find("updated_at");
681 if (updated_it != row.end() && !updated_it->second.empty()) {
682 rule.updated_at = parse_timestamp(updated_it->second);
683 }
684
685 return rule;
686}

688auto routing_repository::entity_to_row(const client::routing_rule& entity) const
689 -> std::map<std::string, database_value> {
690 std::map<std::string, database_value> row;
691
692 row["rule_id"] = entity.rule_id;
693 row["name"] = entity.name;
694 row["description"] = entity.description;
695 row["enabled"] = static_cast<int64_t>(entity.enabled ? 1 : 0);
696 row["priority"] = static_cast<int64_t>(entity.priority);
697 row["conditions_json"] = serialize_conditions(entity.conditions);
698 row["actions_json"] = serialize_actions(entity.actions);
699 row["schedule_cron"] = entity.schedule_cron.value_or("");
700 row["effective_from"] = format_optional_timestamp(entity.effective_from);
701 row["effective_until"] = format_optional_timestamp(entity.effective_until);
702 row["triggered_count"] = static_cast<int64_t>(entity.triggered_count);
703 row["success_count"] = static_cast<int64_t>(entity.success_count);

855[[nodiscard]] std::string escape_json_string(const std::string& str) {
856 std::ostringstream oss;
857 for (char c : str) {
858 switch (c) {
859 case '"': oss << "\\\""; break;
860 case '\\': oss << "\\\\"; break;
861 case '\b': oss << "\\b"; break;
862 case '\f': oss << "\\f"; break;
863 case '\n': oss << "\\n"; break;
864 case '\r': oss << "\\r"; break;
865 case '\t': oss << "\\t"; break;
866 default: oss << c; break;
867 }
868 }
869 return oss.str();

873[[nodiscard]] std::string unescape_json_string(std::string_view str) {
874 std::string result;
875 result.reserve(str.size());
876 for (size_t i = 0; i < str.size(); ++i) {
877 if (str[i] == '\\' && i + 1 < str.size()) {
878 ++i;
879 switch (str[i]) {
880 case '"': result += '"'; break;
881 case '\\': result += '\\'; break;
882 case 'b': result += '\b'; break;
883 case 'f': result += '\f'; break;
884 case 'n': result += '\n'; break;
885 case 'r': result += '\r'; break;
886 case 't': result += '\t'; break;
887 default: result += str[i]; break;
888 }
889 } else {
890 result += str[i];
891 }
892 }
893 return result;
894}

897[[nodiscard]] std::pair<std::string, size_t> extract_json_string(
898 std::string_view json, size_t pos) {
899 auto start = json.find('"', pos);
900 if (start == std::string_view::npos) return {"", std::string_view::npos};
901
902 size_t end = start + 1;
903 while (end < json.size()) {
904 if (json[end] == '\\' && end + 1 < json.size()) {
905 end += 2;
906 } else if (json[end] == '"') {
907 break;
908 } else {
909 ++end;
910 }
911 }
912
913 if (end >= json.size()) return {"", std::string_view::npos};
914
915 auto value = unescape_json_string(json.substr(start + 1, end - start - 1));
916 return {value, end + 1};
917}

945std::vector<client::routing_condition> routing_repository::deserialize_conditions(
946 std::string_view json) {
947 std::vector<client::routing_condition> result;
948 if (json.empty() || json == "[]") return result;
949
950 size_t pos = 0;
951 while (pos < json.size()) {
952 auto obj_start = json.find('{', pos);
953 if (obj_start == std::string_view::npos) break;
954
955 auto obj_end = json.find('}', obj_start);
956 if (obj_end == std::string_view::npos) break;
957
958 auto obj = json.substr(obj_start, obj_end - obj_start + 1);
959
960 client::routing_condition cond;
961
962 // Parse field
963 auto field_pos = obj.find("\"field\"");
964 if (field_pos != std::string_view::npos) {
965 auto [field_value, next] = extract_json_string(obj, field_pos + 7);
966 cond.match_field = client::routing_field_from_string(field_value);
967 }
968
969 // Parse pattern
970 auto pattern_pos = obj.find("\"pattern\"");
971 if (pattern_pos != std::string_view::npos) {
972 auto [pattern_value, next] = extract_json_string(obj, pattern_pos + 9);
973 cond.pattern = pattern_value;
974 }
975
976 // Parse case_sensitive
977 auto case_pos = obj.find("\"case_sensitive\"");
978 if (case_pos != std::string_view::npos) {
979 cond.case_sensitive = (obj.find("true", case_pos) != std::string_view::npos &&
980 obj.find("true", case_pos) < obj.find(',', case_pos));
981 }
982
983 // Parse negate
984 auto negate_pos = obj.find("\"negate\"");
985 if (negate_pos != std::string_view::npos) {
986 cond.negate = (obj.find("true", negate_pos) != std::string_view::npos);
987 }
988
989 result.push_back(std::move(cond));
990 pos = obj_end + 1;
991 }
992
993 return result;

1100 if (!db_) {
1101 return VoidResult(kcenon::common::error_info{
1102 -1, "Database not initialized", "routing_repository"});
1103 }
1104
1105 static constexpr const char* sql = R"(
1106 INSERT INTO routing_rules (
1107 rule_id, name, description, enabled, priority,
1108 conditions_json, actions_json,
1109 schedule_cron, effective_from, effective_until,
1110 triggered_count, success_count, failure_count,
1111 last_triggered, created_at, updated_at
1112 ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
1113 ON CONFLICT(rule_id) DO UPDATE SET
1114 name = excluded.name,
1115 description = excluded.description,
1116 enabled = excluded.enabled,
1117 priority = excluded.priority,
1118 conditions_json = excluded.conditions_json,
1119 actions_json = excluded.actions_json,
1120 schedule_cron = excluded.schedule_cron,
1121 effective_from = excluded.effective_from,
1122 effective_until = excluded.effective_until,
1123 updated_at = CURRENT_TIMESTAMP
1124 )";
1125
1126 sqlite3_stmt* stmt = nullptr;

1128 return VoidResult(kcenon::common::error_info{
1129 -1, "Failed to prepare statement: " + std::string(sqlite3_errmsg(db_)),
1130 "routing_repository"});
1131 }
1132
1133 int idx = 1;
1134 sqlite3_bind_text(stmt, idx++, rule.rule_id.c_str(), -1, SQLITE_TRANSIENT);
1135 sqlite3_bind_text(stmt, idx++, rule.name.c_str(), -1, SQLITE_TRANSIENT);
1136 sqlite3_bind_text(stmt, idx++, rule.description.c_str(), -1, SQLITE_TRANSIENT);
1137 sqlite3_bind_int(stmt, idx++, rule.enabled ? 1 : 0);

1139

1141 sqlite3_bind_text(stmt, idx++, conditions_json.c_str(), -1, SQLITE_TRANSIENT);

1145

1159 }
1160

1162 if (created_str.empty()) {
1163 sqlite3_bind_text(stmt, idx++, "CURRENT_TIMESTAMP", -1, SQLITE_STATIC);

1165 sqlite3_bind_text(stmt, idx++, created_str.c_str(), -1, SQLITE_TRANSIENT);
1166 }
1167
1168 auto updated_str = to_timestamp_string(rule.updated_at);
1169 if (updated_str.empty()) {
1170 sqlite3_bind_text(stmt, idx++, "CURRENT_TIMESTAMP", -1, SQLITE_STATIC);

1172 sqlite3_bind_text(stmt, idx++, updated_str.c_str(), -1, SQLITE_TRANSIENT);
1173 }
1174
1175 auto rc = sqlite3_step(stmt);
1176 sqlite3_finalize(stmt);
1177
1178 if (rc != SQLITE_DONE) {
1179 return VoidResult(kcenon::common::error_info{
1180 -1, "Failed to save rule: " + std::string(sqlite3_errmsg(db_)),
1181 "routing_repository"});
