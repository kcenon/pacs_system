89class MockStorage : public IDicomStorage {
90public:
91 MockStorage() = default;
92 ~MockStorage() override = default;
93
94 // =========================================================================
95 // storage_interface Implementation
96 // =========================================================================
97
98 [[nodiscard]] storage::VoidResult store(const core::dicom_dataset& dataset) override {
99 std::lock_guard<std::mutex> lock(mutex_);
100
101 auto uid = dataset.get_string(core::tags::sop_instance_uid);
102 if (uid.empty()) {
103 return kcenon::common::make_error<std::monostate>(
104 -1, "Dataset missing SOP Instance UID", "MockStorage");
105 }
106
107 datasets_[uid] = dataset;
108 store_calls_.push_back(uid);
109
110 if (on_store_) {
111 on_store_(dataset);
112 }
113
114 return storage::VoidResult::ok({});
115 }
116
117 [[nodiscard]] storage::Result<core::dicom_dataset>
118 retrieve(std::string_view sop_instance_uid) override {
119 std::lock_guard<std::mutex> lock(mutex_);
120
121 auto it = datasets_.find(std::string(sop_instance_uid));
122 if (it == datasets_.end()) {
123 return kcenon::common::make_error<core::dicom_dataset>(
124 -1, "Instance not found", "MockStorage");
125 }
126
127 return storage::Result<core::dicom_dataset>::ok(it->second);
128 }
129
130 [[nodiscard]] storage::VoidResult remove(std::string_view sop_instance_uid) override {
131 std::lock_guard<std::mutex> lock(mutex_);
132 datasets_.erase(std::string(sop_instance_uid));
133 return storage::VoidResult::ok({});
134 }
135
136 [[nodiscard]] bool exists(std::string_view sop_instance_uid) const override {
137 std::lock_guard<std::mutex> lock(mutex_);
138 return datasets_.find(std::string(sop_instance_uid)) != datasets_.end();
139 }
140
141 [[nodiscard]] storage::Result<std::vector<core::dicom_dataset>>
142 find(const core::dicom_dataset& /*query*/) override {
143 std::lock_guard<std::mutex> lock(mutex_);
144
145 std::vector<core::dicom_dataset> results;
146 for (const auto& [uid, dataset] : datasets_) {
147 results.push_back(dataset);
148 }
149 return storage::Result<std::vector<core::dicom_dataset>>::ok(std::move(results));
150 }
151
152 [[nodiscard]] storage::storage_statistics get_statistics() const override {
153 std::lock_guard<std::mutex> lock(mutex_);
154
155 storage::storage_statistics stats;
156 stats.total_instances = datasets_.size();
157 return stats;
158 }
159
160 [[nodiscard]] storage::VoidResult verify_integrity() override {
161 return storage::VoidResult::ok({});
162 }
163
164 // =========================================================================
165 // Test Utilities
166 // =========================================================================
167
169 [[nodiscard]] std::size_t store_count() const {
170 std::lock_guard<std::mutex> lock(mutex_);
171 return store_calls_.size();
172 }
173
175 [[nodiscard]] std::vector<std::string> stored_uids() const {
176 std::lock_guard<std::mutex> lock(mutex_);
177 return store_calls_;
178 }
179
181 void clear() {
182 std::lock_guard<std::mutex> lock(mutex_);
183 datasets_.clear();
184 store_calls_.clear();
185 }
186
188 void on_store(std::function<void(const core::dicom_dataset&)> callback) {
189 on_store_ = std::move(callback);
190 }
191
192private:
193 mutable std::mutex mutex_;
194 std::map<std::string, core::dicom_dataset> datasets_;
195 std::vector<std::string> store_calls_;
196 std::function<void(const core::dicom_dataset&)> on_store_;
197};

211class MockNetwork : public IDicomNetwork {
212public:
213 MockNetwork() = default;
214 ~MockNetwork() override = default;
215
216 [[nodiscard]] std::unique_ptr<network::dicom_server> create_server(
217 const network::server_config& config,
218 const integration::tls_config& /*tls_cfg*/) override {
219 std::lock_guard<std::mutex> lock(mutex_);
220 server_configs_.push_back(config);
221
222 // Return nullptr for mock - actual server creation not needed in tests
223 return nullptr;
224 }
225
226 [[nodiscard]] integration::Result<integration::network_adapter::session_ptr>
227 connect(const integration::connection_config& config) override {
228 std::lock_guard<std::mutex> lock(mutex_);
229 connection_attempts_.push_back({config.host, config.port});
230
231 if (fail_connections_) {
232 return integration::error_info{"Connection refused (mock)"};
233 }
234
235 // Return nullptr session for mock
236 return integration::network_adapter::session_ptr{nullptr};
237 }
238
239 [[nodiscard]] integration::Result<integration::network_adapter::session_ptr>
240 connect(const std::string& host, uint16_t port,
241 std::chrono::milliseconds /*timeout*/) override {
242 integration::connection_config config;
243 config.host = host;
244 config.port = port;
245 return connect(config);
246 }
247
248 // =========================================================================
249 // Test Utilities
250 // =========================================================================
251
253 [[nodiscard]] std::size_t connection_attempt_count() const {
254 std::lock_guard<std::mutex> lock(mutex_);
255 return connection_attempts_.size();
256 }
257
259 [[nodiscard]] std::size_t server_creation_count() const {
260 std::lock_guard<std::mutex> lock(mutex_);
261 return server_configs_.size();
262 }
263
265 void set_fail_connections(bool fail) {
266 std::lock_guard<std::mutex> lock(mutex_);
267 fail_connections_ = fail;
268 }
269
271 void clear() {
272 std::lock_guard<std::mutex> lock(mutex_);
273 connection_attempts_.clear();
274 server_configs_.clear();
275 }
276
277private:
278 mutable std::mutex mutex_;
279 std::vector<std::pair<std::string, uint16_t>> connection_attempts_;
280 std::vector<network::server_config> server_configs_;
281 bool fail_connections_ = false;
282};
