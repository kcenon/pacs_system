63auto receive_network_io_job::execute(pipeline_coordinator& coordinator)
64 -> VoidResult {
65
66 // Validate we have data
67 if (data_.empty()) {
68 if (on_error_) {
69 on_error_(context_.session_id, "Empty data received");
70 }
71 return pacs_void_error(error_codes::invalid_argument,
72 "Empty data received");
73 }
74
75 // Invoke callback if set
76 if (on_data_) {
77 on_data_(context_.session_id, data_);
78 }
79
80 // Generate job ID and update context
81 context_.job_id = coordinator.generate_job_id();
82
83 // Create decode job and submit to next stage
84 auto decode_job = std::make_unique<pdu_decode_job>(
85 context_.session_id,
86 std::move(data_)
87 );
88
89 // Copy context to decode job
90 decode_job->get_context() = context_;
91 decode_job->get_context().stage = pipeline_stage::pdu_decode;
92
93 return coordinator.submit_to_stage(
94 pipeline_stage::pdu_decode,
95 std::move(decode_job)
96 );
97}
