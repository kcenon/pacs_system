100[[nodiscard]] constexpr std::string_view to_string(query_level level) noexcept {
101 switch (level) {
102 case query_level::patient: return "PATIENT";
103 case query_level::study: return "STUDY";
104 case query_level::series: return "SERIES";
105 case query_level::image: return "IMAGE";
106 default: return "UNKNOWN";
107 }
108}

115[[nodiscard]] inline std::optional<query_level> parse_query_level(
116 std::string_view level_str) noexcept {
117 if (level_str == "PATIENT") return query_level::patient;
118 if (level_str == "STUDY") return query_level::study;
119 if (level_str == "SERIES") return query_level::series;
120 if (level_str == "IMAGE") return query_level::image;
121 return std::nullopt;
122}
