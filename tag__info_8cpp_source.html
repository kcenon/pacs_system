42auto value_multiplicity::from_string(std::string_view str)
43 -> std::optional<value_multiplicity> {
44 if (str.empty()) {
45 return std::nullopt;
46 }
47
48 value_multiplicity result;
49
50 // Find the hyphen separator
51 const auto hyphen_pos = str.find('-');
52
53 if (hyphen_pos == std::string_view::npos) {
54 // Single value case: "1", "2", etc.
55 uint32_t val{0};
56 auto [ptr, ec] = std::from_chars(str.data(), str.data() + str.size(), val);
57 if (ec != std::errc{} || ptr != str.data() + str.size()) {
58 return std::nullopt;
59 }
60 result.min = val;
61 result.max = val;
62 return result;
63 }
64
65 // Range case: "min-max" or "min-n" or "min-2n"
66 const auto min_str = str.substr(0, hyphen_pos);
67 const auto max_str = str.substr(hyphen_pos + 1);
68
69 if (min_str.empty() || max_str.empty()) {
70 return std::nullopt;
71 }
72
73 // Parse minimum
74 uint32_t min_val{0};
75 auto [min_ptr, min_ec] = std::from_chars(
76 min_str.data(), min_str.data() + min_str.size(), min_val);
77 if (min_ec != std::errc{} || min_ptr != min_str.data() + min_str.size()) {
78 return std::nullopt;
79 }
80 result.min = min_val;
81
82 // Parse maximum
83 if (max_str == "n") {
84 // Unbounded: "1-n"
85 result.max = std::nullopt;
86 result.multiplier = 1;
87 } else if (max_str.back() == 'n') {
88 // Multiplier pattern: "1-2n", "2-3n"
89 const auto mult_str = max_str.substr(0, max_str.size() - 1);
90 uint32_t mult_val{1};
91 auto [mult_ptr, mult_ec] = std::from_chars(
92 mult_str.data(), mult_str.data() + mult_str.size(), mult_val);
93 if (mult_ec != std::errc{} || mult_ptr != mult_str.data() + mult_str.size()) {
94 return std::nullopt;
95 }
96 result.max = std::nullopt;
97 result.multiplier = mult_val;
98 } else {
99 // Fixed range: "1-2", "1-3"
100 uint32_t max_val{0};
101 auto [max_ptr, max_ec] = std::from_chars(
102 max_str.data(), max_str.data() + max_str.size(), max_val);
103 if (max_ec != std::errc{} || max_ptr != max_str.data() + max_str.size()) {
104 return std::nullopt;
105 }
106 result.max = max_val;
107 }
108
109 return result;
110}

112auto value_multiplicity::to_string() const -> std::string {
113 std::ostringstream oss;
114 oss << min;
115
116 if (!max.has_value()) {
117 // Unbounded
118 oss << '-';
119 if (multiplier > 1) {
120 oss << multiplier;
121 }
122 oss << 'n';
123 } else if (max.value() != min) {
124 // Range
125 oss << '-' << max.value();
126 }
127 // If max == min, just output the single value
128
129 return oss.str();
130}
