61// N-ACTION Request
62// =============================================================================
63
64network::Result<std::monostate> storage_commitment_scu::request_commitment(
65 network::association& assoc,
66 uint8_t context_id,
67 const std::string& transaction_uid,
68 const std::vector<sop_reference>& references) {
69
70 using namespace network::dimse;
71
72 if (transaction_uid.empty() || references.empty()) {
73 return kcenon::pacs::pacs_void_error(
74 kcenon::pacs::error_codes::storage_commitment_missing_transaction_uid,
75 "Transaction UID and references must not be empty");
76 }
77
78 // Build N-ACTION-RQ
79 auto action_rq = make_n_action_rq(
80 1, // message_id
81 storage_commitment_push_model_sop_class_uid,
82 storage_commitment_push_model_sop_instance_uid,
83 storage_commitment_action_type_request);
84
85 // Build and attach action dataset
86 auto action_dataset = build_action_dataset(transaction_uid, references);
87 action_rq.set_dataset(std::move(action_dataset));
88
89 // Send N-ACTION-RQ
90 auto send_result = assoc.send_dimse(context_id, action_rq);
91 if (send_result.is_err()) {
92 return send_result;
93 }
94
95 ++requests_sent_;
96 return kcenon::pacs::ok();
97}

102
103network::Result<commitment_result> storage_commitment_scu::handle_event_report(
104 const network::dimse::dimse_message& event_rq) {
105
106 using namespace network::dimse;
107
108 // Verify command is N-EVENT-REPORT-RQ
109 if (event_rq.command() != command_field::n_event_report_rq) {
110 return kcenon::pacs::pacs_error<commitment_result>(
111 kcenon::pacs::error_codes::storage_commitment_unexpected_command,
112 "Expected N-EVENT-REPORT-RQ, got: " +
113 std::string(to_string(event_rq.command())));
114 }
115
116 // Verify dataset is present
117 if (!event_rq.has_dataset()) {
118 return kcenon::pacs::pacs_error<commitment_result>(
119 kcenon::pacs::error_codes::storage_commitment_missing_sequence,
120 "N-EVENT-REPORT-RQ has no dataset");
121 }
122
123 const auto& dataset = event_rq.dataset().value().get();
124
125 // Parse the event report dataset
126 auto result = parse_event_report_dataset(dataset);
127
128 ++event_reports_received_;
129
130 // Invoke callback if registered
131 if (callback_) {
132 callback_(result.transaction_uid, result);
133 }
134
135 return kcenon::pacs::Result<commitment_result>::ok(std::move(result));
136}

159core::dicom_dataset storage_commitment_scu::build_action_dataset(
160 const std::string& transaction_uid,
161 const std::vector<sop_reference>& references) {
162
163 core::dicom_dataset ds;
164
165 // Transaction UID (0008,1195)
166 ds.set_string(core::tags::transaction_uid, encoding::vr_type::UI,
167 transaction_uid);
168
169 // Referenced SOP Sequence (0008,1199)
170 auto& seq = ds.get_or_create_sequence(core::tags::referenced_sop_sequence);
171
172 for (const auto& ref : references) {
173 core::dicom_dataset item;
174 item.set_string(core::tags::referenced_sop_class_uid,
175 encoding::vr_type::UI, ref.sop_class_uid);
176 item.set_string(core::tags::referenced_sop_instance_uid,
177 encoding::vr_type::UI, ref.sop_instance_uid);
178 seq.push_back(std::move(item));
179 }
180
181 return ds;
182}

184commitment_result storage_commitment_scu::parse_event_report_dataset(
185 const core::dicom_dataset& dataset) {
186
187 commitment_result result;
188
189 // Extract Transaction UID (0008,1195)
190 result.transaction_uid = dataset.get_string(core::tags::transaction_uid);
191 result.timestamp = std::chrono::system_clock::now();
192
193 // Parse Referenced SOP Sequence (0008,1199) — successful instances
194 if (const auto* success_seq =
195 dataset.get_sequence(core::tags::referenced_sop_sequence)) {
196 for (const auto& item : *success_seq) {
197 sop_reference ref;
198 ref.sop_class_uid = item.get_string(
199 core::tags::referenced_sop_class_uid);
200 ref.sop_instance_uid = item.get_string(
201 core::tags::referenced_sop_instance_uid);
202 if (!ref.sop_class_uid.empty() && !ref.sop_instance_uid.empty()) {
203 result.success_references.push_back(std::move(ref));
204 }
205 }
206 }
207
208 // Parse Failed SOP Sequence (0008,1198) — failed instances
209 if (const auto* failed_seq =
210 dataset.get_sequence(core::tags::failed_sop_sequence)) {
211 for (const auto& item : *failed_seq) {
212 sop_reference ref;
213 ref.sop_class_uid = item.get_string(
214 core::tags::referenced_sop_class_uid);
215 ref.sop_instance_uid = item.get_string(
216 core::tags::referenced_sop_instance_uid);
217
218 auto reason = commitment_failure_reason::processing_failure;
219 auto reason_val = item.get_numeric<uint16_t>(
220 core::tags::failure_reason);
221 if (reason_val.has_value()) {
222 reason = static_cast<commitment_failure_reason>(
223 reason_val.value());
224 }
225
226 if (!ref.sop_class_uid.empty() && !ref.sop_instance_uid.empty()) {
227 result.failed_references.emplace_back(std::move(ref), reason);
228 }
229 }
230 }
231
232 return result;
233}
