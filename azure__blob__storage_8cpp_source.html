78auto compute_content_hash(const std::vector<std::uint8_t> &data)
79 -> std::string {
80 std::size_t hash = 0;
81 for (const auto &byte : data) {
82 hash = hash * 31 + byte;
83 }
84
85 std::ostringstream oss;
86 oss << std::hex << std::setfill('0') << std::setw(16) << hash;
87 return oss.str();
88}

95auto generate_block_id(std::size_t block_index) -> std::string {
96 std::ostringstream oss;
97 oss << "block_" << std::setfill('0') << std::setw(6) << block_index;
98 return oss.str();
99}

168class mock_azure_client
169 : public azure_blob_storage::azure_client_interface {
170public:
171 explicit mock_azure_client(const azure_storage_config & /*config*/)
172 : connected_(true) {}
173
174 [[nodiscard]] auto put_blob(const std::string &blob_name,
175 const std::vector<std::uint8_t> &data)
176 -> VoidResult override {
177 if (!connected_) {
178 return make_error<std::monostate>(
179 kConnectionError, "Azure client not connected",
180 "azure_blob_storage");
181 }
182 blob_data blob;
183 blob.data = data;
184 blob.etag = "\"" + compute_content_hash(data) + "\"";
185 blob.content_md5 = compute_content_hash(data);
186 blob.tier = "Hot";
187 blobs_[blob_name] = std::move(blob);
188 return ok();
189 }
190
191 [[nodiscard]] auto get_blob(const std::string &blob_name)
192 -> Result<std::vector<std::uint8_t>> override {
193 if (!connected_) {
194 return make_error<std::vector<std::uint8_t>>(
195 kConnectionError, "Azure client not connected",
196 "azure_blob_storage");
197 }
198 auto it = blobs_.find(blob_name);
199 if (it == blobs_.end()) {
200 return make_error<std::vector<std::uint8_t>>(
201 kBlobNotFound, "Blob not found: " + blob_name,
202 "azure_blob_storage");
203 }
204 return it->second.data;
205 }
206
207 [[nodiscard]] auto delete_blob(const std::string &blob_name)
208 -> VoidResult override {
209 if (!connected_) {
210 return make_error<std::monostate>(
211 kConnectionError, "Azure client not connected",
212 "azure_blob_storage");
213 }
214 blobs_.erase(blob_name);
215 return ok();
216 }
217
218 [[nodiscard]] auto head_blob(const std::string &blob_name) const
219 -> bool override {
220 if (!connected_) {
221 return false;
222 }
223 return blobs_.contains(blob_name);
224 }
225
226 [[nodiscard]] auto get_blob_size(const std::string &blob_name) const
227 -> std::size_t override {
228 auto it = blobs_.find(blob_name);
229 if (it != blobs_.end()) {
230 return it->second.data.size();
231 }
232 return 0;
233 }
234
235 [[nodiscard]] auto get_blob_etag(const std::string &blob_name) const
236 -> std::string override {
237 auto it = blobs_.find(blob_name);
238 if (it != blobs_.end()) {
239 return it->second.etag;
240 }
241 return {};
242 }
243
244 [[nodiscard]] auto get_blob_md5(const std::string &blob_name) const
245 -> std::string override {
246 auto it = blobs_.find(blob_name);
247 if (it != blobs_.end()) {
248 return it->second.content_md5;
249 }
250 return {};
251 }
252
253 [[nodiscard]] auto list_blobs() const
254 -> std::vector<std::string> override {
255 std::vector<std::string> names;
256 names.reserve(blobs_.size());
257 for (const auto &[name, data] : blobs_) {
258 names.push_back(name);
259 }
260 return names;
261 }
262
263 [[nodiscard]] auto is_connected() const -> bool override {
264 return connected_;
265 }
266
267 [[nodiscard]] auto stage_block(const std::string &blob_name,
268 const std::string &block_id,
269 const std::vector<std::uint8_t> &data)
270 -> VoidResult override {
271 if (!connected_) {
272 return make_error<std::monostate>(
273 kConnectionError, "Azure client not connected",
274 "azure_blob_storage");
275 }
276 staged_blocks_[blob_name][block_id] = data;
277 return ok();
278 }
279
280 [[nodiscard]] auto commit_blocks(const std::string &blob_name,
281 const std::vector<std::string> &block_ids)
282 -> VoidResult override {
283 if (!connected_) {
284 return make_error<std::monostate>(
285 kConnectionError, "Azure client not connected",
286 "azure_blob_storage");
287 }
288
289 auto it = staged_blocks_.find(blob_name);
290 if (it == staged_blocks_.end()) {
291 return make_error<std::monostate>(
292 kUploadError, "No staged blocks found for: " + blob_name,
293 "azure_blob_storage");
294 }
295
296 std::vector<std::uint8_t> combined_data;
297 for (const auto &block_id : block_ids) {
298 auto block_it = it->second.find(block_id);
299 if (block_it == it->second.end()) {
300 return make_error<std::monostate>(
301 kUploadError, "Block not found: " + block_id,
302 "azure_blob_storage");
303 }
304 combined_data.insert(combined_data.end(), block_it->second.begin(),
305 block_it->second.end());
306 }
307
308 blob_data blob;
309 blob.data = std::move(combined_data);
310 blob.etag = "\"" + compute_content_hash(blob.data) + "\"";
311 blob.content_md5 = compute_content_hash(blob.data);
312 blob.tier = "Hot";
313 blobs_[blob_name] = std::move(blob);
314
315 staged_blocks_.erase(blob_name);
316 return ok();
317 }
318
319 [[nodiscard]] auto set_tier(const std::string &blob_name,
320 const std::string &tier)
321 -> VoidResult override {
322 if (!connected_) {
323 return make_error<std::monostate>(
324 kConnectionError, "Azure client not connected",
325 "azure_blob_storage");
326 }
327 auto it = blobs_.find(blob_name);
328 if (it == blobs_.end()) {
329 return make_error<std::monostate>(
330 kBlobNotFound, "Blob not found: " + blob_name,
331 "azure_blob_storage");
332 }
333 it->second.tier = tier;
334 return ok();
335 }
336
337private:
338 struct blob_data {
339 std::vector<std::uint8_t> data;
340 std::string etag;
341 std::string content_md5;
342 std::string tier;
343 };
344
345 std::unordered_map<std::string, blob_data> blobs_;
346 std::unordered_map<std::string,
347 std::unordered_map<std::string, std::vector<std::uint8_t>>>
348 staged_blocks_;
349 bool connected_;
350};

364class azure_sdk_client
365 : public azure_blob_storage::azure_client_interface {
366public:
367 explicit azure_sdk_client(const azure_storage_config &config) {
368 if (config.endpoint_url.has_value()) {
369 // Azurite or custom endpoint
370 container_client_ =
371 std::make_unique<Azure::Storage::Blobs::BlobContainerClient>(
372 config.endpoint_url.value() + "/" + config.container_name);
373 } else {
374 container_client_ =
375 std::make_unique<Azure::Storage::Blobs::BlobContainerClient>(
376 Azure::Storage::Blobs::BlobContainerClient::
377 CreateFromConnectionString(config.connection_string,
378 config.container_name));
379 }
380 }
381
382 [[nodiscard]] auto put_blob(const std::string &blob_name,
383 const std::vector<std::uint8_t> &data)
384 -> VoidResult override {
385 try {
386 auto blob_client = container_client_->GetBlockBlobClient(blob_name);
387 Azure::Core::IO::MemoryBodyStream stream(data);
388 blob_client.Upload(stream);
389 return ok();
390 } catch (const Azure::Storage::StorageException &e) {
391 return make_error<std::monostate>(
392 kUploadError,
393 "Azure PutBlob failed: " + std::string(e.what()),
394 "azure_blob_storage");
395 }
396 }
397
398 [[nodiscard]] auto get_blob(const std::string &blob_name)
399 -> Result<std::vector<std::uint8_t>> override {
400 try {
401 auto blob_client = container_client_->GetBlobClient(blob_name);
402 auto response = blob_client.Download();
403 auto &body = response.Value.BodyStream;
404
405 std::vector<std::uint8_t> result(
406 static_cast<std::size_t>(body->Length()));
407 body->ReadToCount(result.data(), result.size());
408 return result;
409 } catch (const Azure::Storage::StorageException &e) {
410 if (e.StatusCode == Azure::Core::Http::HttpStatusCode::NotFound) {
411 return make_error<std::vector<std::uint8_t>>(
412 kBlobNotFound, "Blob not found: " + blob_name,
413 "azure_blob_storage");
414 }
415 return make_error<std::vector<std::uint8_t>>(
416 kDownloadError,
417 "Azure GetBlob failed: " + std::string(e.what()),
418 "azure_blob_storage");
419 }
420 }
421
422 [[nodiscard]] auto delete_blob(const std::string &blob_name)
423 -> VoidResult override {
424 try {
425 auto blob_client = container_client_->GetBlobClient(blob_name);
426 blob_client.Delete();
427 return ok();
428 } catch (const Azure::Storage::StorageException &e) {
429 return make_error<std::monostate>(
430 kUploadError,
431 "Azure DeleteBlob failed: " + std::string(e.what()),
432 "azure_blob_storage");
433 }
434 }
435
436 [[nodiscard]] auto head_blob(const std::string &blob_name) const

439 auto blob_client = container_client_->GetBlobClient(blob_name);

444 }
445 }
446
447 [[nodiscard]] auto get_blob_size(const std::string &blob_name) const

458 [[nodiscard]] auto get_blob_etag(const std::string &blob_name) const

460 try {
461 auto blob_client = container_client_->GetBlobClient(blob_name);

468
469 [[nodiscard]] auto get_blob_md5(const std::string &blob_name) const
470 -> std::string override {
471 try {
472 auto blob_client = container_client_->GetBlobClient(blob_name);
473 auto props = blob_client.GetProperties();
474 if (props.Value.HttpHeaders.ContentHash.Value.empty()) {
475 return {};
476 }
477 // Convert hash bytes to hex string
478 std::ostringstream oss;
479 for (auto byte : props.Value.HttpHeaders.ContentHash.Value) {
480 oss << std::hex << std::setfill('0') << std::setw(2)
481 << static_cast<int>(byte);
482 }
483 return oss.str();
484 } catch (const Azure::Storage::StorageException &) {
485 return {};
486 }
487 }

489 [[nodiscard]] auto list_blobs() const
490 -> std::vector<std::string> override {
491 std::vector<std::string> names;
492 try {
493 for (auto page = container_client_->ListBlobs(); page.HasPage();
494 page.MoveToNextPage()) {
495 for (const auto &blob : page.Blobs) {
496 names.push_back(blob.Name);
497 }
498 }
499 } catch (const Azure::Storage::StorageException &) {
500 // Return whatever was collected
501 }
502 return names;
503 }

507 }

516 blob_client.StageBlock(block_id, stream);

518 } catch (const Azure::Storage::StorageException &e) {

523 }
524 }

541 [[nodiscard]] auto set_tier(const std::string &blob_name,
542 const std::string &tier)
543 -> VoidResult override {
544 try {
545 auto blob_client = container_client_->GetBlobClient(blob_name);
546 Azure::Storage::Blobs::Models::AccessTier access_tier;
547 if (tier == "Hot") {
548 access_tier = Azure::Storage::Blobs::Models::AccessTier::Hot;

551 } else if (tier == "Archive") {
552 access_tier = Azure::Storage::Blobs::Models::AccessTier::Archive;
553 } else {
554 access_tier = Azure::Storage::Blobs::Models::AccessTier::Hot;
555 }
556 blob_client.SetAccessTier(access_tier);
557 return ok();
558 } catch (const Azure::Storage::StorageException &e) {
559 return make_error<std::monostate>(
560 kTierChangeError,
561 "Azure SetAccessTier failed: " + std::string(e.what()),
562 "azure_blob_storage");
563 }
564 }

598auto azure_blob_storage::store_with_progress(
599 const core::dicom_dataset &dataset, azure_progress_callback callback)
600 -> VoidResult {
601 // Extract required UIDs
602 auto study_uid = dataset.get_string(core::tags::study_instance_uid);
603 auto series_uid = dataset.get_string(core::tags::series_instance_uid);
604 auto sop_uid = dataset.get_string(core::tags::sop_instance_uid);
605
606 if (study_uid.empty() || series_uid.empty() || sop_uid.empty()) {
607 return make_error<std::monostate>(
608 kMissingRequiredUid,
609 "Missing required UID (Study, Series, or SOP Instance UID)",
610 "azure_blob_storage");
611 }
612
613 // Build blob name
614 auto blob_name = build_blob_name(study_uid, series_uid, sop_uid);
615
616 // Create DICOM file and serialize to bytes
617 auto dicom_file = core::dicom_file::create(
618 dataset, encoding::transfer_syntax::explicit_vr_little_endian);
619
620 auto data = dicom_file.to_bytes();
621 if (data.empty()) {
622 return make_error<std::monostate>(kSerializationError,
623 "Failed to serialize DICOM dataset",
624 "azure_blob_storage");
625 }
626
627 // Report initial progress
628 if (callback && !callback(0, data.size())) {
629 return make_error<std::monostate>(kUploadError, "Upload cancelled by user",
630 "azure_blob_storage");
631 }
632
633 // Upload to Azure (use block blob for large files)
634 VoidResult upload_result = ok();
635 if (data.size() > config_.block_upload_threshold) {
636 upload_result = upload_block_blob(blob_name, data, callback);
637 } else {
638 upload_result = client_->put_blob(blob_name, data);
639
640 // Report completion progress
641 if (callback) {
642 callback(data.size(), data.size());
643 }
644 }
645
646 if (upload_result.is_err()) {
647 return upload_result;
648 }
649
650 // Update local index
651 {
652 std::unique_lock lock(mutex_);
653 azure_blob_info info;
654 info.blob_name = blob_name;
655 info.sop_instance_uid = sop_uid;
656 info.study_instance_uid = study_uid;
657 info.series_instance_uid = series_uid;
658 info.size_bytes = data.size();
659 info.etag = client_->get_blob_etag(blob_name);
660 info.content_md5 = client_->get_blob_md5(blob_name);
661 index_[sop_uid] = std::move(info);
662 }
663
664 return ok();
665}

672auto azure_blob_storage::retrieve_with_progress(
673 std::string_view sop_instance_uid, azure_progress_callback callback)
674 -> Result<core::dicom_dataset> {
675 std::string blob_name;
676
677 {
678 std::shared_lock lock(mutex_);
679 auto it = index_.find(std::string{sop_instance_uid});
680 if (it == index_.end()) {
681 return make_error<core::dicom_dataset>(
682 kBlobNotFound,
683 "Instance not found: " + std::string{sop_instance_uid},
684 "azure_blob_storage");
685 }
686 blob_name = it->second.blob_name;
687 }
688
689 // Download from Azure
690 auto download_result = client_->get_blob(blob_name);
691 if (download_result.is_err()) {
692 return make_error<core::dicom_dataset>(kDownloadError,
693 "Failed to download from Azure",
694 "azure_blob_storage");
695 }
696
697 const auto &data = download_result.value();
698
699 // Report progress (download complete)
700 if (callback) {
701 callback(data.size(), data.size());
702 }
703
704 // Deserialize DICOM data
705 auto parse_result = core::dicom_file::from_bytes(data);
706 if (parse_result.is_err()) {
707 return make_error<core::dicom_dataset>(
708 kSerializationError,
709 "Failed to parse DICOM data: " + parse_result.error().message,
710 "azure_blob_storage");
711 }
712
713 return parse_result.value().dataset();
714}

716auto azure_blob_storage::remove(std::string_view sop_instance_uid)
717 -> VoidResult {
718 std::string blob_name;
719
720 {
721 std::unique_lock lock(mutex_);
722 auto it = index_.find(std::string{sop_instance_uid});
723 if (it == index_.end()) {
724 // Not found is not an error for remove
725 return ok();
726 }
727 blob_name = it->second.blob_name;
728 index_.erase(it);
729 }
730
731 // Delete from Azure
732 auto delete_result = client_->delete_blob(blob_name);
733 // Ignore delete errors - blob might have been deleted externally
734
735 return ok();
736}

744auto azure_blob_storage::find(const core::dicom_dataset &query)
745 -> Result<std::vector<core::dicom_dataset>> {
746 std::vector<core::dicom_dataset> results;
747
748 std::vector<std::string> blobs_to_retrieve;
749 {
750 std::shared_lock lock(mutex_);
751 blobs_to_retrieve.reserve(index_.size());
752 for (const auto &[uid, info] : index_) {
753 blobs_to_retrieve.push_back(info.blob_name);
754 }
755 }
756
757 for (const auto &blob_name : blobs_to_retrieve) {
758 auto download_result = client_->get_blob(blob_name);
759 if (download_result.is_err()) {
760 continue; // Skip blobs that can't be downloaded
761 }
762

954auto azure_blob_storage::upload_block_blob(
955 const std::string &blob_name, const std::vector<std::uint8_t> &data,
956 azure_progress_callback callback) -> VoidResult {
957 std::size_t total_bytes = data.size();
958 std::size_t bytes_uploaded = 0;
959 std::vector<std::string> block_ids;
960
961 // Stage blocks
962 std::size_t block_index = 0;
963 while (bytes_uploaded < total_bytes) {
964 std::size_t block_size =
965 (std::min)(config_.block_size, total_bytes - bytes_uploaded);
966
967 auto block_id = generate_block_id(block_index);
968 block_ids.push_back(block_id);
969
970 std::vector<std::uint8_t> block_data(data.begin() + bytes_uploaded,
971 data.begin() + bytes_uploaded +
972 block_size);
973
974 auto stage_result = client_->stage_block(blob_name, block_id, block_data);
975 if (stage_result.is_err()) {
976 return make_error<std::monostate>(kUploadError,
977 "Failed to stage block: " + block_id,
978 "azure_blob_storage");
979 }
980
981 bytes_uploaded += block_size;
982 block_index++;
983
984 if (callback && !callback(bytes_uploaded, total_bytes)) {
985 return make_error<std::monostate>(kUploadError,
986 "Upload cancelled by user",
987 "azure_blob_storage");
988 }
989 }
990
991 // Commit blocks
992 auto commit_result = client_->commit_blocks(blob_name, block_ids);
993 if (commit_result.is_err()) {
994 return make_error<std::monostate>(kUploadError, "Failed to commit blocks",
995 "azure_blob_storage");
996 }
997
998 return ok();
999}

1001auto azure_blob_storage::matches_query(const core::dicom_dataset &dataset,
1002 const core::dicom_dataset &query)
1003 -> bool {
1004 // If query is empty, match all
1005 if (query.empty()) {
1006 return true;
1007 }
1008
1009 // Check each query element
1010 for (const auto &[tag, element] : query) {
1011 auto query_value = element.as_string().unwrap_or("");
1012 if (query_value.empty()) {
1013 continue; // Empty value acts as wildcard
1014 }
1015
1016 auto dataset_value = dataset.get_string(tag);
1017
1018 // Support basic wildcard matching (* and ?)
1019 if (query_value.find('*') != std::string::npos ||
1020 query_value.find('?') != std::string::npos) {
1021 // Simple pattern matching
1022 if (query_value.front() == '*' && query_value.back() == '*') {
1023 // Contains
1024 auto inner = query_value.substr(1, query_value.length() - 2);
1025 if (dataset_value.find(inner) == std::string::npos) {
1026 return false;
1027 }
1028 } else if (query_value.front() == '*') {
1029 // Ends with
1030 auto suffix = query_value.substr(1);
1031 if (dataset_value.length() < suffix.length() ||
1032 dataset_value.substr(dataset_value.length() - suffix.length()) !=
1033 suffix) {
1034 return false;
1035 }
1036 } else if (query_value.back() == '*') {
1037 // Starts with
1038 auto prefix = query_value.substr(0, query_value.length() - 1);
1039 if (dataset_value.substr(0, prefix.length()) != prefix) {
1040 return false;
1041 }
1042 }
1043 } else {
1044 // Exact match
1045 if (dataset_value != query_value) {
1046 return false;
1047 }
1048 }
1049 }
1050
1051 return true;
1052}
