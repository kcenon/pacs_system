63void add_cors_headers(crow::response &res, const rest_server_context &ctx) {
64 if (ctx.config && !ctx.config->cors_allowed_origins.empty()) {
65 res.add_header("Access-Control-Allow-Origin",
66 ctx.config->cors_allowed_origins);
67 }

73std::string config_to_json(const rest_server_config &config) {
74 std::ostringstream oss;
75 oss << R"({"bind_address":")" << config.bind_address << R"(",)"
76 << R"("port":)" << config.port << R"(,)"
77 << R"("concurrency":)" << config.concurrency << R"(,)"
78 << R"("enable_cors":)" << (config.enable_cors ? "true" : "false")
79 << R"(,)"
80 << R"("cors_allowed_origins":")" << config.cors_allowed_origins << R"(",)"
81 << R"("enable_tls":)" << (config.enable_tls ? "true" : "false") << R"(,)"
82 << R"("request_timeout_seconds":)" << config.request_timeout_seconds
83 << R"(,)"
84 << R"("max_body_size":)" << config.max_body_size << "}";
85 return oss.str();
86}

88// Helper to check authentication and permission
89// Returns true if allowed, false if denied (and sets response)
90bool check_permission(const std::shared_ptr<rest_server_context> &ctx,
91 const crow::request &req, crow::response &res,
92 security::ResourceType resource, uint32_t action) {
93 if (!ctx->security_manager) {
94 // Fail open or closed? Closed is safer, but for dev maybe open if not
95 // configured? Let's fail closed.
96 res.code = 503;
97 res.body = make_error_json("SECURITY_UNAVAILABLE",
98 "Security manager not configured");
99 return false;
100 }
101
102 // Simple Auth via Header for now (Integration step)
103 auto user_id_header = req.get_header_value("X-User-ID");
104 if (user_id_header.empty()) {
105 res.code = 401;
106 res.body = make_error_json("UNAUTHORIZED", "Missing X-User-ID header");
107 return false;
108 }
109
110 std::string user_id = user_id_header;
111 auto user_res = ctx->security_manager->get_user(user_id);
112 if (user_res.is_err()) {
113 res.code = 401;
114 res.body = make_error_json("UNAUTHORIZED", "Invalid user");
115 return false;
116 }
117
118 auto user = user_res.unwrap();
119 if (!ctx->security_manager->check_permission(user, resource, action)) {
120 res.code = 403;
121 res.body = make_error_json("FORBIDDEN", "Insufficient permissions");
122 return false;
123 }
124
125 return true;
126}

131void register_system_endpoints_impl(crow::SimpleApp &app,
132 std::shared_ptr<rest_server_context> ctx) {
133 // GET /api/v1/system/status - System health status
134 CROW_ROUTE(app, "/api/v1/system/status")
135 .methods(crow::HTTPMethod::GET)([ctx](const crow::request &req) {
136 crow::response res;
137 res.add_header("Content-Type", "application/json");
138 add_cors_headers(res, *ctx);
139
140 // Permission Check: System Read
141 if (!check_permission(ctx, req, res, security::ResourceType::System,
142 security::Action::Read)) {
143 return res;
144 }
145
146 crow::json::wvalue status_json;
147 status_json["status"] = "unknown"; // Default
148 status_json["message"] = "Health checker not configured"; // Default
149 status_json["version"] = "1.0.0"; // Default
150
151#ifdef PACS_WITH_MONITORING
152 if (ctx->health_checker) {
153 auto status = ctx->health_checker->get_status();
154 // to_json is a free function in kcenon::pacs::monitoring namespace
155 status_json = crow::json::load(
156 monitoring::to_json(status)); // Parse existing JSON into wvalue
157 status_json["version"] = "1.0.0"; // Add version
158 } else {
159 status_json["status"] = "unknown";
160 status_json["message"] = "Health checker not configured";
161 }
162#else
163 // Basic status without monitoring module
164 res.code = 200;
165#endif
166 return res;
167 });
168
169 // GET /api/v1/system/metrics - Performance metrics
170 CROW_ROUTE(app, "/api/v1/system/metrics").methods(crow::HTTPMethod::GET)([ctx]() {
171 crow::response res;
172 res.add_header("Content-Type", "application/json");
173 add_cors_headers(res, *ctx);
174
175#ifdef PACS_WITH_MONITORING
176 if (ctx->metrics) {
177 // pacs_metrics provides individual counters, build simple JSON
178 std::ostringstream oss;
179 oss << R"({"message":"Metrics available via pacs_metrics API"})";
180 res.body = oss.str();
181 res.code = 200;
182 } else {
183 res.body =
184 R"({"error":{"code":"METRICS_UNAVAILABLE","message":"Metrics provider not configured"}})";
185 res.code = 503;
186 }
187#else
188 // Basic metrics without monitoring module
189 res.body =
190 R"({"uptime_seconds":0,"requests_total":0,"message":"Metrics module not available"})";
191 res.code = 200;
192#endif
193 return res;
194 });
195
196 // GET /api/v1/system/config - Current configuration
197 CROW_ROUTE(app, "/api/v1/system/config")
198 .methods(crow::HTTPMethod::GET)([ctx]() {
199 crow::response res;
200 res.add_header("Content-Type", "application/json");
201 add_cors_headers(res, *ctx);
202
203 if (ctx->config) {
204 res.body = config_to_json(*ctx->config);
205 res.code = 200;
206 } else {
207 res.body = make_error_json("CONFIG_UNAVAILABLE",
208 "Configuration not available");
209 res.code = 500;
210 }
211 return res;
212 });
213
214 // PUT /api/v1/system/config - Update configuration
215 CROW_ROUTE(app, "/api/v1/system/config")
216 .methods(crow::HTTPMethod::PUT)([ctx](const crow::request &req) {
217 crow::response res;
218 res.add_header("Content-Type", "application/json");
219 add_cors_headers(res, *ctx);
220
221 // Validate content type
222 auto content_type = req.get_header_value("Content-Type");
223 if (content_type.find("application/json") == std::string::npos) {
224 res.body = make_error_json("INVALID_CONTENT_TYPE",
225 "Content-Type must be application/json");
226 res.code = 415;
227 return res;
228 }
229
230 // Validate body is not empty
231 if (req.body.empty()) {
232 res.body = make_error_json("EMPTY_BODY", "Request body is required");
233 res.code = 400;
234 return res;
235 }
236
237 // Note: Full configuration update would require parsing JSON and
238 // updating the config. For now, we return a placeholder success
239 // response. Actual implementation would need a callback or direct
240 // config reference.
241 res.body = make_success_json("Configuration update acknowledged");
242 res.code = 200;
243 return res;
244 });
245
246 // GET /api/v1/system/version - API version info
247 CROW_ROUTE(app, "/api/v1/system/version").methods(crow::HTTPMethod::GET)([ctx]() {
248 crow::response res;
249 res.add_header("Content-Type", "application/json");
250 add_cors_headers(res, *ctx);
251
252 res.body =
253 R"({"api_version":"v1","pacs_version":"1.2.0","crow_version":"1.2.0"})";
254 res.code = 200;
255 return res;
256 });
257}
