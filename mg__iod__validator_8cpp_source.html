115validation_result mg_iod_validator::validate(const dicom_dataset& dataset) const {
116 validation_result result;
117 result.is_valid = true;
118
119 // Validate standard modules
120 if (options_.check_type1 || options_.check_type2) {
121 validate_patient_module(dataset, result.findings);
122 validate_general_study_module(dataset, result.findings);
123 validate_general_series_module(dataset, result.findings);
124 validate_sop_common_module(dataset, result.findings);
125 }
126
127 // Validate mammography series module
128 validate_mammography_series_module(dataset, result.findings);
129
130 if (options_.validate_pixel_data) {
131 validate_image_pixel_module(dataset, result.findings);
132 }
133
134 if (options_.validate_mg_specific) {
135 validate_mammography_image_module(dataset, result.findings);
136 validate_dx_anatomy_imaged_module(dataset, result.findings);
137 validate_dx_detector_module(dataset, result.findings);
138 }
139
140 if (options_.validate_compression || options_.validate_dose_parameters) {
141 validate_acquisition_dose_module(dataset, result.findings);
142 }
143
144 if (options_.validate_implant_attributes) {
145 validate_breast_implant_module(dataset, result.findings);
146 }
147
148 // Check for errors
149 for (const auto& finding : result.findings) {
150 if (finding.severity == validation_severity::error) {
151 result.is_valid = false;
152 break;
153 }
154 if (options_.strict_mode && finding.severity == validation_severity::warning) {
155 result.is_valid = false;
156 break;
157 }
158 }
159
160 return result;
161}

163validation_result
164mg_iod_validator::validate_for_presentation(const dicom_dataset& dataset) const {
165 auto result = validate(dataset);
166
167 // Additional For Presentation validation
168 if (options_.validate_presentation_requirements) {
169 validate_voi_lut_module(dataset, result.findings);
170
171 // Check Presentation Intent Type
172 if (dataset.contains(mg_tags::presentation_intent_type)) {
173 auto intent = dataset.get_string(mg_tags::presentation_intent_type);
174 if (intent != "FOR PRESENTATION") {
175 result.findings.push_back({
176 validation_severity::error,
177 mg_tags::presentation_intent_type,
178 "Presentation Intent Type should be 'FOR PRESENTATION' "
179 "for this SOP Class, found: " + intent,
180 "MG-ERR-010"
181 });
182 }
183 }
184 }
185
186 // Re-check validity
187 for (const auto& finding : result.findings) {
188 if (finding.severity == validation_severity::error) {
189 result.is_valid = false;
190 break;
191 }
192 }
193
194 return result;
195}

197validation_result
198mg_iod_validator::validate_for_processing(const dicom_dataset& dataset) const {
199 auto result = validate(dataset);
200
201 // Additional For Processing validation
202 if (options_.validate_processing_requirements) {
203 // Check Presentation Intent Type
204 if (dataset.contains(mg_tags::presentation_intent_type)) {
205 auto intent = dataset.get_string(mg_tags::presentation_intent_type);
206 if (intent != "FOR PROCESSING") {
207 result.findings.push_back({
208 validation_severity::error,
209 mg_tags::presentation_intent_type,
210 "Presentation Intent Type should be 'FOR PROCESSING' "
211 "for this SOP Class, found: " + intent,
212 "MG-ERR-011"
213 });
214 }
215 }
216
217 // For Processing images should have linear pixel intensity relationship
218 if (dataset.contains(mg_tags::pixel_intensity_relationship)) {
219 auto relationship = dataset.get_string(mg_tags::pixel_intensity_relationship);
220 if (relationship != "LIN") {
221 result.findings.push_back({
222 validation_severity::info,
223 mg_tags::pixel_intensity_relationship,
224 "For Processing mammography images typically have linear (LIN) "
225 "pixel intensity relationship, found: " + relationship,
226 "MG-INFO-002"
227 });
228 }
229 }
230 }
231
232 // Re-check validity
233 for (const auto& finding : result.findings) {
234 if (finding.severity == validation_severity::error) {
235 result.is_valid = false;
236 break;
237 }
238 }
239
240 return result;
241}

294bool mg_iod_validator::quick_check(const dicom_dataset& dataset) const {
295 // Check only critical Type 1 attributes for quick validation
296
297 // General Study Module Type 1
298 if (!dataset.contains(tags::study_instance_uid)) return false;
299
300 // General Series Module Type 1
301 if (!dataset.contains(tags::modality)) return false;
302 if (!dataset.contains(tags::series_instance_uid)) return false;
303
304 // Check modality is MG
305 auto modality = dataset.get_string(tags::modality);
306 if (modality != "MG") return false;
307
308 // Image Pixel Module Type 1
309 if (!dataset.contains(tags::samples_per_pixel)) return false;
310 if (!dataset.contains(tags::photometric_interpretation)) return false;
311 if (!dataset.contains(tags::rows)) return false;
312 if (!dataset.contains(tags::columns)) return false;
313 if (!dataset.contains(tags::bits_allocated)) return false;
314 if (!dataset.contains(tags::bits_stored)) return false;
315 if (!dataset.contains(tags::high_bit)) return false;
316 if (!dataset.contains(tags::pixel_representation)) return false;
317 if (!dataset.contains(tags::pixel_data)) return false;
318
319 // SOP Common Module Type 1
320 if (!dataset.contains(tags::sop_class_uid)) return false;
321 if (!dataset.contains(tags::sop_instance_uid)) return false;
322
323 // Mammography-specific: Laterality should be present
324 if (!dataset.contains(mg_tags::laterality) &&
325 !dataset.contains(mg_tags::image_laterality)) {
326 return false;
327 }
328
329 return true;
330}

395void mg_iod_validator::validate_mammography_series_module(
396 const dicom_dataset& dataset,
397 std::vector<validation_finding>& findings) const {
398

415void mg_iod_validator::validate_mammography_image_module(
416 const dicom_dataset& dataset,
417 std::vector<validation_finding>& findings) const {
418
419 // Image Type (0008,0008) - Type 1
420 if (options_.check_type1) {
421 check_type1_attribute(dataset, mg_tags::image_type, "ImageType", findings);
422
423 if (dataset.contains(mg_tags::image_type)) {
424 auto image_type = dataset.get_string(mg_tags::image_type);
425 // First value should be ORIGINAL or DERIVED
426 if (image_type.find("ORIGINAL") == std::string::npos &&
427 image_type.find("DERIVED") == std::string::npos) {
428 findings.push_back({
429 validation_severity::warning,
430 mg_tags::image_type,
431 "Image Type first value should be ORIGINAL or DERIVED",
432 "MG-WARN-002"
433 });
434 }
435 }
436 }
437
438 // Validate laterality if enabled
439 if (options_.validate_laterality) {
440 check_laterality_consistency(dataset, findings);
441 }
442
443 // Validate view position if enabled
444 if (options_.validate_view_position) {
445 check_view_position_validity(dataset, findings);
446 }
447
448 // Partial View (0028,1350) - Type 3
449 if (dataset.contains(mg_tags::partial_view)) {
450 auto partial = dataset.get_string(mg_tags::partial_view);
451 if (partial == "YES" && !dataset.contains(mg_tags::partial_view_description) &&
452 !dataset.contains(mg_tags::partial_view_code_sequence)) {
453 findings.push_back({
454 validation_severity::info,
455 mg_tags::partial_view_description,
456 "Partial View is YES but no description or code sequence provided",
457 "MG-INFO-003"
458 });
459 }
460 }

463void mg_iod_validator::validate_dx_anatomy_imaged_module(
464 const dicom_dataset& dataset,
465 std::vector<validation_finding>& findings) const {
466
467 // Body Part Examined (0018,0015) - Type 2
468 if (options_.check_type2) {
469 check_type2_attribute(dataset, mg_tags::body_part_examined, "BodyPartExamined", findings);
470
471 if (dataset.contains(mg_tags::body_part_examined)) {
472 auto body_part = dataset.get_string(mg_tags::body_part_examined);
473 if (body_part != "BREAST") {
474 findings.push_back({
475 validation_severity::warning,
476 mg_tags::body_part_examined,
477 "Body Part Examined should be 'BREAST' for mammography, found: " + body_part,
478 "MG-WARN-003"
479 });
480 }
481 }
482 }
483
484 // Anatomic Region Sequence (0008,2218) - Type 1C
485 if (options_.check_conditional) {
486 if (!dataset.contains(mg_tags::body_part_examined) &&
487 !dataset.contains(mg_tags::anatomic_region_sequence)) {
488 findings.push_back({
489 validation_severity::warning,
490 mg_tags::anatomic_region_sequence,
491 "Neither Body Part Examined nor Anatomic Region Sequence present - "
492 "anatomy information may be insufficient for clinical use",
493 "MG-WARN-004"
494 });
495 }
496 }

499void mg_iod_validator::validate_dx_detector_module(
500 const dicom_dataset& dataset,
501 std::vector<validation_finding>& findings) const {
502
503 // Detector Type (0018,7004) - Type 2
504 if (options_.check_type2) {
505 check_type2_attribute(dataset, mg_tags::detector_type, "DetectorType", findings);
506
507 if (dataset.contains(mg_tags::detector_type)) {
508 auto type = dataset.get_string(mg_tags::detector_type);
509 // Mammography typically uses DIRECT (a-Se) or SCINTILLATOR (CsI)
510 if (type != "DIRECT" && type != "SCINTILLATOR" && type != "STORAGE") {
511 findings.push_back({
512 validation_severity::info,
513 mg_tags::detector_type,
514 "Unusual Detector Type for mammography: " + type +
515 " (typical: DIRECT, SCINTILLATOR)",
516 "MG-INFO-004"
517 });
518 }
519 }
520 }
521
522 // Imager Pixel Spacing (0018,1164) - Type 1 for DX/MG
523 if (options_.check_type1) {
524 check_type1_attribute(dataset, mg_tags::imager_pixel_spacing,
525 "ImagerPixelSpacing", findings);
526
527 // Mammography typically has very fine pixel spacing (< 0.1 mm)
528 // This is informational only
529 }

533 const dicom_dataset& dataset,
534 std::vector<validation_finding>& findings) const {
535
536 // Compression Force (0018,11A2) - Type 3 but important for mammography

540
541 // Body Part Thickness (0018,11A0) - Type 3
542 if (options_.validate_dose_parameters) {
543 if (dataset.contains(mg_tags::body_part_thickness)) {
544 auto thickness = dataset.get_numeric<double>(mg_tags::body_part_thickness);

546 findings.push_back({
547 validation_severity::warning,

549 "Compressed breast thickness outside typical range (10-150mm): " +

555 // Recommend including thickness for quality assessment

557 validation_severity::info,

559 "Compressed breast thickness not documented - "
560 "recommended for dose assessment and quality control",
561 "MG-INFO-005"
562 });
563 }
564
565 // KVP (0018,0060) - Type 3 but important for dose
566 if (!dataset.contains(mg_tags::kvp)) {
567 findings.push_back({
568 validation_severity::info,
569 mg_tags::kvp,
570 "KVP not documented - recommended for dose assessment",
571 "MG-INFO-006"
572 });
573 }
574
575 // Entrance Dose (0040,0302) - Type 3
576 // Organ Dose (0040,0316) - Type 3 but important for breast dose tracking
577 }

604void mg_iod_validator::validate_voi_lut_module(
605 const dicom_dataset& dataset,

607
608 // Window Center (0028,1050) and Window Width (0028,1051) - Type 1C for For Presentation
609 if (options_.check_conditional) {
610 bool has_window = dataset.contains(tags::window_center) &&
611 dataset.contains(tags::window_width);
612
613 constexpr dicom_tag voi_lut_sequence{0x0028, 0x3010};
614 bool has_voi_lut = dataset.contains(voi_lut_sequence);
615
616 if (!has_window && !has_voi_lut) {
617 findings.push_back({
618 validation_severity::warning,
619 tags::window_center,
620 "For Presentation mammography images should have Window Center/Width "
621 "or VOI LUT Sequence for proper display",
622 "MG-WARN-008"
623 });
624 }
625 }

652void mg_iod_validator::validate_breast_implant_module(
653 const dicom_dataset& dataset,
654 std::vector<validation_finding>& findings) const {
655
656 // Breast Implant Present (0028,1300) - Type 3
657 if (dataset.contains(mg_tags::breast_implant_present)) {
658 auto implant = dataset.get_string(mg_tags::breast_implant_present);
659 if (implant != "YES" && implant != "NO") {
660 findings.push_back({
661 validation_severity::warning,
662 mg_tags::breast_implant_present,
663 "Breast Implant Present should be 'YES' or 'NO', found: " + implant,
664 "MG-WARN-009"
665 });
666 }
667
668 // If implant is present, check for implant displaced views
669 if (implant == "YES") {
670 // Informational: implant displacement technique (Eklund) may be needed
671 if (dataset.contains(mg_tags::view_position)) {
672 auto view = dataset.get_string(mg_tags::view_position);
673 auto parsed_view = parse_mg_view_position(view);
674 if (parsed_view != mg_view_position::implant &&
675 parsed_view != mg_view_position::id) {
676 findings.push_back({
677 validation_severity::info,
678 mg_tags::view_position,
679 "Breast implant present but view is not implant-displaced (ID). "
680 "Implant displacement technique (Eklund) may improve visualization.",
681 "MG-INFO-007"
682 });
683 }

759void mg_iod_validator::check_laterality_consistency(
760 const dicom_dataset& dataset,
761 std::vector<validation_finding>& findings) const {
762
763 // Check Laterality (0020,0060) - Series level
764 bool has_series_laterality = dataset.contains(mg_tags::laterality);
765 // Check Image Laterality (0020,0062) - Image level
766 bool has_image_laterality = dataset.contains(mg_tags::image_laterality);
767
768 // At least one should be present for mammography
769 if (!has_series_laterality && !has_image_laterality) {
770 findings.push_back({
771 validation_severity::error,
772 mg_tags::laterality,
773 "Neither Laterality (0020,0060) nor Image Laterality (0020,0062) is present. "
774 "Breast laterality must be specified for mammography images.",
775 "MG-ERR-003"
776 });
777 return;
778 }
779
780 // Validate Laterality value if present
781 if (has_series_laterality) {
782 auto lat = dataset.get_string(mg_tags::laterality);
783 if (!is_valid_breast_laterality(lat)) {
784 findings.push_back({
785 validation_severity::error,
786 mg_tags::laterality,
787 "Invalid Laterality value: '" + lat +
788 "'. Must be 'L' (left), 'R' (right), or 'B' (bilateral).",
789 "MG-ERR-004"
790 });
791 }
792 }
793
794 // Validate Image Laterality value if present
795 if (has_image_laterality) {
796 auto img_lat = dataset.get_string(mg_tags::image_laterality);
797 if (!is_valid_breast_laterality(img_lat)) {
798 findings.push_back({
799 validation_severity::error,
800 mg_tags::image_laterality,
801 "Invalid Image Laterality value: '" + img_lat +
802 "'. Must be 'L' (left), 'R' (right), or 'B' (bilateral).",
803 "MG-ERR-005"
804 });
805 }
806 }
807
808 // Check consistency between series and image laterality if both present
809 if (has_series_laterality && has_image_laterality) {
810 auto series_lat = dataset.get_string(mg_tags::laterality);
811 auto image_lat = dataset.get_string(mg_tags::image_laterality);
812
813 if (series_lat != image_lat) {
814 findings.push_back({
815 validation_severity::warning,
816 mg_tags::image_laterality,
817 "Laterality mismatch: Series Laterality is '" + series_lat +
818 "' but Image Laterality is '" + image_lat + "'.",
819 "MG-WARN-001"
820 });
821 }
822 }

825void mg_iod_validator::check_view_position_validity(
826 const dicom_dataset& dataset,
827 std::vector<validation_finding>& findings) const {
828
829 // View Position (0018,5101) - Type 2 for mammography
830 if (!dataset.contains(mg_tags::view_position)) {
831 findings.push_back({
832 validation_severity::warning,
833 mg_tags::view_position,
834 "View Position (0018,5101) is not present. "
835 "This attribute should be specified for mammography images.",
836 "MG-WARN-005"
837 });
838 return;
839 }
840
841 auto view_str = dataset.get_string(mg_tags::view_position);
842 if (view_str.empty()) {
843 findings.push_back({
844 validation_severity::warning,
845 mg_tags::view_position,
846 "View Position is present but empty.",
847 "MG-WARN-006"
848 });
849 return;
850 }
851
852 // Parse and validate view position
853 auto view = parse_mg_view_position(view_str);
854 if (view == mg_view_position::other) {
855 // Check if it's a valid view we just don't recognize
856 auto valid_views = get_valid_mg_view_positions();
857 std::string valid_list;
858 for (size_t i = 0; i < valid_views.size(); ++i) {
859 if (i > 0) valid_list += ", ";
860 valid_list += std::string(valid_views[i]);
861 }
862
863 findings.push_back({
864 validation_severity::warning,
865 mg_tags::view_position,
866 "Unrecognized View Position: '" + view_str +
867 "'. Common mammography views include: " + valid_list,
868 "MG-WARN-010"
869 });
870 }
871
872 // Additional validation: check laterality-view consistency
873 if (dataset.contains(mg_tags::laterality) || dataset.contains(mg_tags::image_laterality)) {
874 auto lat_str = dataset.contains(mg_tags::image_laterality)
875 ? dataset.get_string(mg_tags::image_laterality)
876 : dataset.get_string(mg_tags::laterality);
877
878 auto laterality = parse_breast_laterality(lat_str);
879
880 if (!is_valid_laterality_view_combination(laterality, view)) {
881 findings.push_back({
882 validation_severity::warning,
883 mg_tags::view_position,
884 "Unusual laterality-view combination: " + lat_str + " / " + view_str,
885 "MG-WARN-011"
886 });
887 }
888 }

892 const dicom_dataset& dataset,
893 std::vector<validation_finding>& findings) const {
894
895 if (!dataset.contains(mg_tags::compression_force)) {
896 // Compression force is Type 3 but important for mammography QC
897 findings.push_back({
898 validation_severity::info,
899 mg_tags::compression_force,

906

908 if (!force) {
909 findings.push_back({

915 return;
916 }
917
918 auto [min_typical, max_typical] = get_typical_compression_force_range();
919
920 if (!is_valid_compression_force(*force)) {
921 findings.push_back({

926 "This may indicate a measurement error or non-standard technique.",
927 "MG-WARN-013"
928 });
929 } else if (*force < min_typical || *force > max_typical) {
930 findings.push_back({
931 validation_severity::info,
932 mg_tags::compression_force,
933 "Compression Force (" + std::to_string(*force) +
934 " N) is outside the typical screening range (" +

936 "MG-INFO-009"
937 });

941void mg_iod_validator::check_pixel_data_consistency(
942 const dicom_dataset& dataset,
943 std::vector<validation_finding>& findings) const {
944
945 // Check BitsStored <= BitsAllocated
946 auto bits_allocated = dataset.get_numeric<uint16_t>(tags::bits_allocated);
947 auto bits_stored = dataset.get_numeric<uint16_t>(tags::bits_stored);
948 auto high_bit = dataset.get_numeric<uint16_t>(tags::high_bit);
949
950 if (bits_allocated && bits_stored) {
951 if (*bits_stored > *bits_allocated) {
952 findings.push_back({
953 validation_severity::error,
954 tags::bits_stored,
955 "BitsStored (" + std::to_string(*bits_stored) +
956 ") exceeds BitsAllocated (" + std::to_string(*bits_allocated) + ")",
957 "MG-ERR-006"
958 });
959 }
960
961 // Mammography typically uses 12-16 bits stored for high dynamic range
962 if (*bits_stored < 12 || *bits_stored > 16) {
963 findings.push_back({
964 validation_severity::info,
965 tags::bits_stored,
966 "Mammography images typically use 12-16 bits, found: " +
967 std::to_string(*bits_stored),
968 "MG-INFO-010"
969 });
970 }
971 }
972
973 // Check HighBit == BitsStored - 1
974 if (bits_stored && high_bit) {
975 if (*high_bit != *bits_stored - 1) {
976 findings.push_back({
977 validation_severity::warning,
978 tags::high_bit,
979 "HighBit (" + std::to_string(*high_bit) +
980 ") should typically be BitsStored - 1 (" +
981 std::to_string(*bits_stored - 1) + ")",
982 "MG-WARN-014"
983 });
984 }
985 }
986
987 // Mammography must be grayscale - check SamplesPerPixel
988 auto samples = dataset.get_numeric<uint16_t>(tags::samples_per_pixel);
989 if (samples && *samples != 1) {
990 findings.push_back({
991 validation_severity::error,
992 tags::samples_per_pixel,
993 "Mammography images must be grayscale (SamplesPerPixel = 1), found: " +
994 std::to_string(*samples),
995 "MG-ERR-007"
996 });
997 }
