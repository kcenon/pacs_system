64network::dimse::dimse_message make_n_create_rq(
65 uint16_t message_id,
66 std::string_view sop_class_uid,
67 std::string_view sop_instance_uid,
68 core::dicom_dataset dataset) {
69
70 using namespace network::dimse;
71
72 dimse_message msg{command_field::n_create_rq, message_id};
73 msg.set_affected_sop_class_uid(sop_class_uid);
74 if (!sop_instance_uid.empty()) {
75 msg.set_affected_sop_instance_uid(sop_instance_uid);
76 }
77 msg.set_dataset(std::move(dataset));
78
79 return msg;

180network::Result<print_result> print_scu::create_film_session(
181 network::association& assoc,
182 const print_session_data& data) {
183
184 using namespace network::dimse;
185 using namespace encoding;
186
187 auto start_time = std::chrono::steady_clock::now();
188
189 if (!assoc.is_established()) {
190 return kcenon::pacs::pacs_error<print_result>(
191 kcenon::pacs::error_codes::association_not_established,
192 "Association not established");
193 }
194
195 auto context_id = find_print_context(assoc, basic_film_session_sop_class_uid);
196 if (!context_id) {
197 return kcenon::pacs::pacs_error<print_result>(
198 kcenon::pacs::error_codes::print_invalid_sop_class,
199 "No accepted presentation context for Film Session SOP Class");
200 }
201
202 std::string session_uid = data.sop_instance_uid;
203 if (session_uid.empty() && config_.auto_generate_uid) {
204 session_uid = generate_uid();
205 }
206
207 // Build creation dataset
208 core::dicom_dataset ds;
209 ds.set_string(print_tags::number_of_copies, vr_type::IS,
210 std::to_string(data.number_of_copies));
211 if (!data.print_priority.empty()) {
212 ds.set_string(print_tags::print_priority, vr_type::CS, data.print_priority);
213 }
214 if (!data.medium_type.empty()) {
215 ds.set_string(print_tags::medium_type, vr_type::CS, data.medium_type);
216 }
217 if (!data.film_destination.empty()) {
218 ds.set_string(print_tags::film_destination, vr_type::CS, data.film_destination);
219 }
220 if (!data.film_session_label.empty()) {
221 ds.set_string(print_tags::film_session_label, vr_type::LO, data.film_session_label);
222 }
223
224 auto request = make_n_create_rq(
225 next_message_id(),
226 basic_film_session_sop_class_uid,
227 session_uid,
228 std::move(ds));
229
230 logger_->debug("Sending N-CREATE Film Session: " + session_uid);
231
232 auto send_result = assoc.send_dimse(*context_id, request);
233 if (send_result.is_err()) {
234 logger_->error("Failed to send N-CREATE Film Session: " + send_result.error().message);
235 return send_result.error();
236 }
237
238 auto recv_result = assoc.receive_dimse(config_.timeout);
239 if (recv_result.is_err()) {
240 logger_->error("Failed to receive N-CREATE Film Session response: " +
241 recv_result.error().message);
242 return recv_result.error();
243 }
244
245 const auto& [recv_context_id, response] = recv_result.value();
246
247 if (response.command() != command_field::n_create_rsp) {
248 return kcenon::pacs::pacs_error<print_result>(
249 kcenon::pacs::error_codes::print_unexpected_command,
250 "Expected N-CREATE-RSP but received " +
251 std::string(to_string(response.command())));
252 }
253
254 auto end_time = std::chrono::steady_clock::now();
255
256 print_result result;
257 result.sop_instance_uid = response.affected_sop_instance_uid().empty()
258 ? session_uid : std::string(response.affected_sop_instance_uid());
259 result.status = static_cast<uint16_t>(response.status());
260 result.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
261 end_time - start_time);
262
263 if (response.command_set().contains(tag_error_comment)) {
264 result.error_comment = response.command_set().get_string(tag_error_comment);
265 }
266
267 sessions_created_.fetch_add(1, std::memory_order_relaxed);
268
269 if (result.is_success()) {
270 logger_->info("N-CREATE Film Session successful: " + result.sop_instance_uid);
271 } else {
272 logger_->warn("N-CREATE Film Session status 0x" +
273 std::to_string(result.status) + ": " + result.sop_instance_uid);
274 }
275
276 return result;
277}

355network::Result<print_result> print_scu::create_film_box(
356 network::association& assoc,
357 const print_film_box_data& data) {
358
359 using namespace network::dimse;
360 using namespace encoding;
361
362 auto start_time = std::chrono::steady_clock::now();
363
364 if (!assoc.is_established()) {
365 return kcenon::pacs::pacs_error<print_result>(
366 kcenon::pacs::error_codes::association_not_established,
367 "Association not established");
368 }
369
370 auto context_id = find_print_context(assoc, basic_film_box_sop_class_uid);
371 if (!context_id) {
372 return kcenon::pacs::pacs_error<print_result>(
373 kcenon::pacs::error_codes::print_invalid_sop_class,
374 "No accepted presentation context for Film Box SOP Class");
375 }
376
377 // Build creation dataset
378 core::dicom_dataset ds;
379 if (!data.image_display_format.empty()) {
380 ds.set_string(print_tags::image_display_format, vr_type::ST,
381 data.image_display_format);
382 }
383 if (!data.film_orientation.empty()) {
384 ds.set_string(print_tags::film_orientation, vr_type::CS, data.film_orientation);
385 }
386 if (!data.film_size_id.empty()) {
387 ds.set_string(print_tags::film_size_id, vr_type::CS, data.film_size_id);
388 }
389 if (!data.magnification_type.empty()) {
390 ds.set_string(print_tags::magnification_type, vr_type::CS,
391 data.magnification_type);
392 }
393
394 // Add referenced film session sequence
395 if (!data.film_session_uid.empty()) {
396 auto& seq = ds.get_or_create_sequence(
397 print_tags::referenced_film_session_sequence);
398 core::dicom_dataset ref_item;
399 ref_item.set_string(core::tags::referenced_sop_class_uid, vr_type::UI,
400 std::string(basic_film_session_sop_class_uid));
401 ref_item.set_string(core::tags::referenced_sop_instance_uid, vr_type::UI,
402 data.film_session_uid);
403 seq.push_back(std::move(ref_item));
404 }
405
406 auto request = make_n_create_rq(
407 next_message_id(),
408 basic_film_box_sop_class_uid,
409 "", // SCP generates the UID
410 std::move(ds));
411
412 logger_->debug("Sending N-CREATE Film Box");
413
414 auto send_result = assoc.send_dimse(*context_id, request);
415 if (send_result.is_err()) {
416 logger_->error("Failed to send N-CREATE Film Box: " + send_result.error().message);
417 return send_result.error();
418 }
419
420 auto recv_result = assoc.receive_dimse(config_.timeout);
421 if (recv_result.is_err()) {
422 logger_->error("Failed to receive N-CREATE Film Box response: " +
423 recv_result.error().message);
424 return recv_result.error();
425 }
426
427 const auto& [recv_context_id, response] = recv_result.value();
428
429 if (response.command() != command_field::n_create_rsp) {
430 return kcenon::pacs::pacs_error<print_result>(
431 kcenon::pacs::error_codes::print_unexpected_command,
432 "Expected N-CREATE-RSP but received " +
433 std::string(to_string(response.command())));
434 }
435
436 auto end_time = std::chrono::steady_clock::now();
437
438 print_result result;
439 result.sop_instance_uid = std::string(response.affected_sop_instance_uid());
440 result.status = static_cast<uint16_t>(response.status());
441 result.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
442 end_time - start_time);
443
444 if (response.command_set().contains(tag_error_comment)) {
445 result.error_comment = response.command_set().get_string(tag_error_comment);
446 }
447
448 // Capture response dataset (contains Referenced Image Box Sequence)
449 if (response.has_dataset()) {
450 result.response_data = response.dataset().value().get();
451 }
452
453 film_boxes_created_.fetch_add(1, std::memory_order_relaxed);
454
455 if (result.is_success()) {
456 logger_->info("N-CREATE Film Box successful: " + result.sop_instance_uid);
457 } else {
458 logger_->warn("N-CREATE Film Box status 0x" +
459 std::to_string(result.status));
460 }
461
462 return result;
463}

466 network::association& assoc,
467 std::string_view film_box_uid) {
468
469 using namespace network::dimse;
470
471 auto start_time = std::chrono::steady_clock::now();

507

510 if (response.command() != command_field::n_action_rsp) {

516
517 auto end_time = std::chrono::steady_clock::now();
518

527 }
528
529 prints_executed_.fetch_add(1, std::memory_order_relaxed);
530
531 if (result.is_success()) {
532 logger_->info("N-ACTION Print successful: " + std::string(film_box_uid));

534 logger_->warn("N-ACTION Print status 0x" +

618 network::association& assoc,
619 std::string_view image_box_uid,
620 const print_image_data& data,
621 bool use_color) {
622
623 using namespace network::dimse;
624 using namespace encoding;
625
626 auto start_time = std::chrono::steady_clock::now();
627
628 if (!assoc.is_established()) {
629 return kcenon::pacs::pacs_error<print_result>(
630 kcenon::pacs::error_codes::association_not_established,
631 "Association not established");
632 }
633
634 auto sop_class_uid = use_color
635 ? basic_color_image_box_sop_class_uid
636 : basic_grayscale_image_box_sop_class_uid;
637
638 auto context_id = find_print_context(assoc, sop_class_uid);
639 if (!context_id) {
640 return kcenon::pacs::pacs_error<print_result>(
641 kcenon::pacs::error_codes::print_invalid_sop_class,
642 "No accepted presentation context for Image Box SOP Class");
643 }
644
645 // Build modification dataset with pixel data in the appropriate sequence
646 core::dicom_dataset ds = data.pixel_data;
647
648 if (data.image_position > 0) {
649 ds.set_string(print_tags::image_position, vr_type::US,
650 std::to_string(data.image_position));
651 }
652
653 auto request = make_n_set_rq(
654 next_message_id(),
655 sop_class_uid,
656 image_box_uid,
657 std::move(ds));
658
659 logger_->debug("Sending N-SET Image Box: " + std::string(image_box_uid));
660
661 auto send_result = assoc.send_dimse(*context_id, request);
662 if (send_result.is_err()) {
663 logger_->error("Failed to send N-SET Image Box: " + send_result.error().message);
664 return send_result.error();
665 }
666
667 auto recv_result = assoc.receive_dimse(config_.timeout);
668 if (recv_result.is_err()) {
669 logger_->error("Failed to receive N-SET Image Box response: " +
670 recv_result.error().message);
671 return recv_result.error();
672 }
673
674 const auto& [recv_context_id, response] = recv_result.value();
675
676 if (response.command() != command_field::n_set_rsp) {
677 return kcenon::pacs::pacs_error<print_result>(
678 kcenon::pacs::error_codes::print_unexpected_command,
679 "Expected N-SET-RSP but received " +
680 std::string(to_string(response.command())));
681 }

710
711network::Result<print_result> print_scu::query_printer_status(
712 network::association& assoc) {
713
714 using namespace network::dimse;
715
716 auto start_time = std::chrono::steady_clock::now();
717
718 if (!assoc.is_established()) {
719 return kcenon::pacs::pacs_error<print_result>(
720 kcenon::pacs::error_codes::association_not_established,
721 "Association not established");
722 }
723
724 auto context_id = find_print_context(assoc, printer_sop_class_uid);
725 if (!context_id) {
726 return kcenon::pacs::pacs_error<print_result>(
727 kcenon::pacs::error_codes::print_invalid_sop_class,
728 "No accepted presentation context for Printer SOP Class");
729 }
730
731 // Well-Known Printer SOP Instance UID (PS3.4 H.4.17)
732 constexpr std::string_view printer_instance_uid =
733 "1.2.840.10008.5.1.1.17";
734
735 auto request = make_n_get_rq(
736 next_message_id(),
737 printer_sop_class_uid,
738 printer_instance_uid);
739
740 logger_->debug("Sending N-GET Printer Status");
741
742 auto send_result = assoc.send_dimse(*context_id, request);
743 if (send_result.is_err()) {
744 logger_->error("Failed to send N-GET Printer Status: " + send_result.error().message);
745 return send_result.error();
746 }
747
748 auto recv_result = assoc.receive_dimse(config_.timeout);
749 if (recv_result.is_err()) {
750 logger_->error("Failed to receive N-GET Printer Status response: " +
751 recv_result.error().message);
752 return recv_result.error();
753 }
754
755 const auto& [recv_context_id, response] = recv_result.value();
756
757 if (response.command() != command_field::n_get_rsp) {
758 return kcenon::pacs::pacs_error<print_result>(
759 kcenon::pacs::error_codes::print_unexpected_command,
760 "Expected N-GET-RSP but received " +
761 std::string(to_string(response.command())));
762 }
763
764 auto end_time = std::chrono::steady_clock::now();
765
766 print_result result;
767 result.sop_instance_uid = std::string(printer_instance_uid);
768 result.status = static_cast<uint16_t>(response.status());
769 result.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
770 end_time - start_time);
771
772 if (response.command_set().contains(tag_error_comment)) {
773 result.error_comment = response.command_set().get_string(tag_error_comment);
774 }
775
776 if (response.has_dataset()) {
777 result.response_data = response.dataset().value().get();
778 }
779
780 printer_queries_.fetch_add(1, std::memory_order_relaxed);
781
782 if (result.is_success()) {
783 logger_->info("N-GET Printer Status successful");
784 } else {
785 logger_->warn("N-GET Printer Status returned 0x" +
786 std::to_string(result.status));
787 }
788
789 return result;
790}
