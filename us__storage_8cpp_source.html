84std::string_view to_string(us_photometric_interpretation interp) noexcept {
85 auto index = static_cast<size_t>(interp);
86 if (index < std::size(photometric_strings)) {
87 return photometric_strings[index];
88 }
89 return "MONOCHROME2";
90}

92us_photometric_interpretation
93parse_photometric_interpretation(std::string_view value) noexcept {
94 if (value == "MONOCHROME1") {
95 return us_photometric_interpretation::monochrome1;
96 }
97 if (value == "MONOCHROME2") {
98 return us_photometric_interpretation::monochrome2;
99 }
100 if (value == "PALETTE COLOR") {
101 return us_photometric_interpretation::palette_color;
102 }
103 if (value == "RGB") {
104 return us_photometric_interpretation::rgb;
105 }
106 if (value == "YBR_FULL") {
107 return us_photometric_interpretation::ybr_full;
108 }
109 if (value == "YBR_FULL_422") {
110 return us_photometric_interpretation::ybr_full_422;
111 }
112 // Default to MONOCHROME2 for unknown values
113 return us_photometric_interpretation::monochrome2;
114}

165std::vector<std::string> get_us_storage_sop_classes(bool include_retired) {
166 std::vector<std::string> result;
167 result.reserve(include_retired ? 4 : 2);
168
169 for (const auto& info : us_sop_classes) {
170 if (!info.is_retired || include_retired) {
171 result.emplace_back(info.uid);
172 }
173 }
174
175 return result;
176}

178const us_sop_class_info*
179get_us_sop_class_info(std::string_view uid) noexcept {
180 auto it = std::find_if(
181 us_sop_classes.begin(),
182 us_sop_classes.end(),
183 [uid](const auto& info) { return info.uid == uid; }
184 );
185
186 if (it != us_sop_classes.end()) {
187 return &(*it);
188 }
189 return nullptr;
190}
191
192bool is_us_storage_sop_class(std::string_view uid) noexcept {
193 return get_us_sop_class_info(uid) != nullptr;
194}

196bool is_us_multiframe_sop_class(std::string_view uid) noexcept {
