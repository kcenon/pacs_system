377template <typename T>
378 requires std::is_arithmetic_v<T>
379auto dicom_element::from_numeric_list(dicom_tag tag, encoding::vr_type vr,
380 std::span<const T> values) -> dicom_element {
381 dicom_element elem{tag, vr};
382 std::vector<uint8_t> data(values.size() * sizeof(T));
383
384 for (size_t i = 0; i < values.size(); ++i) {
385 std::memcpy(data.data() + i * sizeof(T), &values[i], sizeof(T));
386 }
387
388 elem.set_value(data);
389 return elem;
390}

392template <typename T>
393 requires std::is_arithmetic_v<T>
394auto dicom_element::as_numeric() const -> kcenon::pacs::Result<T> {
395 if (data_.size() < sizeof(T)) {
396 return kcenon::pacs::pacs_error<T>(
397 kcenon::pacs::error_codes::data_size_mismatch,
398 "Insufficient data for numeric conversion: expected " +
399 std::to_string(sizeof(T)) + " bytes, got " +
400 std::to_string(data_.size()));
401 }
402
403 T result{};
404 std::memcpy(&result, data_.data(), sizeof(T));
405 return kcenon::pacs::ok(result);
406}

408template <typename T>
409 requires std::is_arithmetic_v<T>
410auto dicom_element::as_numeric_list() const -> kcenon::pacs::Result<std::vector<T>> {
411 if (data_.size() % sizeof(T) != 0) {
412 return kcenon::pacs::pacs_error<std::vector<T>>(
413 kcenon::pacs::error_codes::data_size_mismatch,
414 "Data size not aligned for numeric type: " +
415 std::to_string(data_.size()) + " bytes is not divisible by " +
416 std::to_string(sizeof(T)));
417 }
418
419 const size_t count = data_.size() / sizeof(T);
420 std::vector<T> result(count);
421
422 for (size_t i = 0; i < count; ++i) {
423 std::memcpy(&result[i], data_.data() + i * sizeof(T), sizeof(T));
424 }
425
426 return kcenon::pacs::ok(result);
427}
