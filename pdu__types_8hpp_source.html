273[[nodiscard]] constexpr const char* to_string(pdu_type type) noexcept {
274 switch (type) {
275 case pdu_type::associate_rq: return "A-ASSOCIATE-RQ";
276 case pdu_type::associate_ac: return "A-ASSOCIATE-AC";
277 case pdu_type::associate_rj: return "A-ASSOCIATE-RJ";
278 case pdu_type::p_data_tf: return "P-DATA-TF";
279 case pdu_type::release_rq: return "A-RELEASE-RQ";
280 case pdu_type::release_rp: return "A-RELEASE-RP";
281 case pdu_type::abort: return "A-ABORT";
282 default: return "UNKNOWN";
283 }
284}
