81std::string format_timestamp(std::chrono::system_clock::time_point tp) {
82 if (tp == std::chrono::system_clock::time_point{}) {
83 return "";
84 }
85 auto time_t_val = std::chrono::system_clock::to_time_t(tp);
86 std::tm tm_val{};
87#ifdef _WIN32
88 gmtime_s(&tm_val, &time_t_val);
89#else
90 gmtime_r(&time_t_val, &tm_val);
91#endif
92 char buf[32];
93 std::strftime(buf, sizeof(buf), "%Y-%m-%dT%H:%M:%SZ", &tm_val);
94 return buf;
95}

100std::string node_to_json(const client::remote_node& node) {
101 std::ostringstream oss;
102 oss << R"({"node_id":")" << json_escape(node.node_id)
103 << R"(","name":")" << json_escape(node.name)
104 << R"(","ae_title":")" << json_escape(node.ae_title)
105 << R"(","host":")" << json_escape(node.host)
106 << R"(","port":)" << node.port
107 << R"(,"supports_find":)" << (node.supports_find ? "true" : "false")
108 << R"(,"supports_move":)" << (node.supports_move ? "true" : "false")
109 << R"(,"supports_get":)" << (node.supports_get ? "true" : "false")
110 << R"(,"supports_store":)" << (node.supports_store ? "true" : "false")
111 << R"(,"supports_worklist":)" << (node.supports_worklist ? "true" : "false")
112 << R"(,"connection_timeout_sec":)" << node.connection_timeout.count()
113 << R"(,"dimse_timeout_sec":)" << node.dimse_timeout.count()
114 << R"(,"max_associations":)" << node.max_associations
115 << R"(,"status":")" << client::to_string(node.status) << '"';
116
117 auto last_verified = format_timestamp(node.last_verified);
118 if (!last_verified.empty()) {
119 oss << R"(,"last_verified":")" << last_verified << '"';
120 }
121
122 auto last_error_time = format_timestamp(node.last_error);
123 if (!last_error_time.empty()) {
124 oss << R"(,"last_error":")" << last_error_time << '"';
125 }
126
127 if (!node.last_error_message.empty()) {
128 oss << R"(,"last_error_message":")" << json_escape(node.last_error_message) << '"';
129 }
130
131 auto created_at = format_timestamp(node.created_at);
132 if (!created_at.empty()) {
133 oss << R"(,"created_at":")" << created_at << '"';
134 }
135
136 auto updated_at = format_timestamp(node.updated_at);
137 if (!updated_at.empty()) {
138 oss << R"(,"updated_at":")" << updated_at << '"';
139 }
140
141 oss << '}';
142 return oss.str();
143}

148std::string nodes_to_json(const std::vector<client::remote_node>& nodes,
149 size_t total_count) {
150 std::ostringstream oss;
151 oss << R"({"nodes":[)";
152 for (size_t i = 0; i < nodes.size(); ++i) {
153 if (i > 0) {
154 oss << ',';
155 }
156 oss << node_to_json(nodes[i]);
157 }
158 oss << R"(],"total":)" << total_count << '}';
159 return oss.str();
160}

165std::optional<client::remote_node> parse_node_from_json(const std::string& body,
166 std::string& error_message) {
167 client::remote_node node;
168
169 // Simple JSON parsing (in production, use a proper JSON library)
170 auto get_string_value = [&body](const std::string& key) -> std::string {
171 std::string search = "\"" + key + "\":\"";
172 auto pos = body.find(search);
173 if (pos == std::string::npos) {
174 return "";
175 }
176 pos += search.length();
177 auto end_pos = body.find('"', pos);
178 if (end_pos == std::string::npos) {
179 return "";
180 }
181 return body.substr(pos, end_pos - pos);
182 };
183
184 auto get_int_value = [&body](const std::string& key) -> std::optional<int64_t> {
185 std::string search = "\"" + key + "\":";
186 auto pos = body.find(search);
187 if (pos == std::string::npos) {
188 return std::nullopt;
189 }
190 pos += search.length();
191 // Skip whitespace
192 while (pos < body.length() && (body[pos] == ' ' || body[pos] == '\t')) {
193 ++pos;
194 }
195 if (pos >= body.length()) {
196 return std::nullopt;
197 }
198 try {
199 size_t idx = 0;
200 auto val = std::stoll(body.substr(pos), &idx);
201 return val;
202 } catch (...) {
203 return std::nullopt;
204 }
205 };
206
207 auto get_bool_value = [&body](const std::string& key,
208 bool default_val) -> bool {
209 std::string search = "\"" + key + "\":";
210 auto pos = body.find(search);
211 if (pos == std::string::npos) {
212 return default_val;
213 }
214 pos += search.length();
215 // Skip whitespace
216 while (pos < body.length() && (body[pos] == ' ' || body[pos] == '\t')) {
217 ++pos;
218 }
219 if (pos >= body.length()) {
220 return default_val;
221 }
222 if (body.substr(pos, 4) == "true") {
223 return true;
224 }
225 if (body.substr(pos, 5) == "false") {
226 return false;
227 }
228 return default_val;
229 };
230
231 // Required fields
232 node.name = get_string_value("name");
233 node.ae_title = get_string_value("ae_title");
234 node.host = get_string_value("host");
235
236 if (node.ae_title.empty()) {
237 error_message = "ae_title is required";
238 return std::nullopt;
239 }
240
241 if (node.host.empty()) {
242 error_message = "host is required";
243 return std::nullopt;
244 }
245
246 // Optional fields with defaults
247 auto port_val = get_int_value("port");
248 node.port = port_val.value_or(104);
249
250 node.supports_find = get_bool_value("supports_find", true);
251 node.supports_move = get_bool_value("supports_move", true);
252 node.supports_get = get_bool_value("supports_get", false);
253 node.supports_store = get_bool_value("supports_store", true);
254 node.supports_worklist = get_bool_value("supports_worklist", false);
255
256 auto connection_timeout = get_int_value("connection_timeout_sec");
257 if (connection_timeout) {
258 node.connection_timeout = std::chrono::seconds(*connection_timeout);
259 }
260
261 auto dimse_timeout = get_int_value("dimse_timeout_sec");
262 if (dimse_timeout) {
263 node.dimse_timeout = std::chrono::seconds(*dimse_timeout);
264 }
265
266 auto max_assoc = get_int_value("max_associations");
267 if (max_assoc) {
268 node.max_associations = static_cast<size_t>(*max_assoc);
269 }
270
271 return node;
272}

277std::pair<size_t, size_t> parse_pagination(const crow::request& req) {
278 size_t limit = 20;
279 size_t offset = 0;
280
281 auto limit_param = req.url_params.get("limit");
282 if (limit_param) {
283 try {
284 limit = std::stoul(limit_param);
285 if (limit > 100) {
286 limit = 100;
287 }
288 } catch (...) {
289 // Use default
290 }
291 }
292
293 auto offset_param = req.url_params.get("offset");
294 if (offset_param) {
295 try {
296 offset = std::stoul(offset_param);
297 } catch (...) {
298 // Use default
299 }
300 }
301
302 return {limit, offset};
303}

309 std::shared_ptr<rest_server_context> ctx) {
310 // GET /api/v1/remote-nodes - List remote nodes (paginated)
311 CROW_ROUTE(app, "/api/v1/remote-nodes")
312 .methods(crow::HTTPMethod::GET)([ctx](const crow::request& req) {
313 crow::response res;
314 res.add_header("Content-Type", "application/json");
315 add_cors_headers(res, *ctx);
316
317 if (!ctx->node_manager) {
318 res.code = 503;
319 res.body = make_error_json("SERVICE_UNAVAILABLE",
320 "Remote node manager not configured");
321 return res;
322 }
323
324 // Parse pagination
325 auto [limit, offset] = parse_pagination(req);
326
327 // Filter by status if provided
328 auto status_param = req.url_params.get("status");
329 std::vector<client::remote_node> nodes;
330
331 if (status_param) {
332 auto status = client::node_status_from_string(status_param);
333 nodes = ctx->node_manager->list_nodes_by_status(status);
334 } else {
335 nodes = ctx->node_manager->list_nodes();
336 }
337
338 size_t total_count = nodes.size();
339
340 // Apply pagination
341 std::vector<client::remote_node> paginated;
342 for (size_t i = offset; i < nodes.size() && paginated.size() < limit; ++i) {
343 paginated.push_back(nodes[i]);
344 }
345
346 res.code = 200;
347 res.body = nodes_to_json(paginated, total_count);
348 return res;
349 });

351 // POST /api/v1/remote-nodes - Create a new remote node
352 CROW_ROUTE(app, "/api/v1/remote-nodes")
353 .methods(crow::HTTPMethod::POST)([ctx](const crow::request& req) {
354 crow::response res;
355 res.add_header("Content-Type", "application/json");
356 add_cors_headers(res, *ctx);
357
358 if (!ctx->node_manager) {
359 res.code = 503;
360 res.body = make_error_json("SERVICE_UNAVAILABLE",
361 "Remote node manager not configured");
362 return res;
363 }
364
365 std::string error_message;
366 auto node_opt = parse_node_from_json(req.body, error_message);
367 if (!node_opt) {

372

374

379 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
380 now.time_since_epoch()).count();
381 node.node_id = node.ae_title + "_" + std::to_string(ms);

383
384 auto result = ctx->node_manager->add_node(node);
385 if (!result.is_ok()) {
386 res.code = 409;
387 res.body = make_error_json("CONFLICT", result.error().message);
388 return res;
389 }
390
391 // Retrieve the created node to get full details
392 auto created_node = ctx->node_manager->get_node(node.node_id);
393 if (!created_node) {
394 res.code = 201;
395 res.body = node_to_json(node);
396 return res;
397 }
398
399 res.code = 201;
400 res.body = node_to_json(*created_node);
401 return res;
402 });

619 // POST /api/v1/remote-nodes/<nodeId>/query - Query remote PACS
620 CROW_ROUTE(app, "/api/v1/remote-nodes/<string>/query")
621 .methods(crow::HTTPMethod::POST)(
622 [ctx](const crow::request& req, const std::string& node_id) {
623 crow::response res;
624 res.add_header("Content-Type", "application/json");
625 add_cors_headers(res, *ctx);
626
627 if (!ctx->node_manager) {
628 res.code = 503;
629 res.body = make_error_json("SERVICE_UNAVAILABLE",
630 "Remote node manager not configured");
631 return res;
632 }
633
634 auto node = ctx->node_manager->get_node(node_id);
635 if (!node) {
636 res.code = 404;
637 res.body = make_error_json("NOT_FOUND", "Remote node not found");
638 return res;
639 }
640
641 if (!node->supports_find) {
642 res.code = 400;
643 res.body = make_error_json("UNSUPPORTED",
644 "Remote node does not support C-FIND queries");
645 return res;
646 }
647
648 // Parse query parameters from request body
649 auto get_str = [&req](const std::string& key) -> std::string {
650 std::string search = "\"" + key + "\":\"";
651 auto pos = req.body.find(search);
652 if (pos == std::string::npos) return "";
653 pos += search.length();
654 auto end_pos = req.body.find('"', pos);
655 if (end_pos == std::string::npos) return "";
656 return req.body.substr(pos, end_pos - pos);
657 };
658
659 auto level = get_str("level");
660 if (level.empty()) level = "STUDY";
661
662 // Determine calling AE title from dicom_server config
663 std::string calling_ae = "PACS_SCU";
664 if (ctx->dicom_server) {
665 calling_ae = ctx->dicom_server->config().ae_title;
666 }
667
668 // Build association config for C-FIND
669 network::association_config assoc_config;
670 assoc_config.calling_ae_title = calling_ae;
671 assoc_config.called_ae_title = node->ae_title;
672 assoc_config.proposed_contexts.push_back({
673 1,
674 std::string(services::study_root_find_sop_class_uid),
675 {"1.2.840.10008.1.2.1"} // Explicit VR Little Endian
676 });
677
678 auto timeout = std::chrono::milliseconds(
679 node->connection_timeout.count() * 1000);
680 auto assoc_result = network::association::connect(
681 node->host,
682 static_cast<uint16_t>(node->port),
683 assoc_config,
684 timeout);
685
686 if (assoc_result.is_err()) {
687 res.code = 502;
688 res.body = make_error_json("CONNECTION_FAILED",
689 "Failed to connect to remote PACS: " +
690 assoc_result.error().message);
691 return res;
692 }
693
694 auto& assoc = assoc_result.value();
695 services::query_scu scu;
696
697 // Helper to format query results as JSON
698 auto format_query_response = [](
699 const services::query_result& qr) -> std::string {
700 std::ostringstream oss;
701 oss << R"({"matches":[)";
702 for (size_t i = 0; i < qr.matches.size(); ++i) {
703 if (i > 0) oss << ',';
704 oss << '{';
705 bool first_field = true;
706 for (const auto& [tag, element] : qr.matches[i]) {
707 auto val = element.as_string();
708 if (val.is_err()) continue;
709 if (!first_field) oss << ',';
710 first_field = false;

798 // POST /api/v1/remote-nodes/<nodeId>/retrieve - Retrieve from remote PACS
799 CROW_ROUTE(app, "/api/v1/remote-nodes/<string>/retrieve")
800 .methods(crow::HTTPMethod::POST)(
801 [ctx](const crow::request& req, const std::string& node_id) {
802 crow::response res;
803 res.add_header("Content-Type", "application/json");
804 add_cors_headers(res, *ctx);
805
806 if (!ctx->node_manager) {
807 res.code = 503;
808 res.body = make_error_json("SERVICE_UNAVAILABLE",
809 "Remote node manager not configured");
810 return res;
811 }
812
813 if (!ctx->job_manager) {
814 res.code = 503;
815 res.body = make_error_json("SERVICE_UNAVAILABLE",
816 "Job manager not configured");
817 return res;
818 }
819
820 auto node = ctx->node_manager->get_node(node_id);
821 if (!node) {
822 res.code = 404;
823 res.body = make_error_json("NOT_FOUND", "Remote node not found");
824 return res;
825 }
826
827 if (!node->supports_move && !node->supports_get) {
828 res.code = 400;
829 res.body = make_error_json("UNSUPPORTED",
830 "Remote node does not support C-MOVE or C-GET");
831 return res;
832 }
833
834 // Parse retrieve parameters from request body
835 auto get_str = [&req](const std::string& key) -> std::string {
836 std::string search = "\"" + key + "\":\"";
837 auto pos = req.body.find(search);
838 if (pos == std::string::npos) return "";
839 pos += search.length();
840 auto end_pos = req.body.find('"', pos);
841 if (end_pos == std::string::npos) return "";
842 return req.body.substr(pos, end_pos - pos);
843 };
844
845 auto study_uid = get_str("study_uid");
846 if (study_uid.empty()) {
847 res.code = 400;
848 res.body = make_error_json("INVALID_REQUEST",
849 "study_uid is required");
850 return res;
851 }
852
853 auto series_uid = get_str("series_uid");
854 auto priority_str = get_str("priority");
855
856 // Map priority string to enum
857 auto priority = client::job_priority::normal;
858 if (priority_str == "low") {
859 priority = client::job_priority::low;
860 } else if (priority_str == "high") {
861 priority = client::job_priority::high;
862 } else if (priority_str == "urgent") {
863 priority = client::job_priority::urgent;
864 }
865
866 // Create retrieve job
867 std::optional<std::string_view> series_opt;
868 if (!series_uid.empty()) {
869 series_opt = series_uid;
870 }
871
872 auto job_id = ctx->job_manager->create_retrieve_job(
873 node_id, study_uid, series_opt, priority);
874
875 // Start the job
876 auto start_result = ctx->job_manager->start_job(job_id);
877 if (start_result.is_err()) {
878 res.code = 500;
879 res.body = make_error_json("JOB_START_FAILED",
880 "Failed to start retrieve job: " +
881 start_result.error().message);
882 return res;
883 }
884
885 std::ostringstream oss;
886 oss << R"({"job_id":")" << json_escape(job_id)
887 << R"(","status":"pending"})";
888
889 res.code = 202;
890 res.body = oss.str();
891 return res;
892 });
