103document_retrieval_result imaging_document_consumer::extract_references(
104 const core::dicom_dataset& kos_dataset) const {
105
106 document_retrieval_result result;
107 result.success = true;
108
109 // Parse Current Requested Procedure Evidence Sequence (0040,A375)
110 const auto* evidence_elem = kos_dataset.get(
111 consumer_tags::current_requested_procedure_evidence_sequence);
112
113 if (!evidence_elem || !evidence_elem->is_sequence()) {
114 result.error_message = "KOS document does not contain evidence sequence";
115 result.success = false;
116 return result;
117 }
118
119 for (const auto& study_item : evidence_elem->sequence_items()) {
120 // Extract Study Instance UID
121 auto study_uid = study_item.get_string(tags::study_instance_uid);
122 if (!study_uid.empty()) {
123 if (std::find(result.referenced_study_uids.begin(),
124 result.referenced_study_uids.end(),
125 study_uid) == result.referenced_study_uids.end()) {
126 result.referenced_study_uids.push_back(study_uid);
127 }
128 }
129
130 // Parse Referenced Series Sequence
131 const auto* series_elem = study_item.get(
132 consumer_tags::referenced_series_sequence);
133 if (!series_elem || !series_elem->is_sequence()) {
134 continue;
135 }
136
137 for (const auto& series_item : series_elem->sequence_items()) {
138 auto series_uid = series_item.get_string(tags::series_instance_uid);
139 if (!series_uid.empty()) {
140 if (std::find(result.referenced_series_uids.begin(),
141 result.referenced_series_uids.end(),
142 series_uid) == result.referenced_series_uids.end()) {
143 result.referenced_series_uids.push_back(series_uid);
144 }
145 }
146
147 // Parse Referenced SOP Sequence
148 const auto* sop_elem = series_item.get(
149 consumer_tags::referenced_sop_sequence);
150 if (!sop_elem || !sop_elem->is_sequence()) {
151 continue;
152 }
153
154 for (const auto& sop_item : sop_elem->sequence_items()) {
155 auto instance_uid = sop_item.get_string(
156 consumer_tags::referenced_sop_instance_uid);
157 if (!instance_uid.empty()) {
158 result.referenced_instance_uids.push_back(instance_uid);
159 }
160 }
161 }
162 }
163
164 return result;
165}

167std::string imaging_document_consumer::build_wado_rs_url(
168 const std::string& study_uid,
169 const std::string& series_uid,
170 const std::string& instance_uid) const {
171
172 std::ostringstream url;
173 url << config_.wado_rs_url;
174
175 // Ensure base URL doesn't end with '/'
176 if (!config_.wado_rs_url.empty() && config_.wado_rs_url.back() == '/') {
177 // Remove the trailing '/' already added
178 } else {
179 url << '/';
180 }
181
182 url << "studies/" << study_uid
183 << "/series/" << series_uid
184 << "/instances/" << instance_uid;
185
186 return url.str();
187}
