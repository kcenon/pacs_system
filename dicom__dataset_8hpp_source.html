456template <typename T>
457 requires std::is_arithmetic_v<T>
458auto dicom_dataset::get_numeric(dicom_tag tag) const -> std::optional<T> {
459 const auto* elem = get(tag);
460 if (elem == nullptr) {
461 return std::nullopt;
462 }
463
464 auto result = elem->as_numeric<T>();
465 if (result.is_ok()) {
466 return result.value();
467 }
468 return std::nullopt;
469}

473void dicom_dataset::set_numeric(dicom_tag tag, encoding::vr_type vr, T value) {
474 insert(dicom_element::from_numeric<T>(tag, vr, value));
475}
