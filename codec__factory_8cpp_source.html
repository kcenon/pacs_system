74std::unique_ptr<compression_codec> codec_factory::create(
75 std::string_view transfer_syntax_uid) {
76
77 // RLE Lossless (1.2.840.10008.1.2.5)
78 if (transfer_syntax_uid == rle_codec::kTransferSyntaxUID) {
79 return std::make_unique<rle_codec>();
80 }
81
82 // JPEG Baseline (Process 1)
83 if (transfer_syntax_uid == jpeg_baseline_codec::kTransferSyntaxUID) {
84 return std::make_unique<jpeg_baseline_codec>();
85 }
86
87 // JPEG Lossless (Process 14, Selection Value 1)
88 if (transfer_syntax_uid == jpeg_lossless_codec::kTransferSyntaxUID) {
89 return std::make_unique<jpeg_lossless_codec>();
90 }
91
92 // JPEG-LS Lossless (1.2.840.10008.1.2.4.80)
93 if (transfer_syntax_uid == jpeg_ls_codec::kTransferSyntaxUIDLossless) {
94 return std::make_unique<jpeg_ls_codec>(true); // lossless = true
95 }
96
97 // JPEG-LS Near-Lossless (1.2.840.10008.1.2.4.81)
98 if (transfer_syntax_uid == jpeg_ls_codec::kTransferSyntaxUIDNearLossless) {
99 return std::make_unique<jpeg_ls_codec>(false); // lossless = false
100 }
101
102 // JPEG 2000 Lossless Only (1.2.840.10008.1.2.4.90)
103 if (transfer_syntax_uid == jpeg2000_codec::kTransferSyntaxUIDLossless) {
104 return std::make_unique<jpeg2000_codec>(true); // lossless = true
105 }
106
107 // JPEG 2000 (1.2.840.10008.1.2.4.91) - can be lossy or lossless
108 if (transfer_syntax_uid == jpeg2000_codec::kTransferSyntaxUIDLossy) {
109 return std::make_unique<jpeg2000_codec>(false); // lossless = false (default lossy)
110 }
111
112 // HEVC/H.265 Main Profile (1.2.840.10008.1.2.4.107)
113 if (transfer_syntax_uid == hevc_codec::kTransferSyntaxUIDMain) {
114 return std::make_unique<hevc_codec>(false); // main10 = false
115 }
116
117 // HEVC/H.265 Main 10 Profile (1.2.840.10008.1.2.4.108)
118 if (transfer_syntax_uid == hevc_codec::kTransferSyntaxUIDMain10) {
119 return std::make_unique<hevc_codec>(true); // main10 = true
120 }
121
122 // JPEG XL Lossless (1.2.840.10008.1.2.4.110)
123 if (transfer_syntax_uid == jpegxl_codec::kTransferSyntaxUIDLossless) {
124 return std::make_unique<jpegxl_codec>(true, false);
125 }
126
127 // JPEG XL JPEG Recompression (1.2.840.10008.1.2.4.111)
128 if (transfer_syntax_uid == jpegxl_codec::kTransferSyntaxUIDJPEGRecompression) {
129 return std::make_unique<jpegxl_codec>(true, true);
130 }
131
132 // JPEG XL (1.2.840.10008.1.2.4.112) - lossy
133 if (transfer_syntax_uid == jpegxl_codec::kTransferSyntaxUIDLossy) {
134 return std::make_unique<jpegxl_codec>(false);
135 }
136
137 // HTJ2K Lossless Only (1.2.840.10008.1.2.4.201)
138 if (transfer_syntax_uid == htj2k_codec::kTransferSyntaxUIDLossless) {
139 return std::make_unique<htj2k_codec>(true, false);
140 }
141
142 // HTJ2K with RPCL Options (1.2.840.10008.1.2.4.202)
143 if (transfer_syntax_uid == htj2k_codec::kTransferSyntaxUIDRPCL) {
144 return std::make_unique<htj2k_codec>(true, true);
145 }
146
147 // HTJ2K (1.2.840.10008.1.2.4.203) - lossy
148 if (transfer_syntax_uid == htj2k_codec::kTransferSyntaxUIDLossy) {
149 return std::make_unique<htj2k_codec>(false);
150 }
151
152 // Frame Deflate (1.2.840.10008.1.2.11)
153 if (transfer_syntax_uid == frame_deflate_codec::kTransferSyntaxUID) {
154 return std::make_unique<frame_deflate_codec>();
155 }
156
157 return nullptr;
158}

160std::unique_ptr<compression_codec> codec_factory::create(

163}

165std::vector<std::string_view> codec_factory::supported_transfer_syntaxes() {
166 return {kSupportedTransferSyntaxes.begin(), kSupportedTransferSyntaxes.end()};
167}

169bool codec_factory::is_supported(std::string_view transfer_syntax_uid) {
170 for (const auto& uid : kSupportedTransferSyntaxes) {
171 if (uid == transfer_syntax_uid) {
172 return true;
173 }
174 }
175 return false;
176}
177
178bool codec_factory::is_supported(const transfer_syntax& ts) {
179 return is_supported(ts.uid());
180}
