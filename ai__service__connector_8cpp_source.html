86[[nodiscard]] inline std::string escape_string(std::string_view str) {
87 std::string result;
88 result.reserve(str.size() + 10);
89
90 for (char c : str) {
91 switch (c) {
92 case '"':
93 result += "\\\"";
94 break;
95 case '\\':
96 result += "\\\\";
97 break;
98 case '\b':
99 result += "\\b";
100 break;
101 case '\f':
102 result += "\\f";
103 break;
104 case '\n':
105 result += "\\n";
106 break;
107 case '\r':
108 result += "\\r";
109 break;
110 case '\t':
111 result += "\\t";
112 break;
113 default:
114 if (static_cast<unsigned char>(c) < 0x20) {
115 char buf[8];
116 std::snprintf(buf, sizeof(buf), "\\u%04x",
117 static_cast<unsigned char>(c));
118 result += buf;
119 } else {
120 result += c;
121 }
122 break;
123 }
124 }
125
126 return result;
127}

132[[nodiscard]] inline std::string to_iso8601(
133 std::chrono::system_clock::time_point tp) {
134 const auto time_t_val = std::chrono::system_clock::to_time_t(tp);
135 std::tm tm_val{};
136
137#if defined(_MSC_VER)
138 gmtime_s(&tm_val, &time_t_val);
139#else
140 gmtime_r(&time_t_val, &tm_val);
141#endif
142
143 std::ostringstream oss;
144 oss << std::put_time(&tm_val, "%Y-%m-%dT%H:%M:%SZ");
145 return oss.str();
146}

151[[nodiscard]] inline std::optional<std::chrono::system_clock::time_point>
152from_iso8601(const std::string& str) {
153 std::tm tm_val{};
154 std::istringstream iss(str);
155 iss >> std::get_time(&tm_val, "%Y-%m-%dT%H:%M:%SZ");
156
157 if (iss.fail()) {
158 return std::nullopt;
159 }
160
161#if defined(_MSC_VER)
162 auto time_t_val = _mkgmtime(&tm_val);
163#else
164 auto time_t_val = timegm(&tm_val);
165#endif
166
167 return std::chrono::system_clock::from_time_t(time_t_val);
168}

173[[nodiscard]] inline std::optional<std::string> extract_string(
174 const std::string& json,
175 const std::string& key) {
176 std::string search_key = "\"" + key + "\"";
177 auto pos = json.find(search_key);
178 if (pos == std::string::npos) {
179 return std::nullopt;
180 }
181
182 pos = json.find(':', pos);
183 if (pos == std::string::npos) {
184 return std::nullopt;
185 }
186
187 pos = json.find('"', pos);
188 if (pos == std::string::npos) {
189 return std::nullopt;
190 }
191
192 auto start = pos + 1;
193 auto end = json.find('"', start);
194 if (end == std::string::npos) {
195 return std::nullopt;
196 }
197
198 return json.substr(start, end - start);
199}

204[[nodiscard]] inline std::optional<int> extract_int(
205 const std::string& json,
206 const std::string& key) {
207 std::string search_key = "\"" + key + "\"";
208 auto pos = json.find(search_key);
209 if (pos == std::string::npos) {
210 return std::nullopt;
211 }
212
213 pos = json.find(':', pos);
214 if (pos == std::string::npos) {
215 return std::nullopt;
216 }
217
218 // Skip whitespace
219 pos++;
220 while (pos < json.size() && std::isspace(json[pos])) {
221 pos++;
222 }
223
224 if (pos >= json.size()) {
225 return std::nullopt;
226 }
227
228 try {
229 std::size_t end_pos;
230 int value = std::stoi(json.substr(pos), &end_pos);
231 return value;
232 } catch (...) {
233 return std::nullopt;
234 }
235}

240[[nodiscard]] inline std::string build_request_json(const inference_request& request) {
241 std::ostringstream oss;
242 oss << "{"
243 << R"("study_instance_uid":")" << escape_string(request.study_instance_uid) << "\"";
244
245 if (request.series_instance_uid) {
246 oss << R"(,"series_instance_uid":")"
247 << escape_string(*request.series_instance_uid) << "\"";
248 }
249
250 oss << R"(,"model_id":")" << escape_string(request.model_id) << "\""
251 << R"(,"priority":)" << request.priority;
252
253 if (request.callback_url) {
254 oss << R"(,"callback_url":")" << escape_string(*request.callback_url) << "\"";
255 }
256
257 if (!request.parameters.empty()) {
258 oss << R"(,"parameters":{)";
259 bool first = true;
260 for (const auto& [key, value] : request.parameters) {
261 if (!first) oss << ",";
262 oss << "\"" << escape_string(key) << "\":\"" << escape_string(value) << "\"";
263 first = false;
264 }
265 oss << "}";
266 }
267
268 if (!request.metadata.empty()) {
269 oss << R"(,"metadata":{)";
270 bool first = true;
271 for (const auto& [key, value] : request.metadata) {
272 if (!first) oss << ",";
273 oss << "\"" << escape_string(key) << "\":\"" << escape_string(value) << "\"";
274 first = false;
275 }
276 oss << "}";
277 }
278
279 oss << "}";
280 return oss.str();
281}

289

303 status.status = inference_status_code::completed;

305 status.status = inference_status_code::failed;

309 status.status = inference_status_code::timeout;
310 }
311 }
312
313 auto progress = extract_int(json, "progress");
314 if (progress) {
315 status.progress = *progress;
316 }
317
318 auto message = extract_string(json, "message");
319 if (message) {
320 status.message = *message;
321 }
322
323 auto error = extract_string(json, "error_message");
324 if (error) {
325 status.error_message = *error;
326 }
327
328 auto created_at = extract_string(json, "created_at");
329 if (created_at) {
330 auto tp = from_iso8601(*created_at);
331 if (tp) {
332 status.created_at = *tp;

334 }
335
336 auto started_at = extract_string(json, "started_at");
337 if (started_at) {
338 auto tp = from_iso8601(*started_at);

343

345 if (completed_at) {

347 if (tp) {
348 status.completed_at = *tp;

389[[nodiscard]] inline std::string base64_encode(std::string_view input) {
390 static constexpr char table[] =
391 "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
392
393 std::string result;
394 result.reserve(((input.size() + 2) / 3) * 4);
395
396 auto ptr = reinterpret_cast<const unsigned char*>(input.data());
397 auto len = input.size();
398
399 for (std::size_t i = 0; i < len; i += 3) {
400 uint32_t octet_a = ptr[i];
401 uint32_t octet_b = (i + 1 < len) ? ptr[i + 1] : 0;
402 uint32_t octet_c = (i + 2 < len) ? ptr[i + 2] : 0;
403
404 uint32_t triple = (octet_a << 16) | (octet_b << 8) | octet_c;
405
406 result += table[(triple >> 18) & 0x3F];
407 result += table[(triple >> 12) & 0x3F];
408 result += (i + 1 < len) ? table[(triple >> 6) & 0x3F] : '=';
409 result += (i + 2 < len) ? table[triple & 0x3F] : '=';
410 }
411
412 return result;
413}

418[[nodiscard]] inline std::vector<std::string> extract_json_array_objects(
419 const std::string& json) {
420 std::vector<std::string> objects;
421
422 // Find array start
423 auto arr_start = json.find('[');
424 if (arr_start == std::string::npos) {
425 return objects;
426 }
427
428 int depth = 0;
429 std::size_t obj_start = std::string::npos;
430
431 for (std::size_t i = arr_start + 1; i < json.size(); ++i) {
432 char c = json[i];
433
434 if (c == '"') {
435 // Skip string content
436 ++i;
437 while (i < json.size() && json[i] != '"') {
438 if (json[i] == '\\') ++i; // Skip escaped character
439 ++i;
440 }
441 continue;
442 }
443
444 if (c == '{') {
445 if (depth == 0) {
446 obj_start = i;
447 }
448 ++depth;
449 } else if (c == '}') {
450 --depth;
451 if (depth == 0 && obj_start != std::string::npos) {
452 objects.push_back(json.substr(obj_start, i - obj_start + 1));
453 obj_start = std::string::npos;
454 }
455 } else if (c == ']' && depth == 0) {
456 break;
457 }
458 }
459
460 return objects;
461}

485class socket_handle {
486public:
487#ifdef _WIN32
488 using native_type = SOCKET;
489 static constexpr native_type invalid_value = INVALID_SOCKET;
490#else
491 using native_type = int;
492 static constexpr native_type invalid_value = -1;
493#endif
494
495 socket_handle() = default;
496 explicit socket_handle(native_type fd) : fd_(fd) {}
497 ~socket_handle() { close(); }
498
499 socket_handle(const socket_handle&) = delete;
500 socket_handle& operator=(const socket_handle&) = delete;
501
502 socket_handle(socket_handle&& other) noexcept : fd_(other.fd_) {
503 other.fd_ = invalid_value;
504 }
505
506 socket_handle& operator=(socket_handle&& other) noexcept {
507 if (this != &other) {
508 close();
509 fd_ = other.fd_;
510 other.fd_ = invalid_value;
511 }
512 return *this;
513 }
514
515 [[nodiscard]] bool valid() const noexcept { return fd_ != invalid_value; }
516 [[nodiscard]] native_type get() const noexcept { return fd_; }
517
518 void close() noexcept {
519 if (fd_ != invalid_value) {
520#ifdef _WIN32
521 closesocket(fd_);
522#else
523 ::close(fd_);
524#endif
525 fd_ = invalid_value;
526 }
527 }
528
529private:
530 native_type fd_{invalid_value};
531};

536struct parsed_url {
537 std::string host;
538 std::string port;
539 std::string path;
540
541 static std::optional<parsed_url> parse(const std::string& url) {
542 parsed_url result;
543
544 // Find scheme end
545 auto scheme_end = url.find("://");
546 if (scheme_end == std::string::npos) return std::nullopt;
547
548 auto host_start = scheme_end + 3;
549 auto path_start = url.find('/', host_start);
550
551 std::string host_port;
552 if (path_start == std::string::npos) {
553 host_port = url.substr(host_start);
554 result.path = "/";
555 } else {
556 host_port = url.substr(host_start, path_start - host_start);
557 result.path = url.substr(path_start);
558 }
559
560 auto colon = host_port.find(':');
561 if (colon != std::string::npos) {
562 result.host = host_port.substr(0, colon);
563 result.port = host_port.substr(colon + 1);
564 } else {
565 result.host = host_port;
566 result.port = "80";
567 }
568
569 return result;
570 }

635 [[nodiscard]] auto send_request(
636 const std::string& method,
637 const std::string& path,
638 const std::map<std::string, std::string>& headers,
639 const std::string& body) -> Result<http_response> {
640
641 std::string full_url = config_.base_url + path;
642 auto url = parsed_url::parse(full_url);
643 if (!url) {
644 return error_info(-1, "Invalid URL: " + full_url, "ai_service_connector");
645 }
646
647 // Resolve hostname
648 struct addrinfo hints{};
649 hints.ai_family = AF_UNSPEC;
650 hints.ai_socktype = SOCK_STREAM;
651 hints.ai_protocol = IPPROTO_TCP;
652
653 struct addrinfo* addr_result = nullptr;
654 int gai_err = getaddrinfo(url->host.c_str(), url->port.c_str(),
655 &hints, &addr_result);
656 if (gai_err != 0) {
657 return error_info(-1,
658 "DNS resolution failed for " + url->host + ": " +
659 gai_strerror(gai_err),
660 "ai_service_connector");
661 }
662
663 // RAII cleanup for addrinfo
664 auto addr_cleanup = [](struct addrinfo* p) { freeaddrinfo(p); };
665 std::unique_ptr<struct addrinfo, decltype(addr_cleanup)>
666 addr_guard(addr_result, addr_cleanup);
667
668 // Create and connect socket
669 socket_handle sock(
670 ::socket(addr_result->ai_family, addr_result->ai_socktype,
671 addr_result->ai_protocol));
672 if (!sock.valid()) {
673 return error_info(-1, "Failed to create socket", "ai_service_connector");
674 }
675
676 // Set socket timeout
677 auto timeout_sec = std::chrono::duration_cast<std::chrono::seconds>(
678 config_.connection_timeout).count();
679 if (timeout_sec <= 0) timeout_sec = 30;
680
681#ifdef _WIN32
682 DWORD tv = static_cast<DWORD>(timeout_sec * 1000);
683 setsockopt(sock.get(), SOL_SOCKET, SO_RCVTIMEO,
684 reinterpret_cast<const char*>(&tv), sizeof(tv));
685 setsockopt(sock.get(), SOL_SOCKET, SO_SNDTIMEO,
686 reinterpret_cast<const char*>(&tv), sizeof(tv));
687#else
688 struct timeval tv{};
689 tv.tv_sec = timeout_sec;
690 setsockopt(sock.get(), SOL_SOCKET, SO_RCVTIMEO, &tv, sizeof(tv));
691 setsockopt(sock.get(), SOL_SOCKET, SO_SNDTIMEO, &tv, sizeof(tv));
692#endif
693
694 // Connect
695 if (::connect(sock.get(), addr_result->ai_addr,
696 static_cast<int>(addr_result->ai_addrlen)) != 0) {
697 return error_info(-1,
698 "Connection failed to " + url->host + ":" + url->port,
699 "ai_service_connector");
700 }
701
702 // Build HTTP request
703 std::ostringstream request_stream;
704 request_stream << method << " " << url->path << " HTTP/1.1\r\n";
705 request_stream << "Host: " << url->host << "\r\n";
706 request_stream << "Connection: close\r\n";
707
708 for (const auto& [name, value] : headers) {
709 request_stream << name << ": " << value << "\r\n";
710 }
711
712 if (!body.empty()) {
713 request_stream << "Content-Length: " << body.size() << "\r\n";
714 }
715 request_stream << "\r\n";
716 request_stream << body;
717
718 std::string request_data = request_stream.str();
719
720 // Send request
721 auto total_sent = std::size_t{0};
722 while (total_sent < request_data.size()) {
723 auto sent = ::send(sock.get(),
724 request_data.data() + total_sent,
725 static_cast<int>(request_data.size() - total_sent),
726 0);
727 if (sent <= 0) {
728 return error_info(-1, "Failed to send HTTP request",
729 "ai_service_connector");
730 }
731 total_sent += static_cast<std::size_t>(sent);
732 }
733
734 // Receive response
735 std::string response_data;
736 char buffer[4096];
737 while (true) {
738 auto received = ::recv(sock.get(), buffer, sizeof(buffer), 0);
739 if (received < 0) {
740 return error_info(-1, "Failed to receive HTTP response",
741 "ai_service_connector");
742 }
743 if (received == 0) break; // Connection closed
744 response_data.append(buffer, static_cast<std::size_t>(received));
745 }
746
747 // Parse HTTP response
748 return parse_http_response(response_data);
749 }

751 [[nodiscard]] static auto parse_http_response(const std::string& raw)
752 -> Result<http_response> {
753 http_response response;
754
755 auto header_end = raw.find("\r\n\r\n");
756 if (header_end == std::string::npos) {
757 return error_info(-1, "Malformed HTTP response", "ai_service_connector");
758 }
759
760 // Parse status line
761 auto first_line_end = raw.find("\r\n");
762 std::string status_line = raw.substr(0, first_line_end);
763
764 // Parse "HTTP/1.x STATUS_CODE REASON"
765 auto first_space = status_line.find(' ');
766 if (first_space == std::string::npos) {
767 return error_info(-1, "Invalid HTTP status line", "ai_service_connector");
768 }
769 auto code_start = first_space + 1;
770 try {
771 response.status_code = std::stoi(status_line.substr(code_start));
772 } catch (...) {
773 return error_info(-1, "Invalid HTTP status code", "ai_service_connector");
774 }
775
776 // Parse headers
777 auto headers_str = raw.substr(first_line_end + 2,
778 header_end - first_line_end - 2);
779 std::istringstream header_stream(headers_str);
780 std::string line;
781 while (std::getline(header_stream, line)) {
782 if (!line.empty() && line.back() == '\r') line.pop_back();
783 auto colon = line.find(':');
784 if (colon != std::string::npos) {
785 auto name = line.substr(0, colon);
786 auto value = line.substr(colon + 1);
787 // Trim leading whitespace from value
788 auto val_start = value.find_first_not_of(' ');
789 if (val_start != std::string::npos) {
790 value = value.substr(val_start);
791 }
792 response.headers[name] = value;
793 }
794 }
795
796 // Extract body
797 response.body = raw.substr(header_end + 4);
798
799 return response;
800 }

802 [[nodiscard]] std::map<std::string, std::string> build_headers() const {
803 std::map<std::string, std::string> headers;
804 headers["Content-Type"] = "application/json";
805
806 switch (config_.auth_type) {
807 case authentication_type::api_key:
808 headers["X-API-Key"] = config_.api_key;
809 break;
810 case authentication_type::bearer_token:

812 break;
813 case authentication_type::basic: {

815 headers["Authorization"] = "Basic " + json_util::base64_encode(credentials);

839 inference_status status;

854 std::lock_guard lock(mutex_);
855
856 auto it = jobs_.find(job_id);
857 if (it != jobs_.end()) {
858 it->second = status;
859
860 // If job completed or failed, record metrics
861 if (status.status == inference_status_code::completed ||
862 status.status == inference_status_code::failed) {
863
864 // Note: request_map_ entry may be used for future logging
865 (void)request_map_[job_id];
866 auto duration = status.completed_at.value_or(std::chrono::system_clock::now())
867 - status.created_at;
868
869 kcenon::pacs::integration::monitoring_adapter::record_timing(
870 metrics::inference_duration,
871 std::chrono::duration_cast<std::chrono::nanoseconds>(duration));
872
873 if (status.status == inference_status_code::completed) {
874 kcenon::pacs::integration::monitoring_adapter::increment_counter(
875 metrics::inference_requests_success);
876 } else {
877 kcenon::pacs::integration::monitoring_adapter::increment_counter(
878 metrics::inference_requests_failed);
879 }
880 }
881 }

992
993 // Validate request

1001 // Build request JSON
1002 std::string body = json_util::build_request_json(request);
1003
1004 // Send request

1009 response.error().message);

1011 }
1012
1013 if (!response.value().is_success()) {
1014 return error_info(
1015 response.value().status_code,
1016 "AI service returned error: " + response.value().body,
1017 "ai_service_connector");
1018 }
1019
1020 // Parse response to get job ID
1021 auto job_id = json_util::extract_string(response.value().body, "job_id");
1022 if (!job_id) {
1023 return error_info(-1, "Failed to parse job ID from response", "ai_service_connector");
1024 }
1025
1026 // Track the job
1027 job_tracker_->add_job(*job_id, request);
1028
1029 kcenon::pacs::integration::logger_adapter::info(
1030 "Inference request submitted: job_id={}, study={}, model={}",
1031 *job_id, request.study_instance_uid, request.model_id);

1033 kcenon::pacs::integration::monitoring_adapter::increment_counter(
1034 metrics::inference_requests_total);

1039 auto check_status(const std::string& job_id) -> Result<inference_status> {
1040 if (!initialized_) {
1041 return error_info(-1, "AI service connector not initialized", "ai_service_connector");
1042 }
1043
1044 // Check local cache first
1045 auto cached = job_tracker_->get_status(job_id);
1046 if (cached && (cached->status == inference_status_code::completed ||
1047 cached->status == inference_status_code::failed ||
1048 cached->status == inference_status_code::cancelled)) {
1049 return *cached;
1050 }
1051
1052 // Query remote service
1053 auto response = http_client_->get("/inference/" + job_id);
1054 if (response.is_err()) {
1055 return response.error();
1056 }
1057
1058 if (!response.value().is_success()) {
1059 if (response.value().status_code == 404) {
1060 return error_info(404, "Job not found: " + job_id, "ai_service_connector");
1061 }
1062 return error_info(
1063 response.value().status_code,
1064 "Failed to get job status",
1065 "ai_service_connector");
1066 }
1067
1068 // Parse status
1069 auto status = json_util::parse_status_json(response.value().body);
1070 if (!status) {
1071 return error_info(-1, "Failed to parse status response", "ai_service_connector");
1072 }
1073
1074 // Update local cache
1075 job_tracker_->update_status(job_id, *status);
1076
1077 return *status;
1078 }

1080 auto cancel(const std::string& job_id) -> Result<std::monostate> {

1110 auto wait_for_completion(const std::string& job_id,
1111 std::chrono::milliseconds timeout,
1112 status_callback callback) -> Result<inference_status> {
1113 if (!initialized_) {
1114 return error_info(-1, "AI service connector not initialized", "ai_service_connector");
1115 }
1116
1117 auto start_time = std::chrono::steady_clock::now();
1118 auto deadline = start_time + timeout;
1119
1120 while (std::chrono::steady_clock::now() < deadline) {
1121 auto result = check_status(job_id);
1122 if (result.is_err()) {
1123 return result;
1124 }
1125
1126 auto& status = result.value();
1127
1128 // Call callback if provided
1129 if (callback) {
1130 callback(status);
1131 }
1132
1133 // Check if job is complete
1134 if (status.status == inference_status_code::completed ||
1135 status.status == inference_status_code::failed ||
1136 status.status == inference_status_code::cancelled ||
1137 status.status == inference_status_code::timeout) {
1138 return status;
1139 }
1140
1141 // Wait before polling again
1142 std::this_thread::sleep_for(config_.polling_interval);
1143 }
1144
1145 // Timeout
1146 inference_status timeout_status;
1147 timeout_status.job_id = job_id;
1148 timeout_status.status = inference_status_code::timeout;
1149 timeout_status.message = "Timed out waiting for job completion";
1150 return timeout_status;
1151 }
