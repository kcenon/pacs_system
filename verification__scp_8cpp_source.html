46network::Result<std::monostate> verification_scp::handle_message(
47 network::association& assoc,
48 uint8_t context_id,
49 const network::dimse::dimse_message& request) {
50
51 using namespace network::dimse;
52
53 // Verify the message is a C-ECHO request
54 if (request.command() != command_field::c_echo_rq) {
55 return kcenon::pacs::pacs_void_error(
56 kcenon::pacs::error_codes::echo_unexpected_command,
57 "Expected C-ECHO-RQ but received " +
58 std::string(to_string(request.command())));
59 }
60
61 // Build C-ECHO response using the factory function
62 auto response = make_c_echo_rsp(
63 request.message_id(),
64 status_success,
65 verification_sop_class_uid
66 );
67
68 // Send the response
69 return assoc.send_dimse(context_id, response);
70}
