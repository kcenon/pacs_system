38std::string_view to_string(tls_profile profile) noexcept {
39 switch (profile) {
40 case tls_profile::bcp195_basic:
41 return "BCP195-Basic";
42 case tls_profile::bcp195_non_downgrading:
43 return "BCP195-NonDowngrading";
44 case tls_profile::bcp195_extended:
45 return "BCP195-Extended";
46 }
47 return "Unknown";
48}
49
50std::optional<tls_profile> parse_tls_profile(std::string_view str) noexcept {
51 if (str == "BCP195-Basic" || str == "basic")
52 return tls_profile::bcp195_basic;
53 if (str == "BCP195-NonDowngrading" || str == "non-downgrading")
54 return tls_profile::bcp195_non_downgrading;
55 if (str == "BCP195-Extended" || str == "extended")
56 return tls_profile::bcp195_extended;
57 return std::nullopt;
58}

105tls_policy tls_policy::from_profile(tls_profile profile) {
106 switch (profile) {
107 case tls_profile::bcp195_basic:
108 return bcp195_basic_profile();
109 case tls_profile::bcp195_non_downgrading:
110 return bcp195_non_downgrading_profile();
111 case tls_profile::bcp195_extended:
112 return bcp195_extended_profile();
113 }
114 return bcp195_non_downgrading_profile();
115}
