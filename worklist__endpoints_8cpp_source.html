76std::string format_datetime(
77 const std::chrono::system_clock::time_point &tp) {
78 auto time = std::chrono::system_clock::to_time_t(tp);
79 std::ostringstream oss;
80 oss << std::put_time(std::gmtime(&time), "%Y-%m-%dT%H:%M:%SZ");
81 return oss.str();
82}

95 << R"(","sex":")" << json_escape(item.sex)
96 << R"(","accession_no":")" << json_escape(item.accession_no)
97 << R"(","requested_proc_id":")" << json_escape(item.requested_proc_id)
98 << R"(","study_uid":")" << json_escape(item.study_uid)
99 << R"(","scheduled_datetime":")" << json_escape(item.scheduled_datetime)
100 << R"(","station_ae":")" << json_escape(item.station_ae)
101 << R"(","station_name":")" << json_escape(item.station_name)
102 << R"(","modality":")" << json_escape(item.modality)
103 << R"(","procedure_desc":")" << json_escape(item.procedure_desc)
104 << R"(","protocol_code":")" << json_escape(item.protocol_code)
105 << R"(","referring_phys":")" << json_escape(item.referring_phys)
106 << R"(","referring_phys_id":")" << json_escape(item.referring_phys_id)
107 << R"(","created_at":")" << format_datetime(item.created_at)
108 << R"(","updated_at":")" << format_datetime(item.updated_at)
109 << R"("})";
110 return oss.str();

116std::string worklist_items_to_json(
117 const std::vector<storage::worklist_item> &items,
118 size_t total_count) {
119 std::ostringstream oss;
120 oss << R"({"data":[)";
121 for (size_t i = 0; i < items.size(); ++i) {
122 if (i > 0) {
123 oss << ",";
124 }
125 oss << worklist_item_to_json(items[i]);
126 }
127 oss << R"(],"pagination":{"total":)" << total_count
128 << R"(,"count":)" << items.size() << "}}";
129 return oss.str();
130}

135std::pair<size_t, size_t> parse_pagination(const crow::request &req) {
136 size_t limit = 20; // Default limit
137 size_t offset = 0;
138
139 auto limit_param = req.url_params.get("limit");
140 if (limit_param) {
141 try {
142 limit = std::stoul(limit_param);
143 if (limit > 100) {
144 limit = 100; // Cap at 100
145 }
146 } catch (...) {
147 // Use default
148 }
149 }
150
151 auto offset_param = req.url_params.get("offset");
152 if (offset_param) {
153 try {
154 offset = std::stoul(offset_param);
155 } catch (...) {
156 // Use default
157 }
158 }
159
160 return {limit, offset};
161}

166std::optional<storage::worklist_item> parse_worklist_item_json(
167 const std::string &body) {
168 storage::worklist_item item;
169
170 // Simple JSON parsing for required fields
171 auto get_json_string = [&body](const std::string &key) -> std::string {
172 std::string search_key = "\"" + key + "\":\"";
173 auto pos = body.find(search_key);
174 if (pos == std::string::npos) {
175 return "";
176 }
177 pos += search_key.length();
178 auto end_pos = body.find("\"", pos);
179 if (end_pos == std::string::npos) {
180 return "";
181 }
182 return body.substr(pos, end_pos - pos);
183 };
184
185 item.step_id = get_json_string("step_id");
186 item.patient_id = get_json_string("patient_id");
187 item.patient_name = get_json_string("patient_name");
188 item.birth_date = get_json_string("birth_date");
189 item.sex = get_json_string("sex");
190 item.accession_no = get_json_string("accession_no");
191 item.requested_proc_id = get_json_string("requested_proc_id");
192 item.study_uid = get_json_string("study_uid");
193 item.scheduled_datetime = get_json_string("scheduled_datetime");
194 item.station_ae = get_json_string("station_ae");
195 item.station_name = get_json_string("station_name");
196 item.modality = get_json_string("modality");
197 item.procedure_desc = get_json_string("procedure_desc");
198 item.protocol_code = get_json_string("protocol_code");
199 item.referring_phys = get_json_string("referring_phys");
200 item.referring_phys_id = get_json_string("referring_phys_id");
201 item.step_status = get_json_string("step_status");
202
203 // Set default status if not provided
204 if (item.step_status.empty()) {
205 item.step_status = "SCHEDULED";
206 }
207
208 // Validate required fields
209 if (!item.is_valid()) {
210 return std::nullopt;
211 }
212
213 return item;
214}

219void register_worklist_endpoints_impl(
220 crow::SimpleApp &app,
221 std::shared_ptr<rest_server_context> ctx) {
222 // GET /api/v1/worklist - List scheduled procedures (paginated)
223 CROW_ROUTE(app, "/api/v1/worklist")
224 .methods(crow::HTTPMethod::GET)([ctx](const crow::request &req) {
225 crow::response res;
226 res.add_header("Content-Type", "application/json");
227 add_cors_headers(res, *ctx);
228
229 if (!ctx->database) {
230 res.code = 503;
231 res.body =
232 make_error_json("DATABASE_UNAVAILABLE", "Database not configured");
233 return res;
234 }
235
236 // Parse pagination
237 auto [limit, offset] = parse_pagination(req);
238
239 // Build query from URL parameters
240 storage::worklist_query query;
241 query.limit = limit;
242 query.offset = offset;
243
244 auto station_ae = req.url_params.get("station_ae");
245 if (station_ae) {
246 query.station_ae = station_ae;
247 }
248
249 auto modality = req.url_params.get("modality");
250 if (modality) {
251 query.modality = modality;
252 }
253
254 auto scheduled_date_from = req.url_params.get("scheduled_date_from");
255 if (scheduled_date_from) {
256 query.scheduled_date_from = scheduled_date_from;
257 }
258
259 auto scheduled_date_to = req.url_params.get("scheduled_date_to");
260 if (scheduled_date_to) {
261 query.scheduled_date_to = scheduled_date_to;
262 }
263
264 auto patient_id = req.url_params.get("patient_id");
265 if (patient_id) {
266 query.patient_id = patient_id;
267 }
268
269 auto patient_name = req.url_params.get("patient_name");
270 if (patient_name) {
271 query.patient_name = patient_name;
272 }
273
274 auto accession_no = req.url_params.get("accession_no");
275 if (accession_no) {
276 query.accession_no = accession_no;
277 }
278
279 auto step_id = req.url_params.get("step_id");
280 if (step_id) {
281 query.step_id = step_id;
282 }
283
284 // Check if we should include all statuses
285 auto include_all = req.url_params.get("include_all_status");
286 if (include_all && std::string(include_all) == "true") {
287 query.include_all_status = true;
288 }
289
290 // Get total count (without pagination)
291 storage::worklist_query count_query = query;
292 count_query.limit = 0;
293 count_query.offset = 0;
294 auto all_items_result = ctx->database->query_worklist(count_query);
295 if (!all_items_result.is_ok()) {
296 res.code = 500;
297 res.body = make_error_json("QUERY_ERROR",
298 all_items_result.error().message);
299 return res;
300 }
301 size_t total_count = all_items_result.value().size();
302
303 // Get paginated results
304 auto items_result = ctx->database->query_worklist(query);
305 if (!items_result.is_ok()) {
306 res.code = 500;
307 res.body = make_error_json("QUERY_ERROR",
308 items_result.error().message);
309 return res;
310 }
311
312 res.code = 200;
313 res.body = worklist_items_to_json(items_result.value(), total_count);
314 return res;
315 });

317 // POST /api/v1/worklist - Create worklist item
318 CROW_ROUTE(app, "/api/v1/worklist")
319 .methods(crow::HTTPMethod::POST)([ctx](const crow::request &req) {
320 crow::response res;
321 res.add_header("Content-Type", "application/json");
322 add_cors_headers(res, *ctx);
323
324 if (!ctx->database) {
325 res.code = 503;
326 res.body =
327 make_error_json("DATABASE_UNAVAILABLE", "Database not configured");
328 return res;
329 }
330
331 // Parse request body
332 auto item = parse_worklist_item_json(req.body);
333 if (!item) {
334 res.code = 400;
335 res.body = make_error_json(
336 "INVALID_REQUEST",
337 "Missing required fields: step_id, patient_id, modality, "
338 "scheduled_datetime");
339 return res;
340 }
341
342 // Add worklist item
343 auto result = ctx->database->add_worklist_item(*item);
344 if (result.is_err()) {
345 res.code = 500;
346 res.body = make_error_json("CREATE_FAILED", result.error().message);
347 return res;
348 }
349
350 // Fetch the created item to return
351 auto created_item = ctx->database->find_worklist_by_pk(result.value());
352 if (created_item) {
353 res.code = 201;
354 res.body = worklist_item_to_json(*created_item);
355 } else {
356 res.code = 201;
357 res.body = make_success_json("Worklist item created");
358 }
359 return res;
360 });

389 // PUT /api/v1/worklist/:id - Update worklist item
390 CROW_ROUTE(app, "/api/v1/worklist/<int>")
391 .methods(crow::HTTPMethod::PUT)(
392 [ctx](const crow::request &req, int pk) {
393 crow::response res;
394 res.add_header("Content-Type", "application/json");
395 add_cors_headers(res, *ctx);
396
397 if (!ctx->database) {
398 res.code = 503;
399 res.body = make_error_json("DATABASE_UNAVAILABLE",
400 "Database not configured");
401 return res;
402 }
403
404 // Verify item exists
405 auto existing_item = ctx->database->find_worklist_by_pk(pk);
406 if (!existing_item) {
407 res.code = 404;
408 res.body = make_error_json("NOT_FOUND", "Worklist item not found");
409 return res;
410 }
411
412 // Parse new status from request body
413 auto get_json_string = [&req](const std::string &key) -> std::string {
414 std::string search_key = "\"" + key + "\":\"";
415 auto pos = req.body.find(search_key);
416 if (pos == std::string::npos) {
417 return "";
418 }
419 pos += search_key.length();
420 auto end_pos = req.body.find("\"", pos);
421 if (end_pos == std::string::npos) {
422 return "";
423 }
424 return req.body.substr(pos, end_pos - pos);
425 };
426
427 std::string new_status = get_json_string("step_status");
428 if (new_status.empty()) {
429 res.code = 400;
430 res.body = make_error_json("INVALID_REQUEST",
431 "step_status is required for update");
432 return res;
433 }
434
435 // Update status
436 auto result = ctx->database->update_worklist_status(
437 existing_item->step_id,
438 existing_item->accession_no,
439 new_status);
440
441 if (result.is_err()) {
442 res.code = 500;
443 res.body = make_error_json("UPDATE_FAILED", result.error().message);
444 return res;
445 }
446
447 // Fetch updated item
448 auto updated_item = ctx->database->find_worklist_by_pk(pk);
449 if (updated_item) {
450 res.code = 200;
451 res.body = worklist_item_to_json(*updated_item);
452 } else {
453 res.code = 200;
454 res.body = make_success_json("Worklist item updated");
455 }
456 return res;
457 });
