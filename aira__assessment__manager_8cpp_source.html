66bool assessment_manager::store_assessment(
67 const core::dicom_dataset& assessment_sr) {
68
69 auto uid = assessment_sr.get_string(tags::sop_instance_uid);
70 if (uid.empty()) {
71 return false;
72 }
73
74 auto info = parse_assessment_info(assessment_sr);
75 datasets_[uid] = assessment_sr;
76 metadata_[uid] = std::move(info);
77
78 return true;
79}

171assessment_info assessment_manager::parse_assessment_info(
172 const core::dicom_dataset& sr) const {
173
174 assessment_info info;
175 info.assessment_uid = sr.get_string(tags::sop_instance_uid);
176 info.assessment_time = std::chrono::system_clock::now();
177
178 // Parse completion flag to determine status
179 auto completion = sr.get_string(manager_tags::completion_flag);
180 if (completion == "COMPLETE") {
181 info.status = assessment_status::final_;
182 } else {
183 info.status = assessment_status::draft;
184 }
185
186 // Parse Content Sequence for assessment type and assessor
187 const auto* content_elem = sr.get(manager_tags::content_sequence);
188 if (content_elem && content_elem->is_sequence()) {
189 for (const auto& item : content_elem->sequence_items()) {
190 auto vt = item.get_string(manager_tags::value_type);
191
192 // Extract assessment type from CODE items
193 if (vt == "CODE") {
194 const auto* concept_name = item.get(
195 manager_tags::concept_name_code_sequence);
196 if (concept_name && concept_name->is_sequence() &&
197 !concept_name->sequence_items().empty()) {
198 auto code = concept_name->sequence_items()[0].get_string(
199 manager_tags::code_value);
200 if (code == "AIRA-010") {
201 // Assessment Decision item
202 const auto* concept_code = item.get(
203 manager_tags::concept_code_sequence);
204 if (concept_code && concept_code->is_sequence() &&
205 !concept_code->sequence_items().empty()) {
206 auto type_code =
207 concept_code->sequence_items()[0].get_string(
208 manager_tags::code_value);
209 if (type_code == "AIRA-ACCEPT") {
210 info.type = assessment_type::accept;
211 } else if (type_code == "AIRA-MODIFY") {
212 info.type = assessment_type::modify;
213 } else if (type_code == "AIRA-REJECT") {
214 info.type = assessment_type::reject;
215 }
216 }
217 }
218 }
219 }
220
221 // Extract assessor name from PNAME items
222 if (vt == "PNAME") {
223 auto name = item.get_string(manager_tags::person_name);
224 if (!name.empty()) {
225 info.assessor_name = name;
226 }
227 }
228 }
229 }
230
231 // Parse Referenced SOP Sequence for AI result UID
232 const auto* ref_elem = sr.get(manager_tags::referenced_sop_sequence);
233 if (ref_elem && ref_elem->is_sequence() &&
234 !ref_elem->sequence_items().empty()) {
235 info.ai_result_uid = ref_elem->sequence_items()[0].get_string(
236 manager_tags::referenced_sop_instance_uid);
237 }
238
239 return info;
240}
