120struct operation_counter {
121 std::atomic<std::uint64_t> success_count{0};
122 std::atomic<std::uint64_t> failure_count{0};
123 std::atomic<std::uint64_t> total_duration_us{0};
124 std::atomic<std::uint64_t> min_duration_us{UINT64_MAX};
125 std::atomic<std::uint64_t> max_duration_us{0};
126
128 [[nodiscard]] std::uint64_t total_count() const noexcept {
129 return success_count.load(std::memory_order_relaxed) +
130 failure_count.load(std::memory_order_relaxed);
131 }
132
134 [[nodiscard]] std::uint64_t average_duration_us() const noexcept {
135 const auto total = total_count();
136 if (total == 0) {
137 return 0;
138 }
139 return total_duration_us.load(std::memory_order_relaxed) / total;
140 }
141
143 void record_success(std::chrono::microseconds duration) noexcept {
144 success_count.fetch_add(1, std::memory_order_relaxed);
145 const auto duration_us = static_cast<std::uint64_t>(duration.count());
146 total_duration_us.fetch_add(duration_us, std::memory_order_relaxed);
147
148 // Update min/max with CAS loops
149 auto current_min = min_duration_us.load(std::memory_order_relaxed);
150 while (duration_us < current_min &&
151 !min_duration_us.compare_exchange_weak(current_min, duration_us,
152 std::memory_order_relaxed)) {
153 // Loop until successful
154 }
155
156 auto current_max = max_duration_us.load(std::memory_order_relaxed);
157 while (duration_us > current_max &&
158 !max_duration_us.compare_exchange_weak(current_max, duration_us,
159 std::memory_order_relaxed)) {
160 // Loop until successful
161 }
162 }
163
165 void record_failure(std::chrono::microseconds duration) noexcept {
166 failure_count.fetch_add(1, std::memory_order_relaxed);
167 const auto duration_us = static_cast<std::uint64_t>(duration.count());
168 total_duration_us.fetch_add(duration_us, std::memory_order_relaxed);
169
170 // Update min/max with CAS loops
171 auto current_min = min_duration_us.load(std::memory_order_relaxed);
172 while (duration_us < current_min &&
173 !min_duration_us.compare_exchange_weak(current_min, duration_us,
174 std::memory_order_relaxed)) {
175 }
176
177 auto current_max = max_duration_us.load(std::memory_order_relaxed);
178 while (duration_us > current_max &&
179 !max_duration_us.compare_exchange_weak(current_max, duration_us,
180 std::memory_order_relaxed)) {
181 }
182 }
183
185 void reset() noexcept {
186 success_count.store(0, std::memory_order_relaxed);
187 failure_count.store(0, std::memory_order_relaxed);
188 total_duration_us.store(0, std::memory_order_relaxed);
189 min_duration_us.store(UINT64_MAX, std::memory_order_relaxed);
190 max_duration_us.store(0, std::memory_order_relaxed);
191 }
192};

242struct association_counters {
243 std::atomic<std::uint64_t> total_established{0};
244 std::atomic<std::uint64_t> total_rejected{0};
245 std::atomic<std::uint64_t> total_aborted{0};
246 std::atomic<std::uint32_t> current_active{0};
247 std::atomic<std::uint32_t> peak_active{0};
248
250 void record_established() noexcept {
251 total_established.fetch_add(1, std::memory_order_relaxed);
252 const auto active = current_active.fetch_add(1, std::memory_order_relaxed) + 1;
253
254 // Update peak with CAS loop
255 auto current_peak = peak_active.load(std::memory_order_relaxed);
256 while (active > current_peak &&
257 !peak_active.compare_exchange_weak(current_peak, active,
258 std::memory_order_relaxed)) {
259 }
260 }
261
263 void record_released() noexcept {
264 const auto prev = current_active.load(std::memory_order_relaxed);
265 if (prev > 0) {
266 current_active.fetch_sub(1, std::memory_order_relaxed);
267 }
268 }
269
271 void record_rejected() noexcept {
272 total_rejected.fetch_add(1, std::memory_order_relaxed);
273 }
274
276 void record_aborted() noexcept {
277 total_aborted.fetch_add(1, std::memory_order_relaxed);
278 const auto prev = current_active.load(std::memory_order_relaxed);
279 if (prev > 0) {
280 current_active.fetch_sub(1, std::memory_order_relaxed);
281 }
282 }
283
285 void reset() noexcept {
286 total_established.store(0, std::memory_order_relaxed);
287 total_rejected.store(0, std::memory_order_relaxed);
288 total_aborted.store(0, std::memory_order_relaxed);
289 current_active.store(0, std::memory_order_relaxed);
290 peak_active.store(0, std::memory_order_relaxed);
291 }

303struct pool_counters {
304 std::atomic<std::uint64_t> total_acquisitions{0};
305 std::atomic<std::uint64_t> pool_hits{0};
306 std::atomic<std::uint64_t> pool_misses{0};
307 std::atomic<std::uint64_t> total_releases{0};
308 std::atomic<std::uint32_t> current_pool_size{0};
309
311 [[nodiscard]] double hit_ratio() const noexcept {
312 const auto total = total_acquisitions.load(std::memory_order_relaxed);
313 if (total == 0) {
314 return 0.0;
315 }
316 return static_cast<double>(pool_hits.load(std::memory_order_relaxed))
317 / static_cast<double>(total);
318 }
319
321 void record_acquisition(bool was_pool_hit) noexcept {
322 total_acquisitions.fetch_add(1, std::memory_order_relaxed);
323 if (was_pool_hit) {
324 pool_hits.fetch_add(1, std::memory_order_relaxed);
325 } else {
326 pool_misses.fetch_add(1, std::memory_order_relaxed);
327 }
328 }
329
331 void record_release() noexcept {
332 total_releases.fetch_add(1, std::memory_order_relaxed);
333 }
334
336 void set_pool_size(std::uint32_t size) noexcept {
337 current_pool_size.store(size, std::memory_order_relaxed);
338 }
339
341 void reset() noexcept {
342 total_acquisitions.store(0, std::memory_order_relaxed);
343 pool_hits.store(0, std::memory_order_relaxed);
344 pool_misses.store(0, std::memory_order_relaxed);
345 total_releases.store(0, std::memory_order_relaxed);
346 current_pool_size.store(0, std::memory_order_relaxed);
347 }
