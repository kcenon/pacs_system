87network::Result<std::monostate> retrieve_scp::handle_message(
88 network::association& assoc,
89 uint8_t context_id,
90 const network::dimse::dimse_message& request) {
91
92 using namespace network::dimse;
93
94 // Route to appropriate handler based on command type
95 switch (request.command()) {
96 case command_field::c_move_rq:
97 return handle_c_move(assoc, context_id, request);
98
99 case command_field::c_get_rq:
100 return handle_c_get(assoc, context_id, request);
101
102 default:
103 return kcenon::pacs::pacs_void_error(
104 kcenon::pacs::error_codes::retrieve_unexpected_command,
105 "Expected C-MOVE-RQ or C-GET-RQ but received " +
106 std::string(to_string(request.command())));
107 }
108}

140network::Result<std::monostate> retrieve_scp::handle_c_move(
141 network::association& assoc,
142 uint8_t context_id,
143 const network::dimse::dimse_message& request) {
144
145 using namespace network::dimse;
146
147 // Verify we have a retrieve handler
148 if (!retrieve_handler_) {
149 return kcenon::pacs::pacs_void_error(
150 kcenon::pacs::error_codes::retrieve_handler_not_set,
151 "No retrieve handler configured");
152 }
153
154 // Verify we have a destination resolver for C-MOVE
155 if (!destination_resolver_) {
156 return kcenon::pacs::pacs_void_error(
157 kcenon::pacs::error_codes::retrieve_missing_destination,
158 "No destination resolver configured for C-MOVE");
159 }
160
161 auto sop_class_uid = request.affected_sop_class_uid();
162 auto message_id = request.message_id();
163
164 // Verify we have a dataset (query keys)
165 if (!request.has_dataset()) {
166 sub_operation_stats stats;
167 return send_final_response(
168 assoc, context_id, message_id,
169 sop_class_uid, true, stats, false);
170 }
171
172 // Get move destination AE title
173 auto dest_ae = get_move_destination(request);
174 if (dest_ae.empty()) {
175 // Missing Move Destination - return error status
176 ++move_operations_;
177
178 dimse_message response{command_field::c_move_rsp, 0};
179 response.set_affected_sop_class_uid(sop_class_uid);
180 response.set_message_id_responded_to(message_id);
181 response.set_status(status_refused_move_destination_unknown);
182
183 return assoc.send_dimse(context_id, response);
184 }
185
186 // Resolve destination to network address
187 auto dest_addr = destination_resolver_(dest_ae);
188 if (!dest_addr.has_value()) {
189 // Unknown destination AE - return error status
190 ++move_operations_;
191
192 dimse_message response{command_field::c_move_rsp, 0};
193 response.set_affected_sop_class_uid(sop_class_uid);
194 response.set_message_id_responded_to(message_id);
195 response.set_status(status_refused_move_destination_unknown);
196
197 return assoc.send_dimse(context_id, response);
198 }
199
200 // Get calling AE for move originator information
201 std::string calling_ae{assoc.calling_ae()};
202
203 // Retrieve matching files
204 const auto& query_keys = request.dataset().value().get();
205 auto files = retrieve_handler_(query_keys);
206 auto start_time = std::chrono::steady_clock::now();
207
208 // Get study UID for event
209 std::string study_uid = query_keys.get_string(core::tags::study_instance_uid);
210
211 // Publish retrieve started event
212 kcenon::common::get_event_bus().publish(
213 kcenon::pacs::events::retrieve_started_event{
214 kcenon::pacs::events::retrieve_operation::c_move,
215 calling_ae,
216 dest_ae,
217 study_uid,
218 static_cast<uint16_t>(files.size())
219 }
220 );
221
222 // Initialize sub-operation statistics
223 sub_operation_stats stats;
224 stats.remaining = static_cast<uint16_t>(files.size());
225 bool was_cancelled = false;
226
227 // Process each file (C-STORE sub-operations)
228 for (const auto& file : files) {
229 // Check for cancel request
230 if (cancel_check_ && cancel_check_()) {
231 was_cancelled = true;
232 break;
233 }
234
235 // Send pending response with current progress
236 auto pending_result = send_pending_response(
237 assoc, context_id, message_id,
238 sop_class_uid, true, stats);
239
240 if (pending_result.is_err()) {
241 return pending_result;
242 }
243
244 // Perform C-STORE sub-operation
245 // Note: In a full implementation, this would establish a sub-association
246 // to the destination and perform C-STORE. For now, we simulate success
247 // or use the provided store handler.
248 status_code store_status = status_success;
249
250 if (store_handler_) {
251 // Use custom store handler
252 // The handler should establish connection to dest_addr and perform C-STORE
253 // For now, we pass the current association (which isn't correct for C-MOVE)
254 // A real implementation would create a separate association to destination
255 store_status = store_handler_(
256 assoc, context_id, file,
257 calling_ae, message_id);
258 }
259
260 // Update statistics based on store result
261 stats.remaining--;
262 if (is_success(store_status)) {
263 stats.completed++;
264 ++images_transferred_;
265 } else if (is_warning(store_status)) {
266 stats.warning++;
267 ++images_transferred_;
268 } else {
269 stats.failed++;
270 }
271 }
272
273 // Update operation count
274 ++move_operations_;
275
276 // Calculate duration and publish completed event
277 auto end_time = std::chrono::steady_clock::now();
278 auto duration_ms = std::chrono::duration_cast<std::chrono::milliseconds>(
279 end_time - start_time).count();
280
281 kcenon::common::get_event_bus().publish(
282 kcenon::pacs::events::retrieve_completed_event{
283 kcenon::pacs::events::retrieve_operation::c_move,
284 calling_ae,
285 dest_ae,
286 stats.completed,
287 stats.failed,
288 stats.warning,
289 static_cast<uint64_t>(duration_ms)
290 }
291 );
292
293 // Send final response
294 return send_final_response(
295 assoc, context_id, message_id,
296 sop_class_uid, true, stats, was_cancelled);
297}

303network::Result<std::monostate> retrieve_scp::handle_c_get(
304 network::association& assoc,
305 uint8_t context_id,
306 const network::dimse::dimse_message& request) {
307
308 using namespace network::dimse;
309
310 // Verify we have a retrieve handler
311 if (!retrieve_handler_) {
312 return kcenon::pacs::pacs_void_error(
313 kcenon::pacs::error_codes::retrieve_handler_not_set,
314 "No retrieve handler configured");
315 }
316
317 auto sop_class_uid = request.affected_sop_class_uid();
318 auto message_id = request.message_id();
319
320 // Verify we have a dataset (query keys)
321 if (!request.has_dataset()) {
322 sub_operation_stats stats;
323 return send_final_response(
324 assoc, context_id, message_id,
325 sop_class_uid, false, stats, false);
326 }
327
328 // Get calling AE for logging
329 std::string calling_ae{assoc.calling_ae()};
330
331 // Retrieve matching files
332 const auto& query_keys = request.dataset().value().get();
333 auto files = retrieve_handler_(query_keys);
334 auto start_time = std::chrono::steady_clock::now();
335
336 // Get study UID for event
337 std::string study_uid = query_keys.get_string(core::tags::study_instance_uid);
338
339 // Publish retrieve started event (C-GET has no destination AE)
340 kcenon::common::get_event_bus().publish(
341 kcenon::pacs::events::retrieve_started_event{
342 kcenon::pacs::events::retrieve_operation::c_get,
343 calling_ae,
344 "", // No destination for C-GET
345 study_uid,
346 static_cast<uint16_t>(files.size())
347 }
348 );
349
350 // Initialize sub-operation statistics
351 sub_operation_stats stats;
352 stats.remaining = static_cast<uint16_t>(files.size());
353 bool was_cancelled = false;
354
355 // Process each file (C-STORE sub-operations on same association)
356 for (const auto& file : files) {
357 // Check for cancel request
358 if (cancel_check_ && cancel_check_()) {
359 was_cancelled = true;
360 break;
361 }
362
363 // Send pending response with current progress
364 auto pending_result = send_pending_response(
365 assoc, context_id, message_id,
366 sop_class_uid, false, stats);
367
368 if (pending_result.is_err()) {
369 return pending_result;
370 }
371
372 // Perform C-STORE sub-operation on the same association
373 // For C-GET, images are sent back on the same association
374 status_code store_status = status_success;
375
376 if (store_handler_) {
377 // Use custom store handler
378 store_status = store_handler_(
379 assoc, context_id, file,
380 calling_ae, message_id);
381 } else {
382 // Default implementation: build and send C-STORE-RQ
383 // Get the SOP Class and Instance UIDs from the file
384 auto file_sop_class = file.sop_class_uid();
385 auto file_sop_instance = file.sop_instance_uid();
386
387 // Create C-STORE request
388 dimse_message store_rq{command_field::c_store_rq, message_id};
389 store_rq.set_affected_sop_class_uid(file_sop_class);
390 store_rq.set_affected_sop_instance_uid(file_sop_instance);
391 store_rq.set_priority(priority_medium);
392
393 // For C-GET, include Move Originator information
394 // (0000,1030) Move Originator Application Entity Title
395 // (0000,1031) Move Originator Message ID
396 store_rq.command_set().set_string(
397 tag_move_originator_aet,
398 encoding::vr_type::AE,
399 calling_ae);
400 store_rq.command_set().set_numeric<uint16_t>(
401 tag_move_originator_message_id,
402 encoding::vr_type::US,
403 message_id);
404
405 // Attach the dataset
406 store_rq.set_dataset(file.dataset());
407
408 // Find the presentation context for the SOP Class
409 auto store_context_id = assoc.accepted_context_id(file_sop_class);
410 if (!store_context_id.has_value()) {
411 // No accepted context for this SOP Class
412 stats.remaining--;
413 stats.failed++;
414 continue;
415 }
416
417 // Send the C-STORE request
418 auto send_result = assoc.send_dimse(store_context_id.value(), store_rq);
419 if (send_result.is_err()) {
420 stats.remaining--;
421 stats.failed++;
422 continue;
423 }
424
425 // Wait for C-STORE response
426 auto recv_result = assoc.receive_dimse();
427 if (recv_result.is_err()) {
428 stats.remaining--;
429 stats.failed++;
430 continue;
431 }
432
433 auto& [recv_ctx, store_rsp] = recv_result.value();
434 store_status = store_rsp.status();
435 }
436
437 // Update statistics based on store result
438 stats.remaining--;
439 if (is_success(store_status)) {
440 stats.completed++;
441 ++images_transferred_;
442 } else if (is_warning(store_status)) {
443 stats.warning++;
444 ++images_transferred_;
445 } else {
446 stats.failed++;
447 }
448 }
449
450 // Update operation count
451 ++get_operations_;
452
453 // Calculate duration and publish completed event
454 auto end_time = std::chrono::steady_clock::now();
455 auto duration_ms = std::chrono::duration_cast<std::chrono::milliseconds>(
456 end_time - start_time).count();
457
458 kcenon::common::get_event_bus().publish(
459 kcenon::pacs::events::retrieve_completed_event{
460 kcenon::pacs::events::retrieve_operation::c_get,
461 calling_ae,
462 "", // No destination for C-GET
463 stats.completed,
464 stats.failed,
465 stats.warning,
466 static_cast<uint64_t>(duration_ms)
467 }
468 );
469
470 // Send final response
471 return send_final_response(
472 assoc, context_id, message_id,
473 sop_class_uid, false, stats, was_cancelled);
474}

480network::Result<std::monostate> retrieve_scp::send_pending_response(

508network::Result<std::monostate> retrieve_scp::send_final_response(
509 network::association& assoc,
510 uint8_t context_id,
511 uint16_t message_id,
512 std::string_view sop_class_uid,
513 bool is_move,
514 const sub_operation_stats& stats,
515 bool was_cancelled) {
516
517 using namespace network::dimse;
518
519 // Create response message
520 auto cmd = is_move ? command_field::c_move_rsp : command_field::c_get_rsp;
521 dimse_message response{cmd, 0};
522
523 response.set_affected_sop_class_uid(sop_class_uid);
524 response.set_message_id_responded_to(message_id);
525
526 // Determine final status
527 status_code final_status;
528 if (was_cancelled) {
529 final_status = status_cancel;
530 } else if (stats.failed > 0 && stats.completed == 0 && stats.warning == 0) {
531 // All sub-operations failed
532 final_status = status_refused_out_of_resources_subops;
533 } else if (stats.failed > 0 || stats.warning > 0) {
534 // Some sub-operations had issues
535 final_status = status_warning_subops_complete_failures;
536 } else {
537 // All successful
538 final_status = status_success;
539 }
540
541 response.set_status(final_status);
542
543 // Set final sub-operation counts
544 response.set_remaining_subops(stats.remaining);
545 response.set_completed_subops(stats.completed);
546 response.set_failed_subops(stats.failed);
547 response.set_warning_subops(stats.warning);
548
549 // Send the response
550 return assoc.send_dimse(context_id, response);
551}
