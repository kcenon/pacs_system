236struct job_progress {
237 size_t total_items{0};
238 size_t completed_items{0};
239 size_t failed_items{0};
240 size_t skipped_items{0};
241 size_t bytes_transferred{0};
242
243 float percent_complete{0.0f};
244
245 std::string current_item;
246 std::string current_item_description;
247
248 std::chrono::milliseconds elapsed{0};
249 std::chrono::milliseconds estimated_remaining{0};
250
254 void calculate_percent() noexcept {
255 if (total_items > 0) {
256 percent_complete = static_cast<float>(completed_items + failed_items + skipped_items)
257 / static_cast<float>(total_items) * 100.0f;
258 }
259 }
260
264 [[nodiscard]] bool is_complete() const noexcept {
265 return total_items > 0 &&
266 (completed_items + failed_items + skipped_items) >= total_items;
267 }
268};

351 [[nodiscard]] bool is_finished() const noexcept {
352 return is_terminal_status(status);
353 }
354
358 [[nodiscard]] bool can_start() const noexcept {
359 return status == job_status::pending ||
360 status == job_status::queued ||
361 status == job_status::paused;
362 }
363
367 [[nodiscard]] bool can_cancel() const noexcept {
368 return !is_terminal_status(status);
369 }
370
374 [[nodiscard]] bool can_pause() const noexcept {
375 return status == job_status::running ||
376 status == job_status::queued;
377 }
378
382 [[nodiscard]] bool can_retry() const noexcept {
383 return status == job_status::failed && retry_count < max_retries;
384 }

390 [[nodiscard]] std::chrono::milliseconds duration() const noexcept {
391 if (!started_at.has_value()) {
392 return std::chrono::milliseconds{0};
393 }
394 auto end_time = completed_at.value_or(std::chrono::system_clock::now());
395 return std::chrono::duration_cast<std::chrono::milliseconds>(
396 end_time - started_at.value());
