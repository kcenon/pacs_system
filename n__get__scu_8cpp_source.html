62network::Result<n_get_result> n_get_scu::get(
63 network::association& assoc,
64 std::string_view sop_class_uid,
65 std::string_view sop_instance_uid,
66 const std::vector<core::dicom_tag>& attribute_tags) {
67
68 using namespace network::dimse;
69
70 auto start_time = std::chrono::steady_clock::now();
71
72 // Verify association is established
73 if (!assoc.is_established()) {
74 return kcenon::pacs::pacs_error<n_get_result>(
75 kcenon::pacs::error_codes::association_not_established,
76 "Association not established");
77 }
78
79 // Validate UIDs
80 if (sop_class_uid.empty()) {
81 return kcenon::pacs::pacs_error<n_get_result>(
82 kcenon::pacs::error_codes::n_get_missing_uid,
83 "SOP Class UID is required for N-GET");
84 }
85
86 if (sop_instance_uid.empty()) {
87 return kcenon::pacs::pacs_error<n_get_result>(
88 kcenon::pacs::error_codes::n_get_missing_uid,
89 "SOP Instance UID is required for N-GET");
90 }
91
92 // Get accepted presentation context
93 auto context_id = assoc.accepted_context_id(sop_class_uid);
94 if (!context_id) {
95 return kcenon::pacs::pacs_error<n_get_result>(
96 kcenon::pacs::error_codes::n_get_context_not_accepted,
97 "No accepted presentation context for SOP Class: " +
98 std::string(sop_class_uid));
99 }
100
101 // Build N-GET request using factory function
102 auto request = make_n_get_rq(
103 next_message_id(), sop_class_uid, sop_instance_uid, attribute_tags);
104
105 logger_->debug("Sending N-GET request for instance: " +
106 std::string(sop_instance_uid) +
107 " (attributes: " +
108 (attribute_tags.empty() ? "all" :
109 std::to_string(attribute_tags.size())) + ")");
110
111 // Send the request
112 auto send_result = assoc.send_dimse(*context_id, request);
113 if (send_result.is_err()) {
114 logger_->error("Failed to send N-GET: " + send_result.error().message);
115 return send_result.error();
116 }
117
118 // Receive the response
119 auto recv_result = assoc.receive_dimse(config_.timeout);
120 if (recv_result.is_err()) {
121 logger_->error("Failed to receive N-GET response: " +
122 recv_result.error().message);
123 return recv_result.error();
124 }
125
126 const auto& [recv_context_id, response] = recv_result.value();
127
128 // Verify it's an N-GET response
129 if (response.command() != command_field::n_get_rsp) {
130 return kcenon::pacs::pacs_error<n_get_result>(
131 kcenon::pacs::error_codes::n_get_unexpected_command,
132 "Expected N-GET-RSP but received " +
133 std::string(to_string(response.command())));
134 }
135
136 auto end_time = std::chrono::steady_clock::now();
137 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
138 end_time - start_time);
139
140 // Build result
141 n_get_result result;
142 result.status = static_cast<uint16_t>(response.status());
143 result.elapsed = elapsed;
144
145 // Extract error comment if present
146 if (response.command_set().contains(tag_error_comment)) {
147 result.error_comment = response.command_set().get_string(
148 tag_error_comment);
149 }
150
151 // Extract dataset (returned attributes) if present
152 if (response.has_dataset()) {
153 auto dataset_result = response.dataset();
154 if (dataset_result.is_ok()) {
155 result.attributes = dataset_result.value().get();
156 }
157 }
158
159 // Update statistics
160 gets_performed_.fetch_add(1, std::memory_order_relaxed);
161
162 if (result.is_success()) {
163 logger_->info("N-GET successful for instance: " +
164 std::string(sop_instance_uid));
165 } else {
166 logger_->warn("N-GET returned status 0x" +
167 std::to_string(result.status) +
168 " for instance: " + std::string(sop_instance_uid));
169 }
170
171 return result;
172}
