76thumbnail_result thumbnail_service::get_thumbnail(
77 std::string_view sop_instance_uid, const thumbnail_params& params) {
78 // Validate parameters
79 if (params.size != 64 && params.size != 128 && params.size != 256 &&
80 params.size != 512) {
81 return thumbnail_result::error("Invalid size: must be 64, 128, 256, or 512");
82 }
83
84 if (params.format != "jpeg" && params.format != "png") {
85 return thumbnail_result::error("Invalid format: must be jpeg or png");
86 }
87
88 if (params.quality < 1 || params.quality > 100) {
89 return thumbnail_result::error("Invalid quality: must be 1-100");
90 }
91
92 // Check database is available
93 if (database_ == nullptr) {
94 return thumbnail_result::error("Database not configured");
95 }
96
97 // Build cache key
98 cache_key key{
99 std::string(sop_instance_uid), params.size, params.format,
100 params.quality, params.frame};
101
102 // Check cache first (read lock)
103 {
104 std::shared_lock lock(cache_mutex_);
105 auto it = cache_.find(key);
106 if (it != cache_.end()) {
107 // Update last accessed time
108 const_cast<thumbnail_cache_entry&>(it->second).last_accessed =
109 std::chrono::system_clock::now();
110 return thumbnail_result::ok(it->second);
111 }
112 }
113
114 // Find instance in database
115 auto instance = database_->find_instance(sop_instance_uid);
116 if (!instance) {
117 return thumbnail_result::error("Instance not found");
118 }
119
120 // Check if file exists
121 if (!std::filesystem::exists(instance->file_path)) {
122 return thumbnail_result::error("DICOM file not found");
123 }
124
125 // Generate thumbnail
126 auto thumbnail_data = generate_thumbnail(instance->file_path, params);
127 if (thumbnail_data.empty()) {
128 return thumbnail_result::error("Failed to generate thumbnail");
129 }
130
131 // Create cache entry
132 thumbnail_cache_entry entry;
133 entry.data = std::move(thumbnail_data);
134 entry.content_type = get_content_type(params.format);
135 entry.created_at = std::chrono::system_clock::now();
136 entry.last_accessed = entry.created_at;
137
138 // Store in cache (write lock)
139 {
140 std::unique_lock lock(cache_mutex_);
141
142 // Check if we need to evict entries
143 while (current_cache_size_ + entry.data.size() > max_cache_size_ &&
144 !cache_.empty()) {
145 evict_lru();
146 }
147
148 current_cache_size_ += entry.data.size();
149 cache_[key] = entry;
150 }
151
152 return thumbnail_result::ok(std::move(entry));
153}

179void thumbnail_service::clear_cache() {
180 std::unique_lock lock(cache_mutex_);
181 cache_.clear();
182 current_cache_size_ = 0;
183}
184
185void thumbnail_service::clear_cache(std::string_view sop_instance_uid) {
186 std::unique_lock lock(cache_mutex_);
187
188 // Remove all entries for this instance (any size/format/quality)
189 for (auto it = cache_.begin(); it != cache_.end();) {
190 if (it->first.uid == sop_instance_uid) {
191 current_cache_size_ -= it->second.data.size();
192 it = cache_.erase(it);
193 } else {
194 ++it;
195 }
196 }
197}

228std::vector<uint8_t> thumbnail_service::generate_thumbnail(
229 std::string_view file_path, const thumbnail_params& params) {
230 using namespace kcenon::pacs::core;
231
232 // Open DICOM file
233 auto result = dicom_file::open(std::filesystem::path(file_path));
234 if (result.is_err()) {
235 return {};
236 }
237
238 auto& file = result.value();
239 const auto& dataset = file.dataset();
240
241 // Get image dimensions
242 auto rows = dataset.get_numeric<uint16_t>(tags::rows);
243 auto columns = dataset.get_numeric<uint16_t>(tags::columns);
244
245 if (!rows || !columns || *rows == 0 || *columns == 0) {
246 return {};
247 }
248
249 // Get bits information
250 auto bits_allocated =
251 dataset.get_numeric<uint16_t>(dicom_tag{0x0028, 0x0100});
252 auto bits_stored = dataset.get_numeric<uint16_t>(dicom_tag{0x0028, 0x0101});
253 auto samples_per_pixel = dataset.get_numeric<uint16_t>(tags::samples_per_pixel);
254 auto pixel_representation =
255 dataset.get_numeric<uint16_t>(dicom_tag{0x0028, 0x0103});
256
257 if (!bits_allocated || !bits_stored) {
258 return {};
259 }
260
261 uint16_t spp = samples_per_pixel.value_or(1);
262 uint16_t repr = pixel_representation.value_or(0);
263
264 // Get photometric interpretation
265 auto photometric = dataset.get_string(tags::photometric_interpretation);
266
267 // Get pixel data
268 auto* pixel_element = dataset.get(tags::pixel_data);
269 if (pixel_element == nullptr) {
270 return {};
271 }
272
273 auto raw_data = pixel_element->raw_data();
274 if (raw_data.empty()) {
275 return {};
276 }
277
278 // Calculate frame offset for multi-frame images
279 size_t frame_size = static_cast<size_t>(*rows) * (*columns) * spp *
280 ((*bits_allocated + 7) / 8);
281
282 uint32_t frame_index = params.frame > 0 ? params.frame - 1 : 0;
283 size_t frame_offset = frame_index * frame_size;
284
285 if (frame_offset + frame_size > raw_data.size()) {
286 // Fall back to first frame
287 frame_offset = 0;
288 }
289
290 // Convert to 8-bit grayscale or RGB
291 std::vector<uint8_t> pixels;
292 size_t num_pixels = static_cast<size_t>(*rows) * (*columns) * spp;
293
294 if (*bits_allocated == 16) {
295 // 16-bit to 8-bit conversion with auto window/level
296 int min_val = 65535;
297 int max_val = -65536;
298
299 // Find min/max for windowing
300 for (size_t i = 0; i < num_pixels; ++i) {
301 size_t idx = frame_offset + i * 2;
302 if (idx + 1 >= raw_data.size()) break;
303
304 int16_t pixel;
305 if (repr == 0) {
306 pixel = static_cast<int16_t>(raw_data[idx] |
307 (raw_data[idx + 1] << 8));
308 } else {
309 pixel = static_cast<int16_t>(raw_data[idx] |
310 (raw_data[idx + 1] << 8));
311 }
312 min_val = std::min(min_val, static_cast<int>(pixel));
313 max_val = std::max(max_val, static_cast<int>(pixel));
314 }
315
316 double window_width = static_cast<double>(max_val - min_val);
317 double window_center = static_cast<double>(min_val + max_val) / 2.0;
318 if (window_width < 1) window_width = 1;
319
320 pixels.resize(num_pixels);
321 for (size_t i = 0; i < num_pixels; ++i) {
322 size_t idx = frame_offset + i * 2;
323 if (idx + 1 >= raw_data.size()) break;
324
325 int16_t pixel = static_cast<int16_t>(raw_data[idx] |
326 (raw_data[idx + 1] << 8));
327
328 double lower = window_center - window_width / 2.0;
329 double upper = window_center + window_width / 2.0;
330
331 if (pixel <= lower) {
332 pixels[i] = 0;
333 } else if (pixel >= upper) {
334 pixels[i] = 255;
335 } else {
336 pixels[i] = static_cast<uint8_t>(
337 ((pixel - lower) / window_width) * 255.0);
338 }
339 }
340 } else {
341 // 8-bit: direct copy
342 pixels.resize(num_pixels);
343 for (size_t i = 0; i < num_pixels; ++i) {
344 size_t idx = frame_offset + i;
345 if (idx < raw_data.size()) {
346 pixels[i] = raw_data[idx];
347 }
348 }
349 }
350
351 // Handle MONOCHROME1 inversion
352 if (photometric == "MONOCHROME1") {
353 for (auto& p : pixels) {
354 p = static_cast<uint8_t>(255 - p);
355 }
356 }
357
358 // Resize to target size
359 uint16_t src_width = *columns;
360 uint16_t src_height = *rows;
361 uint16_t dst_size = params.size;
362
363 // Calculate destination dimensions maintaining aspect ratio
364 uint16_t dst_width, dst_height;
365 if (src_width > src_height) {
366 dst_width = dst_size;
367 dst_height =
368 static_cast<uint16_t>(static_cast<float>(src_height) / src_width * dst_size);
369 } else {
370 dst_height = dst_size;
371 dst_width =
372 static_cast<uint16_t>(static_cast<float>(src_width) / src_height * dst_size);
373 }
374
375 if (dst_width == 0) dst_width = 1;
376 if (dst_height == 0) dst_height = 1;
377
378 // Simple bilinear resize
379 std::vector<uint8_t> resized(static_cast<size_t>(dst_width) * dst_height * spp);
380
381 float x_ratio = static_cast<float>(src_width) / dst_width;
382 float y_ratio = static_cast<float>(src_height) / dst_height;
383
384 for (uint16_t y = 0; y < dst_height; ++y) {
385 for (uint16_t x = 0; x < dst_width; ++x) {
386 float src_x = x * x_ratio;
387 float src_y = y * y_ratio;
388
389 int x0 = static_cast<int>(src_x);
390 int y0 = static_cast<int>(src_y);
391 int x1 = std::min(x0 + 1, static_cast<int>(src_width - 1));
392 int y1 = std::min(y0 + 1, static_cast<int>(src_height - 1));
393
394 float x_diff = src_x - x0;
395 float y_diff = src_y - y0;
396
397 for (uint16_t c = 0; c < spp; ++c) {
398 size_t idx00 = (static_cast<size_t>(y0) * src_width + x0) * spp + c;
399 size_t idx01 = (static_cast<size_t>(y0) * src_width + x1) * spp + c;
400 size_t idx10 = (static_cast<size_t>(y1) * src_width + x0) * spp + c;
401 size_t idx11 = (static_cast<size_t>(y1) * src_width + x1) * spp + c;
402
403 float v00 = idx00 < pixels.size() ? pixels[idx00] : 0;
404 float v01 = idx01 < pixels.size() ? pixels[idx01] : 0;
405 float v10 = idx10 < pixels.size() ? pixels[idx10] : 0;
406 float v11 = idx11 < pixels.size() ? pixels[idx11] : 0;
407
408 float value = v00 * (1 - x_diff) * (1 - y_diff) +
409 v01 * x_diff * (1 - y_diff) +
410 v10 * (1 - x_diff) * y_diff +
411 v11 * x_diff * y_diff;
412
413 size_t dst_idx =
414 (static_cast<size_t>(y) * dst_width + x) * spp + c;
415 resized[dst_idx] = static_cast<uint8_t>(std::clamp(value, 0.0f, 255.0f));
416 }
417 }
418 }
419
420 // Encode to output format
421 std::vector<uint8_t> output;
422
423 if (params.format == "jpeg") {
424#ifdef PACS_JPEG_FOUND
425 // JPEG encoding
426 struct jpeg_compress_struct cinfo {};
427 struct jpeg_error_mgr jerr {};
428
429 cinfo.err = jpeg_std_error(&jerr);
430 jpeg_create_compress(&cinfo);
431
432 // Memory destination
433 unsigned char* outbuffer = nullptr;
434 unsigned long outsize = 0;
435 jpeg_mem_dest(&cinfo, &outbuffer, &outsize);
436
437 cinfo.image_width = dst_width;
438 cinfo.image_height = dst_height;
439 cinfo.input_components = static_cast<int>(spp);
440 cinfo.in_color_space = (spp == 1) ? JCS_GRAYSCALE : JCS_RGB;
441
442 jpeg_set_defaults(&cinfo);
443 jpeg_set_quality(&cinfo, params.quality, TRUE);
444 jpeg_start_compress(&cinfo, TRUE);
445
446 JSAMPROW row_pointer[1];
447 int row_stride = dst_width * spp;
448
449 while (cinfo.next_scanline < cinfo.image_height) {
450 row_pointer[0] = &resized[cinfo.next_scanline * static_cast<size_t>(row_stride)];
451 jpeg_write_scanlines(&cinfo, row_pointer, 1);
452 }
453
454 jpeg_finish_compress(&cinfo);
455 jpeg_destroy_compress(&cinfo);
456
457 if (outbuffer != nullptr && outsize > 0) {
458 output.assign(outbuffer, outbuffer + outsize);
459 free(outbuffer);
460 }
461#else
462 // Fallback: return empty if JPEG not available
463 return {};
464#endif
465 } else if (params.format == "png") {
466#ifdef PACS_PNG_FOUND
467 // PNG encoding using libpng memory I/O
468 struct png_mem_buffer {
469 std::vector<uint8_t> data;
470 };
471
472 auto write_callback = [](png_structp png_ptr, png_bytep data,
473 png_size_t length) {
474 auto* buffer =
475 static_cast<png_mem_buffer*>(png_get_io_ptr(png_ptr));
476 buffer->data.insert(buffer->data.end(), data, data + length);
477 };
478
479 auto flush_callback = [](png_structp /*png_ptr*/) {
480 // No-op for memory buffer
481 };
482
483 png_mem_buffer buffer;
484
485 png_structp png_ptr = png_create_write_struct(
486 PNG_LIBPNG_VER_STRING, nullptr, nullptr, nullptr);
487 if (png_ptr == nullptr) {
488 return {};
489 }
490
491 png_infop info_ptr = png_create_info_struct(png_ptr);
492 if (info_ptr == nullptr) {
493 png_destroy_write_struct(&png_ptr, nullptr);
494 return {};
495 }
496
497 if (setjmp(png_jmpbuf(png_ptr))) {
498 png_destroy_write_struct(&png_ptr, &info_ptr);
499 return {};
500 }
501
502 png_set_write_fn(png_ptr, &buffer, write_callback, flush_callback);
503
504 int color_type = (spp == 1) ? PNG_COLOR_TYPE_GRAY : PNG_COLOR_TYPE_RGB;
505 png_set_IHDR(png_ptr, info_ptr, dst_width, dst_height, 8, color_type,
506 PNG_INTERLACE_NONE, PNG_COMPRESSION_TYPE_DEFAULT,
507 PNG_FILTER_TYPE_DEFAULT);
508
509 png_write_info(png_ptr, info_ptr);
510
511 int row_stride = dst_width * spp;
512 for (uint16_t y = 0; y < dst_height; ++y) {
513 png_bytep row = &resized[static_cast<size_t>(y) * row_stride];
514 png_write_row(png_ptr, row);
515 }
516
517 png_write_end(png_ptr, nullptr);
518 png_destroy_write_struct(&png_ptr, &info_ptr);
519
520 output = std::move(buffer.data);
521#else
522 // Fallback: return empty if PNG not available
523 return {};
524#endif
525 }
526
527 return output;
528}

530std::optional<std::string> thumbnail_service::select_representative_instance(
531 std::string_view series_uid) {
532 auto instances_result = database_->list_instances(series_uid);
533 if (instances_result.is_err() || instances_result.value().empty()) {
534 return std::nullopt;
535 }
536
537 const auto& instances = instances_result.value();
538
539 // Sort by instance number and select middle instance
540 std::vector<std::pair<int, std::string>> sorted;
541 for (const auto& inst : instances) {
542 int num = inst.instance_number.value_or(1);
543 sorted.emplace_back(num, inst.sop_uid);
544 }
545
546 std::sort(sorted.begin(), sorted.end(),
547 [](const auto& a, const auto& b) { return a.first < b.first; });
548
549 // Select middle instance
550 size_t middle = sorted.size() / 2;
551 return sorted[middle].second;
552}

554std::optional<std::string> thumbnail_service::select_representative_series(
555 std::string_view study_uid) {
556 auto series_result = database_->list_series(study_uid);
557 if (series_result.is_err() || series_result.value().empty()) {
558 return std::nullopt;
559 }
560
561 const auto& series_list = series_result.value();
562
563 // Prefer series with images (CT, MR, CR, DX, etc.)
564 // Use first series with most instances as representative
565 const kcenon::pacs::storage::series_record* best = nullptr;
566 for (const auto& s : series_list) {
567 // Skip structured reports, KOS, etc.
568 if (s.modality == "SR" || s.modality == "KO" || s.modality == "PR") {
569 continue;
570 }
571
572 if (best == nullptr || s.num_instances > best->num_instances) {
573 best = &s;
574 }
575 }
576
577 if (best == nullptr && !series_list.empty()) {
578 // Fall back to first series
579 best = &series_list[0];
580 }
581
582 return best ? std::make_optional(best->series_uid) : std::nullopt;
583}

585void thumbnail_service::evict_lru() {
586 // Find least recently used entry
587 auto lru_it = cache_.end();
588 auto oldest_time = std::chrono::system_clock::time_point::max();
589
590 for (auto it = cache_.begin(); it != cache_.end(); ++it) {
591 if (it->second.last_accessed < oldest_time) {
592 oldest_time = it->second.last_accessed;
593 lru_it = it;
594 }
595 }
596
597 if (lru_it != cache_.end()) {
598 current_cache_size_ -= lru_it->second.data.size();
599 cache_.erase(lru_it);
600 }
601}
