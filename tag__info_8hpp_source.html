66struct value_multiplicity {

74 constexpr value_multiplicity() noexcept = default;
75
82 constexpr value_multiplicity(uint32_t min_val,
83 std::optional<uint32_t> max_val,
84 uint32_t mult = 1) noexcept
85 : min{min_val}, max{max_val}, multiplier{mult} {}
86
92 [[nodiscard]] constexpr auto is_valid(uint32_t count) const noexcept -> bool {
93 if (count < min) {
94 return false;
95 }
96 if (max.has_value()) {
97 return count <= max.value();
98 }
99 // Unbounded case - check multiplier constraint
100 if (multiplier > 1) {
101 return (count - min) % multiplier == 0;
102 }
103 return true;
104 }
105
110 [[nodiscard]] constexpr auto allows_multiple() const noexcept -> bool {
111 return !max.has_value() || max.value() > 1;
112 }
113
118 [[nodiscard]] constexpr auto is_unbounded() const noexcept -> bool {
119 return !max.has_value();
120 }
121
134 [[nodiscard]] static auto from_string(std::string_view str)
135 -> std::optional<value_multiplicity>;
136
141 [[nodiscard]] auto to_string() const -> std::string;
142};
