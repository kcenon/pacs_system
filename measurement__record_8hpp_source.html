67[[nodiscard]] inline auto to_string(measurement_type type) -> std::string {
68 switch (type) {
69 case measurement_type::length: return "length";
70 case measurement_type::area: return "area";
71 case measurement_type::angle: return "angle";
72 case measurement_type::hounsfield: return "hounsfield";
73 case measurement_type::suv: return "suv";
74 case measurement_type::ellipse_area: return "ellipse_area";
75 case measurement_type::polygon_area: return "polygon_area";
76 }
77 return "unknown";
78}

83[[nodiscard]] inline auto measurement_type_from_string(std::string_view str)
84 -> std::optional<measurement_type> {
85 if (str == "length") return measurement_type::length;
86 if (str == "area") return measurement_type::area;
87 if (str == "angle") return measurement_type::angle;
88 if (str == "hounsfield") return measurement_type::hounsfield;
89 if (str == "suv") return measurement_type::suv;
90 if (str == "ellipse_area") return measurement_type::ellipse_area;
91 if (str == "polygon_area") return measurement_type::polygon_area;
92 return std::nullopt;
93}
