117// Clamp resolution levels to valid range for image dimensions
118int compute_resolution_levels(int requested, uint16_t width, uint16_t height) {
119 int max_levels = 1;
120 auto min_dim = std::min(width, height);
121 while (min_dim > 1 && max_levels < requested) {
122 min_dim >>= 1;
123 ++max_levels;
124 }
125 return std::clamp(max_levels, 1, 32);
126}

130codec_result htj2k_codec::encode(
131 std::span<const uint8_t> pixel_data,
132 const image_params& params,
133 [[maybe_unused]] const compression_options& options) const {
134
135 if (pixel_data.empty()) {
136 return kcenon::pacs::pacs_error<compression_result>(
137 kcenon::pacs::error_codes::compression_error, "Empty pixel data");
138 }
139
140 if (params.width == 0 || params.height == 0) {
141 return kcenon::pacs::pacs_error<compression_result>(
142 kcenon::pacs::error_codes::compression_error, "Invalid image dimensions");
143 }
144
145 const int bytes_per_sample = (params.bits_stored <= 8) ? 1 : 2;
146 const size_t expected_size = static_cast<size_t>(params.width)
147 * params.height * params.samples_per_pixel * bytes_per_sample;
148 if (pixel_data.size() < expected_size) {
149 return kcenon::pacs::pacs_error<compression_result>(
150 kcenon::pacs::error_codes::compression_error,
151 "Pixel data too small: expected " + std::to_string(expected_size)
152 + " bytes, got " + std::to_string(pixel_data.size()));
153 }
154
155 try {
156 ojph::codestream codestream;
157
158 // Configure SIZ marker (image dimensions and components)
159 ojph::param_siz siz = codestream.access_siz();
160 siz.set_image_extent(ojph::point{
161 static_cast<ojph::ui32>(params.width),
162 static_cast<ojph::ui32>(params.height)});
163 siz.set_image_offset(ojph::point{0, 0});
164 siz.set_tile_size(ojph::size{0, 0});
165 siz.set_tile_offset(ojph::point{0, 0});
166 siz.set_num_components(params.samples_per_pixel);
167
168 bool is_signed = (params.pixel_representation != 0);
169 for (ojph::ui32 c = 0; c < params.samples_per_pixel; ++c) {
170 siz.set_component(c, ojph::point{1, 1}, params.bits_stored, is_signed);
171 }
172
173 // Configure COD marker (coding parameters)
174 ojph::param_cod cod = codestream.access_cod();
175 int num_levels = compute_resolution_levels(
176 resolution_levels_, params.width, params.height);
177 cod.set_num_decomposition(static_cast<ojph::ui32>(num_levels));
178
179 // Block dimensions (64x64 is standard for HTJ2K)
180 cod.set_block_dims(64, 64);
181
182 // Reversible for lossless, irreversible for lossy
183 cod.set_reversible(lossless_);
184
185 // Color transform for multi-component images
186 cod.set_color_transform(params.samples_per_pixel == 3);
187
188 // Progression order
189 if (use_rpcl_) {
190 cod.set_progression_order("RPCL");
191 } else {
192 cod.set_progression_order("CPRL");
193 }
194
195 // For lossy mode, set quantization step
196 if (!lossless_) {
197 ojph::param_qcd qcd = codestream.access_qcd();
198 // Use a quality-preserving delta scaled by compression ratio.
199 // JPEG 2000 quantization cascades across subbands, so the
200 // base delta must be small enough for acceptable reconstruction.
201 float delta = compression_ratio_ / 10000.0f;
202 qcd.set_irrev_quant(delta);
203 }
204
205 // Set planar mode based on component count
206 // Planar = process one component fully before the next
207 // Non-planar needed when color transform is used
208 codestream.set_planar(params.samples_per_pixel == 1);
209
210 // Write to memory buffer
211 ojph::mem_outfile output;
212 output.open();
213
214 codestream.write_headers(&output);
215
216 // Push pixel data row by row
217 const auto width = static_cast<ojph::ui32>(params.width);
218 const auto height = static_cast<ojph::ui32>(params.height);
219 const auto num_comps = static_cast<ojph::ui32>(params.samples_per_pixel);
220
221 ojph::ui32 next_comp = 0;
222 ojph::line_buf* line = codestream.exchange(nullptr, next_comp);
223
224 for (ojph::ui32 y = 0; y < height; ++y) {
225 for (ojph::ui32 c = 0; c < num_comps; ++c) {
226 // Fill line buffer with pixel data for this row/component.
227 // OpenJPH uses i32 at the API boundary for both lossless and
228 // lossy modes; the library handles float conversion internally
229 // during the irreversible wavelet transform.
230 auto* dst = line->i32;
231 if (bytes_per_sample == 1) {
232 if (num_comps == 1) {
233 const uint8_t* src = pixel_data.data()
234 + static_cast<size_t>(y) * width;
235 for (ojph::ui32 x = 0; x < width; ++x) {
236 dst[x] = is_signed
237 ? static_cast<ojph::si32>(static_cast<int8_t>(src[x]))
238 : static_cast<ojph::si32>(src[x]);
239 }
240 } else {
241 const uint8_t* src = pixel_data.data()
242 + static_cast<size_t>(y) * width * num_comps;
243 for (ojph::ui32 x = 0; x < width; ++x) {
244 dst[x] = is_signed
245 ? static_cast<ojph::si32>(
246 static_cast<int8_t>(src[x * num_comps + next_comp]))
247 : static_cast<ojph::si32>(src[x * num_comps + next_comp]);
248 }
249 }
250 } else {
251 if (num_comps == 1) {
252 const auto* src = reinterpret_cast<const uint16_t*>(
253 pixel_data.data() + static_cast<size_t>(y) * width * 2);
254 for (ojph::ui32 x = 0; x < width; ++x) {
255 dst[x] = is_signed
256 ? static_cast<ojph::si32>(static_cast<int16_t>(src[x]))
257 : static_cast<ojph::si32>(src[x]);
258 }
259 } else {
260 const auto* src = reinterpret_cast<const uint16_t*>(
261 pixel_data.data()
262 + static_cast<size_t>(y) * width * num_comps * 2);
263 for (ojph::ui32 x = 0; x < width; ++x) {
264 dst[x] = is_signed
265 ? static_cast<ojph::si32>(
266 static_cast<int16_t>(src[x * num_comps + next_comp]))
267 : static_cast<ojph::si32>(src[x * num_comps + next_comp]);
268 }
269 }
270 }
271
272 line = codestream.exchange(line, next_comp);
273 }
274 }
275
276 codestream.flush();
277
278 // Copy compressed data BEFORE close (close deallocates the buffer)
279 auto compressed_size = static_cast<size_t>(output.tell());
280 std::vector<uint8_t> result_data(compressed_size);
281 if (compressed_size > 0) {
282 std::memcpy(result_data.data(), output.get_data(), compressed_size);
283 }
284
285 codestream.close();
286
287 return kcenon::pacs::ok<compression_result>(
288 compression_result{std::move(result_data), params});
289
290 } catch (const std::exception& e) {
291 return kcenon::pacs::pacs_error<compression_result>(
292 kcenon::pacs::error_codes::compression_error,
293 std::string("HTJ2K encoding failed: ") + e.what());
294 }
295}

297codec_result htj2k_codec::decode(
298 std::span<const uint8_t> compressed_data,
299 const image_params& params) const {
300
301 if (compressed_data.empty()) {
302 return kcenon::pacs::pacs_error<compression_result>(
303 kcenon::pacs::error_codes::decompression_error, "Empty compressed data");
304 }
305
306 try {
307 ojph::codestream codestream;
308
309 // Read from memory
310 ojph::mem_infile input;
311 input.open(compressed_data.data(), compressed_data.size());
312
313 codestream.read_headers(&input);
314
315 // Get image information from the codestream headers
316 ojph::param_siz siz = codestream.access_siz();
317 ojph::point extent = siz.get_image_extent();
318 auto decoded_width = static_cast<uint16_t>(extent.x);
319 auto decoded_height = static_cast<uint16_t>(extent.y);
320 auto num_comps = static_cast<uint16_t>(siz.get_num_components());
321 auto bit_depth = static_cast<uint16_t>(siz.get_bit_depth(0));
322 bool is_signed = siz.is_signed(0);
323
324 // Build output parameters
325 image_params output_params = params;
326 output_params.width = decoded_width;
327 output_params.height = decoded_height;
328 output_params.samples_per_pixel = num_comps;
329 output_params.bits_stored = bit_depth;
330 output_params.bits_allocated = (bit_depth <= 8) ? 8 : 16;
331 output_params.high_bit = bit_depth - 1;
332 output_params.pixel_representation = is_signed ? 1 : 0;
333 output_params.planar_configuration = 0;
334
335 if (num_comps == 1) {
336 output_params.photometric = photometric_interpretation::monochrome2;
337 } else if (num_comps == 3) {
338 output_params.photometric = photometric_interpretation::rgb;
339 }
340
341 // Validate dimensions if provided
342 if (params.width > 0 && params.width != decoded_width) {
343 return kcenon::pacs::pacs_error<compression_result>(
344 kcenon::pacs::error_codes::decompression_error,
345 "Image width mismatch: expected " + std::to_string(params.width)
346 + ", got " + std::to_string(decoded_width));
347 }
348 if (params.height > 0 && params.height != decoded_height) {
349 return kcenon::pacs::pacs_error<compression_result>(
350 kcenon::pacs::error_codes::decompression_error,
351 "Image height mismatch: expected " + std::to_string(params.height)
352 + ", got " + std::to_string(decoded_height));
353 }
354
355 // Set planar mode matching the encoder
356 codestream.set_planar(num_comps == 1);
357
358 codestream.create();
359
360 // Allocate output buffer
361 const int bytes_per_sample = (bit_depth <= 8) ? 1 : 2;
362 const size_t output_size = static_cast<size_t>(decoded_width)
363 * decoded_height * num_comps * bytes_per_sample;
364 std::vector<uint8_t> output_data(output_size);
365
366 // Pull decoded data row by row.
367 // OpenJPH returns i32 from pull() for both reversible and
368 // irreversible modes (internal float-to-int conversion is done).
369 ojph::ui32 comp_num = 0;
370 for (ojph::ui32 y = 0; y < decoded_height; ++y) {
371 for (ojph::ui32 c = 0; c < num_comps; ++c) {
372 ojph::line_buf* line = codestream.pull(comp_num);
373
374 if (bytes_per_sample == 1) {
375 if (num_comps == 1) {
376 auto* dst = output_data.data()
377 + static_cast<size_t>(y) * decoded_width;
378 for (ojph::ui32 x = 0; x < decoded_width; ++x) {
379 auto val = line->i32[x];
380 dst[x] = static_cast<uint8_t>(std::clamp(val, 0, 255));
381 }
382 } else {
383 auto* dst = output_data.data()
384 + static_cast<size_t>(y) * decoded_width * num_comps;
385 for (ojph::ui32 x = 0; x < decoded_width; ++x) {
386 auto val = line->i32[x];
387 dst[x * num_comps + comp_num] =
388 static_cast<uint8_t>(std::clamp(val, 0, 255));
389 }
390 }
391 } else {
392 if (num_comps == 1) {
393 auto* dst = reinterpret_cast<uint16_t*>(
394 output_data.data()
395 + static_cast<size_t>(y) * decoded_width * 2);
396 for (ojph::ui32 x = 0; x < decoded_width; ++x) {
397 auto val = line->i32[x];
398 if (is_signed) {
399 dst[x] = static_cast<uint16_t>(
400 static_cast<int16_t>(
401 std::clamp(val, -32768, 32767)));
402 } else {
403 dst[x] = static_cast<uint16_t>(
404 std::clamp(val, 0, 65535));
405 }
406 }
407 } else {
408 auto* dst = reinterpret_cast<uint16_t*>(
409 output_data.data()
410 + static_cast<size_t>(y) * decoded_width * num_comps * 2);
411 for (ojph::ui32 x = 0; x < decoded_width; ++x) {
412 auto val = line->i32[x];
413 if (is_signed) {
414 dst[x * num_comps + comp_num] =
415 static_cast<uint16_t>(
416 static_cast<int16_t>(
417 std::clamp(val, -32768, 32767)));
418 } else {
419 dst[x * num_comps + comp_num] =
420 static_cast<uint16_t>(
421 std::clamp(val, 0, 65535));
422 }
423 }
424 }
425 }
426 }
427 }
428
429 codestream.close();
430
431 return kcenon::pacs::ok<compression_result>(
432 compression_result{std::move(output_data), output_params});
433
434 } catch (const std::exception& e) {
435 return kcenon::pacs::pacs_error<compression_result>(
436 kcenon::pacs::error_codes::decompression_error,
437 std::string("HTJ2K decoding failed: ") + e.what());
438 }
