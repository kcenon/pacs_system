100enum class pdu_decode_error : int {

114[[nodiscard]] constexpr const char* to_string(pdu_decode_error err) noexcept {
115 switch (err) {
116 case pdu_decode_error::success: return "Success";
117 case pdu_decode_error::incomplete_header: return "Incomplete PDU header";
118 case pdu_decode_error::incomplete_pdu: return "Incomplete PDU data";
119 case pdu_decode_error::invalid_pdu_type: return "Invalid PDU type";
120 case pdu_decode_error::invalid_protocol_version: return "Invalid protocol version";
121 case pdu_decode_error::invalid_item_type: return "Invalid item type";
122 case pdu_decode_error::malformed_pdu: return "Malformed PDU";
123 case pdu_decode_error::buffer_overflow: return "Buffer overflow";
124 default: return "Unknown error";
125 }
126}
