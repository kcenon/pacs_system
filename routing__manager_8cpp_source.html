68[[nodiscard]] std::string to_lower(std::string_view str) {
69 std::string result;
70 result.reserve(str.size());
71 for (char c : str) {
72 result += static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
73 }
74 return result;

126kcenon::pacs::VoidResult routing_manager::add_rule(const routing_rule& rule) {
127 // Validate
128 if (rule.rule_id.empty()) {
129 return kcenon::pacs::pacs_void_error(-1, "Rule ID cannot be empty");
130 }
131
132 {
133 std::shared_lock lock(rules_mutex_);
134 if (rules_.size() >= config_.max_rules) {
135 return kcenon::pacs::pacs_void_error(-1, "Maximum number of rules reached");
136 }
137 }
138
139 // Save to repository
140 auto result = repo_->save(rule);
141 if (result.is_err()) {
142 return kcenon::pacs::pacs_void_error(
143 result.error().code,
144 "Failed to save rule: " + result.error().message);
145 }
146
147 // Update cache
148 {
149 std::unique_lock lock(rules_mutex_);
150 rules_.push_back(rule);
151
152 // Sort by priority (descending)
153 std::sort(rules_.begin(), rules_.end(),
154 [](const routing_rule& a, const routing_rule& b) {
155 return a.priority > b.priority;
156 });
157 }
158
159 if (logger_) {
160 logger_->info_fmt("routing_manager: Added rule: {} ({})", rule.rule_id, rule.name);
161 }
162
163 return kcenon::pacs::ok();
164}

166kcenon::pacs::VoidResult routing_manager::update_rule(const routing_rule& rule) {
167 // Save to repository
168 auto result = repo_->save(rule);
169 if (result.is_err()) {
170 return kcenon::pacs::pacs_void_error(
171 result.error().code,
172 "Failed to update rule: " + result.error().message);
173 }
174
175 // Update cache
176 {
177 std::unique_lock lock(rules_mutex_);
178 auto it = std::find_if(rules_.begin(), rules_.end(),
179 [&](const routing_rule& r) {
180 return r.rule_id == rule.rule_id;
181 });
182 if (it != rules_.end()) {
183 *it = rule;
184 } else {
185 rules_.push_back(rule);
186 }
187
188 // Re-sort by priority
189 std::sort(rules_.begin(), rules_.end(),
190 [](const routing_rule& a, const routing_rule& b) {
191 return a.priority > b.priority;
192 });
193 }
194
195 if (logger_) {
196 logger_->info_fmt("routing_manager: Updated rule: {} ({})", rule.rule_id, rule.name);
197 }
198
199 return kcenon::pacs::ok();
200}

202kcenon::pacs::VoidResult routing_manager::remove_rule(std::string_view rule_id) {
203 // Remove from repository
204 auto result = repo_->remove(std::string(rule_id));
205 if (!result.is_ok()) {
206 return result;
207 }
208
209 // Remove from cache
210 {
211 std::unique_lock lock(rules_mutex_);
212 rules_.erase(
213 std::remove_if(rules_.begin(), rules_.end(),
214 [&](const routing_rule& r) {
215 return r.rule_id == rule_id;
216 }),
217 rules_.end());
218 }
219
220 if (logger_) {
221 logger_->info_fmt("routing_manager: Removed rule: {}", rule_id);
222 }
223
224 return kcenon::pacs::ok();
225}

282kcenon::pacs::VoidResult routing_manager::reorder_rules(const std::vector<std::string>& rule_ids) {
283 std::unique_lock lock(rules_mutex_);
284
285 // Assign priorities based on order (first = highest)
286 int priority = static_cast<int>(rule_ids.size());
287 for (const auto& rule_id : rule_ids) {
288 auto repo_result = repo_->update_priority(rule_id, priority);
289 if (!repo_result.is_ok()) {
290 return repo_result;
291 }
292
293 auto it = std::find_if(rules_.begin(), rules_.end(),
294 [&](const routing_rule& r) {
295 return r.rule_id == rule_id;
296 });
297 if (it != rules_.end()) {
298 it->priority = priority;
299 }
300 --priority;
301 }
302
303 // Re-sort
304 std::sort(rules_.begin(), rules_.end(),
305 [](const routing_rule& a, const routing_rule& b) {
306 return a.priority > b.priority;
307 });
308
309 return kcenon::pacs::ok();
310}

316std::vector<routing_action> routing_manager::evaluate(const core::dicom_dataset& dataset) {
317 std::vector<routing_action> result;
318
319 if (!enabled_.load()) {
320 return result;
321 }
322
323 ++total_evaluated_;
324
325 std::shared_lock lock(rules_mutex_);
326
327 for (const auto& rule : rules_) {
328 if (!rule.is_effective_now()) {
329 continue;
330 }
331
332 // Check all conditions (AND logic)
333 bool all_match = !rule.conditions.empty();
334 for (const auto& condition : rule.conditions) {
335 if (!match_condition(condition, dataset)) {
336 all_match = false;
337 break;
338 }
339 }
340
341 if (all_match) {
342 ++total_matched_;
343
344 // Add all actions from this rule
345 for (const auto& action : rule.actions) {
346 result.push_back(action);
347 }
348
349 // Only first matching rule is executed (stop after first match)
350 // Uncomment the break below if you want first-match-only behavior
351 // break;
352 }
353 }
354
355 return result;
356}

396void routing_manager::route(const core::dicom_dataset& dataset) {
397 if (!enabled_.load()) {
398 return;
399 }
400
401 auto sop_instance_uid = dataset.get_string(core::tags::sop_instance_uid);
402 if (sop_instance_uid.empty()) {
403 if (logger_) {
404 logger_->warn("routing_manager: Cannot route dataset without SOP Instance UID");
405 }
406 return;
407 }
408
409 auto matches = evaluate_with_rule_ids(dataset);
410
411 for (const auto& [rule_id, actions] : matches) {
412 // Update statistics
413 auto stat_result = repo_->increment_triggered(rule_id);
414 if (!stat_result.is_ok() && logger_) {
415 logger_->warn_fmt("routing_manager: Failed to update statistics for rule: {}",
416 rule_id);
417 }
418
419 // Notify callback
420 if (routing_callback_) {
421 routing_callback_(rule_id, sop_instance_uid, actions);
422 }
423
424 // Execute actions
425 execute_actions(sop_instance_uid, actions);
426 }
427}

607bool routing_manager::match_condition(const routing_condition& condition,
608 const core::dicom_dataset& dataset) const {
609 auto value = get_field_value(condition.match_field, dataset);
610 bool matched = match_pattern(condition.pattern, value, condition.case_sensitive);
611
612 // Apply negation if needed
613 if (condition.negate) {
614 matched = !matched;
615 }
616
617 return matched;
618}
619
620bool routing_manager::match_pattern(std::string_view pattern,
621 std::string_view value,
622 bool case_sensitive) const {
623 // Convert to lowercase if case-insensitive
624 std::string pat_str = case_sensitive ? std::string(pattern) : to_lower(pattern);
625 std::string val_str = case_sensitive ? std::string(value) : to_lower(value);
626
627 // Simple wildcard matching (* and ?)
628 // * matches any sequence of characters
629 // ? matches any single character
630
631 const char* p = pat_str.c_str();
632 const char* v = val_str.c_str();
633
634 const char* star_p = nullptr;
635 const char* star_v = nullptr;
636
637 while (*v) {
638 if (*p == '*') {
639 star_p = p++;
640 star_v = v;
641 } else if (*p == '?' || *p == *v) {
642 ++p;
643 ++v;
644 } else if (star_p) {
645 p = star_p + 1;
646 v = ++star_v;
647 } else {
648 return false;
649 }
650 }
651
652 while (*p == '*') {
653 ++p;
654 }
655
656 return *p == '\0';
657}

697void routing_manager::execute_actions(const std::string& sop_instance_uid,
698 const std::vector<routing_action>& actions) {
699 for (const auto& action : actions) {
700 if (action.destination_node_id.empty()) {
701 if (logger_) {
702 logger_->warn_fmt("routing_manager: Skipping action with empty destination for UID: {}",
703 sop_instance_uid);
704 }
705 continue;
706 }
707
708 // Create a store job via job_manager
709 std::vector<std::string> instance_uids{sop_instance_uid};
710
711 auto job_id = job_manager_->create_store_job(
712 action.destination_node_id,
713 instance_uids,
714 action.priority);
715
716 ++total_forwarded_;
717
718 if (logger_) {
719 logger_->info_fmt("routing_manager: Created forward job {} for UID {} -> {}",
720 job_id, sop_instance_uid, action.destination_node_id);
721 }
722
723 // Note: Delayed forwarding would require a scheduler/timer
724 // For now, jobs are created immediately
725 if (action.delay.count() > 0 && logger_) {
726 logger_->debug_fmt("routing_manager: Delayed forwarding ({} min) not yet implemented",
727 action.delay.count());
728 }
729 }

753 std::sort(rules_.begin(), rules_.end(),
754 [](const routing_rule& a, const routing_rule& b) {

757}
