110accept_worker::result_void accept_worker::before_start() {
111 // Initialize TCP socket for accepting connections
112
113#ifdef _WIN32
114 // Initialize Winsock
115 WSADATA wsa_data;
116 int wsa_result = WSAStartup(MAKEWORD(2, 2), &wsa_data);
117 if (wsa_result != 0) {
118 return kcenon::common::error_info("WSAStartup failed: " + std::to_string(wsa_result));
119 }
120 wsa_initialized_ = true;
121
122 // Create socket
123 listen_socket_ = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
124 if (listen_socket_ == INVALID_SOCKET) {
125 int err = WSAGetLastError();
126 WSACleanup();
127 wsa_initialized_ = false;
128 return kcenon::common::error_info("Failed to create socket: " + std::to_string(err));
129 }
130
131 // Set socket options
132 int opt = 1;
133 setsockopt(listen_socket_, SOL_SOCKET, SO_REUSEADDR,
134 reinterpret_cast<const char*>(&opt), sizeof(opt));
135
136 // Set non-blocking mode
137 u_long mode = 1;
138 ioctlsocket(listen_socket_, FIONBIO, &mode);
139
140 // Bind
141 sockaddr_in addr{};
142 addr.sin_family = AF_INET;
143 addr.sin_addr.s_addr = INADDR_ANY;
144 addr.sin_port = htons(port_);
145
146 if (bind(listen_socket_, reinterpret_cast<sockaddr*>(&addr), sizeof(addr)) == SOCKET_ERROR) {
147 int err = WSAGetLastError();
148 closesocket(listen_socket_);
149 listen_socket_ = INVALID_SOCKET;
150 WSACleanup();
151 wsa_initialized_ = false;
152 return kcenon::common::error_info("Failed to bind to port " +
153 std::to_string(port_) + ": " + std::to_string(err));
154 }
155
156 // Listen
157 if (listen(listen_socket_, backlog_) == SOCKET_ERROR) {
158 int err = WSAGetLastError();
159 closesocket(listen_socket_);
160 listen_socket_ = INVALID_SOCKET;
161 WSACleanup();
162 wsa_initialized_ = false;
163 return kcenon::common::error_info("Failed to listen: " + std::to_string(err));
164 }
165#else
166 // Create socket
167 listen_socket_ = socket(AF_INET, SOCK_STREAM, 0);
168 if (listen_socket_ < 0) {
169 return kcenon::common::error_info("Failed to create socket: " +
170 std::string(std::strerror(errno)));
171 }
172
173 // Set socket options
174 int opt = 1;
175 setsockopt(listen_socket_, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
176
177 // Set non-blocking mode
178 int flags = fcntl(listen_socket_, F_GETFL, 0);
179 if (flags >= 0) {
180 fcntl(listen_socket_, F_SETFL, flags | O_NONBLOCK);
181 }
182
183 // Bind
184 sockaddr_in addr{};
185 addr.sin_family = AF_INET;
186 addr.sin_addr.s_addr = INADDR_ANY;
187 addr.sin_port = htons(port_);
188
189 if (bind(listen_socket_, reinterpret_cast<sockaddr*>(&addr), sizeof(addr)) < 0) {
190 std::string err_msg = std::strerror(errno);
191 close(listen_socket_);
192 listen_socket_ = -1;
193 return kcenon::common::error_info("Failed to bind to port " +
194 std::to_string(port_) + ": " + err_msg);
195 }
196
197 // Listen
198 if (listen(listen_socket_, backlog_) < 0) {
199 std::string err_msg = std::strerror(errno);
200 close(listen_socket_);
201 listen_socket_ = -1;
202 return kcenon::common::error_info("Failed to listen: " + err_msg);
203 }
204#endif
205
206 accepting_.store(true, std::memory_order_release);
207 return kcenon::common::ok();
208}

210accept_worker::result_void accept_worker::do_work() {
211 // Main work routine for the accept loop
212
213#ifdef _WIN32
214 if (listen_socket_ == INVALID_SOCKET) {
215 return kcenon::common::ok();
216 }
217
218 // Use select with short timeout to check for incoming connections
219 fd_set readfds;
220 FD_ZERO(&readfds);
221 FD_SET(listen_socket_, &readfds);
222
223 timeval tv{};
224 tv.tv_sec = 0;
225 tv.tv_usec = 10000; // 10ms timeout for responsive maintenance callbacks
226
227 int result = select(0, &readfds, nullptr, nullptr, &tv);
228
229 if (result > 0 && FD_ISSET(listen_socket_, &readfds)) {
230 sockaddr_in client_addr{};
231 int addr_len = sizeof(client_addr);
232
233 SOCKET client_socket = accept(listen_socket_,
234 reinterpret_cast<sockaddr*>(&client_addr),
235 &addr_len);
236
237 if (client_socket != INVALID_SOCKET) {
238 // Generate session ID and invoke callback
239 uint64_t session_id = next_session_id();
240
241 if (on_connection_) {
242 on_connection_(session_id);
243 }
244
245 // Note: For now, we accept and immediately close the connection
246 // since association doesn't support real network I/O yet.
247 // The connection callback can be extended to handle the socket
248 // when network_system integration is complete.
249 closesocket(client_socket);
250 }
251 }
252#else
253 if (listen_socket_ < 0) {
254 return kcenon::common::ok();
255 }
256
257 // Use select with short timeout to check for incoming connections
258 fd_set readfds;
259 FD_ZERO(&readfds);
260 FD_SET(listen_socket_, &readfds);
261
262 timeval tv{};
263 tv.tv_sec = 0;
264 tv.tv_usec = 10000; // 10ms timeout for responsive maintenance callbacks
265
266 int result = select(listen_socket_ + 1, &readfds, nullptr, nullptr, &tv);
267
268 if (result > 0 && FD_ISSET(listen_socket_, &readfds)) {
269 sockaddr_in client_addr{};
270 socklen_t addr_len = sizeof(client_addr);
271
272 int client_socket = accept(listen_socket_,
273 reinterpret_cast<sockaddr*>(&client_addr),
274 &addr_len);
275
276 if (client_socket >= 0) {
277 // Generate session ID and invoke callback
278 uint64_t session_id = next_session_id();
279
280 if (on_connection_) {
281 on_connection_(session_id);
282 }
283
284 // Note: For now, we accept and immediately close the connection
285 // since association doesn't support real network I/O yet.
286 // The connection callback can be extended to handle the socket
287 // when network_system integration is complete.
288 close(client_socket);
289 }
290 }
291#endif
292
293 // Invoke maintenance callback for periodic tasks (e.g., idle timeout checks)
294 if (on_maintenance_) {
295 on_maintenance_();
296 }
297
298 return kcenon::common::ok();
299}

301accept_worker::result_void accept_worker::after_stop() {
302 // Clean up resources after the accept loop stops
303
304 accepting_.store(false, std::memory_order_release);
305
306#ifdef _WIN32
307 if (listen_socket_ != INVALID_SOCKET) {
308 closesocket(listen_socket_);
309 listen_socket_ = INVALID_SOCKET;
310 }
311 if (wsa_initialized_) {
312 WSACleanup();
313 wsa_initialized_ = false;
314 }
315#else
316 if (listen_socket_ >= 0) {
317 close(listen_socket_);
318 listen_socket_ = -1;
319 }
320#endif
321
322 return kcenon::common::ok();
323}

325bool accept_worker::should_continue_work() const {
326 // Return false to indicate no pending work that must complete before shutdown.
327 // The accept loop will exit promptly when stop() is called.
328 // Note: returning true here would prevent graceful shutdown.
329 return false;
330}

332void accept_worker::on_stop_requested() {
333 // Called when stop() is requested, before thread actually stops.
334 // Close the socket to unblock any pending accept operations.
335
336#ifdef _WIN32
337 if (listen_socket_ != INVALID_SOCKET) {
338 closesocket(listen_socket_);
339 listen_socket_ = INVALID_SOCKET;
340 }
341#else
342 if (listen_socket_ >= 0) {
343 close(listen_socket_);
344 listen_socket_ = -1;
345 }
346#endif
347}
