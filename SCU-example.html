config.calling_ae_title = "MY_SCU";
config.called_ae_title = "REMOTE_SCP";

{"1.2.840.10008.1.2.1"} // Explicit VR LE
});

if (result.is_ok()) {
auto& assoc = result.value();
// Send DIMSE messages...

}
