63auto pdu_decode_job::execute(pipeline_coordinator& coordinator) -> VoidResult {
64 // Validate we have data
65 if (raw_data_.size() < 6) { // Minimum PDU header size
66 if (on_error_) {
67 on_error_(context_.session_id, "PDU data too short");
68 }
69 return pacs_void_error(error_codes::insufficient_data,
70 "PDU data too short for header");
71 }
72
73 // Decode the PDU
74 auto decode_result = decode_pdu();
75 if (!decode_result.is_ok()) {
76 if (on_error_) {
77 auto err = decode_result.error();
78 on_error_(context_.session_id, err.message);
79 }
80 return VoidResult(decode_result.error());
81 }
82
83 auto pdu = decode_result.value();
84
85 // Invoke callback if set
86 if (on_decoded_) {
87 on_decoded_(pdu);
88 }
89
90 // Update context based on PDU type
91 if (pdu.type == kcenon::pacs::network::pdu_type::associate_rq ||
92 pdu.type == kcenon::pacs::network::pdu_type::associate_ac ||
93 pdu.type == kcenon::pacs::network::pdu_type::associate_rj ||
94 pdu.type == kcenon::pacs::network::pdu_type::release_rq ||
95 pdu.type == kcenon::pacs::network::pdu_type::release_rp ||
96 pdu.type == kcenon::pacs::network::pdu_type::abort) {
97 context_.category = job_category::association;
98 }
99
100 // Create DIMSE process job and submit to next stage
101 auto dimse_job = std::make_unique<dimse_process_job>(std::move(pdu));
102
103 // Copy context to DIMSE job
104 dimse_job->get_context() = context_;
105 dimse_job->get_context().stage = pipeline_stage::dimse_process;
106
107 return coordinator.submit_to_stage(
108 pipeline_stage::dimse_process,
109 std::move(dimse_job)
110 );
111}

132auto pdu_decode_job::decode_pdu() -> Result<decoded_pdu> {
133 decoded_pdu result;
134 result.session_id = context_.session_id;
135
136 // PDU Header: Type (1 byte) + Reserved (1 byte) + Length (4 bytes)
137 if (raw_data_.size() < 6) {
138 return pacs_error<decoded_pdu>(error_codes::insufficient_data,
139 "Insufficient data for PDU header");
140 }
141
142 // Extract PDU type
143 result.type = static_cast<kcenon::pacs::network::pdu_type>(raw_data_[0]);
144
145 // Extract length (big-endian, 4 bytes starting at offset 2)
146 uint32_t length = (static_cast<uint32_t>(raw_data_[2]) << 24) |
147 (static_cast<uint32_t>(raw_data_[3]) << 16) |
148 (static_cast<uint32_t>(raw_data_[4]) << 8) |
149 static_cast<uint32_t>(raw_data_[5]);
150
151 // Validate we have complete PDU
152 if (raw_data_.size() < 6 + length) {
153 return pacs_error<decoded_pdu>(error_codes::incomplete_pdu,
154 "Incomplete PDU data");
155 }
156
157 // Copy PDU data (excluding header)
158 result.data.assign(raw_data_.begin() + 6,
159 raw_data_.begin() + 6 + length);
160
161 // For P-DATA-TF, extract presentation context ID and flags
162 if (result.type == kcenon::pacs::network::pdu_type::p_data_tf && !result.data.empty()) {
163 // P-DATA-TF contains PDV items
164 // PDV Item: Length (4) + Presentation Context ID (1) + Message Control Header (1) + Data
165 if (result.data.size() >= 6) {
166 result.presentation_context_id = result.data[4];
167 uint8_t control_header = result.data[5];
168 result.is_last_fragment = (control_header & 0x02) != 0;
169 }
170 }
171
172 return ok(std::move(result));
173}
