104network::Result<std::monostate> query_scp::handle_message(
105 network::association& assoc,
106 uint8_t context_id,
107 const network::dimse::dimse_message& request) {
108
109 using namespace network::dimse;
110 auto start_time = std::chrono::steady_clock::now();
111 std::string calling_ae{assoc.calling_ae()};
112
113 // Verify the message is a C-FIND request
114 if (request.command() != command_field::c_find_rq) {
115 return kcenon::pacs::pacs_void_error(
116 kcenon::pacs::error_codes::find_unexpected_command,
117 "Expected C-FIND-RQ but received " +
118 std::string(to_string(request.command())));
119 }
120
121 // Verify we have a handler
122 if (!handler_) {
123 kcenon::common::get_event_bus().publish(
124 kcenon::pacs::events::query_failed_event{
125 calling_ae,
126 static_cast<int>(kcenon::pacs::error_codes::find_handler_not_set),
127 "No query handler configured"
128 }
129 );
130 return kcenon::pacs::pacs_void_error(
131 kcenon::pacs::error_codes::find_handler_not_set,
132 "No query handler configured");
133 }
134
135 // Get the SOP Class UID from the request
136 auto sop_class_uid = request.affected_sop_class_uid();
137
138 // Verify we have a dataset
139 if (!request.has_dataset()) {
140 return send_final_response(
141 assoc, context_id, request.message_id(),
142 sop_class_uid, status_error_cannot_understand);
143 }
144
145 // Extract query level
146 const auto& query_keys = request.dataset().value().get();
147 auto level = extract_query_level(query_keys);
148
149 if (!level.has_value()) {
150 // Invalid or missing query level
151 return send_final_response(
152 assoc, context_id, request.message_id(),
153 sop_class_uid, status_error_cannot_understand);
154 }
155
156 // Call the handler to get matching results
157 auto results = handler_(level.value(), query_keys, calling_ae);
158
159 // Apply max results limit if configured
160 if (max_results_ > 0 && results.size() > max_results_) {
161 results.resize(max_results_);
162 }
163
164 // Send pending responses for each result
165 for (const auto& result : results) {
166 // Check for cancel request
167 if (cancel_check_ && cancel_check_()) {
168 // Increment statistics
169 ++queries_processed_;
170
171 // Send cancel response
172 return send_final_response(
173 assoc, context_id, request.message_id(),
174 sop_class_uid, status_cancel);
175 }
176
177 // Send pending response with matching dataset
178 auto send_result = send_pending_response(
179 assoc, context_id, request.message_id(),
180 sop_class_uid, result);
181
182 if (send_result.is_err()) {
183 return send_result;
184 }
185 }
186
187 // Increment statistics
188 ++queries_processed_;
189
190 // Calculate execution time
191 auto end_time = std::chrono::steady_clock::now();
192 auto execution_time_ms = std::chrono::duration_cast<std::chrono::milliseconds>(
193 end_time - start_time).count();
194
195 // Publish query executed event
196 kcenon::common::get_event_bus().publish(
197 kcenon::pacs::events::query_executed_event{
198 to_event_level(level.value()),
199 calling_ae,
200 results.size(),
201 static_cast<uint64_t>(execution_time_ms)
202 }
203 );
204
205 // Emit ATNA audit event
206 if (auditor_) {
207 auditor_->audit_query(
208 calling_ae,
209 std::string(assoc.called_ae()),
210 std::string(to_string(level.value())),
211 true);
212 }
213
214 // Send final success response (no dataset)
215 return send_final_response(
216 assoc, context_id, request.message_id(),
217 sop_class_uid, status_success);
218}

240std::optional<query_level> query_scp::extract_query_level(
241 const core::dicom_dataset& dataset) const {
242
243 // Get Query/Retrieve Level tag (0008,0052)
244 auto level_str = dataset.get_string(core::tags::query_retrieve_level);
245
246 if (level_str.empty()) {
247 return std::nullopt;
248 }
249
250 return parse_query_level(level_str);
251}

253network::Result<std::monostate> query_scp::send_pending_response(
254 network::association& assoc,
255 uint8_t context_id,
256 uint16_t message_id,
257 std::string_view sop_class_uid,
258 const core::dicom_dataset& result) {
259
260 using namespace network::dimse;
261
262 // Create C-FIND response with pending status
263 auto response = make_c_find_rsp(
264 message_id,
265 sop_class_uid,
266 status_pending
267 );
268
269 // Attach the matching dataset
270 response.set_dataset(result);
271
272 // Send the response
273 return assoc.send_dimse(context_id, response);
274}
