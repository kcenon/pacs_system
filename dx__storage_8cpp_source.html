278std::vector<std::string> get_dx_storage_sop_classes(bool include_mammography,
279 bool include_intraoral) {
280 std::vector<std::string> result;
281 result.reserve(6);
282
283 for (const auto& info : dx_sop_classes) {
284 // Always include general DX
285 if (!info.is_mammography && !info.is_intraoral) {
286 result.emplace_back(info.uid);
287 }
288 // Include mammography if requested
289 else if (info.is_mammography && include_mammography) {
290 result.emplace_back(info.uid);
291 }
292 // Include intra-oral if requested
293 else if (info.is_intraoral && include_intraoral) {
294 result.emplace_back(info.uid);
295 }
296 }
297
298 return result;
299}

301const dx_sop_class_info*
302get_dx_sop_class_info(std::string_view uid) noexcept {
303 auto it = std::find_if(
304 dx_sop_classes.begin(),
305 dx_sop_classes.end(),
306 [uid](const auto& info) { return info.uid == uid; }
307 );
308
309 if (it != dx_sop_classes.end()) {
310 return &(*it);
311 }
312 return nullptr;
313}
314
315bool is_dx_storage_sop_class(std::string_view uid) noexcept {
316 return get_dx_sop_class_info(uid) != nullptr;
317}
318
319bool is_dx_for_processing_sop_class(std::string_view uid) noexcept {
320 const auto* info = get_dx_sop_class_info(uid);
321 return info != nullptr && info->image_type == dx_image_type::for_processing;
322}
323
324bool is_dx_for_presentation_sop_class(std::string_view uid) noexcept {
325 const auto* info = get_dx_sop_class_info(uid);
326 return info != nullptr && info->image_type == dx_image_type::for_presentation;
327}
328
329bool is_mammography_sop_class(std::string_view uid) noexcept {
330 const auto* info = get_dx_sop_class_info(uid);
331 return info != nullptr && info->is_mammography;
332}
