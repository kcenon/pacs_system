194std::string generate_uuid() {
195 static std::random_device rd;
196 static std::mt19937 gen(rd());
197 static std::uniform_int_distribution<> dis(0, 15);
198 static const char *hex = "0123456789abcdef";
199
200 std::string uuid = "xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx";
201 for (char &c : uuid) {
202 if (c == 'x') {
203 c = hex[dis(gen)];
204 } else if (c == 'y') {
205 c = hex[(dis(gen) & 0x3) | 0x8];
206 }
207 }
208 return uuid;
209}

214std::string format_timestamp(
215 const std::chrono::system_clock::time_point &tp) {
216 auto time = std::chrono::system_clock::to_time_t(tp);
217 std::ostringstream oss;
218 oss << std::put_time(std::gmtime(&time), "%Y-%m-%dT%H:%M:%SZ");
219 return oss.str();

239std::string annotation_to_json(const storage::annotation_record &ann) {
240 std::ostringstream oss;
241 oss << R"({"annotation_id":")" << json_escape(ann.annotation_id)
242 << R"(","study_uid":")" << json_escape(ann.study_uid)
243 << R"(","series_uid":")" << json_escape(ann.series_uid)
244 << R"(","sop_instance_uid":")" << json_escape(ann.sop_instance_uid)
245 << R"(","frame_number":)";
246 if (ann.frame_number.has_value()) {
247 oss << ann.frame_number.value();
248 } else {
249 oss << "null";
250 }
251 oss << R"(,"user_id":")" << json_escape(ann.user_id)
252 << R"(","annotation_type":")" << json_escape(to_string(ann.type))
253 << R"(","geometry":)" << ann.geometry_json
254 << R"(,"text":")" << json_escape(ann.text)
255 << R"(","style":)" << style_to_json(ann.style)
256 << R"(,"created_at":")" << format_timestamp(ann.created_at)
257 << R"(","updated_at":")" << format_timestamp(ann.updated_at) << R"("})";
258 return oss.str();
259}

283std::pair<size_t, size_t> parse_pagination(const crow::request &req) {
284 size_t limit = 20;
285 size_t offset = 0;
286
287 auto limit_param = req.url_params.get("limit");
288 if (limit_param) {
289 try {
290 limit = std::stoul(limit_param);
291 if (limit > 100) {
292 limit = 100;
293 }
294 } catch (...) {
295 // Use default
296 }
297 }
298
299 auto offset_param = req.url_params.get("offset");
300 if (offset_param) {
301 try {
302 offset = std::stoul(offset_param);
303 } catch (...) {
304 // Use default
305 }
306 }
307
308 return {limit, offset};
309}

314std::string parse_json_string(const std::string &json, const std::string &key) {
315 std::string search = "\"" + key + "\":\"";
316 auto pos = json.find(search);
317 if (pos == std::string::npos) {
318 return "";
319 }
320 pos += search.length();
321 auto end = json.find("\"", pos);
322 if (end == std::string::npos) {
323 return "";
324 }
325 return json.substr(pos, end - pos);
326}

331std::optional<int> parse_json_int(const std::string &json,
332 const std::string &key) {
333 std::string search = "\"" + key + "\":";
334 auto pos = json.find(search);
335 if (pos == std::string::npos) {
336 return std::nullopt;
337 }
338 pos += search.length();
339 while (pos < json.size() && (json[pos] == ' ' || json[pos] == '\t')) {
340 ++pos;
341 }
342 if (pos >= json.size()) {
343 return std::nullopt;
344 }
345 if (json.substr(pos, 4) == "null") {
346 return std::nullopt;
347 }
348 try {
349 return std::stoi(json.substr(pos));
350 } catch (...) {
351 return std::nullopt;
352 }
353}
354
358std::string parse_json_object(const std::string &json, const std::string &key) {
359 std::string search = "\"" + key + "\":";
360 auto pos = json.find(search);
361 if (pos == std::string::npos) {
362 return "{}";
363 }
364 pos += search.length();
365 while (pos < json.size() && (json[pos] == ' ' || json[pos] == '\t')) {
366 ++pos;
367 }
368 if (pos >= json.size() || json[pos] != '{') {
369 return "{}";
370 }
371
372 int depth = 0;
373 size_t start = pos;
374 for (; pos < json.size(); ++pos) {
375 if (json[pos] == '{') {
376 ++depth;
377 } else if (json[pos] == '}') {
378 --depth;
379 if (depth == 0) {
380 return json.substr(start, pos - start + 1);
381 }
382 }
383 }
384 return "{}";
385}

390storage::annotation_style parse_style(const std::string &style_json) {
391 storage::annotation_style style;
392 auto color = parse_json_string(style_json, "color");
393 if (!color.empty()) {
394 style.color = color;
395 }
396 auto line_width = parse_json_int(style_json, "line_width");
397 if (line_width.has_value()) {
398 style.line_width = line_width.value();
399 }
400 auto fill_color = parse_json_string(style_json, "fill_color");
401 style.fill_color = fill_color;
402
403 std::string search = "\"fill_opacity\":";
404 auto pos = style_json.find(search);
405 if (pos != std::string::npos) {
406 pos += search.length();
407 try {
408 style.fill_opacity = std::stof(style_json.substr(pos));
409 } catch (...) {
410 // Use default
411 }
412 }
413
414 auto font_family = parse_json_string(style_json, "font_family");
415 if (!font_family.empty()) {
416 style.font_family = font_family;
417 }
418 auto font_size = parse_json_int(style_json, "font_size");
419 if (font_size.has_value()) {
420 style.font_size = font_size.value();
421 }
422 return style;
423}

428void register_annotation_endpoints_impl(crow::SimpleApp &app,
429 std::shared_ptr<rest_server_context> ctx) {
430 // POST /api/v1/annotations - Create annotation
431 CROW_ROUTE(app, "/api/v1/annotations")
432 .methods(crow::HTTPMethod::POST)([ctx](const crow::request &req) {
433 crow::response res;
434 res.add_header("Content-Type", "application/json");
435 add_cors_headers(res, *ctx);
436
437 if (!ctx->database) {
438 res.code = 503;
439 res.body =
440 make_error_json("DATABASE_UNAVAILABLE", "Database not configured");
441 return res;
442 }
443
444 std::string body = req.body;
445 if (body.empty()) {
446 res.code = 400;
447 res.body = make_error_json("INVALID_REQUEST", "Request body is empty");
448 return res;
449 }
450
451 storage::annotation_record ann;
452 ann.annotation_id = generate_uuid();
453 ann.study_uid = parse_json_string(body, "study_uid");
454 ann.series_uid = parse_json_string(body, "series_uid");
455 ann.sop_instance_uid = parse_json_string(body, "sop_instance_uid");
456 ann.frame_number = parse_json_int(body, "frame_number");
457 ann.user_id = parse_json_string(body, "user_id");
458
459 auto type_str = parse_json_string(body, "annotation_type");
460 auto type_opt = storage::annotation_type_from_string(type_str);
461 if (!type_opt.has_value()) {
462 res.code = 400;
463 res.body = make_error_json("INVALID_TYPE", "Invalid annotation type");
464 return res;
465 }
466 ann.type = type_opt.value();
467
468 ann.geometry_json = parse_json_object(body, "geometry");
469 ann.text = parse_json_string(body, "text");
470 ann.style = parse_style(parse_json_object(body, "style"));
471 ann.created_at = std::chrono::system_clock::now();
472 ann.updated_at = ann.created_at;
473
474 if (ann.study_uid.empty()) {
475 res.code = 400;
476 res.body = make_error_json("MISSING_FIELD", "study_uid is required");
477 return res;
478 }
479
480 auto repo = make_annotation_repo(ctx->database.get());
481 auto save_result = repo.save(ann);
482 if (!save_result.is_ok()) {
483 res.code = 500;
484 res.body =
485 make_error_json("SAVE_ERROR", save_result.error().message);
486 return res;
487 }
488
489 res.code = 201;
490 std::ostringstream oss;
491 oss << R"({"annotation_id":")" << json_escape(ann.annotation_id)
492 << R"(","created_at":")" << format_timestamp(ann.created_at)
493 << R"("})";
494 res.body = oss.str();
495 return res;
496 });

498 // GET /api/v1/annotations - List annotations
499 CROW_ROUTE(app, "/api/v1/annotations")
500 .methods(crow::HTTPMethod::GET)([ctx](const crow::request &req) {
501 crow::response res;
502 res.add_header("Content-Type", "application/json");
503 add_cors_headers(res, *ctx);
504
505 if (!ctx->database) {
506 res.code = 503;
507 res.body =
508 make_error_json("DATABASE_UNAVAILABLE", "Database not configured");
509 return res;
510 }
511
512 auto [limit, offset] = parse_pagination(req);
513
514 storage::annotation_query query;
515 query.limit = limit;
516 query.offset = offset;
517
518 auto study_uid = req.url_params.get("study_uid");
519 if (study_uid) {
520 query.study_uid = study_uid;
521 }
522 auto series_uid = req.url_params.get("series_uid");
523 if (series_uid) {
524 query.series_uid = series_uid;
525 }
526 auto sop_instance_uid = req.url_params.get("sop_instance_uid");
527 if (sop_instance_uid) {
528 query.sop_instance_uid = sop_instance_uid;
529 }
530 auto user_id = req.url_params.get("user_id");
531 if (user_id) {
532 query.user_id = user_id;
533 }
534
535 auto repo = make_annotation_repo(ctx->database.get());
536
537 storage::annotation_query count_query = query;
538 count_query.limit = 0;
539 count_query.offset = 0;
540 size_t total_count = count_annotations(repo, count_query);
541
542 auto annotations = search_annotations(repo, query);
543
544 res.code = 200;
545 res.body = annotations_to_json(annotations, total_count);
546 return res;
547 });
