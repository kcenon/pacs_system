56[[nodiscard]] std::string to_timestamp_string(
57 std::chrono::system_clock::time_point tp) {
58 if (tp == std::chrono::system_clock::time_point{}) {
59 return "";
60 }
61 auto time = std::chrono::system_clock::to_time_t(tp);
62 std::tm tm{};
63#ifdef _WIN32
64 gmtime_s(&tm, &time);
65#else
66 gmtime_r(&time, &tm);
67#endif
68 char buf[32];
69 std::strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M:%S", &tm);
70 return buf;
71}
72
74[[nodiscard]] std::chrono::system_clock::time_point from_timestamp_string(
75 const std::string& str) {
76 if (str.empty()) {
77 return {};
78 }
79 std::tm tm{};
80 if (std::sscanf(str.c_str(), "%d-%d-%d %d:%d:%d",
81 &tm.tm_year, &tm.tm_mon, &tm.tm_mday,
82 &tm.tm_hour, &tm.tm_min, &tm.tm_sec) != 6) {
83 return {};
84 }
85 tm.tm_year -= 1900;
86 tm.tm_mon -= 1;
87#ifdef _WIN32
88 auto time = _mkgmtime(&tm);
89#else
90 auto time = timegm(&tm);
91#endif
92 return std::chrono::system_clock::from_time_t(time);
93}

101std::string sync_repository::serialize_vector(const std::vector<std::string>& vec) {
102 if (vec.empty()) return "[]";
103
104 std::ostringstream oss;
105 oss << "[";
106 for (size_t i = 0; i < vec.size(); ++i) {
107 if (i > 0) oss << ",";
108 oss << "\"";
109 for (char c : vec[i]) {
110 if (c == '"') oss << "\\\"";
111 else if (c == '\\') oss << "\\\\";
112 else oss << c;
113 }
114 oss << "\"";
115 }
116 oss << "]";
117 return oss.str();
118}

120std::vector<std::string> sync_repository::deserialize_vector(std::string_view json) {
121 std::vector<std::string> result;
122 if (json.empty() || json == "[]") return result;
123
124 size_t pos = 0;
125 while (pos < json.size()) {
126 auto start = json.find('"', pos);
127 if (start == std::string_view::npos) break;
128
129 size_t end = start + 1;
130 while (end < json.size()) {
131 if (json[end] == '\\' && end + 1 < json.size()) {
132 end += 2;
133 } else if (json[end] == '"') {
134 break;
135 } else {
136 ++end;
137 }
138 }
139
140 if (end < json.size()) {
141 std::string value{json.substr(start + 1, end - start - 1)};
142 std::string unescaped;
143 for (size_t i = 0; i < value.size(); ++i) {
144 if (value[i] == '\\' && i + 1 < value.size()) {
145 unescaped += value[++i];
146 } else {
147 unescaped += value[i];
148 }
149 }
150 result.push_back(std::move(unescaped));
151 }
152
153 pos = end + 1;
154 }
155
156 return result;
157}

218 << "'" << format_timestamp(config.last_successful_sync) << "', "
219 << config.total_syncs << ", "
220 << config.studies_synced << R"()

222 source_node_id = excluded.source_node_id,
223 name = excluded.name,
224 enabled = excluded.enabled,
225 lookback_hours = excluded.lookback_hours,
226 modalities_json = excluded.modalities_json,

686client::sync_config sync_repository::map_row_to_config(
687 const database_row& row) const {
688 client::sync_config config;
689
690 config.pk = std::stoll(row.at("pk"));
691 config.config_id = row.at("config_id");
692 config.source_node_id = row.at("source_node_id");
693 config.name = row.at("name");
694 config.enabled = (row.at("enabled") == "1");
695 config.lookback = std::chrono::hours(std::stoi(row.at("lookback_hours")));
696 config.modalities = deserialize_vector(row.at("modalities_json"));
697 config.patient_id_patterns = deserialize_vector(row.at("patient_patterns_json"));
698 config.direction = client::sync_direction_from_string(row.at("sync_direction"));
699 config.delete_missing = (row.at("delete_missing") == "1");
700 config.overwrite_existing = (row.at("overwrite_existing") == "1");
701 config.sync_metadata_only = (row.at("sync_metadata_only") == "1");

775[[nodiscard]] std::string to_timestamp_string(
776 std::chrono::system_clock::time_point tp) {
777 if (tp == std::chrono::system_clock::time_point{}) {
778 return "";
779 }
780 auto time = std::chrono::system_clock::to_time_t(tp);
781 std::tm tm{};
782#ifdef _WIN32
783 gmtime_s(&tm, &time);
784#else
785 gmtime_r(&time, &tm);
786#endif
787 char buf[32];
788 std::strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M:%S", &tm);
789 return buf;
790}

793[[nodiscard]] std::chrono::system_clock::time_point from_timestamp_string(
794 const char* str) {
795 if (!str || str[0] == '\0') {
796 return {};
797 }
798 std::tm tm{};
799 if (std::sscanf(str, "%d-%d-%d %d:%d:%d",
800 &tm.tm_year, &tm.tm_mon, &tm.tm_mday,
801 &tm.tm_hour, &tm.tm_min, &tm.tm_sec) != 6) {
802 return {};
803 }
804 tm.tm_year -= 1900;
805 tm.tm_mon -= 1;
806#ifdef _WIN32
807 auto time = _mkgmtime(&tm);
808#else
809 auto time = timegm(&tm);
810#endif
811 return std::chrono::system_clock::from_time_t(time);
812}

842std::string sync_repository::serialize_vector(const std::vector<std::string>& vec) {
843 if (vec.empty()) return "[]";
844
845 std::ostringstream oss;
846 oss << "[";
847 for (size_t i = 0; i < vec.size(); ++i) {
848 if (i > 0) oss << ",";
849 oss << "\"";
850 for (char c : vec[i]) {
851 if (c == '"') oss << "\\\"";
852 else if (c == '\\') oss << "\\\\";
853 else oss << c;
854 }
855 oss << "\"";
856 }
857 oss << "]";
858 return oss.str();
859}

861std::vector<std::string> sync_repository::deserialize_vector(std::string_view json) {
862 std::vector<std::string> result;
863 if (json.empty() || json == "[]") return result;
864
865 size_t pos = 0;
866 while (pos < json.size()) {
867 auto start = json.find('"', pos);
868 if (start == std::string_view::npos) break;
869
870 size_t end = start + 1;
871 while (end < json.size()) {
872 if (json[end] == '\\' && end + 1 < json.size()) {
873 end += 2;
874 } else if (json[end] == '"') {
875 break;
876 } else {
877 ++end;
878 }
879 }
880
881 if (end < json.size()) {
882 std::string value{json.substr(start + 1, end - start - 1)};
883 std::string unescaped;
884 for (size_t i = 0; i < value.size(); ++i) {
885 if (value[i] == '\\' && i + 1 < value.size()) {
886 unescaped += value[++i];
887 } else {
888 unescaped += value[i];
889 }
890 }
891 result.push_back(std::move(unescaped));
892 }
893
894 pos = end + 1;
895 }
896
897 return result;
898}
