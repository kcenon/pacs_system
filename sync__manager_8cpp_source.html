72std::string generate_uuid() {
73 static thread_local std::random_device rd;
74 static thread_local std::mt19937_64 gen(rd());
75 static thread_local std::uniform_int_distribution<uint64_t> dis;
76
77 uint64_t ab = dis(gen);
78 uint64_t cd = dis(gen);
79
80 ab = (ab & 0xFFFFFFFFFFFF0FFFULL) | 0x0000000000004000ULL;
81 cd = (cd & 0x3FFFFFFFFFFFFFFFULL) | 0x8000000000000000ULL;
82
83 std::ostringstream oss;
84 oss << std::hex << std::setfill('0');
85 oss << std::setw(8) << (ab >> 32);
86 oss << '-';
87 oss << std::setw(4) << ((ab >> 16) & 0xFFFF);
88 oss << '-';
89 oss << std::setw(4) << (ab & 0xFFFF);
90 oss << '-';
91 oss << std::setw(4) << (cd >> 48);
92 oss << '-';
93 oss << std::setw(12) << (cd & 0xFFFFFFFFFFFFULL);
94
95 return oss.str();
96}

327 sync_result perform_sync(const sync_config& cfg, bool full_sync) {
328 sync_result result;
329 result.config_id = cfg.config_id;
330 result.job_id = generate_uuid();
331 result.started_at = std::chrono::system_clock::now();
332
333 if (logger) {
334 logger->info_fmt("Starting {} sync for config '{}' from node '{}'",
335 full_sync ? "full" : "incremental", cfg.config_id, cfg.source_node_id);
336 }
337
338 // Get the source node
339 auto node_opt = node_manager->get_node(cfg.source_node_id);
340 if (!node_opt) {
341 result.success = false;
342 result.errors.push_back("Source node not found: " + cfg.source_node_id);
343 result.completed_at = std::chrono::system_clock::now();
344 result.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
345 result.completed_at - result.started_at);
346 return result;
347 }
348
349 const auto& node = *node_opt;
350 if (!node.is_online()) {
351 result.success = false;
352 result.errors.push_back("Source node is offline: " + cfg.source_node_id);
353 result.completed_at = std::chrono::system_clock::now();
354 result.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
355 result.completed_at - result.started_at);
356 return result;
357 }
358
359 if (!node.supports_find) {
360 result.success = false;
361 result.errors.push_back("Node does not support C-FIND");
362 result.completed_at = std::chrono::system_clock::now();
363 result.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
364 result.completed_at - result.started_at);
365 return result;
366 }
367
368 // Determine sync start time based on full_sync or incremental
369 auto sync_start_time = full_sync
370 ? std::chrono::system_clock::now() - cfg.lookback
371 : cfg.last_successful_sync;
372
373 // If no successful sync and not full sync, use lookback
374 if (!full_sync && sync_start_time == std::chrono::system_clock::time_point{}) {
375 sync_start_time = std::chrono::system_clock::now() - cfg.lookback;
376 }
377
378 // Query remote studies
379 auto remote_studies = query_remote_studies(cfg, sync_start_time, result);
380 if (!result.errors.empty()) {
381 result.success = false;
382 result.completed_at = std::chrono::system_clock::now();
383 result.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
384 result.completed_at - result.started_at);
385 return result;
386 }
387
388 result.studies_checked = remote_studies.size();
389
390 // Compare with local studies
391 auto comparison = compare_with_local(cfg, remote_studies, result);
392
393 // Process differences based on sync direction
394 if (cfg.direction == sync_direction::pull ||
395 cfg.direction == sync_direction::bidirectional) {
396 // Pull missing studies from remote
397 for (const auto& conflict : comparison) {
398 if (conflict.conflict_type == sync_conflict_type::missing_local) {
399 // Create retrieve job
400 if (job_mgr && node.supports_query_retrieve()) {
401 auto job_id = job_mgr->create_retrieve_job(
402 cfg.source_node_id, conflict.study_uid, std::nullopt,
403 job_priority::normal);
404 if (!job_id.empty()) {
405 result.studies_synced++;
406 notify_progress(cfg.config_id, result.studies_synced,
407 result.studies_checked);
408 }
409 }
410 } else if (conflict.conflict_type == sync_conflict_type::count_mismatch ||
411 conflict.conflict_type == sync_conflict_type::modified) {
412 // Handle conflicts
413 if (config.auto_resolve_conflicts) {
414 resolve_conflict_internal(conflict, config.default_resolution, result);
415 } else {
416 add_conflict(conflict);
417 result.conflicts.push_back(conflict);
418 }
419 }
420 }
421 }
422
423 if (cfg.direction == sync_direction::push ||
424 cfg.direction == sync_direction::bidirectional) {
425 // Push studies missing on remote
426 for (const auto& conflict : comparison) {
427 if (conflict.conflict_type == sync_conflict_type::missing_remote) {
428 // Store job would be created here for push
429 if (logger) {
430 logger->info_fmt("Study {} is missing on remote, would create store job",
431 conflict.study_uid);
432 }
433 result.studies_synced++;
434 }
435 }
436 }
437
438 // Update statistics
439 total_syncs.fetch_add(1, std::memory_order_relaxed);
440 total_studies_synced.fetch_add(result.studies_synced, std::memory_order_relaxed);
441
442 result.success = result.errors.empty();
443 result.completed_at = std::chrono::system_clock::now();
444 result.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
445 result.completed_at - result.started_at);
446
447 if (result.success) {
448 successful_syncs.fetch_add(1, std::memory_order_relaxed);
449 } else {
450 failed_syncs.fetch_add(1, std::memory_order_relaxed);
451 }
452
453 // Update config statistics
454 update_config_stats(cfg.config_id, result.success, result.studies_synced);
455
456 // Store result
457 store_last_result(cfg.config_id, result);
458
459 // Save history
460 sync_history history;
461 history.config_id = cfg.config_id;
462 history.job_id = result.job_id;
463 history.success = result.success;
464 history.studies_checked = result.studies_checked;
465 history.studies_synced = result.studies_synced;
466 history.conflicts_found = result.conflicts.size();
467 history.errors = result.errors;
468 history.started_at = result.started_at;
469 history.completed_at = result.completed_at;
470
471 #ifdef PACS_WITH_DATABASE_SYSTEM
472 if (repositories.history) {
473 [[maybe_unused]] auto save_result = repositories.history->save(history);
474 } else if (compatibility_repo) {
475 [[maybe_unused]] auto save_result =
476 compatibility_repo->save_history(history);
477 }
478 #else
479 if (compatibility_repo) {
480 [[maybe_unused]] auto save_result =
481 compatibility_repo->save_history(history);
482 }
483 #endif
484
485 if (logger) {
486 logger->info_fmt("Sync completed for config '{}': {} checked, {} synced, {} conflicts",
487 cfg.config_id, result.studies_checked, result.studies_synced,
488 result.conflicts.size());
489 }
490
491 return result;
492 }

494 std::vector<std::string> query_remote_studies(
495 const sync_config& cfg,
496 std::chrono::system_clock::time_point since,
497 sync_result& result) {
498
499 std::vector<std::string> study_uids;
500
501 // Get association for query
502 std::vector<std::string> sop_classes = {
503 std::string(services::study_root_find_sop_class_uid)
504 };
505
506 auto assoc_result = node_manager->acquire_association(
507 cfg.source_node_id, sop_classes);
508 if (assoc_result.is_err()) {
509 result.errors.push_back("Failed to acquire association: " +
510 assoc_result.error().message);
511 return study_uids;
512 }
513
514 auto assoc = std::move(assoc_result.value());
515
516 // Build query
517 core::dicom_dataset query_keys;
518 query_keys.set_string(core::tags::query_retrieve_level, encoding::vr_type::CS, "STUDY");
519 query_keys.set_string(core::tags::study_instance_uid, encoding::vr_type::UI, "");
520
521 // Apply filters
522 if (!cfg.modalities.empty()) {
523 std::string modality_filter;
524 for (size_t i = 0; i < cfg.modalities.size(); ++i) {
525 if (i > 0) modality_filter += "\\";
526 modality_filter += cfg.modalities[i];
527 }
528 query_keys.set_string(core::tags::modality, encoding::vr_type::CS, modality_filter);
529 }
530
531 // Date range filter
532 if (since != std::chrono::system_clock::time_point{}) {
533 auto since_time_t = std::chrono::system_clock::to_time_t(since);
534 std::tm tm_buf;
535#ifdef _WIN32
536 localtime_s(&tm_buf, &since_time_t);
537#else
538 localtime_r(&since_time_t, &tm_buf);
539#endif
540 std::ostringstream date_oss;
541 date_oss << std::put_time(&tm_buf, "%Y%m%d") << "-";
542 query_keys.set_string(core::tags::study_date, encoding::vr_type::DA, date_oss.str());
543 }
544
545 // Execute query
546 services::query_scu_config query_config;
547 query_config.model = services::query_model::study_root;
548 query_config.level = services::query_level::study;
549
550 services::query_scu scu(query_config, logger);
551 auto query_result = scu.find(*assoc, query_keys);
552
553 // Release association
554 node_manager->release_association(cfg.source_node_id, std::move(assoc));
555
556 if (query_result.is_err()) {
557 result.errors.push_back("Query failed: " + query_result.error().message);
558 return study_uids;
559 }
560
561 const auto& qr = query_result.value();
562 for (const auto& match : qr.matches) {
563 auto uid = match.get_string(core::tags::study_instance_uid);
564 if (!uid.empty()) {
565 study_uids.push_back(uid);
566 }
567 }
568
569 return study_uids;
570 }

600 void resolve_conflict_internal(const sync_conflict& conflict,
601 conflict_resolution resolution,
602 sync_result& result) {
603 switch (resolution) {
604 case conflict_resolution::prefer_remote:
605 // Create retrieve job
606 if (job_mgr) {
607 auto job_id = job_mgr->create_retrieve_job(
608 conflict.config_id, conflict.study_uid, std::nullopt,
609 job_priority::normal);
610 if (!job_id.empty()) {
611 result.studies_synced++;
612 }
613 }
614 break;
615
616 case conflict_resolution::prefer_local:
617 // Keep local, skip
618 result.studies_skipped++;
619 break;
620
621 case conflict_resolution::prefer_newer:
622 // Compare timestamps and choose
623 if (conflict.remote_modified > conflict.local_modified) {
624 if (job_mgr) {
625 auto job_id = job_mgr->create_retrieve_job(
626 conflict.config_id, conflict.study_uid, std::nullopt,
627 job_priority::normal);
628 if (!job_id.empty()) {
629 result.studies_synced++;
630 }
631 }
632 } else {
633 result.studies_skipped++;
634 }
635 break;
636 }
637
638 total_conflicts_resolved.fetch_add(1, std::memory_order_relaxed);
639 }

645 void scheduler_loop() {
646 while (scheduler_running.load()) {
647 std::unique_lock lock(scheduler_mutex);
648 scheduler_cv.wait_for(lock, std::chrono::minutes(1), [this] {
649 return !scheduler_running.load();
650 });
651
652 if (!scheduler_running.load()) {
653 break;
654 }
655
656 // Check each config for scheduled syncs
657 std::vector<sync_config> configs_copy;
658 {
659 std::shared_lock cfg_lock(configs_mutex);
660 configs_copy = configs;
661 }
662
663 for (const auto& cfg : configs_copy) {
664 if (!cfg.enabled || cfg.schedule_cron.empty()) {
665 continue;
666 }
667
668 // Simple cron check - in production would parse cron expression
669 // For now, check if it's time based on last sync
670 auto now = std::chrono::system_clock::now();
671 auto since_last = now - cfg.last_sync;
672
673 // Default to hourly if cron not parsed
674 if (since_last >= std::chrono::hours(1)) {
675 if (!is_sync_active(cfg.config_id)) {
676 if (logger) {
677 logger->info_fmt("Scheduler triggering sync for config '{}'",
678 cfg.config_id);
679 }
680
681 mark_sync_active(cfg.config_id);
682 auto result = perform_sync(cfg, false);
683 mark_sync_inactive(cfg.config_id);
684 notify_completion(cfg.config_id, result);
685 }
686 }
687 }
688 }
