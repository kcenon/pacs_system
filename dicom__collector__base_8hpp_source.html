72struct dicom_metric {

128template <typename Derived>
129class dicom_collector_base {
130public:
131 dicom_collector_base() = default;
132 virtual ~dicom_collector_base() = default;
133
134 // Non-copyable, non-moveable
135 dicom_collector_base(const dicom_collector_base&) = delete;
136 dicom_collector_base& operator=(const dicom_collector_base&) = delete;
137 dicom_collector_base(dicom_collector_base&&) = delete;
138 dicom_collector_base& operator=(dicom_collector_base&&) = delete;
139
147 bool initialize(const config_map& config) {
148 // Parse common configuration
149 if (auto it = config.find("enabled"); it != config.end()) {
150 enabled_ = (it->second == "true" || it->second == "1");
151 }
152
153 if (auto it = config.find("ae_title"); it != config.end()) {
154 ae_title_ = it->second;
155 }
156
157 init_time_ = std::chrono::steady_clock::now();
158
159 // Delegate to derived class for specific initialization
160 return derived().do_initialize(config);
161 }
162
167 std::vector<dicom_metric> collect() {
168 if (!enabled_) {
169 return {};
170 }
171
172 try {
173 auto metrics = derived().do_collect();
174 ++collection_count_;
175 last_collection_time_ = std::chrono::system_clock::now();
176 return metrics;
177 } catch (...) {
178 ++collection_errors_;
179 return {};
180 }
181 }
182
187 [[nodiscard]] std::string get_name() const {
188 return Derived::collector_name;
189 }
190
195 [[nodiscard]] std::vector<std::string> get_metric_types() const {
196 return derived().do_get_metric_types();
197 }
198
203 [[nodiscard]] bool is_healthy() const {
204 if (!enabled_) {
205 return true; // Disabled collectors are considered healthy
206 }
207 return derived().is_available();
208 }
209
214 [[nodiscard]] stats_map get_statistics() const {
215 std::lock_guard<std::mutex> lock(stats_mutex_);
216 stats_map stats;
217
218 // Common statistics
219 stats["enabled"] = enabled_ ? 1.0 : 0.0;
220 stats["collection_count"] = static_cast<double>(collection_count_.load());
221 stats["collection_errors"] = static_cast<double>(collection_errors_.load());
222
223 // Uptime if initialized
224 if (init_time_.time_since_epoch().count() > 0) {
225 const auto uptime = std::chrono::duration_cast<std::chrono::seconds>(
226 std::chrono::steady_clock::now() - init_time_);
227 stats["uptime_seconds"] = static_cast<double>(uptime.count());
228 }
229
230 // Let derived class add specific statistics
231 derived().do_add_statistics(stats);
232
233 return stats;
234 }
235
240 [[nodiscard]] bool is_enabled() const { return enabled_; }
241
246 [[nodiscard]] std::size_t get_collection_count() const {
247 return collection_count_.load();
248 }
249
254 [[nodiscard]] std::size_t get_collection_errors() const {
255 return collection_errors_.load();
256 }
257
262 [[nodiscard]] std::string get_ae_title() const { return ae_title_; }
263
268 void set_ae_title(std::string ae_title) { ae_title_ = std::move(ae_title); }
269
270protected:
279 [[nodiscard]] dicom_metric create_base_metric(
280 const std::string& name,
281 double value,
282 const std::string& type,
283 const std::unordered_map<std::string, std::string>& extra_tags = {}) const {
284
285 std::unordered_map<std::string, std::string> tags = extra_tags;
286 tags["collector"] = Derived::collector_name;
287 if (!ae_title_.empty()) {
288 tags["ae_title"] = ae_title_;
289 }
290
291 return dicom_metric(name, value, type, std::move(tags));
292 }
293
294 // Configuration
295 bool enabled_{true};
296 std::string ae_title_{"PACS_SCP"};
297
298 // Statistics
299 mutable std::mutex stats_mutex_;
300 std::atomic<std::size_t> collection_count_{0};
301 std::atomic<std::size_t> collection_errors_{0};
302 std::chrono::steady_clock::time_point init_time_;
303 std::chrono::system_clock::time_point last_collection_time_;
304
305private:
309 Derived& derived() { return static_cast<Derived&>(*this); }
310 const Derived& derived() const { return static_cast<const Derived&>(*this); }
311};
