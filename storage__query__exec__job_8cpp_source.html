46 service_handler handler,

48 error_callback on_error)
49 : request_(std::move(request))
50 , handler_(std::move(handler))

65auto storage_query_exec_job::execute(pipeline_coordinator& coordinator)
66 -> VoidResult {
67
68 // Check if handler is set
69 if (!handler_) {
70 // Create default response (no handler means we need to reject)
71 service_result result;
72 result.session_id = request_.session_id;
73 result.message_id = request_.message_id;
74 result.presentation_context_id = request_.presentation_context_id;
75 result.sop_class_uid = request_.sop_class_uid;
76 result.sop_instance_uid = request_.sop_instance_uid;
77 result.status = dimse_status::failure_refused_sop_class_not_supported;
78 result.error_comment = "No service handler configured";
79
80 // Determine response type
81 uint16_t cmd = static_cast<uint16_t>(request_.command_type);
82 result.response_type = static_cast<dimse_command_type>(cmd | 0x8000);
83
84 if (on_error_) {
85 on_error_(request_.session_id, "No service handler configured");
86 }
87
88 // Still submit to encode stage for proper response
89 auto encode_job = std::make_unique<response_encode_job>(std::move(result));
90 encode_job->get_context() = context_;
91 encode_job->get_context().stage = pipeline_stage::response_encode;
92
93 return coordinator.submit_to_stage(
94 pipeline_stage::response_encode,
95 std::move(encode_job)
96 );
97 }
98
99 // Execute the service handler
100 auto result = handler_(request_);
101
102 if (!result.is_ok()) {
103 if (on_error_) {
104 auto err = result.error();
105 on_error_(request_.session_id, err.message);
106 }
107
108 // Create error response
109 service_result error_response;
110 error_response.session_id = request_.session_id;
111 error_response.message_id = request_.message_id;
112 error_response.presentation_context_id = request_.presentation_context_id;
113 error_response.sop_class_uid = request_.sop_class_uid;
114 error_response.sop_instance_uid = request_.sop_instance_uid;
115 error_response.status = dimse_status::failure_unable_to_process;
116 error_response.error_comment = result.error().message;
117
118 uint16_t cmd = static_cast<uint16_t>(request_.command_type);
119 error_response.response_type = static_cast<dimse_command_type>(cmd | 0x8000);
120
121 auto encode_job = std::make_unique<response_encode_job>(std::move(error_response));
122 encode_job->get_context() = context_;
123 encode_job->get_context().stage = pipeline_stage::response_encode;
124
125 return coordinator.submit_to_stage(
126 pipeline_stage::response_encode,
127 std::move(encode_job)
128 );
129 }
130
131 auto service_result_value = result.value();
132
133 // Invoke completion callback
134 if (on_complete_) {
135 on_complete_(service_result_value);
136 }
137
138 // Submit to encode stage
139 auto encode_job = std::make_unique<response_encode_job>(std::move(service_result_value));
140 encode_job->get_context() = context_;
141 encode_job->get_context().stage = pipeline_stage::response_encode;
142
143 return coordinator.submit_to_stage(
144 pipeline_stage::response_encode,
145 std::move(encode_job)
146 );
147}

168auto storage_query_exec_job::get_category_for_command(dimse_command_type type)
169 -> job_category {
170 switch (type) {
171 case dimse_command_type::c_echo_rq:
172 case dimse_command_type::c_echo_rsp:
173 return job_category::echo;
174
175 case dimse_command_type::c_store_rq:
176 case dimse_command_type::c_store_rsp:
177 return job_category::store;
178
179 case dimse_command_type::c_find_rq:
180 case dimse_command_type::c_find_rsp:
181 return job_category::find;
182
183 case dimse_command_type::c_get_rq:
184 case dimse_command_type::c_get_rsp:
185 return job_category::get;
186
187 case dimse_command_type::c_move_rq:
188 case dimse_command_type::c_move_rsp:
189 return job_category::move;
190
191 default:
192 return job_category::other;
193 }
194}
