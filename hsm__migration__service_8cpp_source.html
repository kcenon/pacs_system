73void hsm_migration_service::start() {
74 std::lock_guard lock(mutex_);
75
76 if (running_.load()) {
77 return; // Already running
78 }
79
80 stop_requested_.store(false);
81 running_.store(true);
82
83 // Set next cycle time
84 next_cycle_time_ =
85 std::chrono::steady_clock::now() + config_.migration_interval;
86
87 // Start worker thread
88 worker_thread_ = std::thread([this]() { run_loop(); });
89}

91void hsm_migration_service::stop(bool wait_for_completion) {
92 {
93 std::lock_guard lock(mutex_);
94
95 if (!running_.load()) {
96 return; // Not running
97 }
98
99 stop_requested_.store(true);
100 }
101
102 // Wake up the worker thread
103 cv_.notify_all();
104
105 // Wait for thread to finish
106 if (worker_thread_.joinable()) {
107 if (wait_for_completion) {
108 worker_thread_.join();
109 } else {
110 worker_thread_.detach();
111 }
112 }
113
114 running_.store(false);
115}

158auto hsm_migration_service::time_until_next_cycle() const
159 -> std::optional<std::chrono::seconds> {
160 std::lock_guard lock(mutex_);
161
162 if (!running_.load()) {
163 return std::nullopt;
164 }
165
166 auto now = std::chrono::steady_clock::now();
167 if (next_cycle_time_ <= now) {
168 return std::chrono::seconds{0};
169 }
170
171 return std::chrono::duration_cast<std::chrono::seconds>(next_cycle_time_ -
172 now);
173}

210void hsm_migration_service::run_loop() {
211 while (!stop_requested_.load()) {
212 std::unique_lock lock(mutex_);
213
214 // Wait until next cycle time or stop requested
215 cv_.wait_until(lock, next_cycle_time_, [this]() {
216 return stop_requested_.load() ||
217 std::chrono::steady_clock::now() >= next_cycle_time_;
218 });
219
220 if (stop_requested_.load()) {
221 break;
222 }
223
224 // Release lock during migration
225 lock.unlock();
226
227 // Execute migration cycle
228 auto result = execute_cycle();
229
230 // Update statistics
231 update_stats(result);
232
233 // Store result
234 lock.lock();
235 last_result_ = result;
236 cycles_count_++;
237
238 // Schedule next cycle
239 next_cycle_time_ =
240 std::chrono::steady_clock::now() + config_.migration_interval;
241
242 // Call progress callback (outside of lock)
243 auto callback = config_.on_cycle_complete;
244 lock.unlock();
245
246 if (callback) {
247 callback(result);
248 }
249 }
250}

252auto hsm_migration_service::execute_cycle() -> migration_result {
253 cycle_in_progress_.store(true);
254
255 auto result = storage_.run_migration_cycle();
256
257 // Call error callbacks for failures
258 if (!result.failed_uids.empty() && config_.on_migration_error) {
259 for (const auto& uid : result.failed_uids) {
260 config_.on_migration_error(uid, "Migration failed");
261 }
262 }
263
264 cycle_in_progress_.store(false);
265
266 return result;
267}

269void hsm_migration_service::update_stats(const migration_result& result) {
270 std::lock_guard lock(mutex_);
271
272 cumulative_stats_.instances_migrated += result.instances_migrated;
273 cumulative_stats_.bytes_migrated += result.bytes_migrated;
274 cumulative_stats_.duration += result.duration;
275 cumulative_stats_.instances_skipped += result.instances_skipped;
276
277 // Append failed UIDs (keep last N)
278 constexpr std::size_t kMaxFailedUids = 100;
279 for (const auto& uid : result.failed_uids) {
280 if (cumulative_stats_.failed_uids.size() >= kMaxFailedUids) {
281 cumulative_stats_.failed_uids.erase(
282 cumulative_stats_.failed_uids.begin());
283 }
284 cumulative_stats_.failed_uids.push_back(uid);
285 }
286}
