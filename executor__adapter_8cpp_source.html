63thread_pool_executor_adapter::thread_pool_executor_adapter(std::size_t worker_count) {
64 kcenon::thread::thread_context context;
65 pool_ = std::make_shared<kcenon::thread::thread_pool>("executor_pool", context);
66
67 // Create and enqueue workers
68 std::vector<std::unique_ptr<kcenon::thread::thread_worker>> workers;
69 workers.reserve(worker_count);
70
71 for (std::size_t i = 0; i < worker_count; ++i) {
72 workers.push_back(std::make_unique<kcenon::thread::thread_worker>(false, context));
73 }
74
75 auto enqueue_result = pool_->enqueue_batch(std::move(workers));
76 if (enqueue_result.is_err()) {
77 throw std::runtime_error("Failed to enqueue workers to thread pool");
78 }
79
80 auto start_result = pool_->start();
81 if (start_result.is_err()) {
82 throw std::runtime_error("Failed to start thread pool");
83 }
84}

90kcenon::common::Result<std::future<void>> thread_pool_executor_adapter::execute(
91 std::unique_ptr<kcenon::common::interfaces::IJob>&& job) {
92
93 if (!running_.load()) {
94 return kcenon::common::Result<std::future<void>>(
95 kcenon::common::error_info{-1, "Executor is not running", "executor"});
96 }
97
98 if (!job) {
99 return kcenon::common::Result<std::future<void>>(
100 kcenon::common::error_info{-2, "Job cannot be null", "executor"});
101 }
102
103 auto promise = std::make_shared<std::promise<void>>();
104 auto future = promise->get_future();
105
106 // Capture job by shared_ptr for thread safety
107 auto shared_job = std::shared_ptr<kcenon::common::interfaces::IJob>(std::move(job));
108
109 ++pending_count_;
110
111 try {
112 auto job_obj = kcenon::thread::job_builder()
113 .name("executor_task")
114 .work([this, shared_job, promise]() -> kcenon::common::VoidResult {
115 try {
116 auto result = shared_job->execute();
117 if (result.is_ok()) {
118 promise->set_value();
119 } else {
120 promise->set_exception(std::make_exception_ptr(
121 std::runtime_error(result.error().message)));
122 }
123 } catch (...) {
124 promise->set_exception(std::current_exception());
125 }
126 --pending_count_;
127 return kcenon::common::ok();
128 })
129 .build();
130
131 auto enqueue_result = pool_->enqueue(std::move(job_obj));
132 if (enqueue_result.is_err()) {
133 --pending_count_;
134 return kcenon::common::Result<std::future<void>>(
135 kcenon::common::error_info{-3, "Failed to enqueue task to thread pool", "executor"});
136 }
137 } catch (const std::exception&) {
138 --pending_count_;
139 return kcenon::common::Result<std::future<void>>(
140 kcenon::common::error_info{-3, "Failed to enqueue task to thread pool", "executor"});
141 }
142
143 return kcenon::common::Result<std::future<void>>::ok(std::move(future));
144}

146kcenon::common::Result<std::future<void>> thread_pool_executor_adapter::execute_delayed(
147 std::unique_ptr<kcenon::common::interfaces::IJob>&& job,
148 std::chrono::milliseconds delay) {
149
150 if (!running_.load()) {
151 return kcenon::common::Result<std::future<void>>(
152 kcenon::common::error_info{-1, "Executor is not running", "executor"});
153 }
154
155 if (!job) {
156 return kcenon::common::Result<std::future<void>>(
157 kcenon::common::error_info{-2, "Job cannot be null", "executor"});
158 }
159
160 auto promise = std::make_shared<std::promise<void>>();
161 auto future = promise->get_future();
162
163 // Capture job and schedule for later execution
164 auto shared_job = std::shared_ptr<kcenon::common::interfaces::IJob>(std::move(job));
165 auto execute_at = std::chrono::steady_clock::now() + delay;
166
167 ++pending_count_;
168
169 {
170 std::lock_guard<std::mutex> lock(delay_mutex_);
171 delayed_tasks_.push({execute_at,
172 [this, shared_job, promise]() mutable {
173 try {
174 auto result = shared_job->execute();
175 if (result.is_ok()) {
176 promise->set_value();
177 } else {
178 promise->set_exception(std::make_exception_ptr(
179 std::runtime_error(result.error().message)));
180 }
181 } catch (...) {
182 promise->set_exception(std::current_exception());
183 }
184 --pending_count_;
185 }});
186 }
187
188 // Start delay thread if not already running
189 if (!delay_thread_.joinable()) {
190 delay_thread_ = std::thread([this]() {
191 while (!shutdown_requested_.load()) {
192 std::function<void()> task;
193
194 {
195 std::unique_lock<std::mutex> lock(delay_mutex_);
196
197 if (delayed_tasks_.empty()) {
198 delay_cv_.wait_for(lock, std::chrono::milliseconds(100));
199 continue;
200 }
201
202 auto now = std::chrono::steady_clock::now();
203 const auto& top_task = delayed_tasks_.top();
204
205 if (top_task.execute_at > now) {
206 delay_cv_.wait_until(lock, top_task.execute_at);
207 continue;
208 }
209
210 task = top_task.task;
211 delayed_tasks_.pop();
212 }
213
214 if (task && pool_ && pool_->is_running()) {
215 try {
216 auto job_obj = kcenon::thread::job_builder()
217 .name("delayed_task")
218 .work([task = std::move(task)]() -> kcenon::common::VoidResult {
219 task();
220 return kcenon::common::ok();
221 })
222 .build();
223
224 (void)pool_->enqueue(std::move(job_obj));
225 } catch (...) {
226 // Ignore enqueue failures in delayed task scheduler
227 }
228 }
229 }
230 });
231 }
232
233 delay_cv_.notify_one();
234
235 return kcenon::common::Result<std::future<void>>::ok(std::move(future));
236}

251void thread_pool_executor_adapter::shutdown(bool wait_for_completion) {
252 running_.store(false);
253 shutdown_requested_.store(true);
254
255 delay_cv_.notify_all();
256
257 if (delay_thread_.joinable()) {
258 delay_thread_.join();
259 }
260
261 if (pool_) {
262 pool_->stop(!wait_for_completion);
263 }
264}

281 // Create an executor that uses the pool interface
282 // This requires creating a custom adapter that works with the interface
283
284 class interface_executor_adapter : public kcenon::common::interfaces::IExecutor {
285 public:
286 explicit interface_executor_adapter(std::shared_ptr<thread_pool_interface> pool)
287 : pool_(std::move(pool)) {}
288
289 kcenon::common::Result<std::future<void>> execute(
290 std::unique_ptr<kcenon::common::interfaces::IJob>&& job) override {
291
292 if (!pool_ || !pool_->is_running()) {
293 return kcenon::common::Result<std::future<void>>(
294 kcenon::common::error_info{-1, "Pool is not running", "executor"});
295 }
296
297 auto shared_job = std::shared_ptr<kcenon::common::interfaces::IJob>(std::move(job));
298
299 auto future = pool_->submit([shared_job]() {
300 auto result = shared_job->execute();
301 if (result.is_err()) {
302 throw std::runtime_error(result.error().message);
303 }
304 });
305
306 return kcenon::common::Result<std::future<void>>::ok(std::move(future));
307 }
308
309 kcenon::common::Result<std::future<void>> execute_delayed(
310 std::unique_ptr<kcenon::common::interfaces::IJob>&& /*job*/,
311 std::chrono::milliseconds /*delay*/) override {
312 // Not supported through interface
313 return kcenon::common::Result<std::future<void>>(
314 kcenon::common::error_info{-4, "Delayed execution not supported through pool interface", "executor"});
315 }
316
317 std::size_t worker_count() const override {
318 return pool_ ? pool_->get_thread_count() : 0;
319 }
320
321 bool is_running() const override {
322 return pool_ && pool_->is_running();
323 }
324
325 std::size_t pending_tasks() const override {
326 return pool_ ? pool_->get_pending_task_count() : 0;
327 }
328
329 void shutdown(bool wait_for_completion) override {
330 if (pool_) {
331 pool_->shutdown(wait_for_completion);
332 }
333 }
334
335 private:
336 std::shared_ptr<thread_pool_interface> pool_;
337 };
338
339 return std::make_shared<interface_executor_adapter>(std::move(pool_interface));
340}
