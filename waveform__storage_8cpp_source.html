149std::vector<std::string>
150get_waveform_storage_sop_classes(bool include_presentation_state,
151 bool include_annotation) {
152 std::vector<std::string> result;
153 result.reserve(kWaveformRegistry.size());
154
155 for (const auto& entry : kWaveformRegistry) {
156 if (entry.type == waveform_type::presentation_state &&
157 !include_presentation_state) {
158 continue;
159 }
160 if (entry.type == waveform_type::annotation && !include_annotation) {
161 continue;
162 }
163 result.emplace_back(entry.uid);
164 }
165 return result;
166}

168const waveform_sop_class_info*
169get_waveform_sop_class_info(std::string_view uid) noexcept {
170 auto it = std::find_if(kWaveformRegistry.begin(), kWaveformRegistry.end(),
171 [&uid](const waveform_sop_class_info& info) {
172 return info.uid == uid;
173 });
174 return (it != kWaveformRegistry.end()) ? &(*it) : nullptr;
175}
