80query_cache::size_type query_cache::invalidate_by_query_level(const std::string& query_level) {
81 // Keys are formatted as "LEVEL:params" or "AE/LEVEL:params"
82 // Match both formats
83 return cache_.invalidate_if(
84 [&query_level](const std::string& key, const cached_query_result&) {
85 // Direct match: "LEVEL:..."
86 if (key.starts_with(query_level + ":")) {
87 return true;
88 }
89 // AE prefix match: "AE/LEVEL:..."
90 auto slash_pos = key.find('/');
91 if (slash_pos != std::string::npos) {
92 auto level_start = slash_pos + 1;
93 auto colon_pos = key.find(':', level_start);
94 if (colon_pos != std::string::npos) {
95 auto key_level = key.substr(level_start, colon_pos - level_start);
96 return key_level == query_level;
97 }
98 }
99 return false;
100 });
101}

135std::string query_cache::build_key(
136 const std::string& query_level,
137 const std::vector<std::pair<std::string, std::string>>& params) {
138
139 // Sort parameters for consistent key generation
140 auto sorted_params = params;
141 std::sort(sorted_params.begin(), sorted_params.end(),
142 [](const auto& a, const auto& b) { return a.first < b.first; });
143
144 std::ostringstream oss;
145 oss << query_level << ":";
146
147 bool first = true;
148 for (const auto& [name, value] : sorted_params) {
149 if (!first) {
150 oss << ";";
151 }
152 first = false;
153 oss << name << "=" << value;
154 }
155
156 return oss.str();
157}

179 if (g_config_pending) {
180 g_query_cache = std::make_unique<query_cache>(g_pending_config);

182 g_query_cache = std::make_unique<query_cache>();
183 }

188query_cache& global_query_cache() {
189 std::call_once(g_cache_init_flag, init_global_cache);
190 return *g_query_cache;
191}

193bool configure_global_cache(const query_cache_config& config) {
194 // This is a simple check - if init already happened, return false
195 // Not perfectly thread-safe for the pending config, but configure
196 // should be called at startup before any access
197 if (g_query_cache) {
198 return false;
199 }
200 g_pending_config = config;
201 g_config_pending = true;
202 return true;
203}
