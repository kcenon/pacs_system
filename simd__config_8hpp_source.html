112enum class simd_feature : uint32_t {
113 none = 0,
114 sse2 = 1 << 0,
115 ssse3 = 1 << 1,
116 sse41 = 1 << 2,
117 avx = 1 << 3,
118 avx2 = 1 << 4,
119 avx512f = 1 << 5,
120 neon = 1 << 6
121};
122
126[[nodiscard]] constexpr simd_feature operator|(simd_feature a,
127 simd_feature b) noexcept {
128 return static_cast<simd_feature>(static_cast<uint32_t>(a) |
129 static_cast<uint32_t>(b));
130}
131
135[[nodiscard]] constexpr simd_feature operator&(simd_feature a,
136 simd_feature b) noexcept {
137 return static_cast<simd_feature>(static_cast<uint32_t>(a) &
138 static_cast<uint32_t>(b));
139}
140
144[[nodiscard]] constexpr bool has_feature(simd_feature features,
145 simd_feature check) noexcept {
146 return (features & check) == check;
147}

189[[nodiscard]] inline simd_feature detect_features() noexcept {
190 simd_feature features = simd_feature::none;
191
192#if defined(PACS_ARCH_X64) || defined(PACS_ARCH_X86)
193 int info[4] = {0};
194
195 // Check basic CPUID
196 detail::cpuid(info, 0);
197 int max_function = info[0];
198
199 if (max_function >= 1) {
200 detail::cpuid(info, 1);
201
202 // EDX features
203 if (info[3] & (1 << 26)) {
204 features = features | simd_feature::sse2;
205 }
206
207 // ECX features
208 if (info[2] & (1 << 0)) {
209 features = features | simd_feature::ssse3;
210 }
211 if (info[2] & (1 << 19)) {
212 features = features | simd_feature::sse41;
213 }
214 if (info[2] & (1 << 28)) {
215 features = features | simd_feature::avx;
216 }
217 }
218
219 if (max_function >= 7) {
220 detail::cpuid_ex(info, 7, 0);
221
222 // EBX features
223 if (info[1] & (1 << 5)) {
224 features = features | simd_feature::avx2;
225 }
226 if (info[1] & (1 << 16)) {
227 features = features | simd_feature::avx512f;
228 }
229 }
230
231#elif defined(PACS_ARCH_ARM64)
232 // ARM64 always has NEON
233 features = features | simd_feature::neon;
234
235#elif defined(PACS_ARCH_ARM32)
236 // ARM32: Check for NEON via auxiliary vector or runtime detection
237 // For simplicity, assume NEON is available if compiled with NEON support
238#if defined(PACS_SIMD_NEON)
239 features = features | simd_feature::neon;
240#endif
241
242#endif // Architecture
243
244 return features;
245}

309[[nodiscard]] inline size_t optimal_vector_width() noexcept {
310 if (has_avx512f()) {
311 return 64;
312 }
313 if (has_avx2()) {
314 return 32;
315 }
316 if (has_sse2() || has_neon()) {
317 return 16;
318 }
319 return 0; // No SIMD
320}
