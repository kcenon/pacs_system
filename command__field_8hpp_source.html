136[[nodiscard]] constexpr bool is_request(command_field cmd) noexcept {
137 return (static_cast<uint16_t>(cmd) & 0x8000) == 0;
138}
139
145[[nodiscard]] constexpr bool is_response(command_field cmd) noexcept {
146 return (static_cast<uint16_t>(cmd) & 0x8000) != 0;
147}

154[[nodiscard]] constexpr bool is_dimse_c(command_field cmd) noexcept {
155 // Remove response bit (0x8000) and check range
156 const auto value = static_cast<uint16_t>(cmd) & 0x7FFF;
157 return value <= 0x0030 || value == 0x0FFF;
158}
159
165[[nodiscard]] constexpr bool is_dimse_n(command_field cmd) noexcept {
166 // Remove response bit (0x8000) and check range
167 const auto value = static_cast<uint16_t>(cmd) & 0x7FFF;
168 return value >= 0x0100 && value <= 0x0150;
169}

177[[nodiscard]] constexpr command_field get_response_command(
178 command_field request) noexcept {
179 return static_cast<command_field>(static_cast<uint16_t>(request) | 0x8000);
180}
181
187[[nodiscard]] constexpr command_field get_request_command(
188 command_field response) noexcept {
189 return static_cast<command_field>(static_cast<uint16_t>(response) & 0x7FFF);
190}
