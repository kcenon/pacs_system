65bool frame_deflate_codec::can_encode(const image_params& params) const noexcept {
66 if (params.width == 0 || params.height == 0) {
67 return false;
68 }
69
70 // Frame Deflate supports grayscale only (optimized for segmentation masks)
71 if (params.samples_per_pixel != 1) {
72 return false;
73 }
74
75 // Support 1-bit through 16-bit
76 if (params.bits_stored < 1 || params.bits_stored > 16) {
77 return false;
78 }
79
80 return true;
81}

93codec_result frame_deflate_codec::encode(
94 std::span<const uint8_t> pixel_data,
95 const image_params& params,
96 [[maybe_unused]] const compression_options& options) const {
97
98 if (pixel_data.empty()) {
99 return kcenon::pacs::pacs_error<compression_result>(
100 kcenon::pacs::error_codes::compression_error, "Empty pixel data");
101 }
102
103 if (params.width == 0 || params.height == 0) {
104 return kcenon::pacs::pacs_error<compression_result>(
105 kcenon::pacs::error_codes::compression_error, "Invalid image dimensions");
106 }
107
108 // Calculate expected data size
109 const size_t expected_size = params.frame_size_bytes();
110 if (pixel_data.size() < expected_size) {
111 return kcenon::pacs::pacs_error<compression_result>(
112 kcenon::pacs::error_codes::compression_error,
113 "Pixel data too small: expected " + std::to_string(expected_size)
114 + " bytes, got " + std::to_string(pixel_data.size()));
115 }
116
117 // Estimate compressed output size (zlib recommends deflateBound)
118 uLong source_len = static_cast<uLong>(expected_size);
119 uLong bound = compressBound(source_len);
120 std::vector<uint8_t> compressed(bound);
121
122 uLong dest_len = bound;
123 int ret = compress2(compressed.data(), &dest_len,
124 pixel_data.data(), source_len,
125 compression_level_);
126
127 if (ret != Z_OK) {
128 return kcenon::pacs::pacs_error<compression_result>(
129 kcenon::pacs::error_codes::compression_error,
130 "zlib compress2 failed with error code: " + std::to_string(ret));
131 }
132
133 compressed.resize(dest_len);
134
135 return kcenon::pacs::ok<compression_result>(
136 compression_result{std::move(compressed), params});
137}

139codec_result frame_deflate_codec::decode(
140 std::span<const uint8_t> compressed_data,
141 const image_params& params) const {
142
143 if (compressed_data.empty()) {
144 return kcenon::pacs::pacs_error<compression_result>(
145 kcenon::pacs::error_codes::decompression_error, "Empty compressed data");
146 }
147
148 // Calculate expected decompressed size
149 const size_t expected_size = params.frame_size_bytes();
150 if (expected_size == 0) {
151 return kcenon::pacs::pacs_error<compression_result>(
152 kcenon::pacs::error_codes::decompression_error,
153 "Cannot determine output size from image parameters");
154 }
155
156 std::vector<uint8_t> decompressed(expected_size);
157 uLong dest_len = static_cast<uLong>(expected_size);
158 uLong source_len = static_cast<uLong>(compressed_data.size());
159
160 int ret = uncompress(decompressed.data(), &dest_len,
161 compressed_data.data(), source_len);
162
163 if (ret != Z_OK) {
164 return kcenon::pacs::pacs_error<compression_result>(
165 kcenon::pacs::error_codes::decompression_error,
166 "zlib uncompress failed with error code: " + std::to_string(ret));
167 }
168
169 if (dest_len != expected_size) {
170 return kcenon::pacs::pacs_error<compression_result>(
171 kcenon::pacs::error_codes::decompression_error,
172 "Decompressed size mismatch: expected " + std::to_string(expected_size)
173 + " bytes, got " + std::to_string(dest_len));
174 }
175
176 return kcenon::pacs::ok<compression_result>(
177 compression_result{std::move(decompressed), params});
178}
