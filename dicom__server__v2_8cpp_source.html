82void dicom_server_v2::register_service(services::scp_service_ptr service) {
83 if (!service) {
84 return;
85 }
86
87 std::lock_guard<std::mutex> lock(services_mutex_);
88
89 // Register SOP Class mappings
90 for (const auto& sop_class : service->supported_sop_classes()) {
91 sop_class_to_service_[sop_class] = service.get();
92 }
93
94 services_.push_back(std::move(service));
95}
96
97std::vector<std::string> dicom_server_v2::supported_sop_classes() const {
98 std::lock_guard<std::mutex> lock(services_mutex_);
99
100 std::vector<std::string> sop_classes;
101 sop_classes.reserve(sop_class_to_service_.size());
102
103 for (const auto& [uid, _] : sop_class_to_service_) {
104 sop_classes.push_back(uid);
105 }
106
107 return sop_classes;
108}

114Result<std::monostate> dicom_server_v2::start() {
115 if (running_.exchange(true)) {
116 return error_info("Server already running");
117 }
118
119 // Validate configuration
120 if (config_.ae_title.empty()) {
121 running_ = false;
122 return error_info("AE Title cannot be empty");
123 }
124
125 if (config_.ae_title.length() > AE_TITLE_LENGTH) {
126 running_ = false;
127 return error_info("AE Title exceeds 16 characters");
128 }
129
130 if (config_.port == 0) {
131 running_ = false;
132 return error_info("Invalid port number");
133 }
134
135 // Check that at least one service is registered
136 {
137 std::lock_guard<std::mutex> lock(services_mutex_);
138 if (services_.empty()) {
139 running_ = false;
140 return error_info("No services registered");
141 }
142 }
143
144 // Reset statistics
145 {
146 std::lock_guard<std::mutex> lock(stats_mutex_);
147 stats_.start_time = clock::now();
148 stats_.last_activity = stats_.start_time;
149 stats_.total_associations = 0;
150 stats_.active_associations = 0;
151 stats_.rejected_associations = 0;
152 stats_.messages_processed = 0;
153 stats_.bytes_received = 0;
154 stats_.bytes_sent = 0;
155 }
156
157#ifdef PACS_WITH_NETWORK_SYSTEM
158 try {
159 // Create TCP server via tcp_facade
160 kcenon::network::facade::tcp_facade facade;
161 kcenon::network::facade::tcp_facade::server_config srv_cfg;
162 srv_cfg.server_id = config_.ae_title;
163 srv_cfg.port = config_.port;
164 server_ = facade.create_server(srv_cfg);
165
166 // Set up server-level callbacks
167 server_->set_connection_callback(
168 [this](std::shared_ptr<kcenon::network::interfaces::i_session> session) {
169 on_connection(std::move(session));
170 });
171
172 server_->set_disconnection_callback(
173 [this](std::string_view session_id) {
174 on_disconnection(session_id);
175 });
176
177 server_->set_receive_callback(
178 [this](std::string_view session_id,
179 const std::vector<uint8_t>& data) {
180 on_receive(session_id, data);
181 });
182
183 server_->set_error_callback(
184 [this](std::string_view session_id,
185 std::error_code ec) {
186 on_network_error(session_id, ec);
187 });
188
189 // Start the server on configured port
190 auto result = server_->start(config_.port);
191 if (result.is_err()) {
192 running_ = false;
193 server_.reset();
194 return error_info("Failed to start server");
195 }
196
197 return std::monostate{};
198 } catch (const std::exception& e) {
199 running_ = false;
200 if (server_) {
201 server_.reset();
202 }
203 return error_info(std::string("Exception during server start: ") + e.what());
204 } catch (...) {
205 running_ = false;
206 if (server_) {
207 server_.reset();
208 }
209 return error_info("Unknown exception during server start");
210 }
211#else
212 running_ = false;
213 return error_info("dicom_server_v2 requires PACS_WITH_NETWORK_SYSTEM");
214#endif
215}

217void dicom_server_v2::stop(duration timeout) {
218 if (!running_.exchange(false)) {
219 return; // Already stopped
220 }
221
222#ifdef PACS_WITH_NETWORK_SYSTEM
223 // Phase 1: Stop accepting new connections
224 if (server_) {
225 try {
226 // Stop the server - this closes the acceptor
227 (void)server_->stop();
228 } catch (...) {
229 // Suppress exceptions during server stop
230 }
231 }
232
233 // Phase 2: Wait for handlers to complete gracefully
234 auto deadline = clock::now() + timeout;
235 {
236 std::unique_lock<std::mutex> lock(handlers_mutex_);
237 while (!handlers_.empty() && clock::now() < deadline) {
238 // Release lock while waiting
239 lock.unlock();
240 std::this_thread::sleep_for(std::chrono::milliseconds{100});
241 lock.lock();
242 }
243 }
244
245 // Phase 3: Force stop remaining handlers
246 // Collect handlers first, then stop them without holding the lock
247 // to avoid potential deadlock if handler->stop() triggers callbacks
248 std::vector<std::shared_ptr<dicom_association_handler>> handlers_to_stop;
249 {
250 std::lock_guard<std::mutex> lock(handlers_mutex_);
251 handlers_to_stop.reserve(handlers_.size());
252 for (auto& [session_id, handler] : handlers_) {
253 handlers_to_stop.push_back(handler);
254 }
255 handlers_.clear();
256 }
257
258 // Stop handlers without holding the lock
259 for (auto& handler : handlers_to_stop) {
260 try {
261 handler->stop(false); // Force abort
262 } catch (...) {
263 // Suppress exceptions during handler stop
264 }
265 }
266 handlers_to_stop.clear();
267
268 // Allow any pending callbacks to complete
269 std::this_thread::sleep_for(std::chrono::milliseconds{50});
270
271 // Clear callbacks to break reference cycles before server is destroyed.
272 // The server object itself is kept alive until dicom_server_v2 is destroyed
273 // to avoid use-after-free in the adapter's background I/O threads.
274 if (server_) {
275 server_->set_connection_callback(nullptr);
276 server_->set_disconnection_callback(nullptr);
277 server_->set_receive_callback(nullptr);
278 server_->set_error_callback(nullptr);
279 }
280#else
281 (void)timeout; // Unused without network_system
282#endif
283
284 // Notify shutdown waiters
285 {
286 std::lock_guard<std::mutex> lock(shutdown_mutex_);
287 shutdown_cv_.notify_all();
288 }
289}
290
291void dicom_server_v2::wait_for_shutdown() {
292 std::unique_lock<std::mutex> lock(shutdown_mutex_);
293 shutdown_cv_.wait(lock, [this]() { return !running_; });
294}

343void dicom_server_v2::on_connection(
344 std::shared_ptr<kcenon::network::interfaces::i_session> session) {
345
346 if (!running_ || !session) {
347 return;
348 }
349
350 // Check max associations limit
351 {
352 std::lock_guard<std::mutex> lock(handlers_mutex_);
353 if (config_.max_associations > 0 &&
354 handlers_.size() >= config_.max_associations) {
355 // Reject due to resource limit
356 report_error("Max associations limit reached, rejecting connection");
357
358 std::lock_guard<std::mutex> stats_lock(stats_mutex_);
359 stats_.rejected_associations++;
360
361#ifdef PACS_WITH_NETWORK_SYSTEM
362 session->close();
363#endif
364 return;
365 }
366 }
367
368 // Create handler for this session
369 create_handler(std::move(session));
370}
371
372void dicom_server_v2::on_disconnection(std::string_view session_id) {
373 // Skip if server is shutting down
374 if (!running_) {
375 return;
376 }
377
378 std::string sid(session_id);
379
380 // Notify handler of disconnection before removing it
381 auto handler = find_handler(sid);
382 if (handler) {
383 handler->handle_disconnect();
384 }
385
386 // Remove handler for this session
387 remove_handler(sid);
388}
389
390void dicom_server_v2::on_receive(
391 std::string_view session_id,
392 const std::vector<uint8_t>& data) {
393
394 std::string sid(session_id);
395
396 // Find handler and forward data
397 auto handler = find_handler(sid);
398 if (handler) {
399 // Forward data to handler for PDU processing
400 handler->feed_data(data);
401
402 // Update statistics
403 {
404 std::lock_guard<std::mutex> lock(stats_mutex_);
405 stats_.bytes_received += data.size();
406 stats_.last_activity = clock::now();
407 }
408 }
409}
410
411void dicom_server_v2::on_network_error(
412 std::string_view session_id,
413 std::error_code ec) {
414
415 // Skip if server is shutting down
416 if (!running_) {
417 return;
418 }
419
420 std::string sid(session_id);
421
422 std::ostringstream oss;
423 oss << "Network error on session " << sid
424 << ": " << ec.message() << " (" << ec.value() << ")";
425 report_error(oss.str());
426
427 // Notify handler of the error
428 auto handler = find_handler(sid);
429 if (handler) {
430 handler->handle_error(ec);
431 }
432
433 // Remove the handler - it will clean itself up
434 remove_handler(sid);
435}

441void dicom_server_v2::create_handler(
442 std::shared_ptr<kcenon::network::interfaces::i_session> session) {
443
444#ifdef PACS_WITH_NETWORK_SYSTEM
445 const std::string session_id(session->id());
446#else
447 const std::string session_id;
448 (void)session;
449#endif
450
451 // Build service map for handler
452 auto service_map = build_service_map();
453
454 // Create handler
455 auto handler = std::make_shared<dicom_association_handler>(
456 std::move(session), config_, service_map);
457
458 // Set up access control if configured
459 {
460 std::lock_guard<std::mutex> acl_lock(acl_mutex_);
461 if (access_control_) {
462 handler->set_access_control(access_control_);
463 handler->set_access_control_enabled(access_control_enabled_);
464 }
465 }
466
467 // Set up handler callbacks
468 auto weak_this = std::weak_ptr<dicom_server_v2*>(
469 std::shared_ptr<dicom_server_v2*>(nullptr, [](dicom_server_v2**) {}));
470 // Note: We can't use shared_from_this() since dicom_server_v2 doesn't inherit
471 // from enable_shared_from_this. Instead, capture 'this' directly since the
472 // handler's lifetime is bounded by the server's lifetime.
473
474 handler->set_established_callback(
475 [this](const std::string& sid, const std::string& calling_ae,
476 const std::string& called_ae) {
477 // Update statistics
478 {
479 std::lock_guard<std::mutex> lock(stats_mutex_);
480 stats_.total_associations++;
481 stats_.last_activity = clock::now();
482 }
483
484 // Forward to user callback
485 {
486 std::lock_guard<std::mutex> lock(callback_mutex_);
487 if (on_established_cb_) {
488 on_established_cb_(sid, calling_ae, called_ae);
489 }
490 }
491 });
492
493 handler->set_closed_callback(
494 [this](const std::string& sid, bool graceful) {
495 // Forward to user callback
496 {
497 std::lock_guard<std::mutex> lock(callback_mutex_);
498 if (on_closed_cb_) {
499 on_closed_cb_(sid, graceful);
500 }
501 }
502
503 // Remove handler from map
504 remove_handler(sid);
505 });
506
507 handler->set_error_callback(
508 [this](const std::string& /*sid*/, const std::string& error) {
509 report_error(error);
510 });
511
512 // Register handler
513 {
514 std::lock_guard<std::mutex> lock(handlers_mutex_);
515 handlers_[session_id] = handler;
516 }
517
518 // Start handler (begins processing PDUs)
519 handler->start();
520}
521
522void dicom_server_v2::remove_handler(const std::string& session_id) {
523 // Skip if server is shutting down - handlers are cleaned up in stop()
524 if (!running_) {
525 return;
526 }
527
528 std::lock_guard<std::mutex> lock(handlers_mutex_);
529 auto it = handlers_.find(session_id);
530 if (it != handlers_.end()) {
531 // Handler cleans itself up via stop() in destructor
532 handlers_.erase(it);
533 }
534}

536std::shared_ptr<dicom_association_handler>
537dicom_server_v2::find_handler(const std::string& session_id) const {
538 std::lock_guard<std::mutex> lock(handlers_mutex_);
539 auto it = handlers_.find(session_id);
540 if (it != handlers_.end()) {
541 return it->second;
542 }
543 return nullptr;
544}

546void dicom_server_v2::check_idle_timeouts() {
547 if (config_.idle_timeout.count() == 0) {
548 return; // No timeout configured
549 }
550
551 auto now = clock::now();
552 std::vector<std::string> timed_out;
553
554 {
555 std::lock_guard<std::mutex> lock(handlers_mutex_);
556 for (const auto& [session_id, handler] : handlers_) {
557 auto idle_duration = std::chrono::duration_cast<std::chrono::seconds>(
558 now - handler->last_activity());
559
560 if (idle_duration >= config_.idle_timeout) {
561 timed_out.push_back(session_id);
562 }
563 }
564 }
565
566 // Stop timed-out handlers
567 for (const auto& session_id : timed_out) {
568 auto handler = find_handler(session_id);
569 if (handler) {
570 handler->stop(false); // Force abort
571 }
572 }
573}
