65[[nodiscard]] constexpr uint16_t byte_swap16(uint16_t value) noexcept {
66 return static_cast<uint16_t>((value >> 8) | (value << 8));
67}
68
76[[nodiscard]] constexpr uint32_t byte_swap32(uint32_t value) noexcept {
77 return ((value >> 24) & 0x000000FF) |
78 ((value >> 8) & 0x0000FF00) |
79 ((value << 8) & 0x00FF0000) |
80 ((value << 24) & 0xFF000000);
81}
82
90[[nodiscard]] constexpr uint64_t byte_swap64(uint64_t value) noexcept {
91 return ((value >> 56) & 0x00000000000000FF) |
92 ((value >> 40) & 0x000000000000FF00) |
93 ((value >> 24) & 0x0000000000FF0000) |
94 ((value >> 8) & 0x00000000FF000000) |
95 ((value << 8) & 0x000000FF00000000) |
96 ((value << 24) & 0x0000FF0000000000) |
97 ((value << 40) & 0x00FF000000000000) |
98 ((value << 56) & 0xFF00000000000000);
99}

111[[nodiscard]] constexpr uint16_t read_be16(const uint8_t* data) noexcept {
112 return (static_cast<uint16_t>(data[0]) << 8) |
113 static_cast<uint16_t>(data[1]);
114}

121[[nodiscard]] constexpr uint32_t read_be32(const uint8_t* data) noexcept {
122 return (static_cast<uint32_t>(data[0]) << 24) |
123 (static_cast<uint32_t>(data[1]) << 16) |
124 (static_cast<uint32_t>(data[2]) << 8) |
125 static_cast<uint32_t>(data[3]);
126}

133[[nodiscard]] constexpr uint64_t read_be64(const uint8_t* data) noexcept {
134 return (static_cast<uint64_t>(data[0]) << 56) |
135 (static_cast<uint64_t>(data[1]) << 48) |
136 (static_cast<uint64_t>(data[2]) << 40) |
137 (static_cast<uint64_t>(data[3]) << 32) |
138 (static_cast<uint64_t>(data[4]) << 24) |
139 (static_cast<uint64_t>(data[5]) << 16) |
140 (static_cast<uint64_t>(data[6]) << 8) |
141 static_cast<uint64_t>(data[7]);
142}

149inline void write_be16(std::vector<uint8_t>& buffer, uint16_t value) {
150 buffer.push_back(static_cast<uint8_t>((value >> 8) & 0xFF));
151 buffer.push_back(static_cast<uint8_t>(value & 0xFF));
