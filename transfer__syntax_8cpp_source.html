184const ts_entry* find_entry(std::string_view uid) {
185 auto it = std::find_if(TS_REGISTRY.begin(), TS_REGISTRY.end(),
186 [&uid](const ts_entry& entry) {
187 return entry.uid == uid;
188 });
189 return (it != TS_REGISTRY.end()) ? &(*it) : nullptr;
190}

321// Public constructor - looks up UID in registry
322transfer_syntax::transfer_syntax(std::string_view uid)
323 : uid_(uid),
324 name_("Unknown"),
325 endianness_(byte_order::little_endian),
326 vr_type_(vr_encoding::implicit),
327 encapsulated_(false),
328 deflated_(false),
329 valid_(false),
330 supported_(false) {
331 if (const auto* entry = find_entry(uid)) {
332 name_ = entry->name;
333 endianness_ = entry->endian;
334 vr_type_ = entry->vr;
335 encapsulated_ = entry->encapsulated;
336 deflated_ = entry->deflated;
337 valid_ = true;
338 supported_ = entry->supported;
339 }
340}

396std::optional<transfer_syntax> find_transfer_syntax(std::string_view uid) {
397 if (const auto* entry = find_entry(uid)) {
398 return transfer_syntax{entry->uid, entry->name, entry->endian,
399 entry->vr, entry->encapsulated, entry->deflated,
400 entry->supported};
401 }
402 return std::nullopt;

405std::vector<transfer_syntax> supported_transfer_syntaxes() {
406 std::vector<transfer_syntax> result;
407 result.reserve(TS_REGISTRY.size());
408
409 for (const auto& entry : TS_REGISTRY) {
410 if (entry.supported) {
411 result.push_back(transfer_syntax{entry.uid, entry.name, entry.endian,
412 entry.vr, entry.encapsulated,
413 entry.deflated, entry.supported});
414 }
415 }
416 return result;
417}

419std::vector<transfer_syntax> all_transfer_syntaxes() {
