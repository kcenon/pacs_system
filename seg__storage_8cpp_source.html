233std::vector<std::string> get_seg_storage_sop_classes(bool include_surface) {
234 std::vector<std::string> result;
235 result.reserve(include_surface ? 4 : 3);
236
237 for (const auto& info : seg_sop_classes) {
238 if (!info.is_surface || include_surface) {
239 result.emplace_back(info.uid);
240 }
241 }
242
243 return result;
244}
245
246const seg_sop_class_info*
247get_seg_sop_class_info(std::string_view uid) noexcept {
248 auto it = std::find_if(
249 seg_sop_classes.begin(),
250 seg_sop_classes.end(),
251 [uid](const auto& info) { return info.uid == uid; }
252 );
253
254 if (it != seg_sop_classes.end()) {
255 return &(*it);
256 }
257 return nullptr;
258}
259
260bool is_seg_storage_sop_class(std::string_view uid) noexcept {
261 return get_seg_sop_class_info(uid) != nullptr;
262}
263
264bool is_surface_segmentation_sop_class(std::string_view uid) noexcept {
265 return uid == surface_segmentation_storage_uid;
