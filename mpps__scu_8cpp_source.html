66network::dimse::dimse_message make_n_create_rq(

70
71 using namespace network::dimse;
72
73 dimse_message msg{command_field::n_create_rq, message_id};
74 msg.set_affected_sop_class_uid(mpps_sop_class_uid);
75 msg.set_affected_sop_instance_uid(sop_instance_uid);
76 msg.set_dataset(std::move(dataset));

84network::dimse::dimse_message make_n_set_rq(
85 uint16_t message_id,
86 std::string_view sop_instance_uid,
87 core::dicom_dataset modifications) {
88
89 using namespace network::dimse;
90 using namespace encoding;
91
92 dimse_message msg{command_field::n_set_rq, message_id};
93 msg.set_affected_sop_class_uid(mpps_sop_class_uid);
94
95 // For N-SET, use Requested SOP Instance UID in command set
96 msg.command_set().set_string(
97 tag_requested_sop_instance_uid,
98 vr_type::UI,
99 std::string(sop_instance_uid));
100 msg.set_dataset(std::move(modifications));
101
102 return msg;
103}

123network::Result<mpps_result> mpps_scu::create(
124 network::association& assoc,
125 const mpps_create_data& data) {
126
127 using namespace network::dimse;
128
129 auto start_time = std::chrono::steady_clock::now();
130
131 // Verify association is established
132 if (!assoc.is_established()) {
133 return kcenon::pacs::pacs_error<mpps_result>(
134 kcenon::pacs::error_codes::association_not_established,
135 "Association not established");
136 }
137
138 // Get accepted presentation context for MPPS
139 auto context_id = assoc.accepted_context_id(mpps_sop_class_uid);
140 if (!context_id) {
141 return kcenon::pacs::pacs_error<mpps_result>(
142 kcenon::pacs::error_codes::mpps_context_not_accepted,
143 "No accepted presentation context for MPPS SOP Class");
144 }
145
146 // Generate or use provided MPPS UID
147 std::string mpps_uid = data.mpps_sop_instance_uid;
148 if (mpps_uid.empty() && config_.auto_generate_uid) {
149 mpps_uid = generate_mpps_uid();
150 }
151
152 if (mpps_uid.empty()) {
153 return kcenon::pacs::pacs_error<mpps_result>(
154 kcenon::pacs::error_codes::mpps_missing_uid,
155 "MPPS SOP Instance UID is required");
156 }
157
158 // Build the MPPS creation dataset
159 auto dataset = build_create_dataset(data);
160
161 // Create the N-CREATE request
162 auto request = make_n_create_rq(next_message_id(), mpps_uid, std::move(dataset));
163
164 logger_->debug("Sending N-CREATE request for MPPS: " + mpps_uid);
165
166 // Send the request
167 auto send_result = assoc.send_dimse(*context_id, request);
168 if (send_result.is_err()) {
169 logger_->error("Failed to send N-CREATE: " + send_result.error().message);
170 return send_result.error();
171 }
172
173 // Receive the response
174 auto recv_result = assoc.receive_dimse(config_.timeout);
175 if (recv_result.is_err()) {
176 logger_->error("Failed to receive N-CREATE response: " + recv_result.error().message);
177 return recv_result.error();
178 }
179
180 const auto& [recv_context_id, response] = recv_result.value();
181
182 // Verify it's an N-CREATE response
183 if (response.command() != command_field::n_create_rsp) {
184 return kcenon::pacs::pacs_error<mpps_result>(
185 kcenon::pacs::error_codes::mpps_unexpected_command,
186 "Expected N-CREATE-RSP but received " +
187 std::string(to_string(response.command())));
188 }
189
190 auto end_time = std::chrono::steady_clock::now();
191 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
192 end_time - start_time);
193
194 // Build result
195 mpps_result result;
196 result.mpps_sop_instance_uid = mpps_uid;
197 result.status = static_cast<uint16_t>(response.status());
198 result.elapsed = elapsed;
199
200 // Extract error comment if present
201 if (response.command_set().contains(tag_error_comment)) {
202 result.error_comment = response.command_set().get_string(tag_error_comment);
203 }
204
205 // Update statistics
206 creates_performed_.fetch_add(1, std::memory_order_relaxed);
207
208 if (result.is_success()) {
209 logger_->info("N-CREATE successful for MPPS: " + mpps_uid);
210 } else {
211 logger_->warn("N-CREATE returned status 0x" +
212 std::to_string(result.status) + " for MPPS: " + mpps_uid);
213 }
214
215 return result;
216}

222network::Result<mpps_result> mpps_scu::set(
223 network::association& assoc,
224 const mpps_set_data& data) {
225
226 using namespace network::dimse;
227
228 auto start_time = std::chrono::steady_clock::now();
229
230 // Verify association is established
231 if (!assoc.is_established()) {
232 return kcenon::pacs::pacs_error<mpps_result>(
233 kcenon::pacs::error_codes::association_not_established,
234 "Association not established");
235 }
236
237 // Validate MPPS UID is provided
238 if (data.mpps_sop_instance_uid.empty()) {
239 return kcenon::pacs::pacs_error<mpps_result>(
240 kcenon::pacs::error_codes::mpps_missing_uid,
241 "MPPS SOP Instance UID is required for N-SET");
242 }
243
244 // Validate status is not IN PROGRESS (can only set to COMPLETED or DISCONTINUED)
245 if (data.status == mpps_status::in_progress) {
246 return kcenon::pacs::pacs_error<mpps_result>(
247 kcenon::pacs::error_codes::mpps_invalid_status_transition,
248 "Cannot set MPPS status back to IN PROGRESS");
249 }
250
251 // Get accepted presentation context for MPPS
252 auto context_id = assoc.accepted_context_id(mpps_sop_class_uid);
253 if (!context_id) {
254 return kcenon::pacs::pacs_error<mpps_result>(
255 kcenon::pacs::error_codes::mpps_context_not_accepted,
256 "No accepted presentation context for MPPS SOP Class");
257 }
258
259 // Build the modification dataset
260 auto dataset = build_set_dataset(data);
261
262 // Create the N-SET request
263 auto request = make_n_set_rq(
264 next_message_id(),
265 data.mpps_sop_instance_uid,
266 std::move(dataset));
267
268 logger_->debug("Sending N-SET request for MPPS: " + data.mpps_sop_instance_uid);
269
270 // Send the request
271 auto send_result = assoc.send_dimse(*context_id, request);
272 if (send_result.is_err()) {
273 logger_->error("Failed to send N-SET: " + send_result.error().message);
274 return send_result.error();
275 }
276
277 // Receive the response
278 auto recv_result = assoc.receive_dimse(config_.timeout);
279 if (recv_result.is_err()) {
280 logger_->error("Failed to receive N-SET response: " + recv_result.error().message);
281 return recv_result.error();
282 }
283
284 const auto& [recv_context_id, response] = recv_result.value();
285
286 // Verify it's an N-SET response
287 if (response.command() != command_field::n_set_rsp) {
288 return kcenon::pacs::pacs_error<mpps_result>(
289 kcenon::pacs::error_codes::mpps_unexpected_command,
290 "Expected N-SET-RSP but received " +
291 std::string(to_string(response.command())));
292 }
293
294 auto end_time = std::chrono::steady_clock::now();
295 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
296 end_time - start_time);
297
298 // Build result
299 mpps_result result;
300 result.mpps_sop_instance_uid = data.mpps_sop_instance_uid;
301 result.status = static_cast<uint16_t>(response.status());
302 result.elapsed = elapsed;
303
304 // Extract error comment if present
305 if (response.command_set().contains(tag_error_comment)) {
306 result.error_comment = response.command_set().get_string(tag_error_comment);
307 }
308
309 // Update statistics
310 sets_performed_.fetch_add(1, std::memory_order_relaxed);
311
312 if (result.is_success()) {
313 logger_->info("N-SET successful for MPPS: " + data.mpps_sop_instance_uid +
314 " (status: " + std::string(to_string(data.status)) + ")");
315 } else {
316 logger_->warn("N-SET returned status 0x" +
317 std::to_string(result.status) + " for MPPS: " +
318 data.mpps_sop_instance_uid);
319 }
320
321 return result;
322}

375core::dicom_dataset mpps_scu::build_create_dataset(const mpps_create_data& data) const {
376 using namespace core;
377 using namespace encoding;
378
379 dicom_dataset ds;
380
381 // Performed Procedure Step Status - always IN PROGRESS for N-CREATE
382 ds.set_string(mpps_tags::performed_procedure_step_status, vr_type::CS, "IN PROGRESS");
383
384 // Performed Procedure Step Timing
385 std::string start_date = data.procedure_step_start_date.empty()
386 ? get_current_date() : data.procedure_step_start_date;
387 std::string start_time = data.procedure_step_start_time.empty()
388 ? get_current_time() : data.procedure_step_start_time;
389
390 ds.set_string(mpps_tags::performed_procedure_step_start_date, vr_type::DA, start_date);
391 ds.set_string(mpps_tags::performed_procedure_step_start_time, vr_type::TM, start_time);
392
393 // Modality and Station
394 if (!data.modality.empty()) {
395 ds.set_string(tags::modality, vr_type::CS, data.modality);
396 }
397 if (!data.station_ae_title.empty()) {
398 ds.set_string(mpps_tags::performed_station_ae_title, vr_type::AE, data.station_ae_title);
399 }
400 if (!data.station_name.empty()) {
401 ds.set_string(mpps_tags::performed_station_name, vr_type::SH, data.station_name);
402 }
403
404 // Procedure Step ID
405 if (!data.scheduled_procedure_step_id.empty()) {
406 ds.set_string(mpps_tags::performed_procedure_step_id, vr_type::SH,
407 data.scheduled_procedure_step_id);
408 }
409
410 // Procedure Description
411 if (!data.procedure_description.empty()) {
412 ds.set_string(mpps_tags::performed_procedure_step_description, vr_type::LO,
413 data.procedure_description);
414 }
415
416 // Patient Information (required)
417 std::string patient_name = data.patient_name.empty() ? "ANONYMOUS" : data.patient_name;
418 ds.set_string(tags::patient_name, vr_type::PN, patient_name);
419
420 if (!data.patient_id.empty()) {
421 ds.set_string(tags::patient_id, vr_type::LO, data.patient_id);
422 }
423 if (!data.patient_birth_date.empty()) {
424 ds.set_string(tags::patient_birth_date, vr_type::DA, data.patient_birth_date);
425 }
426 if (!data.patient_sex.empty()) {
427 ds.set_string(tags::patient_sex, vr_type::CS, data.patient_sex);
428 }
429
430 // Study reference
431 if (!data.study_instance_uid.empty()) {
432 ds.set_string(tags::study_instance_uid, vr_type::UI, data.study_instance_uid);
433 }
434 if (!data.accession_number.empty()) {
435 ds.set_string(tags::accession_number, vr_type::SH, data.accession_number);
436 }
437
438 // Performing Information
439 if (!data.performing_physician.empty()) {
440 ds.set_string(mpps_tags::performing_physicians_name, vr_type::PN,
441 data.performing_physician);
442 }
443 if (!data.operator_name.empty()) {
444 ds.set_string(mpps_tags::operators_name, vr_type::PN, data.operator_name);
445 }
446
447 return ds;
448}

450core::dicom_dataset mpps_scu::build_set_dataset(const mpps_set_data& data) const {
451 using namespace core;
452 using namespace encoding;
453
454 dicom_dataset ds;
455
456 // Update status
457 ds.set_string(mpps_tags::performed_procedure_step_status, vr_type::CS,
458 std::string(to_string(data.status)));
459
460 // End date/time
461 std::string end_date = data.procedure_step_end_date.empty()
462 ? get_current_date() : data.procedure_step_end_date;
463 std::string end_time = data.procedure_step_end_time.empty()
464 ? get_current_time() : data.procedure_step_end_time;
465
466 ds.set_string(mpps_tags::performed_procedure_step_end_date, vr_type::DA, end_date);
467 ds.set_string(mpps_tags::performed_procedure_step_end_time, vr_type::TM, end_time);
468
469 // Add performed series sequence for COMPLETED status
470 if (data.status == mpps_status::completed && !data.performed_series.empty()) {
471 auto& seq = ds.get_or_create_sequence(mpps_tags::performed_series_sequence);
472 for (const auto& s : data.performed_series) {
473 dicom_dataset series_item;
474
475 if (!s.series_uid.empty()) {
476 series_item.set_string(tags::series_instance_uid, vr_type::UI, s.series_uid);
477 }
478 if (!s.series_description.empty()) {
479 series_item.set_string(mpps_tags::series_description, vr_type::LO,
480 s.series_description);
481 }
482 if (!s.modality.empty()) {
483 series_item.set_string(tags::modality, vr_type::CS, s.modality);
484 }
485 if (!s.performing_physician.empty()) {
486 series_item.set_string(mpps_tags::performing_physicians_name, vr_type::PN,
487 s.performing_physician);
488 }
489 if (!s.operator_name.empty()) {
490 series_item.set_string(mpps_tags::operators_name, vr_type::PN, s.operator_name);
491 }
492
493 seq.push_back(std::move(series_item));
494 }
495 }
496
497 return ds;
498}

504std::string mpps_scu::generate_mpps_uid() const {
505 static std::mt19937_64 gen{std::random_device{}()};
506 static std::uniform_int_distribution<uint64_t> dist;
507
508 auto now = std::chrono::system_clock::now();
509 auto timestamp = std::chrono::duration_cast<std::chrono::milliseconds>(
510 now.time_since_epoch()).count();
511
512 return std::string(uid_root) + "." + std::to_string(timestamp) +
513 "." + std::to_string(dist(gen) % 100000);
514}

542uint16_t mpps_scu::next_message_id() noexcept {
543 uint16_t id = message_id_counter_.fetch_add(1, std::memory_order_relaxed);
544 // Wrap around at 0xFFFF, skip 0 (reserved)
545 if (id == 0) {

547 }
