95enum class association_state {
96 idle,
97 awaiting_associate_ac,
98 awaiting_associate_rq,
99 established,
100 awaiting_release_rp,
101 awaiting_release_rq,
102 released,
103 aborted
104};

111[[nodiscard]] constexpr const char* to_string(association_state state) noexcept {
112 switch (state) {
113 case association_state::idle: return "Idle (Sta1)";
114 case association_state::awaiting_associate_ac: return "Awaiting A-ASSOCIATE-AC (Sta5)";
115 case association_state::awaiting_associate_rq: return "Awaiting A-ASSOCIATE-RQ (Sta2)";
116 case association_state::established: return "Established (Sta6)";
117 case association_state::awaiting_release_rp: return "Awaiting A-RELEASE-RP (Sta7)";
118 case association_state::awaiting_release_rq: return "Awaiting A-RELEASE-RQ (Sta8)";
119 case association_state::released: return "Released";
120 case association_state::aborted: return "Aborted";
121 default: return "Unknown";
122 }
123}

183struct proposed_presentation_context {
184 uint8_t id;
185 std::string abstract_syntax;
186 std::vector<std::string> transfer_syntaxes;
187
188 proposed_presentation_context() : id(0) {}
189 proposed_presentation_context(uint8_t ctx_id, std::string abs_syntax,
190 std::vector<std::string> ts_list)
191 : id(ctx_id)
192 , abstract_syntax(std::move(abs_syntax))
193 , transfer_syntaxes(std::move(ts_list)) {}
194};
195
199struct accepted_presentation_context {
200 uint8_t id;
201 std::string abstract_syntax;
202 std::string transfer_syntax;
203 presentation_context_result result;
204
205 accepted_presentation_context()
206 : id(0), result(presentation_context_result::acceptance) {}
207 accepted_presentation_context(uint8_t ctx_id, std::string abs_syntax,
208 std::string ts, presentation_context_result res)
209 : id(ctx_id)
210 , abstract_syntax(std::move(abs_syntax))
211 , transfer_syntax(std::move(ts))
212 , result(res) {}
213
214 [[nodiscard]] bool is_accepted() const noexcept {
215 return result == presentation_context_result::acceptance;
216 }
217};
