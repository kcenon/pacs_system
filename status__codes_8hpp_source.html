183[[nodiscard]] constexpr bool is_success(status_code status) noexcept {
184 return status == status_success;
185}
186
192[[nodiscard]] constexpr bool is_pending(status_code status) noexcept {
193 return status == status_pending || status == status_pending_warning;
194}
195
201[[nodiscard]] constexpr bool is_cancel(status_code status) noexcept {
202 return status == status_cancel;
203}
204
210[[nodiscard]] constexpr bool is_warning(status_code status) noexcept {
211 return (status & 0xF000) == 0xB000;
212}
213
222[[nodiscard]] constexpr bool is_failure(status_code status) noexcept {
223 const auto high_nibble = (status & 0xF000) >> 12;
224 return high_nibble == 0xA || high_nibble == 0xC ||
225 (status >= 0x0100 && status <= 0x02FF);
226}

233[[nodiscard]] constexpr bool is_final(status_code status) noexcept {
234 return !is_pending(status);
235}
