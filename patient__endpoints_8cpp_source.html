69std::string patient_to_json(const storage::patient_record &patient) {
70 std::ostringstream oss;
71 oss << R"({"pk":)" << patient.pk << R"(,"patient_id":")"
72 << json_escape(patient.patient_id) << R"(","patient_name":")"
73 << json_escape(patient.patient_name) << R"(","birth_date":")"
74 << json_escape(patient.birth_date) << R"(","sex":")"
75 << json_escape(patient.sex) << R"(","other_ids":")"
76 << json_escape(patient.other_ids) << R"(","ethnic_group":")"
77 << json_escape(patient.ethnic_group) << R"(","comments":")"
78 << json_escape(patient.comments) << R"("})";
79 return oss.str();
80}

85std::string patients_to_json(const std::vector<storage::patient_record> &patients,
86 size_t total_count) {
87 std::ostringstream oss;
88 oss << R"({"data":[)";
89 for (size_t i = 0; i < patients.size(); ++i) {
90 if (i > 0) {
91 oss << ",";
92 }
93 oss << patient_to_json(patients[i]);
94 }
95 oss << R"(],"pagination":{"total":)" << total_count << R"(,"count":)"
96 << patients.size() << "}}";
97 return oss.str();
98}

138std::pair<size_t, size_t> parse_pagination(const crow::request &req) {
139 size_t limit = 20; // Default limit
140 size_t offset = 0;
141
142 auto limit_param = req.url_params.get("limit");
143 if (limit_param) {
144 try {
145 limit = std::stoul(limit_param);
146 if (limit > 100) {
147 limit = 100; // Cap at 100
148 }
149 } catch (...) {
150 // Use default
151 }
152 }
153
154 auto offset_param = req.url_params.get("offset");
155 if (offset_param) {
156 try {
157 offset = std::stoul(offset_param);
158 } catch (...) {
159 // Use default
160 }
161 }
162
163 return {limit, offset};
164}

169void register_patient_endpoints_impl(crow::SimpleApp &app,
170 std::shared_ptr<rest_server_context> ctx) {
171 // GET /api/v1/patients - List patients (paginated)
172 CROW_ROUTE(app, "/api/v1/patients")
173 .methods(crow::HTTPMethod::GET)([ctx](const crow::request &req) {
174 crow::response res;
175 res.add_header("Content-Type", "application/json");
176 add_cors_headers(res, *ctx);
177
178 if (!ctx->database) {
179 res.code = 503;
180 res.body =
181 make_error_json("DATABASE_UNAVAILABLE", "Database not configured");
182 return res;
183 }
184
185 // Parse pagination
186 auto [limit, offset] = parse_pagination(req);
187
188 // Build query from URL parameters
189 storage::patient_query query;
190 query.limit = limit;
191 query.offset = offset;
192
193 auto patient_id = req.url_params.get("patient_id");
194 if (patient_id) {
195 query.patient_id = patient_id;
196 }
197
198 auto patient_name = req.url_params.get("patient_name");
199 if (patient_name) {
200 query.patient_name = patient_name;
201 }
202
203 auto birth_date = req.url_params.get("birth_date");
204 if (birth_date) {
205 query.birth_date = birth_date;
206 }
207
208 auto birth_date_from = req.url_params.get("birth_date_from");
209 if (birth_date_from) {
210 query.birth_date_from = birth_date_from;
211 }
212
213 auto birth_date_to = req.url_params.get("birth_date_to");
214 if (birth_date_to) {
215 query.birth_date_to = birth_date_to;
216 }
217
218 auto sex = req.url_params.get("sex");
219 if (sex) {
220 query.sex = sex;
221 }
222
223 // Get total count (without pagination for accurate count)
224 storage::patient_query count_query = query;
225 count_query.limit = 0;
226 count_query.offset = 0;
227 auto all_patients_result = ctx->database->search_patients(count_query);
228 if (!all_patients_result.is_ok()) {
229 res.code = 500;
230 res.body = make_error_json("QUERY_ERROR",
231 all_patients_result.error().message);
232 return res;
233 }
234 size_t total_count = all_patients_result.value().size();
235
236 // Get paginated results
237 auto patients_result = ctx->database->search_patients(query);
238 if (!patients_result.is_ok()) {
239 res.code = 500;
240 res.body = make_error_json("QUERY_ERROR",
241 patients_result.error().message);
242 return res;
243 }
244
245 res.code = 200;
246 res.body = patients_to_json(patients_result.value(), total_count);
247 return res;
248 });

252 .methods(crow::HTTPMethod::GET)(

254 crow::response res;

271

301 if (!studies_result.is_ok()) {
302 res.code = 500;

304 studies_result.error().message);
305 return res;
306 }
307
308 res.code = 200;
309 res.body = studies_to_json(studies_result.value());
310 return res;
311 });
