71[[nodiscard]] constexpr const char* to_string(prefetch_trigger trigger) noexcept {
72 switch (trigger) {
73 case prefetch_trigger::worklist_match: return "worklist_match";
74 case prefetch_trigger::prior_studies: return "prior_studies";
75 case prefetch_trigger::scheduled_exam: return "scheduled_exam";
76 case prefetch_trigger::manual: return "manual";
77 default: return "unknown";
78 }
79}

86[[nodiscard]] inline prefetch_trigger prefetch_trigger_from_string(
87 std::string_view str) noexcept {
88 if (str == "worklist_match") return prefetch_trigger::worklist_match;
89 if (str == "prior_studies") return prefetch_trigger::prior_studies;
90 if (str == "scheduled_exam") return prefetch_trigger::scheduled_exam;
91 if (str == "manual") return prefetch_trigger::manual;
92 return prefetch_trigger::manual;
93}
