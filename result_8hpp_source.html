258template <typename T>
259inline Result<T> pacs_error(int code, const std::string& message,
260 const std::string& details = "") {
261 if (details.empty()) {
262 return kcenon::common::make_error<T>(code, message, "pacs");
263 }
264 return kcenon::common::make_error<T>(code, message, "pacs", details);
265}
266
274inline VoidResult pacs_void_error(int code, const std::string& message,
275 const std::string& details = "") {
276 if (details.empty()) {
277 return VoidResult(error_info{code, message, "pacs"});
278 }
279 return VoidResult(error_info{code, message, "pacs", details});
280}
