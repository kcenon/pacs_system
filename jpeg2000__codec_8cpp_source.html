50void opj_error_callback(const char* msg, void* client_data) {
51 auto* error_msg = static_cast<std::string*>(client_data);
52 if (error_msg && msg) {
53 if (!error_msg->empty()) {
54 error_msg->append("; ");
55 }
56 error_msg->append(msg);
57 // Remove trailing newline if present
58 while (!error_msg->empty() && error_msg->back() == '\n') {
59 error_msg->pop_back();
60 }
61 }
62}

86struct opj_memory_stream {
87 const uint8_t* data;
88 size_t size;
89 size_t offset;
90};
91
95OPJ_SIZE_T opj_memory_stream_read(void* buffer, OPJ_SIZE_T nb_bytes, void* user_data) {
96 auto* stream = static_cast<opj_memory_stream*>(user_data);
97 if (!stream || stream->offset >= stream->size) {
98 return static_cast<OPJ_SIZE_T>(-1);
99 }
100
101 OPJ_SIZE_T bytes_to_read = (std::min)(nb_bytes,
102 static_cast<OPJ_SIZE_T>(stream->size - stream->offset));
103 std::memcpy(buffer, stream->data + stream->offset, bytes_to_read);
104 stream->offset += bytes_to_read;
105 return bytes_to_read;
106}
107
111OPJ_OFF_T opj_memory_stream_skip(OPJ_OFF_T nb_bytes, void* user_data) {
112 auto* stream = static_cast<opj_memory_stream*>(user_data);
113 if (!stream) {
114 return -1;
115 }
116
117 if (nb_bytes < 0) {
118 // Handle negative skip (backward)
119 if (stream->offset < static_cast<size_t>(-nb_bytes)) {
120 nb_bytes = -static_cast<OPJ_OFF_T>(stream->offset);
121 }
122 } else {
123 // Handle positive skip (forward)
124 if (stream->offset + nb_bytes > stream->size) {
125 nb_bytes = static_cast<OPJ_OFF_T>(stream->size - stream->offset);
126 }
127 }
128
129 stream->offset = static_cast<size_t>(static_cast<OPJ_OFF_T>(stream->offset) + nb_bytes);
130 return nb_bytes;
131}
132
136OPJ_BOOL opj_memory_stream_seek(OPJ_OFF_T nb_bytes, void* user_data) {
137 auto* stream = static_cast<opj_memory_stream*>(user_data);
138 if (!stream || nb_bytes < 0 || static_cast<size_t>(nb_bytes) > stream->size) {
139 return OPJ_FALSE;
140 }
141 stream->offset = static_cast<size_t>(nb_bytes);
142 return OPJ_TRUE;
143}

148struct opj_output_buffer {
149 std::vector<uint8_t> data;
150 size_t offset;
151};
152
156OPJ_SIZE_T opj_output_buffer_write(void* buffer, OPJ_SIZE_T nb_bytes, void* user_data) {
157 auto* output = static_cast<opj_output_buffer*>(user_data);
158 if (!output || !buffer) {
159 return static_cast<OPJ_SIZE_T>(-1);
160 }
161
162 size_t new_size = output->offset + nb_bytes;
163 if (new_size > output->data.size()) {
164 output->data.resize(new_size);
165 }
166
167 std::memcpy(output->data.data() + output->offset, buffer, nb_bytes);
168 output->offset += nb_bytes;
169 return nb_bytes;
170}
171
175OPJ_OFF_T opj_output_buffer_skip(OPJ_OFF_T nb_bytes, void* user_data) {
176 auto* output = static_cast<opj_output_buffer*>(user_data);
177 if (!output || nb_bytes < 0) {
178 return -1;
179 }
180
181 size_t new_offset = output->offset + static_cast<size_t>(nb_bytes);
182 if (new_offset > output->data.size()) {
183 output->data.resize(new_offset);
184 }
185 output->offset = new_offset;
186 return nb_bytes;
187}
188
192OPJ_BOOL opj_output_buffer_seek(OPJ_OFF_T nb_bytes, void* user_data) {
193 auto* output = static_cast<opj_output_buffer*>(user_data);
194 if (!output || nb_bytes < 0) {
195 return OPJ_FALSE;
196 }
197
198 size_t new_offset = static_cast<size_t>(nb_bytes);
199 if (new_offset > output->data.size()) {
200 output->data.resize(new_offset);
201 }
202 output->offset = new_offset;
203 return OPJ_TRUE;
204}

210OPJ_CODEC_FORMAT detect_j2k_format(std::span<const uint8_t> data) {
211 if (data.size() < 12) {
212 return OPJ_CODEC_J2K; // Default to raw codestream
213 }
214
215 // JP2 signature: 00 00 00 0C 6A 50 20 20 0D 0A 87 0A
216 static constexpr uint8_t jp2_signature[] = {
217 0x00, 0x00, 0x00, 0x0C, 0x6A, 0x50, 0x20, 0x20, 0x0D, 0x0A, 0x87, 0x0A
218 };
219
220 if (std::memcmp(data.data(), jp2_signature, sizeof(jp2_signature)) == 0) {
221 return OPJ_CODEC_JP2;
222 }
223
224 // J2K codestream signature: FF 4F FF 51 (SOC + SIZ markers)
225 if (data[0] == 0xFF && data[1] == 0x4F) {
226 return OPJ_CODEC_J2K;
227 }
228
229 return OPJ_CODEC_J2K; // Default
230}

258 [[nodiscard]] codec_result encode(std::span<const uint8_t> pixel_data,
259 const image_params& params,
260 const compression_options& options) const {
261#ifndef PACS_WITH_JPEG2000_CODEC
262 (void)pixel_data;
263 (void)params;
264 (void)options;
265 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error,
266 "JPEG 2000 codec not available: OpenJPEG library not found at build time");
267#else
268 // Determine if we should use lossless mode
269 bool use_lossless = lossless_ || options.lossless;
270
271 // Create image component parameters
272 std::vector<opj_image_cmptparm_t> cmptparm(params.samples_per_pixel);
273 for (uint16_t i = 0; i < params.samples_per_pixel; ++i) {
274 std::memset(&cmptparm[i], 0, sizeof(opj_image_cmptparm_t));
275 cmptparm[i].prec = params.bits_stored;
276 cmptparm[i].bpp = params.bits_stored;
277 cmptparm[i].sgnd = params.is_signed() ? 1 : 0;
278 cmptparm[i].dx = 1;
279 cmptparm[i].dy = 1;
280 cmptparm[i].w = params.width;
281 cmptparm[i].h = params.height;
282 }
283
284 // Determine color space
285 OPJ_COLOR_SPACE color_space = OPJ_CLRSPC_GRAY;
286 if (params.samples_per_pixel == 3) {
287 color_space = (params.photometric == photometric_interpretation::rgb)
288 ? OPJ_CLRSPC_SRGB
289 : OPJ_CLRSPC_SYCC;
290 }
291
292 // Create image
293 opj_image_t* image = opj_image_create(
294 params.samples_per_pixel,
295 cmptparm.data(),
296 color_space);
297
298 if (!image) {
299 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to create OpenJPEG image structure");
300 }
301
302 // Set image offset and reference grid
303 image->x0 = 0;
304 image->y0 = 0;
305 image->x1 = params.width;
306 image->y1 = params.height;
307
308 // Copy pixel data to image components
309 size_t bytes_per_sample = (params.bits_allocated + 7) / 8;
310 size_t pixel_count = static_cast<size_t>(params.width) * params.height;
311
312 for (uint16_t c = 0; c < params.samples_per_pixel; ++c) {
313 OPJ_INT32* comp_data = image->comps[c].data;
314
315 for (size_t i = 0; i < pixel_count; ++i) {
316 size_t src_idx;
317 if (params.planar_configuration == 0) {
318 // Interleaved: R0G0B0R1G1B1...
319 src_idx = (i * params.samples_per_pixel + c) * bytes_per_sample;
320 } else {
321 // Planar: RRR...GGG...BBB...
322 src_idx = (c * pixel_count + i) * bytes_per_sample;
323 }
324
325 if (src_idx + bytes_per_sample > pixel_data.size()) {
326 opj_image_destroy(image);
327 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Pixel data buffer too small");
328 }
329
330 OPJ_INT32 value = 0;
331 if (bytes_per_sample == 1) {
332 value = params.is_signed()
333 ? static_cast<OPJ_INT32>(static_cast<int8_t>(pixel_data[src_idx]))
334 : static_cast<OPJ_INT32>(pixel_data[src_idx]);
335 } else {
336 // 16-bit (little-endian)
337 uint16_t raw = static_cast<uint16_t>(pixel_data[src_idx]) |
338 (static_cast<uint16_t>(pixel_data[src_idx + 1]) << 8);
339 value = params.is_signed()
340 ? static_cast<OPJ_INT32>(static_cast<int16_t>(raw))
341 : static_cast<OPJ_INT32>(raw);
342 }
343 comp_data[i] = value;
344 }
345 }
346
347 // Create encoder
348 opj_codec_t* codec = opj_create_compress(OPJ_CODEC_J2K);
349 if (!codec) {
350 opj_image_destroy(image);
351 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to create OpenJPEG encoder");
352 }
353
354 // Set up error handling
355 std::string error_msg;
356 opj_set_error_handler(codec, opj_error_callback, &error_msg);
357 opj_set_warning_handler(codec, opj_warning_callback, nullptr);
358 opj_set_info_handler(codec, opj_info_callback, nullptr);
359
360 // Set encoding parameters
361 opj_cparameters_t parameters;
362 opj_set_default_encoder_parameters(&parameters);
363
364 parameters.tcp_numlayers = 1;
365 parameters.cp_disto_alloc = 1;
366 parameters.numresolution = resolution_levels_;
367
368 if (use_lossless) {
369 // Lossless: use reversible 5/3 wavelet
370 parameters.irreversible = 0;
371 parameters.tcp_rates[0] = 0; // 0 = lossless
372 } else {
373 // Lossy: use irreversible 9/7 wavelet
374 parameters.irreversible = 1;
375
376 // Convert quality (1-100) to compression ratio
377 float ratio = compression_ratio_;
378 if (options.quality > 0 && options.quality <= 100) {
379 // Map quality 100->2 (near lossless), quality 1->100 (high compression)
380 ratio = 2.0f + (100.0f - static_cast<float>(options.quality)) * 0.98f;
381 }
382 parameters.tcp_rates[0] = ratio;
383 }
384
385 // Setup encoder with parameters
386 if (!opj_setup_encoder(codec, &parameters, image)) {
387 opj_destroy_codec(codec);
388 opj_image_destroy(image);
389 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to setup OpenJPEG encoder: " + error_msg);
390 }
391
392 // Create output stream
393 opj_output_buffer output_buf;
394 output_buf.offset = 0;
395 output_buf.data.reserve(pixel_data.size() / 2); // Estimate compressed size
396
397 opj_stream_t* stream = opj_stream_default_create(OPJ_FALSE);
398 if (!stream) {
399 opj_destroy_codec(codec);
400 opj_image_destroy(image);
401 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to create OpenJPEG output stream");
402 }
403
404 opj_stream_set_user_data(stream, &output_buf, nullptr);
405 opj_stream_set_user_data_length(stream, 0);
406 opj_stream_set_write_function(stream, opj_output_buffer_write);
407 opj_stream_set_skip_function(stream, opj_output_buffer_skip);
408 opj_stream_set_seek_function(stream, opj_output_buffer_seek);
409
410 // Encode
411 if (!opj_start_compress(codec, image, stream)) {
412 opj_stream_destroy(stream);
413 opj_destroy_codec(codec);
414 opj_image_destroy(image);
415 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to start JPEG 2000 encoding: " + error_msg);
416 }
417
418 if (!opj_encode(codec, stream)) {
419 opj_stream_destroy(stream);
420 opj_destroy_codec(codec);
421 opj_image_destroy(image);
422 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to encode JPEG 2000 data: " + error_msg);
423 }
424
425 if (!opj_end_compress(codec, stream)) {
426 opj_stream_destroy(stream);
427 opj_destroy_codec(codec);
428 opj_image_destroy(image);
429 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to finalize JPEG 2000 encoding: " + error_msg);
430 }
431
432 // Cleanup
433 opj_stream_destroy(stream);
434 opj_destroy_codec(codec);
435 opj_image_destroy(image);
436
437 // Trim output buffer to actual size
438 output_buf.data.resize(output_buf.offset);
439
440 return kcenon::pacs::ok<compression_result>(compression_result{std::move(output_buf.data), params});
441#endif // PACS_WITH_JPEG2000_CODEC
442 }

444 [[nodiscard]] codec_result decode(std::span<const uint8_t> compressed_data,
445 const image_params& params) const {
446#ifndef PACS_WITH_JPEG2000_CODEC
447 (void)compressed_data;
448 (void)params;
449 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error,
450 "JPEG 2000 codec not available: OpenJPEG library not found at build time");
451#else
452 if (compressed_data.empty()) {
453 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Empty compressed data");
454 }
455
456 // Detect format
457 OPJ_CODEC_FORMAT format = detect_j2k_format(compressed_data);
458
459 // Create decoder
460 opj_codec_t* codec = opj_create_decompress(format);
461 if (!codec) {
462 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to create OpenJPEG decoder");
463 }
464
465 // Set up error handling
466 std::string error_msg;
467 opj_set_error_handler(codec, opj_error_callback, &error_msg);
468 opj_set_warning_handler(codec, opj_warning_callback, nullptr);
469 opj_set_info_handler(codec, opj_info_callback, nullptr);
470
471 // Set decoding parameters
472 opj_dparameters_t parameters;
473 opj_set_default_decoder_parameters(&parameters);
474
475 if (!opj_setup_decoder(codec, &parameters)) {
476 opj_destroy_codec(codec);
477 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to setup OpenJPEG decoder: " + error_msg);
478 }
479
480 // Create input stream
481 opj_memory_stream input_stream;
482 input_stream.data = compressed_data.data();
483 input_stream.size = compressed_data.size();
484 input_stream.offset = 0;
485
486 opj_stream_t* stream = opj_stream_default_create(OPJ_TRUE);
487 if (!stream) {
488 opj_destroy_codec(codec);
489 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to create OpenJPEG input stream");
490 }
491
492 opj_stream_set_user_data(stream, &input_stream, nullptr);
493 opj_stream_set_user_data_length(stream, compressed_data.size());
494 opj_stream_set_read_function(stream, opj_memory_stream_read);
495 opj_stream_set_skip_function(stream, opj_memory_stream_skip);
496 opj_stream_set_seek_function(stream, opj_memory_stream_seek);
497
498 // Read header
499 opj_image_t* image = nullptr;
500 if (!opj_read_header(stream, codec, &image)) {
501 opj_stream_destroy(stream);
502 opj_destroy_codec(codec);
503 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to read JPEG 2000 header: " + error_msg);
504 }
505
506 // Decode
507 if (!opj_decode(codec, stream, image)) {
508 opj_image_destroy(image);
509 opj_stream_destroy(stream);
510 opj_destroy_codec(codec);
511 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to decode JPEG 2000 data: " + error_msg);
512 }
513
514 if (!opj_end_decompress(codec, stream)) {
515 opj_image_destroy(image);
516 opj_stream_destroy(stream);
517 opj_destroy_codec(codec);
518 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Failed to finalize JPEG 2000 decoding: " + error_msg);
519 }
520
521 // Extract image parameters from decoded data
522 image_params output_params = params;
523 output_params.width = static_cast<uint16_t>(image->x1 - image->x0);
524 output_params.height = static_cast<uint16_t>(image->y1 - image->y0);
525 output_params.samples_per_pixel = static_cast<uint16_t>(image->numcomps);
526
527 if (image->numcomps > 0) {
528 output_params.bits_stored = static_cast<uint16_t>(image->comps[0].prec);
529 output_params.bits_allocated = (output_params.bits_stored <= 8) ? 8 : 16;
530 output_params.high_bit = output_params.bits_stored - 1;
531 output_params.pixel_representation = image->comps[0].sgnd ? 1 : 0;
532 }
533
534 // Determine photometric interpretation
535 if (image->numcomps == 1) {
536 output_params.photometric = photometric_interpretation::monochrome2;
537 } else if (image->numcomps == 3) {
538 output_params.photometric = (image->color_space == OPJ_CLRSPC_SRGB)
539 ? photometric_interpretation::rgb
540 : photometric_interpretation::ycbcr_full;
541 }
542
543 // Validate dimensions if provided
544 if (params.width > 0 && params.width != output_params.width) {
545 opj_image_destroy(image);
546 opj_stream_destroy(stream);
547 opj_destroy_codec(codec);
548 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Image width mismatch: expected " +
549 std::to_string(params.width) + ", got " +
550 std::to_string(output_params.width));
551 }
552 if (params.height > 0 && params.height != output_params.height) {
553 opj_image_destroy(image);
554 opj_stream_destroy(stream);
555 opj_destroy_codec(codec);
556 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Image height mismatch: expected " +
557 std::to_string(params.height) + ", got " +
558 std::to_string(output_params.height));
559 }
560
561 // Allocate output buffer
562 size_t pixel_count = static_cast<size_t>(output_params.width) * output_params.height;
563 size_t bytes_per_sample = (output_params.bits_allocated + 7) / 8;
564 size_t total_size = pixel_count * output_params.samples_per_pixel * bytes_per_sample;
565
566 std::vector<uint8_t> output_data(total_size);
567
568 // Copy decoded data to output buffer (interleaved format)
569 for (OPJ_UINT32 c = 0; c < image->numcomps; ++c) {
570 const OPJ_INT32* comp_data = image->comps[c].data;
571
572 for (size_t i = 0; i < pixel_count; ++i) {
573 size_t dst_idx = (i * output_params.samples_per_pixel + c) * bytes_per_sample;
574 OPJ_INT32 value = comp_data[i];
575
576 if (bytes_per_sample == 1) {
577 output_data[dst_idx] = static_cast<uint8_t>(value & 0xFF);
578 } else {
579 // 16-bit little-endian
580 output_data[dst_idx] = static_cast<uint8_t>(value & 0xFF);
581 output_data[dst_idx + 1] = static_cast<uint8_t>((value >> 8) & 0xFF);
582 }
583 }
584 }
585
586 // Cleanup
587 opj_image_destroy(image);
588 opj_stream_destroy(stream);
589 opj_destroy_codec(codec);
590
591 output_params.planar_configuration = 0; // Output is always interleaved
592 return kcenon::pacs::ok<compression_result>(compression_result{std::move(output_data), output_params});
593#endif // PACS_WITH_JPEG2000_CODEC
594 }

599 int resolution_levels_;

626bool jpeg2000_codec::can_encode(const image_params& params) const noexcept {
627 // JPEG 2000 supports wide range of bit depths
628 if (params.bits_stored < 1 || params.bits_stored > 16) {
629 return false;
630 }
631
632 // bits_allocated must be 8 or 16
633 if (params.bits_allocated != 8 && params.bits_allocated != 16) {
634 return false;
635 }
636
637 // Support grayscale (1) and color (3) images
