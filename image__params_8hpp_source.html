65[[nodiscard]] inline std::string to_string(photometric_interpretation pi) {
66 switch (pi) {
67 case photometric_interpretation::monochrome1:
68 return "MONOCHROME1";
69 case photometric_interpretation::monochrome2:
70 return "MONOCHROME2";
71 case photometric_interpretation::rgb:
72 return "RGB";
73 case photometric_interpretation::ycbcr_full:
74 return "YBR_FULL";
75 case photometric_interpretation::ycbcr_full_422:
76 return "YBR_FULL_422";

89[[nodiscard]] inline photometric_interpretation parse_photometric_interpretation(
90 const std::string& str) {
91 if (str == "MONOCHROME1") return photometric_interpretation::monochrome1;
92 if (str == "MONOCHROME2") return photometric_interpretation::monochrome2;
93 if (str == "RGB") return photometric_interpretation::rgb;
94 if (str == "YBR_FULL") return photometric_interpretation::ycbcr_full;
95 if (str == "YBR_FULL_422") return photometric_interpretation::ycbcr_full_422;
96 if (str == "PALETTE COLOR") return photometric_interpretation::palette_color;
97 return photometric_interpretation::unknown;
98}

147 [[nodiscard]] size_t frame_size_bytes() const noexcept {
148 size_t bits_per_pixel = static_cast<size_t>(bits_allocated) * samples_per_pixel;
149 size_t total_bits = static_cast<size_t>(width) * height * bits_per_pixel;
150 return (total_bits + 7) / 8; // Round up to byte boundary
151 }
152
157 [[nodiscard]] bool is_grayscale() const noexcept {
158 return samples_per_pixel == 1;
159 }
160
165 [[nodiscard]] bool is_color() const noexcept {
166 return samples_per_pixel > 1;
167 }
168
173 [[nodiscard]] bool is_signed() const noexcept {
174 return pixel_representation == 1;
175 }
176
185 [[nodiscard]] bool valid_for_jpeg_baseline() const noexcept {
186 if (bits_allocated != 8 || bits_stored != 8) return false;
187 if (samples_per_pixel != 1 && samples_per_pixel != 3) return false;
188 return true;
189 }
190
200 [[nodiscard]] bool valid_for_jpeg_lossless() const noexcept {
201 if (bits_stored < 2 || bits_stored > 16) return false;
202 if (bits_allocated != 8 && bits_allocated != 16) return false;
203 if (samples_per_pixel != 1) return false;
204 return true;
205 }
206
217 [[nodiscard]] bool valid_for_jpeg2000() const noexcept {
218 if (bits_stored < 1 || bits_stored > 16) return false;
219 if (bits_allocated != 8 && bits_allocated != 16) return false;
220 if (samples_per_pixel != 1 && samples_per_pixel != 3) return false;
221 if (width == 0 || height == 0) return false;
222 return true;
223 }
224
235 [[nodiscard]] bool valid_for_jpeg_ls() const noexcept {
236 if (bits_stored < 2 || bits_stored > 16) return false;
237 if (bits_allocated != 8 && bits_allocated != 16) return false;
238 if (samples_per_pixel != 1 && samples_per_pixel != 3) return false;
239 if (width == 0 || height == 0) return false;
240 if (width > 65535 || height > 65535) return false;
241 return true;
242 }
243
254 [[nodiscard]] bool valid_for_rle() const noexcept {
255 if (bits_allocated != 8 && bits_allocated != 16) return false;
256 if (samples_per_pixel < 1 || samples_per_pixel > 3) return false;
257 if (width == 0 || height == 0) return false;
258 if (width > 65535 || height > 65535) return false;
259 // Check segment count limit (max 15 segments)
260 int bytes_per_sample = (bits_allocated + 7) / 8;
261 int num_segments = samples_per_pixel * bytes_per_sample;
262 if (num_segments > 15) return false;
263 return true;
264 }
265};
