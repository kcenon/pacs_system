62network::Result<std::monostate> storage_commitment_scp::handle_message(
63 network::association& assoc,
64 uint8_t context_id,
65 const network::dimse::dimse_message& request) {
66
67 using namespace network::dimse;
68
69 switch (request.command()) {
70 case command_field::n_action_rq:
71 return handle_n_action(assoc, context_id, request);
72
73 default:
74 return kcenon::pacs::pacs_void_error(
75 kcenon::pacs::error_codes::storage_commitment_unexpected_command,
76 "Unexpected command for Storage Commitment SCP: " +
77 std::string(to_string(request.command())));
78 }
79}

111network::Result<std::monostate> storage_commitment_scp::handle_n_action(
112 network::association& assoc,
113 uint8_t context_id,
114 const network::dimse::dimse_message& request) {
115
116 using namespace network::dimse;
117
118 // Validate Requested SOP Class UID
119 auto sop_class_uid = request.command_set().get_string(
120 tag_requested_sop_class_uid);
121 if (sop_class_uid.empty()) {
122 sop_class_uid = request.affected_sop_class_uid();
123 }
124
125 if (sop_class_uid != storage_commitment_push_model_sop_class_uid) {
126 return send_n_action_response(
127 assoc, context_id, request.message_id(),
128 status_refused_sop_class_not_supported);
129 }
130
131 // Validate Action Type ID = 1 (Request Storage Commitment)
132 auto action_type = request.action_type_id();
133 if (!action_type.has_value() ||
134 action_type.value() != storage_commitment_action_type_request) {
135 return send_n_action_response(
136 assoc, context_id, request.message_id(),
137 status_error_no_such_action_type);
138 }
139
140 // Verify we have a dataset
141 if (!request.has_dataset()) {
142 return send_n_action_response(
143 assoc, context_id, request.message_id(),
144 status_error_missing_attribute);
145 }
146
147 const auto& dataset = request.dataset().value().get();
148
149 // Extract Transaction UID (0008,1195)
150 auto transaction_uid = dataset.get_string(core::tags::transaction_uid);
151 if (transaction_uid.empty()) {
152 return send_n_action_response(
153 assoc, context_id, request.message_id(),
154 status_error_missing_attribute);
155 }
156
157 // Parse Referenced SOP Sequence (0008,1199)
158 auto references = parse_referenced_sop_sequence(dataset);
159 if (references.empty()) {
160 return send_n_action_response(
161 assoc, context_id, request.message_id(),
162 status_error_missing_attribute);
163 }
164
165 // Send N-ACTION-RSP with Success (acknowledge the request)
166 auto rsp_result = send_n_action_response(
167 assoc, context_id, request.message_id(),
168 status_success);
169 if (rsp_result.is_err()) {
170 return rsp_result;
171 }
172
173 // Update statistics
174 ++actions_processed_;
175
176 // Verify instances against storage
177 auto result = verify_instances(transaction_uid, references);
178
179 // Update instance statistics
180 instances_committed_ += result.success_references.size();
181 instances_failed_ += result.failed_references.size();
182
183 // Send N-EVENT-REPORT with verification results
184 return send_event_report(assoc, context_id, result);
185}

188// Instance Verification

190
191commitment_result storage_commitment_scp::verify_instances(
192 const std::string& transaction_uid,
193 const std::vector<sop_reference>& references) {
194
195 commitment_result result;
196 result.transaction_uid = transaction_uid;
197 result.timestamp = std::chrono::system_clock::now();
198
199 for (const auto& ref : references) {
200 if (storage_ && storage_->exists(ref.sop_instance_uid)) {
201 result.success_references.push_back(ref);
202 } else {
203 result.failed_references.emplace_back(
204 ref, commitment_failure_reason::no_such_object_instance);
205 }
206 }
207
208 return result;
209}

214
215network::Result<std::monostate> storage_commitment_scp::send_event_report(
216 network::association& assoc,
217 uint8_t context_id,
218 const commitment_result& result) {
219
220 using namespace network::dimse;
221
222 // Determine event type: 1 = all success, 2 = failures exist
223 uint16_t event_type = result.failed_references.empty()
224 ? storage_commitment_event_type_success
225 : storage_commitment_event_type_failure;
226
227 // Build N-EVENT-REPORT-RQ
228 auto event_rq = make_n_event_report_rq(
229 1, // message_id
230 storage_commitment_push_model_sop_class_uid,
231 storage_commitment_push_model_sop_instance_uid,
232 event_type);
233
234 // Build and attach event dataset
235 auto event_dataset = build_event_report_dataset(result);
236 event_rq.set_dataset(std::move(event_dataset));
237
238 // Send N-EVENT-REPORT-RQ
239 return assoc.send_dimse(context_id, event_rq);
240}

268std::vector<sop_reference> storage_commitment_scp::parse_referenced_sop_sequence(
269 const core::dicom_dataset& dataset) {
270
271 std::vector<sop_reference> references;
272
273 const auto* seq = dataset.get_sequence(core::tags::referenced_sop_sequence);
274 if (seq == nullptr) {
275 return references;
276 }
277
278 for (const auto& item : *seq) {
279 sop_reference ref;
280 ref.sop_class_uid = item.get_string(core::tags::referenced_sop_class_uid);
281 ref.sop_instance_uid = item.get_string(core::tags::referenced_sop_instance_uid);
282
283 if (!ref.sop_class_uid.empty() && !ref.sop_instance_uid.empty()) {
284 references.push_back(std::move(ref));
285 }
286 }
287
288 return references;
289}

291core::dicom_dataset storage_commitment_scp::build_event_report_dataset(
292 const commitment_result& result) {
293
294 core::dicom_dataset ds;
295
296 // Transaction UID (0008,1195)
297 ds.set_string(core::tags::transaction_uid, encoding::vr_type::UI,
298 result.transaction_uid);
299
300 // Referenced SOP Sequence (0008,1199) — successful instances
301 if (!result.success_references.empty()) {
302 auto& success_seq = ds.get_or_create_sequence(
303 core::tags::referenced_sop_sequence);
304
305 for (const auto& ref : result.success_references) {
306 core::dicom_dataset item;
307 item.set_string(core::tags::referenced_sop_class_uid,
308 encoding::vr_type::UI, ref.sop_class_uid);
309 item.set_string(core::tags::referenced_sop_instance_uid,
310 encoding::vr_type::UI, ref.sop_instance_uid);
311 success_seq.push_back(std::move(item));
312 }
313 }
314
315 // Failed SOP Sequence (0008,1198) — failed instances
316 if (!result.failed_references.empty()) {
317 auto& failed_seq = ds.get_or_create_sequence(
318 core::tags::failed_sop_sequence);
319
320 for (const auto& [ref, reason] : result.failed_references) {
321 core::dicom_dataset item;
322 item.set_string(core::tags::referenced_sop_class_uid,
323 encoding::vr_type::UI, ref.sop_class_uid);
324 item.set_string(core::tags::referenced_sop_instance_uid,
325 encoding::vr_type::UI, ref.sop_instance_uid);
326 item.set_numeric<uint16_t>(core::tags::failure_reason,
327 encoding::vr_type::US,
328 static_cast<uint16_t>(reason));
329 failed_seq.push_back(std::move(item));
330 }
331 }
332
333 return ds;
334}
