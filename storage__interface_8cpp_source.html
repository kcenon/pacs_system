58auto storage_interface::retrieve_batch(
59 const std::vector<std::string>& sop_instance_uids)
60 -> Result<std::vector<core::dicom_dataset>> {
61 std::vector<core::dicom_dataset> results;
62 results.reserve(sop_instance_uids.size());
63
64 for (const auto& uid : sop_instance_uids) {
65 auto result = retrieve(uid);
66 if (result.is_ok()) {
67 results.push_back(std::move(result.value()));
68 }
69 // Silently skip missing instances as per interface contract
70 }
71
72 return results;
73}
