69[[nodiscard]] inline auto to_string(annotation_type type) -> std::string {
70 switch (type) {
71 case annotation_type::arrow: return "arrow";
72 case annotation_type::line: return "line";
73 case annotation_type::rectangle: return "rectangle";
74 case annotation_type::ellipse: return "ellipse";
75 case annotation_type::polygon: return "polygon";
76 case annotation_type::freehand: return "freehand";
77 case annotation_type::text: return "text";
78 case annotation_type::angle: return "angle";
79 case annotation_type::roi: return "roi";
80 }
81 return "unknown";
82}

87[[nodiscard]] inline auto annotation_type_from_string(std::string_view str)
88 -> std::optional<annotation_type> {
89 if (str == "arrow") return annotation_type::arrow;
90 if (str == "line") return annotation_type::line;
91 if (str == "rectangle") return annotation_type::rectangle;
92 if (str == "ellipse") return annotation_type::ellipse;
93 if (str == "polygon") return annotation_type::polygon;
94 if (str == "freehand") return annotation_type::freehand;
95 if (str == "text") return annotation_type::text;
96 if (str == "angle") return annotation_type::angle;
97 if (str == "roi") return annotation_type::roi;
98 return std::nullopt;
99}
