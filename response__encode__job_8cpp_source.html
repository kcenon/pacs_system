64auto response_encode_job::execute(pipeline_coordinator& coordinator) -> VoidResult {
65 // Encode the response
66 auto encode_result = encode_response();
67 if (!encode_result.is_ok()) {
68 if (on_error_) {
69 auto err = encode_result.error();
70 on_error_(result_.session_id, err.message);
71 }
72 return VoidResult(encode_result.error());
73 }
74
75 auto responses = encode_result.value();
76
77 // Submit each encoded response to the send stage
78 for (auto& response : responses) {
79 // Invoke callback if set
80 if (on_encoded_) {
81 on_encoded_(response);
82 }
83
84 // Create send job
85 // Note: send_fn will be set by the adapter layer
86 auto send_job = std::make_unique<send_network_io_job>(
87 response.session_id,
88 std::move(response.pdu_data),
89 nullptr // send_fn set by adapter
90 );
91
92 send_job->get_context() = context_;
93 send_job->get_context().stage = pipeline_stage::network_send;
94
95 auto submit_result = coordinator.submit_to_stage(
96 pipeline_stage::network_send,
97 std::move(send_job)
98 );
99
100 if (!submit_result.is_ok()) {
101 return submit_result;
102 }
103 }
104
105 return ok();
106}

127auto response_encode_job::encode_response()
128 -> Result<std::vector<encoded_response>> {
129
130 std::vector<encoded_response> responses;
131
132 // Encode DIMSE command
133 auto command_result = encode_dimse_command();
134 if (!command_result.is_ok()) {
135 return Result<std::vector<encoded_response>>(command_result.error());
136 }
137
138 auto command_data = command_result.value();
139
140 // Fragment the command data if needed
141 auto command_fragments = fragment_data(command_data);
142
143 // Create P-DATA-TF PDUs for command
144 for (size_t i = 0; i < command_fragments.size(); ++i) {
145 encoded_response response;
146 response.session_id = result_.session_id;
147 response.message_id = result_.message_id;
148 response.is_final = (i == command_fragments.size() - 1) && result_.data_set.empty();
149
150 // Build P-DATA-TF PDU
151 // PDU: Type (1) + Reserved (1) + Length (4) + PDV Items
152 // PDV: Length (4) + Presentation Context ID (1) + Control Header (1) + Data
153
154 const auto& frag = command_fragments[i];
155 size_t pdv_length = 2 + frag.size(); // Context ID + Control + Data
156 size_t pdu_length = 4 + pdv_length; // PDV Length + PDV
157
158 response.pdu_data.resize(6 + pdu_length);
159
160 // PDU header
161 response.pdu_data[0] = 0x04; // P-DATA-TF
162 response.pdu_data[1] = 0x00; // Reserved
163
164 // PDU length (big-endian)
165 response.pdu_data[2] = static_cast<uint8_t>((pdu_length >> 24) & 0xFF);
166 response.pdu_data[3] = static_cast<uint8_t>((pdu_length >> 16) & 0xFF);
167 response.pdu_data[4] = static_cast<uint8_t>((pdu_length >> 8) & 0xFF);
168 response.pdu_data[5] = static_cast<uint8_t>(pdu_length & 0xFF);
169
170 // PDV length (big-endian)
171 size_t offset = 6;
172 response.pdu_data[offset++] = static_cast<uint8_t>((pdv_length >> 24) & 0xFF);
173 response.pdu_data[offset++] = static_cast<uint8_t>((pdv_length >> 16) & 0xFF);
174 response.pdu_data[offset++] = static_cast<uint8_t>((pdv_length >> 8) & 0xFF);
175 response.pdu_data[offset++] = static_cast<uint8_t>(pdv_length & 0xFF);
176
177 // Presentation context ID
178 response.pdu_data[offset++] = result_.presentation_context_id;
179
180 // Control header: command (bit 0) + last fragment (bit 1)
181 uint8_t control = 0x01; // Command fragment
182 if (i == command_fragments.size() - 1) {
183 control |= 0x02; // Last command fragment
184 }
185 response.pdu_data[offset++] = control;
186
187 // Copy fragment data
188 std::copy(frag.begin(), frag.end(), response.pdu_data.begin() + offset);
189
190 responses.push_back(std::move(response));
191 }
192
193 // Encode data set if present
194 if (!result_.data_set.empty()) {
195 auto data_fragments = fragment_data(result_.data_set);
196
197 for (size_t i = 0; i < data_fragments.size(); ++i) {
198 encoded_response response;
199 response.session_id = result_.session_id;
200 response.message_id = result_.message_id;
201 response.is_final = (i == data_fragments.size() - 1);
202
203 const auto& frag = data_fragments[i];
204 size_t pdv_length = 2 + frag.size();
205 size_t pdu_length = 4 + pdv_length;
206
207 response.pdu_data.resize(6 + pdu_length);
208
209 // PDU header
210 response.pdu_data[0] = 0x04;
211 response.pdu_data[1] = 0x00;
212
213 response.pdu_data[2] = static_cast<uint8_t>((pdu_length >> 24) & 0xFF);
214 response.pdu_data[3] = static_cast<uint8_t>((pdu_length >> 16) & 0xFF);
215 response.pdu_data[4] = static_cast<uint8_t>((pdu_length >> 8) & 0xFF);
216 response.pdu_data[5] = static_cast<uint8_t>(pdu_length & 0xFF);
217
218 size_t offset = 6;
219 response.pdu_data[offset++] = static_cast<uint8_t>((pdv_length >> 24) & 0xFF);
220 response.pdu_data[offset++] = static_cast<uint8_t>((pdv_length >> 16) & 0xFF);
221 response.pdu_data[offset++] = static_cast<uint8_t>((pdv_length >> 8) & 0xFF);
222 response.pdu_data[offset++] = static_cast<uint8_t>(pdv_length & 0xFF);
223
224 response.pdu_data[offset++] = result_.presentation_context_id;
225
226 uint8_t control = 0x00; // Data fragment
227 if (i == data_fragments.size() - 1) {
228 control |= 0x02; // Last data fragment
229 }
230 response.pdu_data[offset++] = control;
231
232 std::copy(frag.begin(), frag.end(), response.pdu_data.begin() + offset);
233
234 responses.push_back(std::move(response));
235 }
236 }
237
238 return ok(std::move(responses));
239}

241auto response_encode_job::encode_dimse_command()
242 -> Result<std::vector<uint8_t>> {
243
244 std::vector<uint8_t> command;
245 command.reserve(256); // Typical command size
246
247 // Helper to append element (Implicit VR Little Endian)
248 auto append_element = [&command](uint16_t group, uint16_t element,
249 const void* data, uint32_t length) {
250 // Tag
251 command.push_back(static_cast<uint8_t>(group & 0xFF));
252 command.push_back(static_cast<uint8_t>((group >> 8) & 0xFF));
253 command.push_back(static_cast<uint8_t>(element & 0xFF));
254 command.push_back(static_cast<uint8_t>((element >> 8) & 0xFF));
255 // Length
256 command.push_back(static_cast<uint8_t>(length & 0xFF));
257 command.push_back(static_cast<uint8_t>((length >> 8) & 0xFF));
258 command.push_back(static_cast<uint8_t>((length >> 16) & 0xFF));
259 command.push_back(static_cast<uint8_t>((length >> 24) & 0xFF));
260 // Data
261 const auto* bytes = static_cast<const uint8_t*>(data);
262 command.insert(command.end(), bytes, bytes + length);
263 };
264
265 auto append_uint16 = [&append_element](uint16_t group, uint16_t element,
266 uint16_t value) {
267 append_element(group, element, &value, 2);
268 };
269
270 auto append_string = [&append_element](uint16_t group, uint16_t element,
271 const std::string& value) {
272 std::string padded = value;
273 if (padded.size() % 2 != 0) {
274 padded.push_back(' '); // Pad to even length
275 }
276 append_element(group, element, padded.data(),
277 static_cast<uint32_t>(padded.size()));
278 };
279
280 // Placeholder for CommandGroupLength (will be filled after encoding)
281 size_t group_length_offset = command.size();
282 uint32_t placeholder = 0;
283 append_element(0x0000, 0x0000, &placeholder, 4);
284
285 // AffectedSOPClassUID (0000,0002) or RequestedSOPClassUID (0000,0003)
286 if (!result_.sop_class_uid.empty()) {
287 append_string(0x0000, 0x0002, result_.sop_class_uid);
288 }
289
290 // CommandField (0000,0100)
291 append_uint16(0x0000, 0x0100, static_cast<uint16_t>(result_.response_type));
292
293 // MessageIDBeingRespondedTo (0000,0120)
294 append_uint16(0x0000, 0x0120, result_.message_id);
295
296 // CommandDataSetType (0000,0800)
297 // 0x0101 = no data set, 0x0001 = data set present
298 uint16_t data_set_type = result_.data_set.empty() ? 0x0101 : 0x0001;
299 append_uint16(0x0000, 0x0800, data_set_type);
300
301 // Status (0000,0900)
302 append_uint16(0x0000, 0x0900, static_cast<uint16_t>(result_.status));
303
304 // AffectedSOPInstanceUID (0000,1000) if present
305 if (!result_.sop_instance_uid.empty()) {
306 append_string(0x0000, 0x1000, result_.sop_instance_uid);
307 }
308
309 // For C-GET/C-MOVE responses, include sub-operation counts
310 if (result_.response_type == dimse_command_type::c_get_rsp ||
311 result_.response_type == dimse_command_type::c_move_rsp) {
312 if (result_.status == dimse_status::pending ||
313 result_.status == dimse_status::pending_warning) {
314 append_uint16(0x0000, 0x1020, result_.remaining_sub_ops);
315 append_uint16(0x0000, 0x1021, result_.completed_sub_ops);
316 append_uint16(0x0000, 0x1022, result_.failed_sub_ops);
317 append_uint16(0x0000, 0x1023, result_.warning_sub_ops);
318 }
319 }
320
321 // Error comment if present
322 if (!result_.error_comment.empty()) {
323 append_string(0x0000, 0x0902, result_.error_comment);
324 }
325
326 // Update CommandGroupLength
327 uint32_t group_length = static_cast<uint32_t>(command.size() - group_length_offset - 8);
328 command[group_length_offset + 4] = static_cast<uint8_t>(group_length & 0xFF);
329 command[group_length_offset + 5] = static_cast<uint8_t>((group_length >> 8) & 0xFF);
330 command[group_length_offset + 6] = static_cast<uint8_t>((group_length >> 16) & 0xFF);
331 command[group_length_offset + 7] = static_cast<uint8_t>((group_length >> 24) & 0xFF);
332
333 return ok(std::move(command));
334}

336auto response_encode_job::fragment_data(const std::vector<uint8_t>& data)
337 -> std::vector<std::vector<uint8_t>> {
338
339 std::vector<std::vector<uint8_t>> fragments;
340
341 // Max fragment size = max_pdu_size - PDU header (6) - PDV header (6)
342 size_t max_fragment_size = max_pdu_size_ - 12;
343 if (max_fragment_size > 16376) {
344 max_fragment_size = 16376; // Reasonable default
345 }
346
347 size_t offset = 0;
348 while (offset < data.size()) {
349 size_t fragment_size = std::min(max_fragment_size, data.size() - offset);
350 fragments.emplace_back(data.begin() + offset,
351 data.begin() + offset + fragment_size);
352 offset += fragment_size;
353 }
354
355 if (fragments.empty()) {
356 fragments.emplace_back(); // Empty fragment for empty data
357 }
358
359 return fragments;
360}
