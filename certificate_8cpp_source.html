83auto read_file(std::string_view path) -> kcenon::common::Result<std::string> {
84 std::ifstream file(std::string(path), std::ios::binary);
85 if (!file.is_open()) {
86 return kcenon::common::make_error<std::string>(
87 1, "Failed to open file: " + std::string(path), "certificate");
88 }
89
90 std::ostringstream ss;
91 ss << file.rdbuf();
92 return ss.str();

108auto asn1_time_to_time_point(const ASN1_TIME* asn1_time)
109 -> std::chrono::system_clock::time_point {
110 if (!asn1_time) {
111 return std::chrono::system_clock::time_point{};
112 }
113
114 struct tm tm_time = {};
115 int offset_sec = 0;
116
117 // Parse the ASN1 time
118 if (ASN1_TIME_to_tm(asn1_time, &tm_time) != 1) {
119 return std::chrono::system_clock::time_point{};
120 }
121
122 // Convert to time_t (using portable function for cross-platform support)
123 time_t time = portable_timegm(&tm_time);
124
125 return std::chrono::system_clock::from_time_t(time);
126}

134class certificate_impl {
135public:
136 certificate_impl() : x509_(nullptr) {}
137
138 explicit certificate_impl(X509* cert) : x509_(cert) {}
139
140 ~certificate_impl() {
141 if (x509_) {
142 X509_free(x509_);
143 }
144 }
145
146 // Non-copyable but movable
147 certificate_impl(const certificate_impl& other) {
148 if (other.x509_) {
149 x509_ = X509_dup(other.x509_);
150 } else {
151 x509_ = nullptr;
152 }
153 }
154
155 certificate_impl(certificate_impl&& other) noexcept
156 : x509_(other.x509_) {
157 other.x509_ = nullptr;
158 }
159
160 auto operator=(const certificate_impl& other) -> certificate_impl& {
161 if (this != &other) {
162 if (x509_) {
163 X509_free(x509_);
164 }
165 if (other.x509_) {
166 x509_ = X509_dup(other.x509_);
167 } else {
168 x509_ = nullptr;
169 }
170 }
171 return *this;
172 }
173
174 auto operator=(certificate_impl&& other) noexcept -> certificate_impl& {
175 if (this != &other) {
176 if (x509_) {
177 X509_free(x509_);
178 }
179 x509_ = other.x509_;
180 other.x509_ = nullptr;
181 }
182 return *this;
183 }
184
185 [[nodiscard]] auto x509() const noexcept -> X509* { return x509_; }
186 [[nodiscard]] auto is_loaded() const noexcept -> bool { return x509_ != nullptr; }
187
188private:
189 X509* x509_;
190};

321auto certificate::subject_common_name() const -> std::string {
322 if (!impl_->is_loaded()) {
323 return "";
324 }
325
326 X509_NAME* name = X509_get_subject_name(impl_->x509());
327 if (!name) {
328 return "";
329 }
330
331 int idx = X509_NAME_get_index_by_NID(name, NID_commonName, -1);
332 if (idx < 0) {
333 return "";
334 }
335
336 X509_NAME_ENTRY* entry = X509_NAME_get_entry(name, idx);
337 if (!entry) {
338 return "";
339 }
340
341 ASN1_STRING* data = X509_NAME_ENTRY_get_data(entry);
342 if (!data) {
343 return "";
344 }
345
346 unsigned char* utf8 = nullptr;
347 int len = ASN1_STRING_to_UTF8(&utf8, data);
348 if (len < 0) {
349 return "";
350 }
351
352 std::string result(reinterpret_cast<char*>(utf8), static_cast<size_t>(len));
353 OPENSSL_free(utf8);
354 return result;
355}

438auto certificate::thumbprint() const -> std::string {
439 if (!impl_->is_loaded()) {
440 return "";
441 }
442
443 unsigned char hash[EVP_MAX_MD_SIZE];
444 unsigned int hash_len = 0;
445
446 if (X509_digest(impl_->x509(), EVP_sha256(), hash, &hash_len) != 1) {
447 return "";
448 }
449
450 std::string result;
451 result.reserve(hash_len * 2);
452
453 static const char hex_chars[] = "0123456789ABCDEF";
454 for (unsigned int i = 0; i < hash_len; ++i) {
455 result += hex_chars[(hash[i] >> 4) & 0x0F];
456 result += hex_chars[hash[i] & 0x0F];
457 }
458
459 return result;
460}

663auto certificate_chain::load_from_pem(std::string_view path)
664 -> kcenon::common::Result<certificate_chain> {
665 auto content_result = read_file(path);
666 if (content_result.is_err()) {
667 return kcenon::common::make_error<certificate_chain>(
668 content_result.error().code,
669 content_result.error().message,
670 "certificate_chain");
671 }
672
673 const std::string& content = content_result.value();
674 certificate_chain chain;
675
676 // Parse multiple certificates from PEM
677 bio_ptr bio(BIO_new_mem_buf(content.data(), static_cast<int>(content.size())));
678 if (!bio) {
679 return kcenon::common::make_error<certificate_chain>(
680 2, "Failed to create BIO: " + get_openssl_error(), "certificate_chain");
681 }
682
683 while (true) {
684 X509* x509 = PEM_read_bio_X509(bio.get(), nullptr, nullptr, nullptr);
685 if (!x509) {
686 // Check if it's EOF or an error
687 unsigned long err = ERR_peek_last_error();
688 if (ERR_GET_REASON(err) == PEM_R_NO_START_LINE) {
689 ERR_clear_error();
690 break; // End of file
691 }
692 break; // Error or EOF
693 }
694
695 certificate cert;
696 cert.impl_ = std::make_unique<certificate_impl>(x509);
697 chain.add(std::move(cert));
698 }
699
700 if (chain.empty()) {
701 return kcenon::common::make_error<certificate_chain>(
702 3, "No certificates found in PEM file", "certificate_chain");
703 }
704
705 return chain;
706}
