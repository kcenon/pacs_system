137class mock_s3_client : public s3_storage::s3_client_interface {
138public:
139 explicit mock_s3_client(const cloud_storage_config & /*config*/)
140 : connected_(true) {}
141
142 [[nodiscard]] auto put_object(const std::string &key,
143 const std::vector<std::uint8_t> &data)
144 -> VoidResult override {
145 if (!connected_) {
146 return make_error<std::monostate>(
147 kConnectionError, "S3 client not connected", "s3_storage");
148 }
149 objects_[key] = data;
150 return ok();
151 }
152
153 [[nodiscard]] auto get_object(const std::string &key)
154 -> Result<std::vector<std::uint8_t>> override {
155 if (!connected_) {
156 return make_error<std::vector<std::uint8_t>>(
157 kConnectionError, "S3 client not connected", "s3_storage");
158 }
159 auto it = objects_.find(key);
160 if (it == objects_.end()) {
161 return make_error<std::vector<std::uint8_t>>(
162 kObjectNotFound, "Object not found: " + key, "s3_storage");
163 }
164 return it->second;
165 }
166
167 [[nodiscard]] auto delete_object(const std::string &key)
168 -> VoidResult override {
169 if (!connected_) {
170 return make_error<std::monostate>(
171 kConnectionError, "S3 client not connected", "s3_storage");
172 }
173 objects_.erase(key);
174 return ok();
175 }
176
177 [[nodiscard]] auto head_object(const std::string &key) const
178 -> bool override {
179 if (!connected_) {
180 return false;
181 }
182 return objects_.contains(key);
183 }
184
185 [[nodiscard]] auto get_object_size(const std::string &key) const
186 -> std::size_t override {
187 auto it = objects_.find(key);
188 if (it != objects_.end()) {
189 return it->second.size();
190 }
191 return 0;
192 }
193
194 [[nodiscard]] auto list_objects() const
195 -> std::vector<std::string> override {
196 std::vector<std::string> keys;
197 keys.reserve(objects_.size());
198 for (const auto &[key, data] : objects_) {
199 keys.push_back(key);
200 }
201 return keys;
202 }
203
204 [[nodiscard]] auto is_connected() const -> bool override {
205 return connected_;
206 }
207
208 [[nodiscard]] auto multipart_upload(const std::string &key,
209 const std::vector<std::uint8_t> &data,
210 std::size_t part_size,
211 progress_callback callback)
212 -> VoidResult override {
213 std::size_t total_bytes = data.size();
214 std::size_t bytes_uploaded = 0;
215
216 while (bytes_uploaded < total_bytes) {
217 std::size_t chunk = (std::min)(part_size, total_bytes - bytes_uploaded);
218 bytes_uploaded += chunk;
219
220 if (callback && !callback(bytes_uploaded, total_bytes)) {
221 return make_error<std::monostate>(
222 kUploadError, "Upload cancelled by user", "s3_storage");
223 }
224 }
225
226 return put_object(key, data);
227 }
228
229private:
230 std::unordered_map<std::string, std::vector<std::uint8_t>> objects_;
231 bool connected_;
232};

265class aws_s3_client : public s3_storage::s3_client_interface {
266public:
267 explicit aws_s3_client(const cloud_storage_config &config)
268 : bucket_(config.bucket_name) {
269 aws_sdk_guard::ensure_initialized();
270
271 Aws::Client::ClientConfiguration client_config;
272 client_config.region = config.region;
273 client_config.connectTimeoutMs = config.connect_timeout_ms;
274 client_config.requestTimeoutMs = config.request_timeout_ms;
275 client_config.maxConnections = static_cast<unsigned>(config.max_connections);
276
277 if (config.endpoint_url.has_value()) {
278 client_config.endpointOverride = config.endpoint_url.value();
279 // For S3-compatible services (MinIO), use path-style addressing
280 use_path_style_ = true;
281 }
282
283 Aws::Auth::AWSCredentials credentials(config.access_key_id,
284 config.secret_access_key);
285
286 Aws::S3::S3ClientConfiguration s3_config(client_config);
287 s3_config.useVirtualAddressing = !use_path_style_;
288 client_ = std::make_unique<Aws::S3::S3Client>(credentials, nullptr,
289 s3_config);
290 }

292 [[nodiscard]] auto put_object(const std::string &key,
293 const std::vector<std::uint8_t> &data)
294 -> VoidResult override {
295 Aws::S3::Model::PutObjectRequest request;
296 request.SetBucket(bucket_);
297 request.SetKey(key);
298 request.SetContentType("application/dicom");
299
300 auto stream = Aws::MakeShared<Aws::StringStream>("PutObjectStream");
301 stream->write(reinterpret_cast<const char *>(data.data()),
302 static_cast<std::streamsize>(data.size()));
303 request.SetBody(stream);
304
305 auto outcome = client_->PutObject(request);
306 if (!outcome.IsSuccess()) {
307 return make_error<std::monostate>(
308 kUploadError,
309 "S3 PutObject failed: " +
310 std::string(outcome.GetError().GetMessage()),
311 "s3_storage");
312 }
313
314 return ok();
315 }
316
317 [[nodiscard]] auto get_object(const std::string &key)
318 -> Result<std::vector<std::uint8_t>> override {
319 Aws::S3::Model::GetObjectRequest request;
320 request.SetBucket(bucket_);
321 request.SetKey(key);
322
323 auto outcome = client_->GetObject(request);
324 if (!outcome.IsSuccess()) {
325 const auto &error = outcome.GetError();
326 if (error.GetErrorType() ==
327 Aws::S3::S3Errors::NO_SUCH_KEY) {
328 return make_error<std::vector<std::uint8_t>>(
329 kObjectNotFound, "Object not found: " + key, "s3_storage");
330 }
331 return make_error<std::vector<std::uint8_t>>(
332 kDownloadError,
333 "S3 GetObject failed: " + std::string(error.GetMessage()),
334 "s3_storage");
335 }
336
337 auto &body = outcome.GetResult().GetBody();
338 std::vector<std::uint8_t> result(
339 (std::istreambuf_iterator<char>(body)),
340 std::istreambuf_iterator<char>());
341 return result;
342 }
343
344 [[nodiscard]] auto delete_object(const std::string &key)
345 -> VoidResult override {
346 Aws::S3::Model::DeleteObjectRequest request;
347 request.SetBucket(bucket_);
348 request.SetKey(key);
349
350 auto outcome = client_->DeleteObject(request);
351 if (!outcome.IsSuccess()) {
352 return make_error<std::monostate>(
353 kUploadError,
354 "S3 DeleteObject failed: " +
355 std::string(outcome.GetError().GetMessage()),
356 "s3_storage");
357 }
358
359 return ok();
360 }
361
362 [[nodiscard]] auto head_object(const std::string &key) const
363 -> bool override {
364 Aws::S3::Model::HeadObjectRequest request;
365 request.SetBucket(bucket_);
366 request.SetKey(key);
367
368 auto outcome = client_->HeadObject(request);
369 return outcome.IsSuccess();
370 }
371
372 [[nodiscard]] auto get_object_size(const std::string &key) const
373 -> std::size_t override {
374 Aws::S3::Model::HeadObjectRequest request;
375 request.SetBucket(bucket_);
376 request.SetKey(key);
377
378 auto outcome = client_->HeadObject(request);
379 if (outcome.IsSuccess()) {
380 return static_cast<std::size_t>(
381 outcome.GetResult().GetContentLength());
382 }
383 return 0;
384 }
385
386 [[nodiscard]] auto list_objects() const
387 -> std::vector<std::string> override {
388 std::vector<std::string> keys;
389
390 Aws::S3::Model::ListObjectsV2Request request;
391 request.SetBucket(bucket_);
392
393 bool has_more = true;
394 while (has_more) {
395 auto outcome = client_->ListObjectsV2(request);
396 if (!outcome.IsSuccess()) {
397 break;
398 }
399
400 const auto &result = outcome.GetResult();
401 for (const auto &object : result.GetContents()) {
402 keys.push_back(object.GetKey());
403 }
404
405 has_more = result.GetIsTruncated();
406 if (has_more) {
407 request.SetContinuationToken(result.GetNextContinuationToken());
408 }
409 }
410
411 return keys;
412 }
413
414 [[nodiscard]] auto is_connected() const -> bool override {
415 return client_ != nullptr;
416 }

418 [[nodiscard]] auto multipart_upload(const std::string &key,
419 const std::vector<std::uint8_t> &data,
420 std::size_t part_size,
421 progress_callback callback)
422 -> VoidResult override {
423 // Initiate multipart upload
424 Aws::S3::Model::CreateMultipartUploadRequest create_request;
425 create_request.SetBucket(bucket_);
426 create_request.SetKey(key);
427 create_request.SetContentType("application/dicom");
428
429 auto create_outcome = client_->CreateMultipartUpload(create_request);
430 if (!create_outcome.IsSuccess()) {
431 return make_error<std::monostate>(
432 kUploadError,
433 "Failed to initiate multipart upload: " +
434 std::string(create_outcome.GetError().GetMessage()),
435 "s3_storage");
436 }
437
438 auto upload_id = create_outcome.GetResult().GetUploadId();
439 Aws::S3::Model::CompletedMultipartUpload completed_upload;
440
441 std::size_t total_bytes = data.size();
442 std::size_t bytes_uploaded = 0;
443 int part_number = 1;
444
445 while (bytes_uploaded < total_bytes) {
446 std::size_t chunk =
447 (std::min)(part_size, total_bytes - bytes_uploaded);
448
449 auto stream = Aws::MakeShared<Aws::StringStream>("UploadPartStream");
450 stream->write(
451 reinterpret_cast<const char *>(data.data() + bytes_uploaded),
452 static_cast<std::streamsize>(chunk));
453
454 Aws::S3::Model::UploadPartRequest part_request;
455 part_request.SetBucket(bucket_);
456 part_request.SetKey(key);
457 part_request.SetUploadId(upload_id);
458 part_request.SetPartNumber(part_number);
459 part_request.SetBody(stream);
460 part_request.SetContentLength(static_cast<long long>(chunk));
461
462 auto part_outcome = client_->UploadPart(part_request);
463 if (!part_outcome.IsSuccess()) {
464 // Abort the multipart upload on failure
465 Aws::S3::Model::AbortMultipartUploadRequest abort_request;
466 abort_request.SetBucket(bucket_);
467 abort_request.SetKey(key);
468 abort_request.SetUploadId(upload_id);
469 client_->AbortMultipartUpload(abort_request);
470
471 return make_error<std::monostate>(
472 kUploadError,
473 "Failed to upload part " + std::to_string(part_number) + ": " +
474 std::string(part_outcome.GetError().GetMessage()),
475 "s3_storage");
476 }
477
478 Aws::S3::Model::CompletedPart completed_part;
479 completed_part.SetPartNumber(part_number);
480 completed_part.SetETag(part_outcome.GetResult().GetETag());
481 completed_upload.AddParts(std::move(completed_part));
482
483 bytes_uploaded += chunk;
484 part_number++;
485
486 if (callback && !callback(bytes_uploaded, total_bytes)) {
487 // Abort on user cancellation
488 Aws::S3::Model::AbortMultipartUploadRequest abort_request;
489 abort_request.SetBucket(bucket_);
490 abort_request.SetKey(key);
491 abort_request.SetUploadId(upload_id);
492 client_->AbortMultipartUpload(abort_request);
493
494 return make_error<std::monostate>(
495 kUploadError, "Upload cancelled by user", "s3_storage");
496 }
497 }
498
499 // Complete multipart upload
500 Aws::S3::Model::CompleteMultipartUploadRequest complete_request;
501 complete_request.SetBucket(bucket_);
502 complete_request.SetKey(key);
503 complete_request.SetUploadId(upload_id);
504 complete_request.SetMultipartUpload(completed_upload);
505
506 auto complete_outcome = client_->CompleteMultipartUpload(complete_request);
507 if (!complete_outcome.IsSuccess()) {
508 return make_error<std::monostate>(
509 kUploadError,
510 "Failed to complete multipart upload: " +
511 std::string(complete_outcome.GetError().GetMessage()),
512 "s3_storage");
513 }
514
515 return ok();
516 }
517
518private:
519 std::string bucket_;
520 std::unique_ptr<Aws::S3::S3Client> client_;
521 bool use_path_style_{false};
522};

550auto s3_storage::store_with_progress(const core::dicom_dataset &dataset,
551 progress_callback callback) -> VoidResult {
552 // Extract required UIDs
553 auto study_uid = dataset.get_string(core::tags::study_instance_uid);
554 auto series_uid = dataset.get_string(core::tags::series_instance_uid);
555 auto sop_uid = dataset.get_string(core::tags::sop_instance_uid);
556
557 if (study_uid.empty() || series_uid.empty() || sop_uid.empty()) {
558 return make_error<std::monostate>(
559 kMissingRequiredUid,
560 "Missing required UID (Study, Series, or SOP Instance UID)",
561 "s3_storage");
562 }
563
564 // Build S3 object key
565 auto object_key = build_object_key(study_uid, series_uid, sop_uid);
566
567 // Create DICOM file and serialize to bytes
568 auto dicom_file = core::dicom_file::create(
569 dataset, encoding::transfer_syntax::explicit_vr_little_endian);
570
571 auto data = dicom_file.to_bytes();
572 if (data.empty()) {
573 return make_error<std::monostate>(
574 kSerializationError, "Failed to serialize DICOM dataset", "s3_storage");
575 }
576
577 // Report initial progress
578 if (callback && !callback(0, data.size())) {
579 return make_error<std::monostate>(kUploadError, "Upload cancelled by user",
580 "s3_storage");
581 }
582
583 // Upload to S3 (use multipart for large files)
584 VoidResult upload_result = ok();
585 if (data.size() > config_.multipart_threshold) {
586 upload_result =
587 client_->multipart_upload(object_key, data, config_.part_size, callback);
588 } else {
589 upload_result = client_->put_object(object_key, data);
590
591 // Report completion progress
592 if (callback) {
593 callback(data.size(), data.size());
594 }
595 }
596
597 if (upload_result.is_err()) {
598 return upload_result;
599 }
600
601 // Update local index
602 {
603 std::unique_lock lock(mutex_);
604 s3_object_info info;
605 info.key = object_key;
606 info.sop_instance_uid = sop_uid;
607 info.study_instance_uid = study_uid;
608 info.series_instance_uid = series_uid;
609 info.size_bytes = data.size();
610 index_[sop_uid] = std::move(info);
611 }
612
613 return ok();
614}

621auto s3_storage::retrieve_with_progress(std::string_view sop_instance_uid,
622 progress_callback callback)
623 -> Result<core::dicom_dataset> {
624 std::string object_key;
625
626 {
627 std::shared_lock lock(mutex_);
628 auto it = index_.find(std::string{sop_instance_uid});
629 if (it == index_.end()) {
630 return make_error<core::dicom_dataset>(
631 kObjectNotFound,
632 "Instance not found: " + std::string{sop_instance_uid}, "s3_storage");
633 }
634 object_key = it->second.key;
635 }
636
637 // Download from S3
638 auto download_result = client_->get_object(object_key);
639 if (download_result.is_err()) {
640 return make_error<core::dicom_dataset>(
641 kDownloadError, "Failed to download from S3", "s3_storage");
642 }
643
644 const auto &data = download_result.value();
645
646 // Report progress (download complete)
647 if (callback) {
648 callback(data.size(), data.size());
649 }
650
651 // Deserialize DICOM data
652 auto parse_result = core::dicom_file::from_bytes(data);
653 if (parse_result.is_err()) {
654 return make_error<core::dicom_dataset>(
655 kSerializationError,
656 "Failed to parse DICOM data: " + parse_result.error().message,
657 "s3_storage");
658 }
659
660 return parse_result.value().dataset();
661}

663auto s3_storage::remove(std::string_view sop_instance_uid) -> VoidResult {
664 std::string object_key;
665
666 {
667 std::unique_lock lock(mutex_);
668 auto it = index_.find(std::string{sop_instance_uid});
669 if (it == index_.end()) {
670 // Not found is not an error for remove
671 return ok();
672 }
673 object_key = it->second.key;
674 index_.erase(it);
675 }
676
677 // Delete from S3
678 auto delete_result = client_->delete_object(object_key);
679 // Ignore delete errors - object might have been deleted externally
680
681 return ok();
682}

689auto s3_storage::find(const core::dicom_dataset &query)
690 -> Result<std::vector<core::dicom_dataset>> {
691 std::vector<core::dicom_dataset> results;
692
693 std::vector<std::string> keys_to_retrieve;
694 {
695 std::shared_lock lock(mutex_);
696 keys_to_retrieve.reserve(index_.size());
697 for (const auto &[uid, info] : index_) {
698 keys_to_retrieve.push_back(info.key);
699 }
700 }
701
702 for (const auto &key : keys_to_retrieve) {
703 auto download_result = client_->get_object(key);
704 if (download_result.is_err()) {
705 continue; // Skip objects that can't be downloaded
706 }
707
708 auto parse_result = core::dicom_file::from_bytes(download_result.value());
709 if (parse_result.is_err()) {
710 continue; // Skip invalid DICOM files
711 }
712
713 const auto &dataset = parse_result.value().dataset();
714 if (matches_query(dataset, query)) {
715 results.push_back(dataset);
716 }
717 }
718
719 return results;
720}

722auto s3_storage::get_statistics() const -> storage_statistics {
723 storage_statistics stats;
724
725 std::set<std::string> studies;
726 std::set<std::string> series;
727 std::set<std::string> patients;
728
729 {
730 std::shared_lock lock(mutex_);
731 stats.total_instances = index_.size();
732
733 for (const auto &[uid, info] : index_) {
734 stats.total_bytes += info.size_bytes;
735
736 if (!info.study_instance_uid.empty()) {
737 studies.insert(info.study_instance_uid);
738 }
739 if (!info.series_instance_uid.empty()) {
740 series.insert(info.series_instance_uid);
741 }
742 }
743 }
744
745 stats.studies_count = studies.size();
746 stats.series_count = series.size();
747 // Note: patient_count requires downloading datasets to extract PatientID
748
749 return stats;
750}

752auto s3_storage::verify_integrity() -> VoidResult {
753 std::vector<std::pair<std::string, std::string>> entries;
754 {
755 std::shared_lock lock(mutex_);
756 entries.reserve(index_.size());
757 for (const auto &[uid, info] : index_) {
758 entries.emplace_back(uid, info.key);
759 }
760 }
761
762 std::vector<std::string> invalid_entries;
763
764 for (const auto &[uid, key] : entries) {
765 if (!client_->head_object(key)) {
766 invalid_entries.push_back(uid + " (object missing)");
767 }
768 }
769
770 if (!invalid_entries.empty()) {
771 std::string message = "Integrity check failed for " +
772 std::to_string(invalid_entries.size()) + " entries";
773 return make_error<std::monostate>(kIntegrityError, message, "s3_storage");
774 }
775
776 return ok();
777}

797auto s3_storage::rebuild_index() -> VoidResult {
798 std::unique_lock lock(mutex_);
799 index_.clear();
800
801 // List all objects from S3
802 auto keys = client_->list_objects();
803
804 for (const auto &key : keys) {
805 // Download and parse each object to rebuild index
806 auto download_result = client_->get_object(key);
807 if (download_result.is_err()) {
808 continue;
809 }
810
811 auto parse_result = core::dicom_file::from_bytes(download_result.value());
812 if (parse_result.is_err()) {
813 continue;
814 }
815
816 const auto &dataset = parse_result.value().dataset();
817 auto sop_uid = dataset.get_string(core::tags::sop_instance_uid);
818 auto study_uid = dataset.get_string(core::tags::study_instance_uid);
819 auto series_uid = dataset.get_string(core::tags::series_instance_uid);
820
821 if (!sop_uid.empty()) {
822 s3_object_info info;
823 info.key = key;
824 info.sop_instance_uid = sop_uid;
825 info.study_instance_uid = study_uid;
826 info.series_instance_uid = series_uid;
827 info.size_bytes = client_->get_object_size(key);
828 index_[sop_uid] = std::move(info);
829 }
830 }
831
832 return ok();
833}

853auto s3_storage::sanitize_uid(std::string_view uid) -> std::string {
854 std::string result;
855 result.reserve(uid.length());
856
857 for (char c : uid) {
858 // UIDs contain digits and dots, which are safe for S3 keys
859 // Replace any other characters with underscore
860 if (std::isalnum(static_cast<unsigned char>(c)) || c == '.') {
861 result += c;
862 } else {
863 result += '_';
864 }
865 }
866
867 return result;

876auto s3_storage::matches_query(const core::dicom_dataset &dataset,
877 const core::dicom_dataset &query) -> bool {
878 // If query is empty, match all
879 if (query.empty()) {
880 return true;
881 }
882
883 // Check each query element
884 for (const auto &[tag, element] : query) {
885 auto query_value = element.as_string().unwrap_or("");
886 if (query_value.empty()) {
887 continue; // Empty value acts as wildcard
888 }
889
890 auto dataset_value = dataset.get_string(tag);
891
892 // Support basic wildcard matching (* and ?)
893 if (query_value.find('*') != std::string::npos ||
894 query_value.find('?') != std::string::npos) {
895 // Simple pattern matching
896 if (query_value.front() == '*' && query_value.back() == '*') {
897 // Contains
898 auto inner = query_value.substr(1, query_value.length() - 2);
899 if (dataset_value.find(inner) == std::string::npos) {
900 return false;
901 }
902 } else if (query_value.front() == '*') {
903 // Ends with
904 auto suffix = query_value.substr(1);
905 if (dataset_value.length() < suffix.length() ||
906 dataset_value.substr(dataset_value.length() - suffix.length()) !=
907 suffix) {
908 return false;
909 }
910 } else if (query_value.back() == '*') {
911 // Starts with
912 auto prefix = query_value.substr(0, query_value.length() - 1);
913 if (dataset_value.substr(0, prefix.length()) != prefix) {
914 return false;
915 }
916 }
917 } else {
918 // Exact match
919 if (dataset_value != query_value) {
920 return false;
921 }
922 }
923 }
924
925 return true;
926}
