69bool jpegxl_codec::can_encode(const image_params& params) const noexcept {
70 if (params.width == 0 || params.height == 0) {
71 return false;
72 }
73
74 if (params.samples_per_pixel != 1 && params.samples_per_pixel != 3) {
75 return false;
76 }
77
78 if (params.bits_stored < 1 || params.bits_stored > 16) {
79 return false;
80 }
81
82 return true;
83}
