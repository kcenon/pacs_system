113template <typename T>
114class Result {
115public:
116 Result(T value) : data_(std::move(value)), has_value_(true) {}
117 Result(const error_info& err) : error_(err), has_value_(false) {}
118
119 [[nodiscard]] bool is_ok() const noexcept { return has_value_; }
120 [[nodiscard]] bool is_err() const noexcept { return !has_value_; }
121 [[nodiscard]] T& value() & { return data_; }
122 [[nodiscard]] const T& value() const& { return data_; }
123 [[nodiscard]] T&& value() && { return std::move(data_); }
124 [[nodiscard]] const error_info& error() const { return error_; }
125
126private:
127 T data_{};
128 error_info error_;
129 bool has_value_;
130};
