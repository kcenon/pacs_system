72[[nodiscard]] inline auto to_string(ups_state state) -> std::string {
73 switch (state) {
74 case ups_state::scheduled:
75 return "SCHEDULED";
76 case ups_state::in_progress:
77 return "IN PROGRESS";
78 case ups_state::completed:
79 return "COMPLETED";
80 case ups_state::canceled:
81 return "CANCELED";
82 default:
83 return "SCHEDULED";
84 }
85}

93[[nodiscard]] inline auto parse_ups_state(std::string_view str)
94 -> std::optional<ups_state> {
95 if (str == "SCHEDULED") {
96 return ups_state::scheduled;
97 }
98 if (str == "IN PROGRESS") {
99 return ups_state::in_progress;
100 }
101 if (str == "COMPLETED") {
102 return ups_state::completed;
103 }
104 if (str == "CANCELED") {
105 return ups_state::canceled;
106 }
107 return std::nullopt;
108}
