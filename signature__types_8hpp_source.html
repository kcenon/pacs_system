94inline std::optional<signature_algorithm> parse_signature_algorithm(std::string_view str) {
95 if (str == "RSA-SHA256") return signature_algorithm::rsa_sha256;
96 if (str == "RSA-SHA384") return signature_algorithm::rsa_sha384;
97 if (str == "RSA-SHA512") return signature_algorithm::rsa_sha512;
98 if (str == "ECDSA-SHA256") return signature_algorithm::ecdsa_sha256;
99 if (str == "ECDSA-SHA384") return signature_algorithm::ecdsa_sha384;
100 return std::nullopt;
101}
