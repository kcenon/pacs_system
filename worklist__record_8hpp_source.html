56enum class worklist_status {
57 scheduled,
58 started,
59 completed
60};

68[[nodiscard]] inline auto to_string(worklist_status status) -> std::string {
69 switch (status) {
70 case worklist_status::scheduled:
71 return "SCHEDULED";
72 case worklist_status::started:
73 return "STARTED";
74 case worklist_status::completed:
75 return "COMPLETED";
76 default:
77 return "SCHEDULED";
78 }
79}
80
87[[nodiscard]] inline auto parse_worklist_status(std::string_view str)
88 -> std::optional<worklist_status> {
89 if (str == "SCHEDULED") {
90 return worklist_status::scheduled;
91 }
92 if (str == "STARTED") {
93 return worklist_status::started;
94 }
95 if (str == "COMPLETED") {
96 return worklist_status::completed;
97 }
98 return std::nullopt;
99}
