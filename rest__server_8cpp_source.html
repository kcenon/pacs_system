225void rest_server::start() {
226 if (impl_->running.exchange(true)) {
227 return; // Already running
228 }
229
230 impl_->app = std::make_unique<crow::SimpleApp>();
231 auto &app = *impl_->app;
232
233 // Register system endpoints
234 endpoints::register_system_endpoints_impl(app, impl_->context);
235
236 // Add CORS preflight handler
237 if (impl_->config.enable_cors) {
238 CROW_ROUTE(app, "/api/<path>")
239 .methods(crow::HTTPMethod::OPTIONS)(
240 [this](const crow::request & /*req*/,
241 const std::string & /*path*/) {
242 crow::response res(204);
243 res.add_header("Access-Control-Allow-Origin",
244 impl_->config.cors_allowed_origins);
245 res.add_header("Access-Control-Allow-Methods",
246 "GET, POST, PUT, DELETE, OPTIONS");
247 res.add_header("Access-Control-Allow-Headers",
248 "Content-Type, Authorization");
249 res.add_header("Access-Control-Max-Age", "86400");
250 return res;
251 });
252 }
253
254 // Start server
255 app.bindaddr(impl_->config.bind_address)
256 .port(impl_->config.port)
257 .concurrency(static_cast<uint16_t>(impl_->config.concurrency))
258 .run();
259
260 impl_->running = false;
261}

263void rest_server::start_async() {
264 if (impl_->running.exchange(true)) {
265 return; // Already running
266 }
267
268 impl_->server_thread = std::thread([this]() {
269 impl_->app = std::make_unique<crow::SimpleApp>();
270 auto &app = *impl_->app;
271
272 // Register endpoints
273 endpoints::register_system_endpoints_impl(app, impl_->context);
274 endpoints::register_security_endpoints_impl(app, impl_->context);
275 endpoints::register_patient_endpoints_impl(app, impl_->context);
276 endpoints::register_study_endpoints_impl(app, impl_->context);
277 endpoints::register_series_endpoints_impl(app, impl_->context);
278 endpoints::register_worklist_endpoints_impl(app, impl_->context);
279 endpoints::register_audit_endpoints_impl(app, impl_->context);
280 endpoints::register_association_endpoints_impl(app, impl_->context);
281 endpoints::register_dicomweb_endpoints_impl(app, impl_->context);
282 endpoints::register_remote_nodes_endpoints_impl(app, impl_->context);
283 endpoints::register_jobs_endpoints_impl(app, impl_->context);
284 endpoints::register_routing_endpoints_impl(app, impl_->context);
285 endpoints::register_thumbnail_endpoints_impl(app, impl_->context);
286 endpoints::register_metadata_endpoints_impl(app, impl_->context);
287 endpoints::register_annotation_endpoints_impl(app, impl_->context);
288 endpoints::register_measurement_endpoints_impl(app, impl_->context);
289 endpoints::register_key_image_endpoints_impl(app, impl_->context);
290 endpoints::register_viewer_state_endpoints_impl(app, impl_->context);
291 endpoints::register_wado_uri_endpoints_impl(app, impl_->context);
292 endpoints::register_storage_commitment_endpoints_impl(app, impl_->context);
293
294#ifdef PACS_WITH_DATABASE_SYSTEM
295 endpoints::register_metrics_endpoints_impl(app, impl_->context);
296#endif
297
298 // Add CORS preflight handler
299 if (impl_->config.enable_cors) {
300 CROW_ROUTE(app, "/api/<path>")
301 .methods(crow::HTTPMethod::OPTIONS)(
302 [this](const crow::request & /*req*/,
303 const std::string & /*path*/) {
304 crow::response res(204);
305 res.add_header("Access-Control-Allow-Origin",
306 impl_->config.cors_allowed_origins);
307 res.add_header("Access-Control-Allow-Methods",
308 "GET, POST, PUT, DELETE, OPTIONS");
309 res.add_header("Access-Control-Allow-Headers",
310 "Content-Type, Authorization");
311 res.add_header("Access-Control-Max-Age", "86400");
312 return res;
313 });
314 }
315
316 // Start server
317 app.bindaddr(impl_->config.bind_address)
318 .port(impl_->config.port)
319 .concurrency(static_cast<uint16_t>(impl_->config.concurrency))
320 .run();
321
322 impl_->running = false;
323 });
324}

326void rest_server::stop() {
327 if (!impl_->running) {
328 return;
329 }
330
331 if (impl_->app) {
332 impl_->app->stop();
333 }
334
335 if (impl_->server_thread.joinable()) {
336 impl_->server_thread.join();
337 }
338
339 impl_->running = false;
340}
341
342bool rest_server::is_running() const noexcept { return impl_->running; }

344void rest_server::wait() {
345 if (impl_->server_thread.joinable()) {
346 impl_->server_thread.join();
347 }
348}

351 return impl_->running ? impl_->config.port : 0;
