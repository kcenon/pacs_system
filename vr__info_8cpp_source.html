87// Build lookup map at runtime
88const std::unordered_map<vr_type, const vr_info*>& get_vr_map() {
89 static const auto map = []() {
90 std::unordered_map<vr_type, const vr_info*> m;
91 for (const auto& info : vr_table) {
92 m[info.type] = &info;
93 }
94 m[vr_type::AT] = &at_info;
95 m[vr_type::SQ] = &sq_info;
96 m[vr_type::UN] = &un_info;
97 return m;
98 }();
99 return map;
100}

102// Character validation helpers
103bool is_cs_char(char c) {
104 return std::isupper(static_cast<unsigned char>(c)) ||
105 std::isdigit(static_cast<unsigned char>(c)) ||
106 c == ' ' || c == '_';
107}
108
109bool is_da_char(char c) {
110 return std::isdigit(static_cast<unsigned char>(c));
111}
112
113bool is_tm_char(char c) {
114 return std::isdigit(static_cast<unsigned char>(c)) ||
115 c == '.' || c == ':';
116}
117
118bool is_ui_char(char c) {
119 return std::isdigit(static_cast<unsigned char>(c)) || c == '.';
120}
121
122bool is_ds_char(char c) {
123 return std::isdigit(static_cast<unsigned char>(c)) ||
124 c == '+' || c == '-' || c == '.' ||
125 c == 'E' || c == 'e' || c == ' ';
126}
127
128bool is_is_char(char c) {
129 return std::isdigit(static_cast<unsigned char>(c)) ||
130 c == '+' || c == '-';
131}
132
133bool is_as_char(char c) {
134 return std::isdigit(static_cast<unsigned char>(c)) ||
135 c == 'D' || c == 'W' || c == 'M' || c == 'Y';
136}
137
138bool is_dt_char(char c) {
139 return std::isdigit(static_cast<unsigned char>(c)) ||
140 c == '.' || c == '+' || c == '-';
141}
142
143bool is_printable(char c) {
144 auto uc = static_cast<unsigned char>(c);
145 // Allow printable ASCII (0x20-0x7E), CR, LF, FF, TAB for text VRs,
146 // ESC (0x1B) for ISO 2022 escape sequences, and high bytes (0x80+)
147 // for multi-byte CJK encodings (EUC-KR, GB2312, ISO-2022-JP, etc.)
148 return (uc >= 0x20 && uc <= 0x7E) ||
149 uc >= 0x80 ||
150 uc == 0x1B ||
151 c == '\r' || c == '\n' || c == '\f' || c == '\t';
152}

156const vr_info& get_vr_info(vr_type vr) {
157 const auto& map = get_vr_map();
158 auto it = map.find(vr);
159 if (it != map.end()) {
160 return *it->second;
161 }
162 // Return UN info for unknown VR types
163 return un_info;
164}

166bool validate_value(vr_type vr, std::span<const uint8_t> data) {
167 const auto& info = get_vr_info(vr);
168
169 // For fixed-length VRs, check that size is a multiple of fixed_size (VM >= 1)
170 if (info.is_fixed_length && info.fixed_size > 0) {
171 if (data.size() % info.fixed_size != 0) {
172 return false;
173 }
174 // Fixed-length VRs pass validation if size is correct multiple
175 return true;
176 }
177
178 // Check maximum length for variable-length VRs (0xFFFFFFFE means unlimited)
179 if (info.max_length != 0xFFFFFFFE && info.max_length != 0xFFFFFFFF) {
180 if (data.size() > info.max_length) {
181 return false;
182 }
183 }
184
185 // String VRs need character validation
186 if (is_string_vr(vr)) {
187 std::string_view str_value(reinterpret_cast<const char*>(data.data()),
188 data.size());
189 return is_valid_charset(vr, str_value);
190 }
191
192 return true;
193}

195bool validate_string(vr_type vr, std::string_view value) {
196 const auto& info = get_vr_info(vr);
197
198 // Non-string VRs cannot be validated as strings
199 if (!is_string_vr(vr)) {
200 return false;
201 }
202
203 // Check maximum length
204 if (info.max_length != 0xFFFFFFFE) {
205 if (value.size() > info.max_length) {
206 return false;
207 }
208 }
209
210 // Fixed-length string VRs (AS, DA) must match exact length
211 if (info.is_fixed_length && info.fixed_size > 0) {
212 if (value.size() != info.fixed_size) {
213 return false;
214 }
215 }
216
217 // Validate character set
218 return is_valid_charset(vr, value);
219}

221std::vector<uint8_t> pad_to_even(vr_type vr, std::span<const uint8_t> data) {
222 std::vector<uint8_t> result(data.begin(), data.end());
223
224 if (result.size() % 2 != 0) {
225 const auto& info = get_vr_info(vr);
226 result.push_back(static_cast<uint8_t>(info.padding_char));
227 }
228
229 return result;
230}

232std::string trim_padding(vr_type vr, std::string_view value) {
233 if (value.empty()) {
234 return std::string{};
235 }
236
237 const auto& info = get_vr_info(vr);
238 char pad = info.padding_char;
239
240 // Find the last non-padding character
241 auto end = value.find_last_not_of(pad);
242 if (end == std::string_view::npos) {
243 // All characters are padding
244 return std::string{};
245 }
246
247 return std::string{value.substr(0, end + 1)};
248}

250bool is_valid_charset(vr_type vr, std::string_view value) {
251 switch (vr) {
252 case vr_type::CS:
253 // Code String: A-Z, 0-9, space, underscore
254 return std::all_of(value.begin(), value.end(), is_cs_char);
255
256 case vr_type::DA:
257 // Date: YYYYMMDD format, digits only
258 return value.size() == 8 &&
259 std::all_of(value.begin(), value.end(), is_da_char);
260
261 case vr_type::TM:
262 // Time: digits, period, colon
263 return std::all_of(value.begin(), value.end(), is_tm_char);
264
265 case vr_type::UI:
266 // UID: digits and periods only, max 64 chars
267 return value.size() <= 64 &&
268 std::all_of(value.begin(), value.end(), is_ui_char);
269
270 case vr_type::DS:
271 // Decimal String: digits, +, -, ., E, e, space
272 return std::all_of(value.begin(), value.end(), is_ds_char);
273
274 case vr_type::IS:
275 // Integer String: digits, +, -
276 return std::all_of(value.begin(), value.end(), is_is_char);
277
278 case vr_type::AS:
279 // Age String: nnnX format where X is D/W/M/Y
280 if (value.size() != 4) {
281 return false;
282 }
283 return std::all_of(value.begin(), value.end(), is_as_char) &&
284 (value[3] == 'D' || value[3] == 'W' ||
285 value[3] == 'M' || value[3] == 'Y');
286
287 case vr_type::DT:
288 // Date Time: digits, period, +, -
289 return std::all_of(value.begin(), value.end(), is_dt_char);
290
291 case vr_type::AE:
292 case vr_type::LO:
293 case vr_type::SH:
294 case vr_type::PN:
295 // These allow most printable characters but no control chars
296 // except leading/trailing spaces
297 return std::all_of(value.begin(), value.end(), is_printable);
298
299 case vr_type::LT:
300 case vr_type::ST:
301 case vr_type::UT:
302 case vr_type::UC:
303 case vr_type::UR:
304 // Text VRs allow all printable plus CR, LF, FF, TAB
305 return std::all_of(value.begin(), value.end(), is_printable);
306
307 default:
308 // Non-string VRs - charset validation not applicable
309 return true;
310 }
