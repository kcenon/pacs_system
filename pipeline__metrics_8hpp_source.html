60struct stage_metrics {
62 std::atomic<uint64_t> jobs_processed{0};
63
65 std::atomic<uint64_t> jobs_queued{0};
66
68 std::atomic<uint64_t> jobs_failed{0};
69
71 std::atomic<uint64_t> total_processing_time_ns{0};
72
74 std::atomic<uint64_t> max_processing_time_ns{0};
75
77 std::atomic<uint32_t> active_workers{0};
78
80 std::atomic<uint32_t> idle_workers{0};
81
87 void record_job_completion(uint64_t processing_time_ns, bool success) noexcept {
88 jobs_processed.fetch_add(1, std::memory_order_relaxed);
89 total_processing_time_ns.fetch_add(processing_time_ns, std::memory_order_relaxed);
90
91 if (!success) {
92 jobs_failed.fetch_add(1, std::memory_order_relaxed);
93 }
94
95 // Update max processing time (compare-exchange loop)
96 uint64_t current_max = max_processing_time_ns.load(std::memory_order_relaxed);
97 while (processing_time_ns > current_max &&
98 !max_processing_time_ns.compare_exchange_weak(
99 current_max, processing_time_ns,
100 std::memory_order_relaxed, std::memory_order_relaxed)) {
101 // Loop until successful or processing_time_ns <= current_max
102 }
103 }
104
109 [[nodiscard]] auto get_avg_processing_time_ns() const noexcept -> uint64_t {
110 uint64_t processed = jobs_processed.load(std::memory_order_relaxed);
111 if (processed == 0) {
112 return 0;
113 }
114 return total_processing_time_ns.load(std::memory_order_relaxed) / processed;
115 }
116
120 void reset() noexcept {
121 jobs_processed.store(0, std::memory_order_relaxed);
122 jobs_queued.store(0, std::memory_order_relaxed);
123 jobs_failed.store(0, std::memory_order_relaxed);
124 total_processing_time_ns.store(0, std::memory_order_relaxed);
125 max_processing_time_ns.store(0, std::memory_order_relaxed);
126 }
127};

133struct category_metrics {
135 std::atomic<uint64_t> total_operations{0};
136
138 std::atomic<uint64_t> successful_operations{0};
139
141 std::atomic<uint64_t> failed_operations{0};
142
144 std::atomic<uint64_t> total_latency_ns{0};
145
147 std::atomic<uint64_t> max_latency_ns{0};
148
150 std::atomic<uint64_t> min_latency_ns{UINT64_MAX};
151
157 void record_operation(uint64_t latency_ns, bool success) noexcept {
158 total_operations.fetch_add(1, std::memory_order_relaxed);
159 total_latency_ns.fetch_add(latency_ns, std::memory_order_relaxed);
160
161 if (success) {
162 successful_operations.fetch_add(1, std::memory_order_relaxed);
163 } else {
164 failed_operations.fetch_add(1, std::memory_order_relaxed);
165 }
166
167 // Update max latency
168 uint64_t current_max = max_latency_ns.load(std::memory_order_relaxed);
169 while (latency_ns > current_max &&
170 !max_latency_ns.compare_exchange_weak(
171 current_max, latency_ns,
172 std::memory_order_relaxed, std::memory_order_relaxed)) {
173 }
174
175 // Update min latency
176 uint64_t current_min = min_latency_ns.load(std::memory_order_relaxed);
177 while (latency_ns < current_min &&
178 !min_latency_ns.compare_exchange_weak(
179 current_min, latency_ns,
180 std::memory_order_relaxed, std::memory_order_relaxed)) {
181 }
182 }
183
187 [[nodiscard]] auto get_avg_latency_ns() const noexcept -> uint64_t {
188 uint64_t total = total_operations.load(std::memory_order_relaxed);
189 if (total == 0) {
190 return 0;
191 }
192 return total_latency_ns.load(std::memory_order_relaxed) / total;
193 }
194
198 void reset() noexcept {
199 total_operations.store(0, std::memory_order_relaxed);
200 successful_operations.store(0, std::memory_order_relaxed);
201 failed_operations.store(0, std::memory_order_relaxed);
202 total_latency_ns.store(0, std::memory_order_relaxed);
203 max_latency_ns.store(0, std::memory_order_relaxed);
204 min_latency_ns.store(UINT64_MAX, std::memory_order_relaxed);
205 }
206};

355 void increment_active_associations() noexcept {
356 uint32_t current = active_associations_.fetch_add(1, std::memory_order_relaxed);
357 // Update peak if necessary
358 uint32_t peak = peak_associations_.load(std::memory_order_relaxed);
359 while (current + 1 > peak &&
360 !peak_associations_.compare_exchange_weak(
361 peak, current + 1,
362 std::memory_order_relaxed, std::memory_order_relaxed)) {
363 }
364 }

371 }

410 [[nodiscard]] auto get_throughput_per_second(job_category category) const noexcept
411 -> double {
412 auto elapsed = std::chrono::steady_clock::now() - start_time_;
413 auto seconds = std::chrono::duration<double>(elapsed).count();
414 if (seconds <= 0.0) {
415 return 0.0;
416 }
417 return static_cast<double>(
418 category_metrics_[static_cast<size_t>(category)]
419 .total_operations.load(std::memory_order_relaxed)) / seconds;
420 }

422private:

425 std::atomic<uint64_t> total_operations_{0};
426 std::atomic<uint32_t> active_associations_{0};
427 std::atomic<uint32_t> peak_associations_{0};
428 std::chrono::steady_clock::time_point start_time_{std::chrono::steady_clock::now()};
429};
