78thumbnail_params parse_thumbnail_params(const crow::request& req) {
79 thumbnail_params params;
80
81 // Parse size
82 auto size_param = req.url_params.get("size");
83 if (size_param != nullptr) {
84 try {
85 int size = std::stoi(size_param);
86 if (size == 64 || size == 128 || size == 256 || size == 512) {
87 params.size = static_cast<uint16_t>(size);
88 }
89 } catch (...) {
90 // Use default
91 }
92 }
93
94 // Parse format
95 auto format_param = req.url_params.get("format");
96 if (format_param != nullptr) {
97 std::string format = format_param;
98 if (format == "jpeg" || format == "png") {
99 params.format = format;
100 }
101 }
102
103 // Parse quality
104 auto quality_param = req.url_params.get("quality");
105 if (quality_param != nullptr) {
106 try {
107 int quality = std::stoi(quality_param);
108 if (quality >= 1 && quality <= 100) {
109 params.quality = quality;
110 }
111 } catch (...) {
112 // Use default
113 }
114 }
115
116 // Parse frame
117 auto frame_param = req.url_params.get("frame");
118 if (frame_param != nullptr) {
119 try {
120 int frame = std::stoi(frame_param);
121 if (frame >= 1) {
122 params.frame = static_cast<uint32_t>(frame);
123 }
124 } catch (...) {
125 // Use default
126 }
127 }
128
129 return params;
130}

146 // GET /api/v1/thumbnails/instances/{sopInstanceUid}
147 CROW_ROUTE(app, "/api/v1/thumbnails/instances/<string>")
148 .methods(crow::HTTPMethod::GET)(
149 [ctx](const crow::request& req, const std::string& sop_uid) {
150 crow::response res;
151 add_cors_headers(res, *ctx);
152
153 if (g_thumbnail_service == nullptr) {
154 res.code = 503;
155 res.add_header("Content-Type", "application/json");
156 res.body = make_error_json("SERVICE_UNAVAILABLE",
157 "Thumbnail service not configured");
158 return res;
159 }
160
161 auto params = parse_thumbnail_params(req);
162 auto result = g_thumbnail_service->get_thumbnail(sop_uid, params);
163
164 if (!result.success) {
165 res.code = 404;
166 res.add_header("Content-Type", "application/json");
167 res.body =
168 make_error_json("NOT_FOUND", result.error_message);
169 return res;
170 }
171
172 res.code = 200;
173 res.add_header("Content-Type", result.entry.content_type);
174 res.add_header("Cache-Control", "max-age=3600");
175 res.body = std::string(
176 reinterpret_cast<const char*>(result.entry.data.data()),
177 result.entry.data.size());
178 return res;
179 });

254 CROW_ROUTE(app, "/api/v1/thumbnails/cache")
255 .methods(crow::HTTPMethod::DELETE)([ctx](const crow::request& /*req*/) {
256 crow::response res;
257 res.add_header("Content-Type", "application/json");
258 add_cors_headers(res, *ctx);

264 return res;
265 }
266
267 g_thumbnail_service->clear_cache();
268
269 res.code = 200;
270 res.body = make_success_json("Cache cleared successfully");
271 return res;
272 });
