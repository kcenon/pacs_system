120template <typename T>
121class Result {
122public:
123 Result(T value) : data_(std::move(value)), has_value_(true) {}
124 Result(const error_info& err) : error_(err), has_value_(false) {}
125
126 [[nodiscard]] bool is_ok() const noexcept { return has_value_; }
127 [[nodiscard]] bool is_err() const noexcept { return !has_value_; }
128 [[nodiscard]] T& value() & { return data_; }
129 [[nodiscard]] const T& value() const& { return data_; }
130 [[nodiscard]] T&& value() && { return std::move(data_); }
131 [[nodiscard]] const error_info& error() const { return error_; }
132
133private:
134 T data_{};
135 error_info error_;
136 bool has_value_;
137};
