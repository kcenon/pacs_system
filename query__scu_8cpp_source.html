68network::Result<query_result> query_scu::find_impl(
69 network::association& assoc,
70 const core::dicom_dataset& query_keys,
71 uint16_t message_id) {
72
73 using namespace network::dimse;
74
75 auto start_time = std::chrono::steady_clock::now();
76
77 // Verify association is established
78 if (!assoc.is_established()) {
79 return kcenon::pacs::pacs_error<query_result>(
80 kcenon::pacs::error_codes::association_not_established,
81 "Association not established");
82 }
83
84 // Get SOP Class UID
85 auto sop_class_uid = get_sop_class_uid();
86
87 // Get accepted presentation context for this SOP class
88 auto context_id = assoc.accepted_context_id(sop_class_uid);
89 if (!context_id) {
90 return kcenon::pacs::pacs_error<query_result>(
91 kcenon::pacs::error_codes::no_acceptable_context,
92 "No accepted presentation context for SOP Class: " +
93 std::string(sop_class_uid));
94 }
95
96 // Build C-FIND-RQ message
97 auto request = make_c_find_rq(message_id, sop_class_uid);
98 request.set_dataset(query_keys);
99
100 logger_->debug_fmt("Sending C-FIND request (message_id={}, sop_class={})",
101 message_id, sop_class_uid);
102
103 // Send the request
104 auto send_result = assoc.send_dimse(*context_id, request);
105 if (send_result.is_err()) {
106 return send_result.error();
107 }
108
109 // Receive responses
110 query_result result;
111 bool query_complete = false;
112
113 while (!query_complete) {
114 auto recv_result = assoc.receive_dimse(config_.timeout);
115 if (recv_result.is_err()) {
116 return recv_result.error();
117 }
118
119 const auto& [recv_context_id, response] = recv_result.value();
120
121 // Verify it's a C-FIND response
122 if (response.command() != command_field::c_find_rsp) {
123 return kcenon::pacs::pacs_error<query_result>(
124 kcenon::pacs::error_codes::find_unexpected_command,
125 "Expected C-FIND-RSP but received " +
126 std::string(to_string(response.command())));
127 }
128
129 auto status = response.status();
130
131 if (status == status_pending || status == status_pending_warning) {
132 ++result.total_pending;
133
134 if (response.has_dataset()) {
135 // Check if we should collect this result
136 if (config_.max_results == 0 ||
137 result.matches.size() < config_.max_results) {
138
139 auto dataset_result = response.dataset();
140 if (dataset_result.is_ok()) {
141 result.matches.push_back(dataset_result.value().get());
142 }
143 }
144
145 // Check if we should cancel due to max_results
146 if (config_.max_results > 0 &&
147 result.matches.size() >= config_.max_results &&
148 config_.cancel_on_max) {
149
150 logger_->debug_fmt(
151 "Max results ({}) reached, sending C-CANCEL",
152 config_.max_results);
153
154 // Send C-CANCEL
155 auto cancel_result = cancel(assoc, message_id);
156 if (cancel_result.is_err()) {
157 logger_->warn_fmt("Failed to send C-CANCEL: {}",
158 cancel_result.error().message);
159 }
160 }
161 }
162 } else if (status == status_success) {
163 query_complete = true;
164 result.status = static_cast<uint16_t>(status);
165 } else if (status == status_cancel) {
166 query_complete = true;
167 result.status = static_cast<uint16_t>(status);
168 } else {
169 // Error status
170 query_complete = true;
171 result.status = static_cast<uint16_t>(status);
172 }
173 }
174
175 auto end_time = std::chrono::steady_clock::now();
176 result.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
177 end_time - start_time);
178
179 // Update statistics
180 queries_performed_.fetch_add(1, std::memory_order_relaxed);
181 total_matches_.fetch_add(result.matches.size(), std::memory_order_relaxed);
182
183 logger_->debug_fmt("C-FIND completed: {} matches in {} ms",
184 result.matches.size(), result.elapsed.count());
185
186 return result;
187}

189network::Result<size_t> query_scu::find_streaming(
190 network::association& assoc,
191 const core::dicom_dataset& query_keys,
192 query_streaming_callback callback) {
193
194 using namespace network::dimse;
195
196 auto message_id = next_message_id();
197 auto start_time = std::chrono::steady_clock::now();
198
199 // Verify association is established
200 if (!assoc.is_established()) {
201 return kcenon::pacs::pacs_error<size_t>(
202 kcenon::pacs::error_codes::association_not_established,
203 "Association not established");
204 }
205
206 // Get SOP Class UID
207 auto sop_class_uid = get_sop_class_uid();
208
209 // Get accepted presentation context for this SOP class
210 auto context_id = assoc.accepted_context_id(sop_class_uid);
211 if (!context_id) {
212 return kcenon::pacs::pacs_error<size_t>(
213 kcenon::pacs::error_codes::no_acceptable_context,
214 "No accepted presentation context for SOP Class: " +
215 std::string(sop_class_uid));
216 }
217
218 // Build C-FIND-RQ message
219 auto request = make_c_find_rq(message_id, sop_class_uid);
220 request.set_dataset(query_keys);
221
222 logger_->debug_fmt("Sending streaming C-FIND request (message_id={})",
223 message_id);
224
225 // Send the request
226 auto send_result = assoc.send_dimse(*context_id, request);
227 if (send_result.is_err()) {
228 return send_result.error();
229 }
230
231 // Receive responses
232 size_t count = 0;
233 bool query_complete = false;
234 bool should_cancel = false;
235
236 while (!query_complete) {
237 auto recv_result = assoc.receive_dimse(config_.timeout);
238 if (recv_result.is_err()) {
239 return recv_result.error();
240 }
241
242 const auto& [recv_context_id, response] = recv_result.value();
243
244 // Verify it's a C-FIND response
245 if (response.command() != command_field::c_find_rsp) {
246 return kcenon::pacs::pacs_error<size_t>(
247 kcenon::pacs::error_codes::find_unexpected_command,
248 "Expected C-FIND-RSP but received " +
249 std::string(to_string(response.command())));
250 }
251
252 auto status = response.status();
253
254 if (status == status_pending || status == status_pending_warning) {
255 if (response.has_dataset()) {
256 auto dataset_result = response.dataset();
257 if (dataset_result.is_ok()) {
258 ++count;
259
260 // Call the callback
261 if (callback) {
262 if (!callback(dataset_result.value().get())) {
263 should_cancel = true;
264 }
265 }
266
267 // Check if we need to cancel
268 if (should_cancel ||
269 (config_.max_results > 0 &&
270 count >= config_.max_results &&
271 config_.cancel_on_max)) {
272
273 logger_->debug("Cancelling streaming query");
274 auto cancel_result = cancel(assoc, message_id);
275 if (cancel_result.is_err()) {
276 logger_->warn_fmt("Failed to send C-CANCEL: {}",
277 cancel_result.error().message);
278 }
279 should_cancel = false; // Already sent cancel
280 }
281 }
282 }
283 } else {
284 query_complete = true;
285 }
286 }
287
288 // Update statistics
289 queries_performed_.fetch_add(1, std::memory_order_relaxed);
290 total_matches_.fetch_add(count, std::memory_order_relaxed);
291
292 auto end_time = std::chrono::steady_clock::now();
293 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
294 end_time - start_time);
295
296 logger_->debug_fmt("Streaming C-FIND completed: {} results in {} ms",
297 count, elapsed.count());
298
299 return count;
300}

366network::Result<std::monostate> query_scu::cancel(
367 network::association& assoc,
368 uint16_t message_id) {
369
370 using namespace network::dimse;
371
372 auto sop_class_uid = get_sop_class_uid();
373
374 auto context_id = assoc.accepted_context_id(sop_class_uid);
375 if (!context_id) {
376 return kcenon::pacs::pacs_error<std::monostate>(
377 kcenon::pacs::error_codes::no_acceptable_context,
378 "No accepted presentation context for cancel");
379 }
380
381 // Build C-CANCEL-RQ manually since there's no factory function
382 dimse_message cancel_rq{command_field::c_cancel_rq, message_id};
383 return assoc.send_dimse(*context_id, cancel_rq);
384}

419uint16_t query_scu::next_message_id() noexcept {
420 uint16_t id = message_id_counter_.fetch_add(1, std::memory_order_relaxed);
421 // Wrap around at 0xFFFF, skip 0 (reserved)
422 if (id == 0) {
423 id = message_id_counter_.fetch_add(1, std::memory_order_relaxed);
424 }
425 return id;
426}

428core::dicom_dataset query_scu::build_query_dataset(
429 const patient_query_keys& keys) const {
430
431 using namespace core;
432 using namespace encoding;
433
434 dicom_dataset ds;
435
436 // Set Query/Retrieve Level
437 ds.set_string(tags::query_retrieve_level, vr_type::CS, "PATIENT");
438
439 // Set query keys (empty string = return key, non-empty = match key)
440 ds.set_string(tags::patient_name, vr_type::PN, keys.patient_name);
441 ds.set_string(tags::patient_id, vr_type::LO, keys.patient_id);
442 ds.set_string(tags::patient_birth_date, vr_type::DA, keys.birth_date);
443 ds.set_string(tags::patient_sex, vr_type::CS, keys.sex);
444
445 // Add standard return keys
446 ds.set_string(tags::number_of_patient_related_studies, vr_type::IS, "");
447 ds.set_string(tags::number_of_patient_related_series, vr_type::IS, "");
448 ds.set_string(tags::number_of_patient_related_instances, vr_type::IS, "");
449
450 return ds;
451}
