144 [[nodiscard]] constexpr auto is_private() const noexcept -> bool {

165 [[nodiscard]] constexpr auto is_private_creator() const noexcept -> bool {
166 if (!is_private()) {
167 return false;
168 }
169 const auto elem = element();

171 }

181 [[nodiscard]] constexpr auto is_private_data() const noexcept -> bool {
182 return is_private() && element() > 0x00FF;
183 }

194 [[nodiscard]] constexpr auto private_block_number() const noexcept
195 -> std::optional<uint8_t> {
196 if (!is_private_data()) {

198 }

200 }

211 [[nodiscard]] constexpr auto private_creator_tag() const noexcept
212 -> std::optional<dicom_tag> {
213 const auto block = private_block_number();
214 if (!block) {
215 return std::nullopt;
216 }
217 return dicom_tag{group(), *block};

228 [[nodiscard]] constexpr auto private_data_range() const noexcept
229 -> std::optional<std::pair<dicom_tag, dicom_tag>> {

234 return std::pair{
235 dicom_tag{group(), block},
236 dicom_tag{group(), static_cast<uint16_t>(block | 0x00FF)}
