64auto dimse_process_job::execute(pipeline_coordinator& coordinator) -> VoidResult {
65 // Handle association PDUs differently
66 if (pdu_.type != kcenon::pacs::network::pdu_type::p_data_tf) {
67 return process_association_pdu();
68 }
69
70 // Process P-DATA-TF (DIMSE message)
71 auto request_result = process_p_data();
72 if (!request_result.is_ok()) {
73 if (on_error_) {
74 auto err = request_result.error();
75 on_error_(context_.session_id, err.message);
76 }
77 return VoidResult(request_result.error());
78 }
79
80 auto request = request_result.value();
81
82 // Update context with message ID and category
83 context_.message_id = request.message_id;
84
85 // Determine category from command type
86 switch (request.command_type) {
87 case dimse_command_type::c_echo_rq:
88 case dimse_command_type::c_echo_rsp:
89 context_.category = job_category::echo;
90 break;
91 case dimse_command_type::c_store_rq:
92 case dimse_command_type::c_store_rsp:
93 context_.category = job_category::store;
94 break;
95 case dimse_command_type::c_find_rq:
96 case dimse_command_type::c_find_rsp:
97 context_.category = job_category::find;
98 break;
99 case dimse_command_type::c_get_rq:
100 case dimse_command_type::c_get_rsp:
101 context_.category = job_category::get;
102 break;
103 case dimse_command_type::c_move_rq:
104 case dimse_command_type::c_move_rsp:
105 context_.category = job_category::move;
106 break;
107 default:
108 context_.category = job_category::other;
109 break;
110 }
111
112 // Invoke callback if set
113 if (on_request_) {
114 on_request_(request);
115 }
116
117 // For response messages, skip execution stage
118 uint16_t cmd_type = static_cast<uint16_t>(request.command_type);
119 if (cmd_type & 0x8000) {
120 // This is a response, not a request - don't submit to execution
121 return ok();
122 }
123
124 // Create execution job for request
125 // Note: The actual service handler would be set by the adapter layer
126 auto exec_job = std::make_unique<storage_query_exec_job>(
127 std::move(request),
128 nullptr // Handler will be set by adapter
129 );
130
131 // Copy context to execution job
132 exec_job->get_context() = context_;
133 exec_job->get_context().stage = pipeline_stage::storage_query_exec;
134
135 return coordinator.submit_to_stage(
136 pipeline_stage::storage_query_exec,
137 std::move(exec_job)
138 );
139}

159auto dimse_process_job::process_p_data() -> Result<dimse_request> {
160 dimse_request request;
161 request.session_id = pdu_.session_id;
162 request.presentation_context_id = pdu_.presentation_context_id;
163
164 // PDV data starts after length (4) + context ID (1) + control header (1)
165 if (pdu_.data.size() < 6) {
166 return pacs_error<dimse_request>(error_codes::insufficient_data,
167 "PDV data too short");
168 }
169
170 // Extract command from PDV data
171 // The command is DICOM dataset starting at offset 6
172 const auto& data = pdu_.data;
173 size_t offset = 6;
174
175 // Parse DICOM command elements
176 // Minimum: (0000,0000) CommandGroupLength + (0000,0100) CommandField
177 while (offset + 8 <= data.size()) {
178 // Tag: group (2) + element (2)
179 uint16_t group = static_cast<uint16_t>(data[offset]) |
180 (static_cast<uint16_t>(data[offset + 1]) << 8);
181 uint16_t element = static_cast<uint16_t>(data[offset + 2]) |
182 (static_cast<uint16_t>(data[offset + 3]) << 8);
183 offset += 4;
184
185 // VR is implicit for command set (Implicit VR Little Endian)
186 // Length (4 bytes)
187 if (offset + 4 > data.size()) break;
188 uint32_t length = static_cast<uint32_t>(data[offset]) |
189 (static_cast<uint32_t>(data[offset + 1]) << 8) |
190 (static_cast<uint32_t>(data[offset + 2]) << 16) |
191 (static_cast<uint32_t>(data[offset + 3]) << 24);
192 offset += 4;
193
194 if (offset + length > data.size()) break;
195
196 // Extract key elements
197 if (group == 0x0000) {
198 switch (element) {
199 case 0x0100: // CommandField
200 if (length >= 2) {
201 request.command_type = static_cast<dimse_command_type>(
202 static_cast<uint16_t>(data[offset]) |
203 (static_cast<uint16_t>(data[offset + 1]) << 8)
204 );
205 }
206 break;
207 case 0x0110: // MessageID
208 if (length >= 2) {
209 request.message_id = static_cast<uint16_t>(data[offset]) |
210 (static_cast<uint16_t>(data[offset + 1]) << 8);
211 }
212 break;
213 case 0x0002: // AffectedSOPClassUID
214 case 0x0003: // RequestedSOPClassUID
215 request.sop_class_uid = std::string(
216 reinterpret_cast<const char*>(&data[offset]), length);
217 // Trim trailing nulls/spaces
218 while (!request.sop_class_uid.empty() &&
219 (request.sop_class_uid.back() == '\0' ||
220 request.sop_class_uid.back() == ' ')) {
221 request.sop_class_uid.pop_back();
222 }
223 break;
224 case 0x1000: // AffectedSOPInstanceUID
225 case 0x1001: // RequestedSOPInstanceUID
226 request.sop_instance_uid = std::string(
227 reinterpret_cast<const char*>(&data[offset]), length);
228 while (!request.sop_instance_uid.empty() &&
229 (request.sop_instance_uid.back() == '\0' ||
230 request.sop_instance_uid.back() == ' ')) {
231 request.sop_instance_uid.pop_back();
232 }
233 break;
234 case 0x0700: // Priority
235 if (length >= 2) {
236 request.priority = static_cast<uint16_t>(data[offset]) |
237 (static_cast<uint16_t>(data[offset + 1]) << 8);
238 }
239 break;
240 }
241 }
242
243 offset += length;
244 }
245
246 // Store full command data for later use
247 request.command_data = pdu_.data;
248
249 return ok(std::move(request));
250}

252auto dimse_process_job::process_association_pdu() -> VoidResult {
253 // Invoke association callback
254 if (on_association_) {
255 on_association_(pdu_.session_id, pdu_.type, pdu_.data);
256 }
257
258 // Association PDUs don't go through the execution pipeline
259 // They are handled directly by the association state machine
260 return ok();
