100struct alignas(SSE_ALIGNMENT) vec128_int {
101#if defined(PACS_SIMD_SSE2)
102 __m128i data;
103
104 vec128_int() noexcept : data(_mm_setzero_si128()) {}
105 explicit vec128_int(__m128i v) noexcept : data(v) {}
106
107 static vec128_int load(const void* ptr) noexcept {
108 return vec128_int(_mm_loadu_si128(reinterpret_cast<const __m128i*>(ptr)));
109 }
110
111 static vec128_int load_aligned(const void* ptr) noexcept {
112 return vec128_int(_mm_load_si128(reinterpret_cast<const __m128i*>(ptr)));
113 }
114
115 void store(void* ptr) const noexcept {
116 _mm_storeu_si128(reinterpret_cast<__m128i*>(ptr), data);
117 }
118
119 void store_aligned(void* ptr) const noexcept {
120 _mm_store_si128(reinterpret_cast<__m128i*>(ptr), data);
121 }
122
123#elif defined(PACS_SIMD_NEON)
124 uint8x16_t data;
125
126 vec128_int() noexcept : data(vdupq_n_u8(0)) {}
127 explicit vec128_int(uint8x16_t v) noexcept : data(v) {}
128
129 static vec128_int load(const void* ptr) noexcept {
130 return vec128_int(vld1q_u8(reinterpret_cast<const uint8_t*>(ptr)));
131 }
132
133 static vec128_int load_aligned(const void* ptr) noexcept {
134 return load(ptr); // NEON handles unaligned loads efficiently
135 }
136
137 void store(void* ptr) const noexcept {
138 vst1q_u8(reinterpret_cast<uint8_t*>(ptr), data);
139 }
140
141 void store_aligned(void* ptr) const noexcept {
142 store(ptr);
143 }
144
145#else
146 // Scalar fallback
147 alignas(16) uint8_t bytes[16]{};
148
149 vec128_int() noexcept = default;
150
151 static vec128_int load(const void* ptr) noexcept {
152 vec128_int result;
153 std::memcpy(result.bytes, ptr, 16);
154 return result;
155 }
156
157 static vec128_int load_aligned(const void* ptr) noexcept {
158 return load(ptr);
159 }
160
161 void store(void* ptr) const noexcept {
162 std::memcpy(ptr, bytes, 16);
163 }
164
165 void store_aligned(void* ptr) const noexcept {
166 store(ptr);
167 }
168#endif
169};

174struct alignas(AVX_ALIGNMENT) vec256_int {
175#if defined(PACS_SIMD_AVX2)
176 __m256i data;
177
178 vec256_int() noexcept : data(_mm256_setzero_si256()) {}
179 explicit vec256_int(__m256i v) noexcept : data(v) {}
180
181 static vec256_int load(const void* ptr) noexcept {
182 return vec256_int(
183 _mm256_loadu_si256(reinterpret_cast<const __m256i*>(ptr)));
184 }
185
186 static vec256_int load_aligned(const void* ptr) noexcept {
187 return vec256_int(
188 _mm256_load_si256(reinterpret_cast<const __m256i*>(ptr)));
189 }
190
191 void store(void* ptr) const noexcept {
192 _mm256_storeu_si256(reinterpret_cast<__m256i*>(ptr), data);
193 }
194
195 void store_aligned(void* ptr) const noexcept {
196 _mm256_store_si256(reinterpret_cast<__m256i*>(ptr), data);
197 }
198
199#else
200 // Fallback using two 128-bit vectors
201 vec128_int low;
202 vec128_int high;
203
204 vec256_int() noexcept = default;
205 vec256_int(vec128_int l, vec128_int h) noexcept : low(l), high(h) {}
206
207 static vec256_int load(const void* ptr) noexcept {
208 const auto* p = reinterpret_cast<const uint8_t*>(ptr);
209 return vec256_int(vec128_int::load(p), vec128_int::load(p + 16));
210 }
211
212 static vec256_int load_aligned(const void* ptr) noexcept {
213 const auto* p = reinterpret_cast<const uint8_t*>(ptr);
214 return vec256_int(vec128_int::load_aligned(p),
215 vec128_int::load_aligned(p + 16));
216 }
217
218 void store(void* ptr) const noexcept {
219 auto* p = reinterpret_cast<uint8_t*>(ptr);
220 low.store(p);
221 high.store(p + 16);
222 }
223
224 void store_aligned(void* ptr) const noexcept {
225 auto* p = reinterpret_cast<uint8_t*>(ptr);
226 low.store_aligned(p);
227 high.store_aligned(p + 16);
228 }
229#endif
230};

235template <size_t Alignment>
236[[nodiscard]] constexpr bool is_aligned(const void* ptr) noexcept {
237 return (reinterpret_cast<uintptr_t>(ptr) % Alignment) == 0;
238}
239
243template <size_t Alignment>
244[[nodiscard]] constexpr size_t align_offset(const void* ptr) noexcept {
245 const auto addr = reinterpret_cast<uintptr_t>(ptr);
246 const auto remainder = addr % Alignment;
247 return remainder == 0 ? 0 : Alignment - remainder;
248}
