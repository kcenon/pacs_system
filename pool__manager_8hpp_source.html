62struct pool_statistics {
63 std::atomic<uint64_t> total_acquisitions{0};
64 std::atomic<uint64_t> pool_hits{0};
65 std::atomic<uint64_t> pool_misses{0};
66 std::atomic<uint64_t> total_releases{0};
67
72 [[nodiscard]] auto hit_ratio() const noexcept -> double {
73 const uint64_t total = total_acquisitions.load(std::memory_order_relaxed);
74 if (total == 0) {
75 return 0.0;
76 }
77 return static_cast<double>(pool_hits.load(std::memory_order_relaxed))
78 / static_cast<double>(total);
79 }
80
84 void reset() noexcept {
85 total_acquisitions.store(0, std::memory_order_relaxed);
86 pool_hits.store(0, std::memory_order_relaxed);
87 pool_misses.store(0, std::memory_order_relaxed);
88 total_releases.store(0, std::memory_order_relaxed);
89 }
90};

96template <typename T>
97class tracked_pool {
98public:
99 using pool_type = kcenon::common::utils::ObjectPool<T>;
100 using unique_ptr_type = std::unique_ptr<T, std::function<void(T*)>>;
101
102 explicit tracked_pool(std::size_t initial_size = 32)
103 : pool_(initial_size) {
104 pool_.reserve(initial_size);
105 }
106
112 template <typename... Args>
113 auto acquire(Args&&... args) -> unique_ptr_type {
114 bool reused = false;
115 auto ptr = pool_.acquire(&reused, std::forward<Args>(args)...);
116
117 stats_.total_acquisitions.fetch_add(1, std::memory_order_relaxed);
118 if (reused) {
119 stats_.pool_hits.fetch_add(1, std::memory_order_relaxed);
120 } else {
121 stats_.pool_misses.fetch_add(1, std::memory_order_relaxed);
122 }
123
124 return ptr;
125 }
126
131 [[nodiscard]] auto statistics() const noexcept -> const pool_statistics& {
132 return stats_;
133 }
134
139 [[nodiscard]] auto available() const -> std::size_t {
140 return pool_.available();
141 }
142
147 void reserve(std::size_t count) {
148 pool_.reserve(count);
149 }
150
154 void clear() {
155 pool_.clear();
156 }
157
158private:
159 pool_type pool_;
160 pool_statistics stats_;
161};
