171enum class dimse_error {
172 success = 0,
173 invalid_command_set,
174 missing_required_field,
175 invalid_data_format,
176 encoding_error,
177 decoding_error,
178};
179
183[[nodiscard]] constexpr std::string_view to_string(dimse_error err) noexcept {
184 switch (err) {
185 case dimse_error::success: return "Success";
186 case dimse_error::invalid_command_set: return "Invalid command set";
187 case dimse_error::missing_required_field: return "Missing required field";
188 case dimse_error::invalid_data_format: return "Invalid data format";
189 case dimse_error::encoding_error: return "Encoding error";
190 case dimse_error::decoding_error: return "Decoding error";
191 default: return "Unknown error";
192 }
193}
