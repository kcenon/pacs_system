70std::optional<auth_result> oauth2_middleware::authenticate(
71 const crow::request& req, crow::response& res) const {
72
73 // Extract Bearer token from Authorization header
74 auto bearer = extract_bearer_token(req);
75 if (!bearer) {
76 set_unauthorized(res, "Missing or invalid Authorization header");
77 return std::nullopt;
78 }
79
80 // Decode JWT
81 auto [token, decode_err] = validator_.decode(*bearer);
82 if (decode_err != jwt_error::none) {
83 set_unauthorized(res, std::string(jwt_error_message(decode_err)));
84 return std::nullopt;
85 }
86
87 // Validate claims (issuer, audience, expiration)
88 auto claims_err = validator_.validate_claims(token.claims);
89 if (claims_err != jwt_error::none) {
90 set_unauthorized(res, std::string(jwt_error_message(claims_err)));
91 return std::nullopt;
92 }
93
94 // Verify signature using JWKS
95 if (!verify_signature(token)) {
96 set_unauthorized(res, "Token signature verification failed");
97 return std::nullopt;
98 }
99
100 // Create user_context from JWT subject
101 // Try to find existing user in RBAC system, or create an OAuth-based context
102 security::User user;
103 user.id = token.claims.sub;
104 user.username = token.claims.sub;
105 user.active = true;
106
107 // If we have an access control manager, try to look up the user
108 if (security_manager_) {
109 auto user_result = security_manager_->get_user(token.claims.sub);
110 if (user_result.is_ok()) {
111 user = user_result.unwrap();
112 } else if (config_.allow_unknown_users) {
113 // Backward compatibility: grant Viewer role to unknown OAuth users
114 user.roles = {security::Role::Viewer};
115 } else {
116 set_unauthorized(res, "Unknown user: not registered in access control system");
117 return std::nullopt;
118 }
119 } else if (config_.allow_unknown_users) {
120 user.roles = {security::Role::Viewer};
121 } else {
122 set_unauthorized(res, "Unknown user: no access control manager configured");
123 return std::nullopt;
124 }
125
126 auto ctx = security::user_context(std::move(user), token.claims.jti);
127 ctx.set_source_ip(std::string(req.remote_ip_address));
128 ctx.touch();
129
130 return auth_result{std::move(ctx), std::move(token.claims)};
131}

133bool oauth2_middleware::require_scope(
134 const jwt_claims& claims,
135 crow::response& res,
136 std::string_view required_scope) const {
137
138 if (!jwt_validator::has_scope(claims, required_scope)) {
139 set_forbidden(res,
140 "Insufficient scope: requires " + std::string(required_scope));
141 return false;
142 }
143 return true;
144}

146bool oauth2_middleware::require_any_scope(
147 const jwt_claims& claims,
148 crow::response& res,
149 const std::vector<std::string>& required_scopes) const {
150
151 if (!jwt_validator::has_any_scope(claims, required_scopes)) {
152 std::string scope_list;
153 for (size_t i = 0; i < required_scopes.size(); ++i) {
154 if (i > 0) scope_list += " or ";
155 scope_list += required_scopes[i];
156 }
157 set_forbidden(res, "Insufficient scope: requires " + scope_list);
158 return false;
159 }
160 return true;
161}

175std::optional<std::string_view> oauth2_middleware::extract_bearer_token(
176 const crow::request& req) const {
177
178 auto auth_header = req.get_header_value("Authorization");
179 if (auth_header.empty()) return std::nullopt;
180
181 std::string_view header_view(auth_header);
182
183 // Check for "Bearer " prefix (case-insensitive for "Bearer")
184 constexpr std::string_view bearer_prefix = "Bearer ";
185 if (header_view.size() <= bearer_prefix.size()) return std::nullopt;
186
187 // Case-sensitive check per RFC 6750
188 if (header_view.substr(0, bearer_prefix.size()) != bearer_prefix) {
189 return std::nullopt;
190 }
191
192 auto token = header_view.substr(bearer_prefix.size());
193 if (token.empty()) return std::nullopt;
194
195 return token;
196}

198bool oauth2_middleware::verify_signature(const jwt_token& token) const {
199 if (!jwks_provider_) return false;
200
201 // Try to get key by kid from the token header
202 std::optional<jwk_key> key;
203 if (!token.header.kid.empty()) {
204 key = jwks_provider_->get_key(token.header.kid);
205 }
206
207 // Fall back to algorithm-based lookup
208 if (!key) {
209 key = jwks_provider_->get_key_by_alg(token.header.alg);
210 }
211
212 if (!key) return false;
213
214 // Verify based on algorithm
215 if (token.header.alg == "RS256") {
216 return validator_.verify_rs256(token, key->pem);
217 } else if (token.header.alg == "ES256") {
218 return validator_.verify_es256(token, key->pem);
219 }
220
221 return false;
222}
