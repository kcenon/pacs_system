73// Internal implementation function called from rest_server.cpp
74void register_association_endpoints_impl(
75 crow::SimpleApp &app,
76 std::shared_ptr<rest_server_context> ctx) {
77 // GET /api/v1/associations/active - List active DICOM associations
78 CROW_ROUTE(app, "/api/v1/associations/active")
79 .methods(crow::HTTPMethod::GET)([ctx](const crow::request & /*req*/) {
80 crow::response res;
81 res.add_header("Content-Type", "application/json");
82 add_cors_headers(res, *ctx);
83
84 if (!ctx->dicom_server) {
85 res.code = 503;
86 res.body = make_error_json("SERVICE_UNAVAILABLE",
87 "DICOM server not configured");
88 return res;
89 }
90
91 auto stats = ctx->dicom_server->get_statistics();
92 auto active_count = ctx->dicom_server->active_associations();
93 auto uptime_sec = stats.uptime().count();
94
95 std::ostringstream oss;
96 oss << R"({"active_count":)" << active_count
97 << R"(,"total_associations":)" << stats.total_associations
98 << R"(,"rejected_associations":)" << stats.rejected_associations
99 << R"(,"messages_processed":)" << stats.messages_processed
100 << R"(,"bytes_received":)" << stats.bytes_received
101 << R"(,"bytes_sent":)" << stats.bytes_sent
102 << R"(,"uptime_seconds":)" << uptime_sec
103 << R"(,"server_running":)"
104 << (ctx->dicom_server->is_running() ? "true" : "false")
105 << '}';
106
107 res.code = 200;
108 res.body = oss.str();
109 return res;
110 });
111
112 // DELETE /api/v1/associations/:id - Terminate a DICOM association
113 CROW_ROUTE(app, "/api/v1/associations/<string>")
114 .methods(crow::HTTPMethod::DELETE)(
115 [ctx](const crow::request & /*req*/,
116 const std::string &association_id) {
117 crow::response res;
118 res.add_header("Content-Type", "application/json");
119 add_cors_headers(res, *ctx);
120
121 if (association_id.empty()) {
122 res.code = 400;
123 res.body = make_error_json("INVALID_REQUEST",
124 "Association ID is required");
125 return res;
126 }
127
128 if (!ctx->dicom_server) {
129 res.code = 503;
130 res.body = make_error_json("SERVICE_UNAVAILABLE",
131 "DICOM server not configured");
132 return res;
133 }
134
135 // Individual association termination is not supported via the
136 // dicom_server public API. The server manages association
137 // lifecycles internally through idle timeouts and graceful
138 // shutdown.
139 res.code = 501;
140 res.body = make_error_json(
141 "NOT_IMPLEMENTED",
142 "Individual association termination is not supported. "
143 "Associations are managed by the DICOM server via idle "
144 "timeouts and graceful shutdown.");
145 return res;
146 });
147
148 // GET /api/v1/associations/:id - Get specific association details
149 CROW_ROUTE(app, "/api/v1/associations/<string>")
150 .methods(crow::HTTPMethod::GET)(
151 [ctx](const crow::request & /*req*/,
152 const std::string &association_id) {
153 crow::response res;
154 res.add_header("Content-Type", "application/json");
155 add_cors_headers(res, *ctx);
156
157 if (association_id.empty()) {
158 res.code = 400;
159 res.body = make_error_json("INVALID_REQUEST",
160 "Association ID is required");
161 return res;
162 }
163
164 if (!ctx->dicom_server) {
165 res.code = 503;
166 res.body = make_error_json("SERVICE_UNAVAILABLE",
167 "DICOM server not configured");
168 return res;
169 }
170
171 // Individual association lookup is not supported via the
172 // dicom_server public API. Use GET /associations/active for
173 // aggregate statistics.
174 res.code = 501;
175 res.body = make_error_json(
176 "NOT_IMPLEMENTED",
177 "Individual association lookup is not supported. "
178 "Use GET /api/v1/associations/active for aggregate "
179 "statistics.");
180 return res;
181 });
182}
