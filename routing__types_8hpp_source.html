257 [[nodiscard]] bool is_effective_now() const noexcept {
258 if (!enabled) return false;
259
260 auto now = std::chrono::system_clock::now();
261
262 if (effective_from.has_value() && now < effective_from.value()) {
263 return false;
264 }
265
266 if (effective_until.has_value() && now > effective_until.value()) {
267 return false;
268 }
269
270 return true;
271 }
