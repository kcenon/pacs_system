83[[nodiscard]] inline std::string to_string(metric_type type) {
84 switch (type) {
85 case metric_type::gauge:
86 return "gauge";
87 case metric_type::counter:
88 return "counter";
89 case metric_type::histogram:
90 return "histogram";
91 case metric_type::summary:
92 return "summary";
93 default:
94 return "unknown";
95 }

102struct metric_value {
103 std::string name;
104 double value;
105 metric_type type{metric_type::gauge};
106 std::chrono::system_clock::time_point timestamp;
107 std::unordered_map<std::string, std::string> tags;
108
109 metric_value() = default;
110
111 metric_value(std::string n,
112 double v,
113 metric_type t = metric_type::gauge,
114 std::unordered_map<std::string, std::string> tg = {})
115 : name(std::move(n))
116 , value(v)
117 , type(t)
118 , timestamp(std::chrono::system_clock::now())
119 , tags(std::move(tg)) {}
120};

145enum class monitor_health_status {
146 healthy = 0,
147 degraded = 1,
148 unhealthy = 2,
149 unknown = 3
150};
151
156struct health_check_result {
157 monitor_health_status status{monitor_health_status::unknown};
158 std::string message;
159 std::chrono::system_clock::time_point timestamp;
160 std::chrono::milliseconds check_duration{0};
161 std::unordered_map<std::string, std::string> metadata;
162
163 health_check_result()
164 : timestamp(std::chrono::system_clock::now()) {}
165
166 [[nodiscard]] bool is_healthy() const {
167 return status == monitor_health_status::healthy;
168 }
169
170 [[nodiscard]] bool is_operational() const {
171 return status == monitor_health_status::healthy ||
172 status == monitor_health_status::degraded;
173 }
174};

438inline void pacs_monitor::initialize_collectors() {
439 association_collector_ = std::make_unique<dicom_association_collector>(config_.ae_title);
440 service_collector_ = std::make_unique<dicom_service_collector>(config_.ae_title);
441 storage_collector_ = std::make_unique<dicom_storage_collector>(config_.ae_title);
442 unified_collector_ = std::make_unique<dicom_metrics_collector>(config_.ae_title);
443
444 // Initialize all collectors
445 std::unordered_map<std::string, std::string> collector_config;
446 collector_config["ae_title"] = config_.ae_title;
447
448 (void)association_collector_->initialize(collector_config);
449 (void)service_collector_->initialize(collector_config);
450 (void)storage_collector_->initialize(collector_config);
451 storage_collector_->set_pool_metrics_enabled(config_.enable_pool_metrics);
452
453 // Initialize unified CRTP-based collector
454 config_map unified_config;
455 unified_config["ae_title"] = config_.ae_title;
456 unified_config["collect_associations"] = config_.enable_association_metrics ? "true" : "false";
457 unified_config["collect_transfers"] = config_.enable_storage_metrics ? "true" : "false";
458 unified_config["collect_storage"] = config_.enable_storage_metrics ? "true" : "false";
459 unified_config["collect_queries"] = config_.enable_service_metrics ? "true" : "false";
460 unified_config["collect_pools"] = config_.enable_pool_metrics ? "true" : "false";
461 (void)unified_collector_->initialize(unified_config);
462}

484inline void pacs_monitor::collect_all_metrics(metrics_snapshot& snapshot) {
485 std::shared_lock<std::shared_mutex> config_lock(config_mutex_);
486
487 // Collect association metrics
488 if (config_.enable_association_metrics && association_collector_) {
489 for (const auto& m : association_collector_->collect()) {
490 snapshot.add_metric(
491 m.name,
492 m.value,
493 m.type == "counter" ? metric_type::counter : metric_type::gauge);
494 }
495 }
496
497 // Collect service metrics
498 if (config_.enable_service_metrics && service_collector_) {
499 for (const auto& m : service_collector_->collect()) {
500 snapshot.add_metric(
501 m.name,
502 m.value,
503 m.type == "counter" ? metric_type::counter : metric_type::gauge);
504 }
505 }
506
507 // Collect storage metrics
508 if (config_.enable_storage_metrics && storage_collector_) {
509 for (const auto& m : storage_collector_->collect()) {
510 snapshot.add_metric(
511 m.name,
512 m.value,
513 m.type == "counter" ? metric_type::counter : metric_type::gauge);
514 }
515 }
516
517 // Add custom metrics
518 {
519 std::lock_guard<std::mutex> lock(custom_metrics_mutex_);
520 for (const auto& m : custom_metrics_) {
521 snapshot.metrics.push_back(m);
522 }
523 }
524}

526inline health_check_result pacs_monitor::check_health() {
527 const auto start = std::chrono::steady_clock::now();
528 health_check_result result;
529 result.status = monitor_health_status::healthy;
530
531 std::lock_guard<std::mutex> lock(health_checks_mutex_);
532
533 bool all_healthy = true;
534 bool any_check_failed = false;
535
536 for (const auto& [component, check] : health_checks_) {
537 try {
538 if (!check()) {
539 all_healthy = false;
540 result.metadata[component] = "unhealthy";
541 } else {
542 result.metadata[component] = "healthy";
543 }
544 } catch (const std::exception& e) {
545 any_check_failed = true;
546 result.metadata[component] = std::string("error: ") + e.what();
547 }
548 }
549
550 // Check collectors health
551 if (association_collector_ && !association_collector_->is_healthy()) {
552 all_healthy = false;
553 result.metadata["association_collector"] = "unhealthy";
554 }
555 if (service_collector_ && !service_collector_->is_healthy()) {
556 all_healthy = false;
557 result.metadata["service_collector"] = "unhealthy";
558 }
559 if (storage_collector_ && !storage_collector_->is_healthy()) {
560 all_healthy = false;
561 result.metadata["storage_collector"] = "unhealthy";
562 }
563 if (unified_collector_ && !unified_collector_->is_healthy()) {
564 all_healthy = false;
565 result.metadata["unified_collector"] = "unhealthy";
566 }
567
568 if (any_check_failed) {
569 result.status = monitor_health_status::unhealthy;
570 result.message = "Some health checks threw exceptions";
571 } else if (!all_healthy) {
572 result.status = monitor_health_status::degraded;
573 result.message = "Some components are unhealthy";
574 } else {
575 result.message = "All components healthy";
576 }
577
578 const auto end = std::chrono::steady_clock::now();
579 result.check_duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
580
581 return result;
582}

641 unified_collector_->set_collect_storage(config_.enable_storage_metrics);
642 unified_collector_->set_collect_queries(config_.enable_service_metrics);
643 unified_collector_->set_collect_pools(config_.enable_pool_metrics);
644 }
