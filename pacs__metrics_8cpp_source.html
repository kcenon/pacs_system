51std::string counter_to_json(const operation_counter& counter) {
52 std::ostringstream oss;
53 const auto total = counter.total_count();
54 const auto success = counter.success_count.load(std::memory_order_relaxed);
55 const auto failure = counter.failure_count.load(std::memory_order_relaxed);
56 const auto avg_us = counter.average_duration_us();
57 const auto min_us = counter.min_duration_us.load(std::memory_order_relaxed);
58 const auto max_us = counter.max_duration_us.load(std::memory_order_relaxed);
59
60 oss << "{"
61 << R"("total":)" << total
62 << R"(,"success":)" << success
63 << R"(,"failure":)" << failure
64 << R"(,"avg_duration_us":)" << avg_us;
65
66 // Only include min/max if there were operations
67 if (total > 0) {
68 oss << R"(,"min_duration_us":)" << min_us
69 << R"(,"max_duration_us":)" << max_us;
70 }
71
72 oss << "}";
73 return oss.str();
74}

83void counter_to_prometheus(std::ostringstream& oss,
84 std::string_view prefix,
85 std::string_view op_name,
86 const operation_counter& counter) {
87 const auto success = counter.success_count.load(std::memory_order_relaxed);
88 const auto failure = counter.failure_count.load(std::memory_order_relaxed);
89 const auto total = counter.total_count();
90 const auto total_duration = counter.total_duration_us.load(std::memory_order_relaxed);
91
92 // Total operations counter
93 oss << "# HELP " << prefix << "_dimse_" << op_name << "_total "
94 << "Total " << op_name << " operations\n"
95 << "# TYPE " << prefix << "_dimse_" << op_name << "_total counter\n"
96 << prefix << "_dimse_" << op_name << "_total " << total << "\n";
97
98 // Success counter
99 oss << "# HELP " << prefix << "_dimse_" << op_name << "_success_total "
100 << "Successful " << op_name << " operations\n"
101 << "# TYPE " << prefix << "_dimse_" << op_name << "_success_total counter\n"
102 << prefix << "_dimse_" << op_name << "_success_total " << success << "\n";
103
104 // Failure counter
105 oss << "# HELP " << prefix << "_dimse_" << op_name << "_failure_total "
106 << "Failed " << op_name << " operations\n"
107 << "# TYPE " << prefix << "_dimse_" << op_name << "_failure_total counter\n"
108 << prefix << "_dimse_" << op_name << "_failure_total " << failure << "\n";
109
110 // Duration sum (for calculating averages)
111 oss << "# HELP " << prefix << "_dimse_" << op_name << "_duration_microseconds_sum "
112 << "Total duration of " << op_name << " operations in microseconds\n"
113 << "# TYPE " << prefix << "_dimse_" << op_name << "_duration_microseconds_sum counter\n"
114 << prefix << "_dimse_" << op_name << "_duration_microseconds_sum " << total_duration << "\n";
115
116 // Duration min/max (only if there were operations)
117 if (total > 0) {
118 const auto min_us = counter.min_duration_us.load(std::memory_order_relaxed);
119 const auto max_us = counter.max_duration_us.load(std::memory_order_relaxed);
120
121 oss << "# HELP " << prefix << "_dimse_" << op_name << "_duration_microseconds_min "
122 << "Minimum duration of " << op_name << " operations in microseconds\n"
123 << "# TYPE " << prefix << "_dimse_" << op_name << "_duration_microseconds_min gauge\n"
124 << prefix << "_dimse_" << op_name << "_duration_microseconds_min " << min_us << "\n";
125
126 oss << "# HELP " << prefix << "_dimse_" << op_name << "_duration_microseconds_max "
127 << "Maximum duration of " << op_name << " operations in microseconds\n"
128 << "# TYPE " << prefix << "_dimse_" << op_name << "_duration_microseconds_max gauge\n"
129 << prefix << "_dimse_" << op_name << "_duration_microseconds_max " << max_us << "\n";
130 }
131}

135std::string pacs_metrics::to_json() const {
136 std::ostringstream oss;
137 oss << "{";
138
139 // DIMSE operations section
140 oss << R"("dimse_operations":{)"
141 << R"("c_echo":)" << counter_to_json(c_echo_)
142 << R"(,"c_store":)" << counter_to_json(c_store_)
143 << R"(,"c_find":)" << counter_to_json(c_find_)
144 << R"(,"c_move":)" << counter_to_json(c_move_)
145 << R"(,"c_get":)" << counter_to_json(c_get_)
146 << R"(,"n_create":)" << counter_to_json(n_create_)
147 << R"(,"n_set":)" << counter_to_json(n_set_)
148 << R"(,"n_get":)" << counter_to_json(n_get_)
149 << R"(,"n_action":)" << counter_to_json(n_action_)
150 << R"(,"n_event":)" << counter_to_json(n_event_)
151 << R"(,"n_delete":)" << counter_to_json(n_delete_)
152 << "}";
153
154 // Data transfer section
155 oss << R"(,"data_transfer":{)"
156 << R"("bytes_sent":)" << transfer_.bytes_sent.load(std::memory_order_relaxed)
157 << R"(,"bytes_received":)" << transfer_.bytes_received.load(std::memory_order_relaxed)
158 << R"(,"images_stored":)" << transfer_.images_stored.load(std::memory_order_relaxed)
159 << R"(,"images_retrieved":)" << transfer_.images_retrieved.load(std::memory_order_relaxed)
160 << "}";
161
162 // Associations section
163 oss << R"(,"associations":{)"
164 << R"("total_established":)" << associations_.total_established.load(std::memory_order_relaxed)
165 << R"(,"total_rejected":)" << associations_.total_rejected.load(std::memory_order_relaxed)
166 << R"(,"total_aborted":)" << associations_.total_aborted.load(std::memory_order_relaxed)
167 << R"(,"current_active":)" << associations_.current_active.load(std::memory_order_relaxed)
168 << R"(,"peak_active":)" << associations_.peak_active.load(std::memory_order_relaxed)
169 << "}";
170
171 oss << "}";
172 return oss.str();
173}
