64auto send_network_io_job::execute(pipeline_coordinator& /*coordinator*/)
65 -> VoidResult {
66
67 // Validate we have data and send function
68 if (data_.empty()) {
69 if (on_error_) {
70 on_error_(context_.session_id, "Empty data to send");
71 }
72 return pacs_void_error(error_codes::invalid_argument,
73 "Empty data to send");
74 }
75
76 if (!send_fn_) {
77 if (on_error_) {
78 on_error_(context_.session_id, "No send function provided");
79 }
80 return pacs_void_error(error_codes::invalid_argument,
81 "No send function provided");
82 }
83
84 // Perform the actual network send
85 auto result = send_fn_(context_.session_id, data_);
86
87 if (!result.is_ok()) {
88 if (on_error_) {
89 auto err = result.error();
90 on_error_(context_.session_id, err.message);
91 }
92 if (on_complete_) {
93 on_complete_(context_.session_id, false, 0);
94 }
95 return result;
96 }
97
98 // Success - invoke completion callback
99 if (on_complete_) {
100 on_complete_(context_.session_id, true, data_.size());
101 }
102
103 return ok();
104}
