114auto anonymizer::anonymize_with_mapping(
115 dicom_dataset& dataset,
116 uid_mapping& mapping
117) -> kcenon::common::Result<anonymization_report> {
118 anonymization_report report;
119 report.profile_name = std::string(to_string(profile_));
120 report.date_offset = date_offset_;
121 report.timestamp = std::chrono::system_clock::now();
122
123 // Get profile actions
124 auto profile_actions = get_profile_actions(profile_);
125
126 // Merge with custom actions (custom takes precedence)
127 for (const auto& [tag, config] : custom_actions_) {
128 profile_actions[tag] = config;
129 }
130
131 // Process each tag in the profile
132 for (const auto& [tag, config] : profile_actions) {
133 if (!dataset.contains(tag)) {
134 continue;
135 }
136
137 auto record = apply_action(dataset, tag, config, &mapping);
138 report.total_tags_processed++;
139
140 switch (config.action) {
141 case tag_action::remove:
142 case tag_action::remove_or_empty:
143 if (record.success) {
144 report.tags_removed++;
145 }
146 break;
147 case tag_action::empty:
148 if (record.success) {
149 report.tags_emptied++;
150 }
151 break;
152 case tag_action::replace:
153 if (record.success) {
154 report.tags_replaced++;
155 }
156 break;
157 case tag_action::replace_uid:
158 if (record.success) {
159 report.uids_replaced++;
160 }
161 break;
162 case tag_action::keep:
163 report.tags_kept++;
164 break;
165 case tag_action::shift_date:
166 if (record.success) {
167 report.dates_shifted++;
168 }
169 break;
170 case tag_action::hash:
171 if (record.success) {
172 report.values_hashed++;
173 }
174 break;
175 case tag_action::encrypt:
176 if (record.success) {
177 report.tags_replaced++;
178 }
179 break;
180 }
181
182 if (!record.success && !record.error_message.empty()) {
183 report.errors.push_back(record.error_message);
184 }
185
186 if (detailed_reporting_) {
187 report.action_records.push_back(std::move(record));
188 }
189 }
190
191 // Remove private tags if configured
192 if (private_tag_action_ != private_tag_action::keep) {
193 std::vector<dicom_tag> tags_to_remove;
194
195 for (auto it = dataset.begin(); it != dataset.end(); ++it) {
196 auto tag = it->first;
197 if (private_tag_action_ == private_tag_action::remove_all) {
198 if (tag.is_private()) {
199 tags_to_remove.push_back(tag);
200 }
201 } else if (private_tag_action_ == private_tag_action::remove_data) {
202 if (tag.is_private_data()) {
203 tags_to_remove.push_back(tag);
204 }
205 }
206 }
207
208 for (const auto& tag : tags_to_remove) {
209 if (dataset.remove(tag)) {
210 report.private_tags_removed++;
211 }
212 }
213 }
214
215 return report;
216}

284auto anonymizer::generate_random_date_offset(
285 std::chrono::days min_days,
286 std::chrono::days max_days
287) -> std::chrono::days {
288 std::random_device rd;
289 std::mt19937 gen(rd());
290 std::uniform_int_distribution<int> dist(
291 static_cast<int>(min_days.count()),
292 static_cast<int>(max_days.count())
293 );
294 return std::chrono::days{dist(gen)};
295}

328auto anonymizer::get_profile_actions(anonymization_profile profile)
329 -> std::map<dicom_tag, tag_action_config> {
330 std::map<dicom_tag, tag_action_config> actions;
331
332 // Common patient identifiers
333 auto add_patient_identifiers = [&actions]() {
334 actions[tags::patient_name] = tag_action_config::make_replace("ANONYMOUS");
335 actions[tags::patient_id] = tag_action_config::make_replace("ANON_ID");
336 actions[tags::patient_birth_date] = tag_action_config::make_empty();
337 actions[tags::patient_sex] = tag_action_config::make_keep();
338 actions[tags::patient_age] = tag_action_config::make_remove();
339 actions[tags::patient_address] = tag_action_config::make_remove();
340 actions[tags::patient_comments] = tag_action_config::make_remove();
341 };
342
343 // Institution identifiers
344 auto add_institution_identifiers = [&actions]() {
345 actions[tags::institution_name] = tag_action_config::make_empty();
346 actions[tags::institution_address] = tag_action_config::make_remove();
347 actions[tags::station_name] = tag_action_config::make_empty();
348 };
349
350 // Personnel identifiers
351 auto add_personnel_identifiers = [&actions]() {
352 actions[tags::referring_physician_name] = tag_action_config::make_empty();
353 actions[tags::performing_physician_name] = tag_action_config::make_empty();
354 actions[tags::name_of_physicians_reading_study] = tag_action_config::make_empty();
355 actions[tags::operators_name] = tag_action_config::make_empty();
356 actions[tags::scheduled_performing_physician_name] = tag_action_config::make_empty();
357 };
358
359 // UID replacements
360 auto add_uid_replacements = [&actions]() {
361 actions[tags::study_instance_uid] =
362 {.action = tag_action::replace_uid};
363 actions[tags::series_instance_uid] =
364 {.action = tag_action::replace_uid};
365 actions[tags::sop_instance_uid] =
366 {.action = tag_action::replace_uid};
367 actions[tags::frame_of_reference_uid] =
368 {.action = tag_action::replace_uid};
369 };
370
371 // Study identifiers
372 auto add_study_identifiers = [&actions]() {
373 actions[tags::accession_number] = tag_action_config::make_empty();
374 actions[tags::study_id] = tag_action_config::make_replace("ANON_STUDY");
375 };
376
377 // Dates for shifting
378 auto add_date_shifting = [&actions]() {
379 actions[tags::study_date] = {.action = tag_action::shift_date};
380 actions[tags::series_date] = {.action = tag_action::shift_date};
381 actions[tags::acquisition_date] = {.action = tag_action::shift_date};
382 actions[tags::content_date] = {.action = tag_action::shift_date};
383 actions[tags::instance_creation_date] = {.action = tag_action::shift_date};
384 actions[tags::patient_birth_date] = {.action = tag_action::shift_date};
385 };
386
387 switch (profile) {
388 case anonymization_profile::basic:
389 add_patient_identifiers();
390 add_institution_identifiers();
391 add_personnel_identifiers();
392 add_uid_replacements();
393 add_study_identifiers();
394 break;
395
396 case anonymization_profile::clean_pixel:
397 add_patient_identifiers();
398 add_institution_identifiers();
399 add_personnel_identifiers();
400 add_uid_replacements();
401 add_study_identifiers();
402 // Note: Pixel data cleaning requires image processing
403 // which is not implemented in this basic version
404 break;
405
406 case anonymization_profile::clean_descriptions:
407 add_patient_identifiers();
408 add_institution_identifiers();
409 add_personnel_identifiers();
410 add_uid_replacements();
411 add_study_identifiers();
412 actions[tags::study_description] = tag_action_config::make_empty();
413 actions[tags::series_description] = tag_action_config::make_empty();
414 break;
415
416 case anonymization_profile::retain_longitudinal:
417 add_patient_identifiers();
418 add_institution_identifiers();
419 add_personnel_identifiers();
420 add_uid_replacements();
421 add_study_identifiers();
422 add_date_shifting();
423 break;
424
425 case anonymization_profile::retain_patient_characteristics:
426 add_patient_identifiers();
427 add_institution_identifiers();
428 add_personnel_identifiers();
429 add_uid_replacements();
430 add_study_identifiers();
431 // Keep demographic data
432 actions[tags::patient_sex] = tag_action_config::make_keep();
433 actions[tags::patient_age] = tag_action_config::make_keep();
434 actions[tags::patient_size] = tag_action_config::make_keep();
435 actions[tags::patient_weight] = tag_action_config::make_keep();
436 break;
437
438 case anonymization_profile::hipaa_safe_harbor:
439 add_patient_identifiers();
440 add_institution_identifiers();
441 add_personnel_identifiers();
442 add_uid_replacements();
443 add_study_identifiers();
444 // Remove all HIPAA identifiers
445 for (const auto& tag : get_hipaa_identifier_tags()) {
446 if (actions.find(tag) == actions.end()) {
447 actions[tag] = tag_action_config::make_remove();
448 }
449 }
450 // Dates - keep only year or remove
451 actions[tags::patient_birth_date] = tag_action_config::make_remove();
452 break;
453
454 case anonymization_profile::gdpr_compliant:
455 add_patient_identifiers();
456 add_institution_identifiers();
457 add_personnel_identifiers();
458 add_uid_replacements();
459 add_study_identifiers();
460 // GDPR allows pseudonymization - use hash for linkage
461 actions[tags::patient_id] = tag_action_config::make_hash();
462 actions[tags::patient_name] = tag_action_config::make_hash();
463 break;
464 }
465
466 return actions;
467}

496auto anonymizer::apply_action(
497 dicom_dataset& dataset,
498 dicom_tag tag,
499 const tag_action_config& config,
500 uid_mapping* mapping
501) -> tag_action_record {
502 tag_action_record record;
503 record.tag = tag;
504 record.action = config.action;
505
506 auto* element = dataset.get(tag);
507 if (element == nullptr) {
508 record.success = false;
509 record.error_message = "Tag not found in dataset";
510 return record;
511 }
512
513 record.original_value = element->as_string().unwrap_or("");
514
515 switch (config.action) {
516 case tag_action::remove:
517 case tag_action::remove_or_empty:
518 record.success = dataset.remove(tag);
519 break;
520
521 case tag_action::empty:
522 dataset.set_string(tag, element->vr(), "");
523 record.new_value = "";
524 record.success = true;
525 break;
526
527 case tag_action::keep:
528 record.new_value = record.original_value;
529 record.success = true;
530 break;
531
532 case tag_action::replace:
533 dataset.set_string(tag, element->vr(), config.replacement_value);
534 record.new_value = config.replacement_value;
535 record.success = true;
536 break;
537
538 case tag_action::replace_uid:
539 if (mapping != nullptr) {
540 auto result = mapping->get_or_create(record.original_value);
541 if (result.is_ok()) {
542 dataset.set_string(tag, element->vr(), result.value());
543 record.new_value = result.value();
544 record.success = true;
545 } else {
546 record.success = false;
547 record.error_message = "Failed to create UID mapping";
548 }
549 } else {
550 // Generate new UID without mapping
551 uid_mapping temp;
552 auto result = temp.get_or_create(record.original_value);
553 if (result.is_ok()) {
554 dataset.set_string(tag, element->vr(), result.value());
555 record.new_value = result.value();
556 record.success = true;
557 } else {
558 record.success = false;
559 record.error_message = "Failed to generate new UID";
560 }
561 }
562 break;
563
564 case tag_action::hash: {
565 auto hashed = hash_value(record.original_value);
566 if (hashed.empty()) {
567 record.success = false;
568 record.error_message = "Hash computation failed";
569 } else {
570 dataset.set_string(tag, element->vr(), hashed);
571 record.new_value = hashed;
572 record.success = true;
573 }
574 break;
575 }
576
577 case tag_action::encrypt: {
578 auto result = encrypt_value(record.original_value);
579 if (result.is_ok()) {
580 dataset.set_string(tag, element->vr(), result.value());
581 record.new_value = result.value();
582 record.success = true;
583 } else {
584 record.success = false;
585 record.error_message = result.error().message;
586 }
587 break;
588 }
589
590 case tag_action::shift_date: {
591 if (date_offset_.has_value()) {
592 auto shifted = shift_date(record.original_value);
593 dataset.set_string(tag, element->vr(), shifted);
594 record.new_value = shifted;
595 record.success = true;
596 } else {
597 // No offset set - empty the date instead
598 dataset.set_string(tag, element->vr(), "");
599 record.new_value = "";
600 record.success = true;
601 }
602 break;
603 }
604 }
605
606 return record;
607}

609auto anonymizer::shift_date(std::string_view date_string) const -> std::string {
610 if (date_string.empty() || !date_offset_.has_value()) {
611 return "";
612 }
613
614 // Parse DICOM date format (YYYYMMDD)
615 if (date_string.length() < 8) {
616 return std::string(date_string);
617 }
618
619 try {
620 int year = std::stoi(std::string(date_string.substr(0, 4)));
621 int month = std::stoi(std::string(date_string.substr(4, 2)));
622 int day = std::stoi(std::string(date_string.substr(6, 2)));
623
624 // Create time point from date
625 std::tm tm = {};
626 tm.tm_year = year - 1900;
627 tm.tm_mon = month - 1;
628 tm.tm_mday = day;
629
630 auto time_point = std::chrono::system_clock::from_time_t(std::mktime(&tm));
631
632 // Apply offset
633 time_point += date_offset_.value();
634
635 // Convert back to date string
636 auto shifted_time = std::chrono::system_clock::to_time_t(time_point);
637 std::tm shifted_tm{};
638#if defined(_WIN32)
639 localtime_s(&shifted_tm, &shifted_time);
640#else
641 localtime_r(&shifted_time, &shifted_tm);
642#endif
643
644 std::ostringstream oss;
645 oss << std::setfill('0')
646 << std::setw(4) << (shifted_tm.tm_year + 1900)
647 << std::setw(2) << (shifted_tm.tm_mon + 1)
648 << std::setw(2) << shifted_tm.tm_mday;
649
650 return oss.str();
651
652 } catch (const std::exception&) {
653 return std::string(date_string);
654 }
655}

657auto anonymizer::hash_value(std::string_view value) const -> std::string {
658 std::string to_hash = std::string(value);
659
660 if (hash_salt_.has_value()) {
661 to_hash = hash_salt_.value() + to_hash;
662 }
663
664#ifdef PACS_WITH_DIGITAL_SIGNATURES
665 auto* ctx = EVP_MD_CTX_new();
666 if (ctx == nullptr) {
667 return {};
668 }
669
670 struct md_ctx_deleter {
671 void operator()(EVP_MD_CTX* c) const { EVP_MD_CTX_free(c); }
672 };
673 std::unique_ptr<EVP_MD_CTX, md_ctx_deleter> ctx_guard(ctx);
674
675 if (EVP_DigestInit_ex(ctx, EVP_sha256(), nullptr) != 1
676 || EVP_DigestUpdate(ctx, to_hash.data(), to_hash.size()) != 1) {
677 return {};
678 }
679
680 std::array<unsigned char, EVP_MAX_MD_SIZE> digest{};
681 unsigned int digest_len = 0;
682 if (EVP_DigestFinal_ex(ctx, digest.data(), &digest_len) != 1) {
683 return {};
684 }
685
686 std::ostringstream oss;
687 oss << std::hex << std::setfill('0');
688 for (unsigned int i = 0; i < digest_len; ++i) {
689 oss << std::setw(2) << static_cast<int>(digest[i]);
690 }
691 return oss.str();
692#else
693 // Non-cryptographic fallback when OpenSSL is unavailable
694 auto hash = std::hash<std::string>{}(to_hash);
695 std::ostringstream oss;
696 oss << std::hex << std::setfill('0') << std::setw(16) << hash;
697 return oss.str();
698#endif
699}

701auto anonymizer::encrypt_value(std::string_view value) const
702 -> kcenon::common::Result<std::string> {
703 if (encryption_key_.empty()) {
704 return kcenon::common::make_error<std::string>(
705 1, "No encryption key configured");
706 }
707
708#ifdef PACS_WITH_DIGITAL_SIGNATURES
709 constexpr int iv_length = 12;
710 constexpr int tag_length = 16;
711
712 // Generate random IV
713 std::array<unsigned char, iv_length> iv{};
714 if (RAND_bytes(iv.data(), iv_length) != 1) {
715 return kcenon::common::make_error<std::string>(
716 2, "Failed to generate random IV");
717 }
718
719 // Create cipher context with RAII
720 struct cipher_ctx_deleter {
721 void operator()(EVP_CIPHER_CTX* c) const { EVP_CIPHER_CTX_free(c); }
722 };
723
724 std::unique_ptr<EVP_CIPHER_CTX, cipher_ctx_deleter> ctx(
725 EVP_CIPHER_CTX_new());
726 if (!ctx) {
727 return kcenon::common::make_error<std::string>(
728 3, "Failed to create cipher context");
729 }
730
731 // Initialize AES-256-GCM
732 if (EVP_EncryptInit_ex(
733 ctx.get(), EVP_aes_256_gcm(), nullptr, nullptr, nullptr)
734 != 1) {
735 return kcenon::common::make_error<std::string>(
736 4, "Failed to initialize AES-256-GCM");
737 }
738
739 if (EVP_CIPHER_CTX_ctrl(
740 ctx.get(), EVP_CTRL_GCM_SET_IVLEN, iv_length, nullptr)
741 != 1) {
742 return kcenon::common::make_error<std::string>(
743 5, "Failed to set IV length");
744 }
745
746 if (EVP_EncryptInit_ex(ctx.get(), nullptr, nullptr,
747 encryption_key_.data(), iv.data())
748 != 1) {
749 return kcenon::common::make_error<std::string>(
750 6, "Failed to set encryption key and IV");
751 }
752
753 // Encrypt
754 std::vector<unsigned char> ciphertext(value.size() + EVP_MAX_BLOCK_LENGTH);
755 int out_len = 0;
756 if (EVP_EncryptUpdate(
757 ctx.get(), ciphertext.data(), &out_len,
758 reinterpret_cast<const unsigned char*>(value.data()),
759 static_cast<int>(value.size()))
760 != 1) {
761 return kcenon::common::make_error<std::string>(
762 7, "Encryption failed");
763 }
764 int ciphertext_len = out_len;
765
766 // Finalize
767 if (EVP_EncryptFinal_ex(
768 ctx.get(), ciphertext.data() + out_len, &out_len)
769 != 1) {
770 return kcenon::common::make_error<std::string>(
771 8, "Encryption finalization failed");
772 }
773 ciphertext_len += out_len;
774
775 // Get authentication tag
776 std::array<unsigned char, tag_length> tag{};
777 if (EVP_CIPHER_CTX_ctrl(
778 ctx.get(), EVP_CTRL_GCM_GET_TAG, tag_length, tag.data())
779 != 1) {
780 return kcenon::common::make_error<std::string>(
781 9, "Failed to get authentication tag");
782 }
783
784 // Encode as hex: IV || ciphertext || tag
785 std::ostringstream oss;
786 oss << std::hex << std::setfill('0');
787 for (int i = 0; i < iv_length; ++i) {
788 oss << std::setw(2) << static_cast<int>(iv[i]);
789 }
790 for (int i = 0; i < ciphertext_len; ++i) {
791 oss << std::setw(2) << static_cast<int>(ciphertext[i]);
792 }
793 for (int i = 0; i < tag_length; ++i) {
794 oss << std::setw(2) << static_cast<int>(tag[i]);
795 }
796
797 return kcenon::common::Result<std::string>(oss.str());
798#else
799 (void)value;
800 return kcenon::common::make_error<std::string>(
801 1, "Encryption requires OpenSSL (build with PACS_WITH_DIGITAL_SIGNATURES)");
802#endif
803}
