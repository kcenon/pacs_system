69auto pipeline_coordinator::start() -> VoidResult {
70 std::lock_guard<std::mutex> lock(mutex_);
71
72 if (running_.load(std::memory_order_acquire)) {
73 return pacs_void_error(error_codes::already_exists,
74 "Pipeline coordinator is already running");
75 }
76
77 // Create and start thread pools for each stage
78 for (size_t i = 0; i < stage_count; ++i) {
79 auto stage = static_cast<pipeline_stage>(i);
80 try {
81 stage_pools_[i] = create_stage_pool(stage);
82 if (!stage_pools_[i]->start()) {
83 // Cleanup already started pools
84 for (size_t j = 0; j < i; ++j) {
85 if (stage_pools_[j]) {
86 stage_pools_[j]->shutdown(false);
87 stage_pools_[j].reset();
88 }
89 }
90 return pacs_void_error(error_codes::internal_error,
91 "Failed to start thread pool for stage: " +
92 std::string(get_stage_name(stage)));
93 }
94
95 // Update metrics
96 if (config_.enable_metrics) {
97 auto& stage_metrics = metrics_.get_stage_metrics(stage);
98 stage_metrics.active_workers.store(
99 static_cast<uint32_t>(config_.get_workers_for_stage(stage)),
100 std::memory_order_relaxed);
101 }
102 } catch (const std::exception& e) {
103 // Cleanup already started pools
104 for (size_t j = 0; j < i; ++j) {
105 if (stage_pools_[j]) {
106 stage_pools_[j]->shutdown(false);
107 stage_pools_[j].reset();
108 }
109 }
110 return pacs_void_error(error_codes::internal_error,
111 "Exception creating thread pool for stage " +
112 std::string(get_stage_name(stage)) + ": " + e.what());
113 }
114 }
115
116 running_.store(true, std::memory_order_release);
117 return ok();
118}

120auto pipeline_coordinator::stop() -> VoidResult {
121 std::lock_guard<std::mutex> lock(mutex_);
122
123 if (!running_.load(std::memory_order_acquire)) {
124 return ok(); // Already stopped
125 }
126
127 running_.store(false, std::memory_order_release);
128
129 // Stop all stage pools with timeout consideration
130 bool all_stopped = true;
131 for (size_t i = 0; i < stage_count; ++i) {
132 if (stage_pools_[i]) {
133 // Use wait_for_completion=true to drain queues gracefully
134 stage_pools_[i]->shutdown(true);
135 stage_pools_[i].reset();
136 }
137 }
138
139 if (!all_stopped) {
140 return pacs_void_error(error_codes::internal_error,
141 "Some stage pools did not shutdown cleanly");
142 }
143
144 return ok();
145}

155auto pipeline_coordinator::submit_to_stage(pipeline_stage stage,
156 std::unique_ptr<pipeline_job_base> job)
157 -> VoidResult {
158 if (!job) {
159 return pacs_void_error(error_codes::invalid_argument,
160 "Cannot submit null job");
161 }
162 // Extract context before moving job to avoid undefined behavior
163 // (evaluation order of function arguments is unspecified in C++)
164 auto ctx = job->get_context();
165 return submit_to_stage(stage, std::move(job), ctx);
166}
167
168auto pipeline_coordinator::submit_to_stage(pipeline_stage stage,
169 std::unique_ptr<pipeline_job_base> job,
170 const job_context& ctx) -> VoidResult {
171 if (!running_.load(std::memory_order_acquire)) {
172 return pacs_void_error(error_codes::not_initialized,
173 "Pipeline coordinator is not running");
174 }
175
176 size_t stage_idx = static_cast<size_t>(stage);
177 if (stage_idx >= stage_count || !stage_pools_[stage_idx]) {
178 return pacs_void_error(error_codes::invalid_argument,
179 "Invalid pipeline stage");
180 }
181
182 // Check backpressure
183 check_backpressure(stage);
184
185 // Update metrics
186 if (config_.enable_metrics) {
187 metrics_.increment_stage_queue(stage);
188 }
189
190 // Capture job for lambda
191 auto job_ptr = job.release();
192 auto start_time = std::chrono::steady_clock::now();
193
194 // Submit to stage pool
195 stage_pools_[stage_idx]->submit_fire_and_forget(
196 [this, job_ptr, stage, ctx_copy = ctx, start_time]() mutable {
197 std::unique_ptr<pipeline_job_base> owned_job(job_ptr);
198
199 // Execute the job
200 auto result = owned_job->execute(*this);
201
202 // Record completion
203 auto end_time = std::chrono::steady_clock::now();
204 auto duration_ns = std::chrono::duration_cast<std::chrono::nanoseconds>(
205 end_time - start_time).count();
206
207 bool success = result.is_ok();
208
209 if (config_.enable_metrics) {
210 metrics_.decrement_stage_queue(stage);
211 metrics_.record_stage_completion(stage,
212 static_cast<uint64_t>(duration_ns), success);
213 }
214
215 notify_job_completion(ctx_copy, success);
216 });
217
218 return ok();
219}

221auto pipeline_coordinator::submit_task(pipeline_stage stage,
222 std::function<void()> task) -> VoidResult {
223 if (!running_.load(std::memory_order_acquire)) {
224 return pacs_void_error(error_codes::not_initialized,
225 "Pipeline coordinator is not running");
226 }
227
228 size_t stage_idx = static_cast<size_t>(stage);
229 if (stage_idx >= stage_count || !stage_pools_[stage_idx]) {
230 return pacs_void_error(error_codes::invalid_argument,
231 "Invalid pipeline stage");
232 }
233
234 if (config_.enable_metrics) {
235 metrics_.increment_stage_queue(stage);
236 }
237
238 stage_pools_[stage_idx]->submit_fire_and_forget(
239 [this, stage, task = std::move(task)]() {
240 task();
241 if (config_.enable_metrics) {
242 metrics_.decrement_stage_queue(stage);
243 }
244 });
245
246 return ok();
247}

262auto pipeline_coordinator::get_queue_depth(pipeline_stage stage) const noexcept
263 -> size_t {
264 size_t stage_idx = static_cast<size_t>(stage);
265 if (stage_idx >= stage_count || !stage_pools_[stage_idx]) {
266 return 0;
267 }

353void pipeline_coordinator::notify_job_completion(const job_context& ctx, bool success) {
354 job_completion_callback callback;
355 {
356 std::lock_guard<std::mutex> lock(mutex_);

359
