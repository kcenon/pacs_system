77auto dicom_element::as_string() const -> kcenon::pacs::Result<std::string> {
78 if (data_.empty()) {
79 return kcenon::pacs::ok(std::string{});
80 }
81
82 // For string VRs, convert bytes to string and remove padding
83 if (encoding::is_string_vr(vr_)) {
84 std::string_view raw{reinterpret_cast<const char*>(data_.data()),
85 data_.size()};
86 return kcenon::pacs::ok(remove_padding(raw, vr_));
87 }
88
89 // For numeric VRs, convert to string representation
90 if (encoding::is_numeric_vr(vr_)) {
91 switch (vr_) {
92 case encoding::vr_type::US:
93 if (data_.size() >= 2) {
94 if (auto val = as_numeric<uint16_t>(); val.is_ok())
95 return kcenon::pacs::ok(std::to_string(val.value()));
96 }
97 break;
98 case encoding::vr_type::SS:
99 if (data_.size() >= 2) {
100 if (auto val = as_numeric<int16_t>(); val.is_ok())
101 return kcenon::pacs::ok(std::to_string(val.value()));
102 }
103 break;
104 case encoding::vr_type::UL:
105 if (data_.size() >= 4) {
106 if (auto val = as_numeric<uint32_t>(); val.is_ok())
107 return kcenon::pacs::ok(std::to_string(val.value()));
108 }
109 break;
110 case encoding::vr_type::SL:
111 if (data_.size() >= 4) {
112 if (auto val = as_numeric<int32_t>(); val.is_ok())
113 return kcenon::pacs::ok(std::to_string(val.value()));
114 }
115 break;
116 case encoding::vr_type::UV:
117 if (data_.size() >= 8) {
118 if (auto val = as_numeric<uint64_t>(); val.is_ok())
119 return kcenon::pacs::ok(std::to_string(val.value()));
120 }
121 break;
122 case encoding::vr_type::SV:
123 if (data_.size() >= 8) {
124 if (auto val = as_numeric<int64_t>(); val.is_ok())
125 return kcenon::pacs::ok(std::to_string(val.value()));
126 }
127 break;
128 case encoding::vr_type::FL:
129 if (data_.size() >= 4) {
130 if (auto val = as_numeric<float>(); val.is_ok())
131 return kcenon::pacs::ok(std::to_string(val.value()));
132 }
133 break;
134 case encoding::vr_type::FD:
135 if (data_.size() >= 8) {
136 if (auto val = as_numeric<double>(); val.is_ok())
137 return kcenon::pacs::ok(std::to_string(val.value()));
138 }
139 break;
140 default:
141 break;
142 }
143 }
144
145 // For binary VRs or unknown, return raw bytes as string
146 return kcenon::pacs::ok(
147 std::string{reinterpret_cast<const char*>(data_.data()), data_.size()});
148}

150auto dicom_element::as_string_list() const
151 -> kcenon::pacs::Result<std::vector<std::string>> {
152 auto str_result = as_string();
153 if (str_result.is_err()) {
154 return kcenon::pacs::pacs_error<std::vector<std::string>>(
155 str_result.error().code, str_result.error().message);
156 }
157 const std::string str = str_result.value();
158
159 std::vector<std::string> result;
160 if (str.empty()) {
161 return kcenon::pacs::ok(result);
162 }
163
164 // Split by backslash (DICOM VM delimiter)
165 std::string::size_type start = 0;
166 std::string::size_type pos = 0;
167
168 while ((pos = str.find('\\', start)) != std::string::npos) {
169 result.push_back(str.substr(start, pos - start));
170 start = pos + 1;
171 }
172
173 // Add the last segment
174 result.push_back(str.substr(start));
175
176 return kcenon::pacs::ok(result);
177}

226auto dicom_element::apply_padding(std::string_view str) const -> std::string {
227 std::string result{str};
228
229 // DICOM requires even length for all values
230 if (result.length() % 2 != 0) {
231 result += encoding::padding_char(vr_);
232 }
233
234 return result;
235}

237auto dicom_element::remove_padding(std::string_view str,
238 encoding::vr_type vr) -> std::string {
239 if (str.empty()) {
240 return {};
241 }
242
243 std::string result{str};
244
245 // Determine padding character based on VR
246 const char pad = encoding::padding_char(vr);
247
248 // Remove trailing padding characters
249 while (!result.empty() && result.back() == pad) {
250 result.pop_back();
251 }
252
253 // For string VRs (except UI), also trim trailing spaces
254 if (encoding::is_string_vr(vr) && vr != encoding::vr_type::UI) {
255 while (!result.empty() && result.back() == ' ') {
256 result.pop_back();
257 }
258 }
259
260 return result;
261}
