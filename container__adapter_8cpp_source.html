47auto container_adapter::to_container_value(const core::dicom_element& element)
48 -> container_module::optimized_value {
49 container_module::optimized_value result;
50 const auto vr = element.vr();
51 const auto tag = element.tag();
52
53 // Use tag string as the value name
54 result.name = make_element_key(tag, vr);
55
56 // Handle empty elements
57 // For string VRs, keep as empty string; for others, treat as null
58 if (element.is_empty()) {
59 if (encoding::is_string_vr(vr)) {
60 result.type = container_module::value_types::string_value;
61 result.data = std::string{};
62 } else {
63 result.type = container_module::value_types::null_value;
64 result.data = std::monostate{};
65 }
66 return result;
67 }
68
69 // Handle sequence type specially
70 if (element.is_sequence()) {
71 result.type = container_module::value_types::container_value;
72 result.data = sequence_to_container(element);
73 return result;
74 }
75
76 // Map VR to appropriate container type
77 if (encoding::is_string_vr(vr)) {
78 result.type = container_module::value_types::string_value;
79 result.data = element.as_string().unwrap_or("");
80 } else if (encoding::is_numeric_vr(vr)) {
81 // Map numeric VRs to appropriate types
82 // Use fallback to bytes if numeric conversion fails
83 bool converted = false;
84 switch (vr) {
85 case encoding::vr_type::SS: { // Signed Short (2 bytes)
86 auto val = element.as_numeric<short>();
87 if (val.is_ok()) {
88 result.type = container_module::value_types::short_value;
89 result.data = val.value();
90 converted = true;
91 }
92 break;
93 }
94 case encoding::vr_type::US: { // Unsigned Short (2 bytes)
95 auto val = element.as_numeric<unsigned short>();
96 if (val.is_ok()) {
97 result.type = container_module::value_types::ushort_value;
98 result.data = val.value();
99 converted = true;
100 }
101 break;
102 }
103 case encoding::vr_type::SL: { // Signed Long (4 bytes)
104 auto val = element.as_numeric<int>();
105 if (val.is_ok()) {
106 result.type = container_module::value_types::int_value;
107 result.data = val.value();
108 converted = true;
109 }
110 break;
111 }
112 case encoding::vr_type::UL: { // Unsigned Long (4 bytes)
113 auto val = element.as_numeric<unsigned int>();
114 if (val.is_ok()) {
115 result.type = container_module::value_types::uint_value;
116 result.data = val.value();
117 converted = true;
118 }
119 break;
120 }
121 case encoding::vr_type::SV: { // Signed 64-bit
122 auto val = element.as_numeric<long long>();
123 if (val.is_ok()) {
124 result.type = container_module::value_types::llong_value;
125 result.data = val.value();
126 converted = true;
127 }
128 break;
129 }
130 case encoding::vr_type::UV: { // Unsigned 64-bit
131 auto val = element.as_numeric<unsigned long long>();
132 if (val.is_ok()) {
133 result.type = container_module::value_types::ullong_value;
134 result.data = val.value();
135 converted = true;
136 }
137 break;
138 }
139 case encoding::vr_type::FL: { // Float
140 auto val = element.as_numeric<float>();
141 if (val.is_ok()) {
142 result.type = container_module::value_types::float_value;
143 result.data = val.value();
144 converted = true;
145 }
146 break;
147 }
148 case encoding::vr_type::FD: { // Double
149 auto val = element.as_numeric<double>();
150 if (val.is_ok()) {
151 result.type = container_module::value_types::double_value;
152 result.data = val.value();
153 converted = true;
154 }
155 break;
156 }
157 default:
158 break;
159 }
160 if (!converted) {
161 // Fallback to bytes for unhandled or failed numeric types
162 result.type = container_module::value_types::bytes_value;
163 auto raw = element.raw_data();
164 result.data = std::vector<uint8_t>(raw.begin(), raw.end());
165 }
166 } else if (encoding::is_binary_vr(vr)) {
167 result.type = container_module::value_types::bytes_value;
168 auto raw = element.raw_data();
169 result.data = std::vector<uint8_t>(raw.begin(), raw.end());
170 } else if (vr == encoding::vr_type::AT) {
171 // Attribute Tag - store as 32-bit unsigned
172 auto val = element.as_numeric<unsigned int>();
173 if (val.is_ok()) {
174 result.type = container_module::value_types::uint_value;
175 result.data = val.value();
176 } else {
177 result.type = container_module::value_types::bytes_value;
178 auto raw = element.raw_data();
179 result.data = std::vector<uint8_t>(raw.begin(), raw.end());
180 }
181 } else {
182 // Default: store as bytes
183 result.type = container_module::value_types::bytes_value;
184 auto raw = element.raw_data();
185 result.data = std::vector<uint8_t>(raw.begin(), raw.end());
186 }
187
188 return result;
189}

191auto container_adapter::from_container_value(
192 core::dicom_tag tag,
193 encoding::vr_type vr,
194 const container_module::optimized_value& val) -> core::dicom_element {
195 // Handle null/empty values
196 if (val.type == container_module::value_types::null_value) {
197 return core::dicom_element{tag, vr};
198 }
199
200 // Handle sequence type
201 if (vr == encoding::vr_type::SQ) {
202 auto elem = core::dicom_element{tag, vr};
203 if (std::holds_alternative<std::shared_ptr<container_module::value_container>>(
204 val.data)) {
205 auto container = std::get<std::shared_ptr<container_module::value_container>>(val.data);
206 if (container) {
207 auto items = container_to_sequence(*container);
208 for (auto& item : items) {
209 elem.sequence_items().push_back(std::move(item));
210 }
211 }
212 }
213 return elem;
214 }
215
216 // Handle string VRs
217 if (encoding::is_string_vr(vr)) {
218 std::string str_val;
219 if (std::holds_alternative<std::string>(val.data)) {
220 str_val = std::get<std::string>(val.data);
221 } else {
222 str_val = container_module::variant_helpers::to_string(val.data, val.type);
223 }
224 return core::dicom_element::from_string(tag, vr, str_val);
225 }
226
227 // Handle numeric VRs
228 if (encoding::is_numeric_vr(vr)) {
229 switch (vr) {
230 case encoding::vr_type::SS:
231 if (std::holds_alternative<short>(val.data)) {
232 return core::dicom_element::from_numeric(tag, vr,
233 std::get<short>(val.data));
234 }
235 break;
236 case encoding::vr_type::US:
237 if (std::holds_alternative<unsigned short>(val.data)) {
238 return core::dicom_element::from_numeric(tag, vr,
239 std::get<unsigned short>(val.data));
240 }
241 break;
242 case encoding::vr_type::SL:
243 if (std::holds_alternative<int>(val.data)) {
244 return core::dicom_element::from_numeric(tag, vr,
245 std::get<int>(val.data));
246 }
247 break;
248 case encoding::vr_type::UL:
249 if (std::holds_alternative<unsigned int>(val.data)) {
250 return core::dicom_element::from_numeric(tag, vr,
251 std::get<unsigned int>(val.data));
252 }
253 break;
254 case encoding::vr_type::SV:
255 if (std::holds_alternative<long long>(val.data)) {
256 return core::dicom_element::from_numeric(tag, vr,
257 std::get<long long>(val.data));
258 }
259 break;
260 case encoding::vr_type::UV:
261 if (std::holds_alternative<unsigned long long>(val.data)) {
262 return core::dicom_element::from_numeric(tag, vr,
263 std::get<unsigned long long>(val.data));
264 }
265 break;
266 case encoding::vr_type::FL:
267 if (std::holds_alternative<float>(val.data)) {
268 return core::dicom_element::from_numeric(tag, vr,
269 std::get<float>(val.data));
270 }
271 break;
272 case encoding::vr_type::FD:
273 if (std::holds_alternative<double>(val.data)) {
274 return core::dicom_element::from_numeric(tag, vr,
275 std::get<double>(val.data));
276 }
277 break;
278 default:
279 break;
280 }
281 }
282
283 // Handle binary VRs or fallback
284 if (std::holds_alternative<std::vector<uint8_t>>(val.data)) {
285 auto& bytes = std::get<std::vector<uint8_t>>(val.data);
286 return core::dicom_element{tag, vr, bytes};
287 }
288
289 // Return empty element as last resort
290 return core::dicom_element{tag, vr};
291}

297auto container_adapter::serialize_dataset(const core::dicom_dataset& dataset)
298 -> std::shared_ptr<container_module::value_container> {
299 auto container = std::make_shared<container_module::value_container>();
300
301 // Add metadata
302 container->set(kVersionKey, std::string(kProtocolVersion));
303 container->set(kElementCountKey, static_cast<unsigned int>(dataset.size()));
304
305 // Serialize each element
306 for (const auto& [tag, element] : dataset) {
307 auto value = to_container_value(element);
308 container->set(value);
309 }
310
311 return container;
312}

314auto container_adapter::deserialize_dataset(
315 const container_module::value_container& container)
316 -> Result<core::dicom_dataset> {
317 core::dicom_dataset dataset;
318
319 // Process each value in the container
320 for (const auto& val : container) {
321 // Skip metadata keys
322 if (val.name.starts_with("_")) {
323 continue;
324 }
325
326 // Parse element key to get tag and VR
327 auto parsed = parse_element_key(val.name);
328 if (!parsed) {
329 continue; // Skip invalid keys
330 }
331
332 auto [tag, vr] = *parsed;
333
334 // Convert value back to DICOM element
335 try {
336 auto element = from_container_value(tag, vr, val);
337 dataset.insert(std::move(element));
338 } catch (const std::exception& e) {
339 return Result<core::dicom_dataset>::err(
340 kcenon::common::error_info{
341 kcenon::pacs::compat::format("Failed to convert element {}: {}", val.name, e.what())
342 });
343 }
344 }
345
346 return Result<core::dicom_dataset>::ok(std::move(dataset));
347}

364auto container_adapter::from_binary(std::span<const uint8_t> data)
365 -> Result<core::dicom_dataset> {
366 container_module::value_container container;
367
368 auto deserialize_result = container.deserialize(data);
369 if (deserialize_result.is_err()) {
370 return Result<core::dicom_dataset>::err(
371 kcenon::common::error_info{
372 "Failed to deserialize binary data to container"
373 });
374 }
375
376 return deserialize_dataset(container);
377}

383auto container_adapter::get_container_type(encoding::vr_type vr) noexcept
384 -> container_module::value_types {
385 if (encoding::is_string_vr(vr)) {
386 return container_module::value_types::string_value;
387 }
388
389 switch (vr) {
390 case encoding::vr_type::SS:
391 return container_module::value_types::short_value;
392 case encoding::vr_type::US:
393 return container_module::value_types::ushort_value;
394 case encoding::vr_type::SL:
395 return container_module::value_types::int_value;
396 case encoding::vr_type::UL:
397 case encoding::vr_type::AT:
398 return container_module::value_types::uint_value;
399 case encoding::vr_type::SV:
400 return container_module::value_types::llong_value;
401 case encoding::vr_type::UV:
402 return container_module::value_types::ullong_value;
403 case encoding::vr_type::FL:
404 return container_module::value_types::float_value;
405 case encoding::vr_type::FD:
406 return container_module::value_types::double_value;
407 case encoding::vr_type::SQ:
408 return container_module::value_types::container_value;
409 default:
410 return container_module::value_types::bytes_value;
411 }

425auto container_adapter::parse_element_key(std::string_view key)
426 -> std::optional<std::pair<core::dicom_tag, encoding::vr_type>> {
427 // Expected format: "GGGG,EEEE:VR"
428 if (key.size() < 12) { // Minimum: "0000,0000:XX"
429 return std::nullopt;
430 }
431
432 // Find the colon separator
433 auto colon_pos = key.find(':');
434 if (colon_pos == std::string_view::npos || colon_pos < 9) {
435 return std::nullopt;
436 }
437
438 // Parse group (first 4 hex chars)
439 uint16_t group = 0;
440 auto group_str = key.substr(0, 4);
441 auto group_result = std::from_chars(group_str.data(),
442 group_str.data() + group_str.size(),
443 group, 16);
444 if (group_result.ec != std::errc{}) {
445 return std::nullopt;
446 }
447
448 // Parse element (chars 5-8, after comma)
449 uint16_t element = 0;
450 auto elem_str = key.substr(5, 4);
451 auto elem_result = std::from_chars(elem_str.data(),
452 elem_str.data() + elem_str.size(),
453 element, 16);
454 if (elem_result.ec != std::errc{}) {
455 return std::nullopt;
456 }
457
458 // Parse VR (2 chars after colon)
459 auto vr_str = key.substr(colon_pos + 1, 2);
460 auto vr_opt = encoding::from_string(vr_str);
461 if (!vr_opt) {
462 return std::nullopt;
463 }
464
465 return std::make_pair(core::dicom_tag{group, element}, *vr_opt);
466}

468auto container_adapter::sequence_to_container(const core::dicom_element& element)
469 -> std::shared_ptr<container_module::value_container> {
470 auto container = std::make_shared<container_module::value_container>();
471
472 const auto& items = element.sequence_items();
473 container->set("_item_count", static_cast<unsigned int>(items.size()));
474
475 size_t item_index = 0;
476 for (const auto& item : items) {
477 auto item_container = serialize_dataset(item);
478 container_module::optimized_value val;
479 val.name = kcenon::pacs::compat::format("item_{}", item_index++);
480 val.type = container_module::value_types::container_value;
481 val.data = item_container;
482 container->set(val);
483 }
484
485 return container;
486}

488auto container_adapter::container_to_sequence(
489 const container_module::value_container& container)
490 -> std::vector<core::dicom_dataset> {
491 std::vector<core::dicom_dataset> items;
492
493 for (const auto& val : container) {
494 if (!val.name.starts_with("item_")) {
495 continue;
496 }
497
498 if (std::holds_alternative<std::shared_ptr<container_module::value_container>>(
499 val.data)) {
500 auto item_container =
501 std::get<std::shared_ptr<container_module::value_container>>(val.data);
502 if (item_container) {
503 auto result = deserialize_dataset(*item_container);
504 if (result.is_ok()) {
505 items.push_back(std::move(result.value()));
506 }
507 }
508 }
509 }
510
511 return items;
512}
