62

67
68 // Create DICOM server with the provided configuration

77Result<network_adapter::session_ptr>
78network_adapter::connect(const connection_config& config) {
79 // Validate configuration
80 if (config.host.empty()) {
81 return Result<session_ptr>(error_info("Connection failed: empty host"));
82 }
83
84 if (config.port == 0) {
85 return Result<session_ptr>(error_info("Connection failed: invalid port"));
86 }
87
88 // Validate TLS configuration if enabled
89 if (config.tls.enabled && !config.tls.is_valid()) {
90 return Result<session_ptr>(error_info("Connection failed: invalid TLS configuration"));
91 }
92
93 try {
94 // Create TCP client via tcp_facade
95 kcenon::network::facade::tcp_facade facade;
96 kcenon::network::facade::tcp_facade::client_config client_cfg;
97 client_cfg.host = config.host;
98 client_cfg.port = config.port;
99 client_cfg.client_id = "pacs_client";
100 client_cfg.timeout = config.timeout;
101 client_cfg.use_ssl = config.tls.enabled;
102 if (config.tls.enabled && !config.tls.ca_path.empty()) {
103 client_cfg.ca_cert_path = config.tls.ca_path.string();
104 }
105 client_cfg.verify_certificate = config.tls.verify_peer;
106
107 auto client = facade.create_client(client_cfg);
108
109 // Set up promise/future for synchronous connection
110 std::promise<std::error_code> connect_promise;
111 auto connect_future = connect_promise.get_future();
112
113 // Use deprecated callbacks for synchronous connect pattern
114 // (observer pattern would be preferred for long-lived clients)
115#if defined(__clang__) || defined(__GNUC__)
116#pragma GCC diagnostic push
117#pragma GCC diagnostic ignored "-Wdeprecated-declarations"
118#endif
119 client->set_connected_callback([&connect_promise]() {
120 connect_promise.set_value(std::error_code{});
121 });
122
123 client->set_error_callback([&connect_promise](std::error_code ec) {
124 try {
125 connect_promise.set_value(ec);
126 } catch (const std::future_error&) {
127 // Promise already satisfied
128 }
129 });
130#if defined(__clang__) || defined(__GNUC__)
131#pragma GCC diagnostic pop
132#endif
133
134 // Start connection
135 auto start_result = client->start(config.host, config.port);
136 if (start_result.is_err()) {
137 return Result<session_ptr>(error_info("Connection failed: unable to start client"));
138 }
139
140 // Wait for connection with timeout
141 auto status = connect_future.wait_for(config.timeout);
142 if (status == std::future_status::timeout) {
143 (void)client->stop();
144 return Result<session_ptr>(error_info("Connection failed: timeout"));
145 }
146
147 // Check connection result
148 auto ec = connect_future.get();
149 if (ec) {
150 (void)client->stop();
151 return Result<session_ptr>(error_info("Connection failed: " + ec.message()));
152 }
153
154 // Connection successful but client-based session wrapping
155 // is not yet fully implemented for the public API
156 return Result<session_ptr>(error_info("Connection not yet fully implemented"));
157
158 } catch (const std::exception& e) {
159 return Result<session_ptr>(
160 error_info(std::string("Connection failed: ") + e.what()));
161 }
162}

177Result<std::monostate>
178network_adapter::configure_tls(const tls_config& config) {
179 if (!config.enabled) {
180 return Result<std::monostate>(std::monostate{});
181 }
182
183 // Validate certificate path
184 if (!std::filesystem::exists(config.cert_path)) {
185 return Result<std::monostate>(error_info(
186 "TLS configuration error: certificate file not found: " +
187 config.cert_path.string()));
188 }
189
190 // Validate key path
191 if (!std::filesystem::exists(config.key_path)) {
192 return Result<std::monostate>(error_info(
193 "TLS configuration error: key file not found: " +
194 config.key_path.string()));
195 }
196
197 // Validate CA path if specified
198 if (!config.ca_path.empty() && !std::filesystem::exists(config.ca_path)) {
199 return Result<std::monostate>(error_info(
200 "TLS configuration error: CA certificate file not found: " +
201 config.ca_path.string()));
202 }
203
204 // TLS version validation
205 if (config.min_version != tls_config::tls_version::v1_2 &&
206 config.min_version != tls_config::tls_version::v1_3) {
207 return Result<std::monostate>(error_info(
208 "TLS configuration error: unsupported TLS version"));
209 }
210
211 return Result<std::monostate>(std::monostate{});
212}
