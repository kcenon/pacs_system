116struct query_result {
118 std::vector<core::dicom_dataset> matches;
119
121 uint16_t status{0};
122
124 std::chrono::milliseconds elapsed{0};
125
128 size_t total_pending{0};
129
131 [[nodiscard]] bool is_success() const noexcept {
132 return status == 0x0000;
133 }
134
136 [[nodiscard]] bool is_cancelled() const noexcept {
137 return status == 0xFE00;
138 }
139};
