61struct pdu_pool_statistics {
62 std::atomic<uint64_t> total_acquisitions{0};
63 std::atomic<uint64_t> pool_hits{0};
64 std::atomic<uint64_t> pool_misses{0};
65 std::atomic<uint64_t> total_releases{0};
66 std::atomic<uint64_t> total_bytes_allocated{0};
67
72 [[nodiscard]] auto hit_ratio() const noexcept -> double {
73 const uint64_t total = total_acquisitions.load(std::memory_order_relaxed);
74 if (total == 0) {
75 return 0.0;
76 }
77 return static_cast<double>(pool_hits.load(std::memory_order_relaxed))
78 / static_cast<double>(total);
79 }
80
84 void reset() noexcept {
85 total_acquisitions.store(0, std::memory_order_relaxed);
86 pool_hits.store(0, std::memory_order_relaxed);
87 pool_misses.store(0, std::memory_order_relaxed);
88 total_releases.store(0, std::memory_order_relaxed);
89 total_bytes_allocated.store(0, std::memory_order_relaxed);
90 }
91};

189template <typename T>
190class tracked_pdu_pool {
191public:
192 using pool_type = kcenon::common::utils::ObjectPool<T>;
193 using unique_ptr_type = std::unique_ptr<T, std::function<void(T*)>>;
194
195 explicit tracked_pdu_pool(std::size_t initial_size = 64)
196 : pool_(initial_size) {
197 pool_.reserve(initial_size);
198 }
199
205 template <typename... Args>
206 auto acquire(Args&&... args) -> unique_ptr_type {
207 bool reused = false;
208 auto ptr = pool_.acquire(&reused, std::forward<Args>(args)...);
209
210 stats_.total_acquisitions.fetch_add(1, std::memory_order_relaxed);
211 if (reused) {
212 stats_.pool_hits.fetch_add(1, std::memory_order_relaxed);
213 } else {
214 stats_.pool_misses.fetch_add(1, std::memory_order_relaxed);
215 }
216
217 return ptr;
218 }
219
224 [[nodiscard]] auto statistics() const noexcept -> const pdu_pool_statistics& {
225 return stats_;
226 }
227
232 [[nodiscard]] auto available() const -> std::size_t {
233 return pool_.available();
234 }
235
240 void reserve(std::size_t count) {
241 pool_.reserve(count);
242 }
243
247 void clear() {
248 pool_.clear();
249 }
250
251private:
252 pool_type pool_;
253 pdu_pool_statistics stats_;
254};
