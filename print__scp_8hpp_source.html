108[[nodiscard]] inline auto to_string(printer_status status) -> std::string_view {
109 switch (status) {
110 case printer_status::normal:
111 return "NORMAL";
112 case printer_status::warning:
113 return "WARNING";
114 case printer_status::failure:
115 return "FAILURE";
116 default:
117 return "NORMAL";
118 }
