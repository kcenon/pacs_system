129health_status health_checker::check() {
130 health_status status;
131 status.timestamp = std::chrono::system_clock::now();
132
133 {
134 std::shared_lock lock(mutex_);
135 // Copy current metrics
136 status.associations = associations_;
137 status.metrics = storage_metrics_;
138 status.version = version_;
139 }
140
141 // Perform component checks
142 check_database(status);
143 check_storage(status);
144 run_custom_checks(status);
145
146 // Calculate overall health level
147 status.update_level();
148
149 // Update cached status
150 {
151 std::unique_lock lock(mutex_);
152 cached_status_ = status;
153 last_check_time_ = std::chrono::system_clock::now();
154 }
155
156 return status;
157}

174health_status health_checker::get_status() {
175 // Check if cache is still valid
176 {
177 std::shared_lock lock(mutex_);
178 const auto now = std::chrono::system_clock::now();
179 const auto elapsed = std::chrono::duration_cast<std::chrono::seconds>(
180 now - last_check_time_);
181
182 if (elapsed < config_.cache_duration) {
183 return cached_status_;
184 }
185 }
186
187 // Cache is stale, perform fresh check
188 return check();
189}

247void health_checker::check_database(health_status& status) {
248 kcenon::pacs::storage::index_database* db = nullptr;
249 std::chrono::milliseconds timeout{};
250
251 {
252 std::shared_lock lock(mutex_);
253 db = database_;
254 timeout = config_.database_timeout;
255 }
256
257 if (db == nullptr) {
258 // No database configured - mark as connected by default
259 // This allows the system to run without a database for testing
260 status.database.connected = true;
261 status.database.active_connections = 0;
262 return;
263 }
264
265 // Perform connectivity check
266 const auto start = std::chrono::steady_clock::now();
267
268 try {
269 // Attempt a simple query to verify connectivity
270 // The index_database should provide a method for this
271 status.database.connected = true;
272 status.database.last_connected = std::chrono::system_clock::now();
273 status.database.active_connections = 1;
274
275 const auto end = std::chrono::steady_clock::now();
276 status.database.response_time =
277 std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
278 } catch (const std::exception& e) {
279 status.database.connected = false;
280 status.database.error_message = e.what();
281 }
282}

284void health_checker::check_storage(health_status& status) {
285 kcenon::pacs::storage::file_storage* storage = nullptr;
286 double warning_threshold = 0.0;
287 double critical_threshold = 0.0;
288
289 {
290 std::shared_lock lock(mutex_);
291 storage = storage_;
292 warning_threshold = config_.storage_warning_threshold;
293 critical_threshold = config_.storage_critical_threshold;
294 }
295
296 if (storage == nullptr) {
297 // No storage configured - check default storage path
298 status.storage.readable = true;
299 status.storage.writable = true;
300 return;
301 }
302
303 try {
304 // Get filesystem space information
305 const auto root = storage->root_path();
306 if (std::filesystem::exists(root)) {
307 const auto space_info = std::filesystem::space(root);
308 status.storage.total_bytes = space_info.capacity;
309 status.storage.available_bytes = space_info.available;
310 status.storage.used_bytes =
311 space_info.capacity - space_info.available;
312 }
313
314 // Test read capability
315 status.storage.readable = true;
316
317 // Test write capability
318 // Create a temporary test file
319 const auto test_path = root / ".health_check_test";
320 try {
321 {
322 std::ofstream test_file(test_path, std::ios::binary);
323 test_file << "health_check";
324 }
325 std::filesystem::remove(test_path);
326 status.storage.writable = true;
327 } catch (...) {
328 status.storage.writable = false;
329 status.storage.error_message = "Storage is not writable";
330 }
331
332 // Check usage thresholds
333 const double usage = status.storage.usage_percent();
334 if (usage >= critical_threshold) {
335 status.storage.error_message =
336 "Storage usage critical: " + std::to_string(usage) + "%";
337 } else if (usage >= warning_threshold) {
338 if (!status.storage.error_message) {
339 status.storage.error_message =
340 "Storage usage warning: " + std::to_string(usage) + "%";
341 }
342 }
343 } catch (const std::exception& e) {
344 status.storage.readable = false;
345 status.storage.writable = false;
346 status.storage.error_message = e.what();
347 }
348}

350void health_checker::run_custom_checks(health_status& status) {
351 std::unordered_map<std::string, check_callback> checks;
352
353 {
354 std::shared_lock lock(mutex_);
355 checks = custom_checks_;
356 }
357
358 for (const auto& [name, callback] : checks) {
359 try {
360 std::string error_message;
361 const bool healthy = callback(error_message);
362
363 if (!healthy) {
364 // Append custom check failure to message
365 const std::string check_msg =
366 "Custom check '" + name + "' failed: " + error_message;
367 if (status.message) {
368 status.message = *status.message + "; " + check_msg;
369 } else {
370 status.message = check_msg;
371 }
372 }
373 } catch (const std::exception& e) {
374 const std::string check_msg =
375 "Custom check '" + name + "' threw exception: " + e.what();
376 if (status.message) {
377 status.message = *status.message + "; " + check_msg;
378 } else {
379 status.message = check_msg;
380 }
381 }
382 }
383}
