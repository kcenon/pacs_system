83[[nodiscard]] inline auto to_string(lock_type type) -> std::string {
84 switch (type) {
85 case lock_type::exclusive: return "exclusive";
86 case lock_type::shared: return "shared";
87 case lock_type::migration: return "migration";
88 default: return "unknown";
89 }
90}
91
95[[nodiscard]] inline auto parse_lock_type(const std::string& str)
96 -> std::optional<lock_type> {
97 if (str == "exclusive") return lock_type::exclusive;
98 if (str == "shared") return lock_type::shared;
99 if (str == "migration") return lock_type::migration;
100 return std::nullopt;
101}

110struct lock_token {
112 std::string token_id;
113
115 std::string study_uid;
116
118 lock_type type{lock_type::exclusive};
119
121 std::chrono::system_clock::time_point acquired_at;
122
124 std::optional<std::chrono::system_clock::time_point> expires_at;
125
129 [[nodiscard]] auto is_valid() const -> bool {
130 if (!expires_at) return true;
131 return std::chrono::system_clock::now() < *expires_at;
132 }
133
137 [[nodiscard]] auto is_expired() const -> bool {
138 return !is_valid();
139 }
140
144 [[nodiscard]] auto remaining_time() const
145 -> std::optional<std::chrono::milliseconds> {
146 if (!expires_at) return std::nullopt;
147 auto now = std::chrono::system_clock::now();
148 if (now >= *expires_at) return std::chrono::milliseconds{0};
149 return std::chrono::duration_cast<std::chrono::milliseconds>(
150 *expires_at - now);
151 }
152};
153
157struct lock_info {
159 std::string study_uid;
160
162 lock_type type{lock_type::exclusive};
163
165 std::string reason;
166
168 std::string holder;
169
171 std::string token_id;
172
174 std::chrono::system_clock::time_point acquired_at;
175
177 std::optional<std::chrono::system_clock::time_point> expires_at;
178
180 std::size_t shared_count{0};
181
185 [[nodiscard]] auto duration() const -> std::chrono::milliseconds {
186 return std::chrono::duration_cast<std::chrono::milliseconds>(
187 std::chrono::system_clock::now() - acquired_at);
188 }
189
193 [[nodiscard]] auto is_expired() const -> bool {
194 if (!expires_at) return false;
195 return std::chrono::system_clock::now() >= *expires_at;
196 }
197};
