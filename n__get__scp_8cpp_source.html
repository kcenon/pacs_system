73network::Result<std::monostate> n_get_scp::handle_message(
74 network::association& assoc,
75 uint8_t context_id,
76 const network::dimse::dimse_message& request) {
77
78 using namespace network::dimse;
79
80 // Verify command type is N-GET-RQ
81 if (request.command() != command_field::n_get_rq) {
82 return kcenon::pacs::pacs_void_error(
83 kcenon::pacs::error_codes::n_get_unexpected_command,
84 "Unexpected command for N-GET SCP: " +
85 std::string(to_string(request.command())));
86 }
87
88 // Verify we have a handler configured
89 if (!handler_) {
90 return kcenon::pacs::pacs_void_error(
91 kcenon::pacs::error_codes::n_get_handler_not_set,
92 "No N-GET handler configured");
93 }
94
95 // Extract Requested SOP Class UID
96 auto sop_class_uid = request.requested_sop_class_uid();
97 if (sop_class_uid.empty()) {
98 return send_n_get_response(
99 assoc, context_id, request.message_id(),
100 "", "", status_error_missing_attribute);
101 }
102
103 // Verify the SOP Class is supported
104 if (!supports_sop_class(sop_class_uid)) {
105 return send_n_get_response(
106 assoc, context_id, request.message_id(),
107 sop_class_uid, "", status_refused_sop_class_not_supported);
108 }
109
110 // Extract Requested SOP Instance UID
111 auto sop_instance_uid = request.requested_sop_instance_uid();
112 if (sop_instance_uid.empty()) {
113 return send_n_get_response(
114 assoc, context_id, request.message_id(),
115 sop_class_uid, "", status_error_missing_attribute);
116 }
117
118 // Extract Attribute Identifier List (empty = all attributes)
119 auto attribute_tags = request.attribute_identifier_list();
120
121 logger_->debug("N-GET request for instance: " + sop_instance_uid +
122 " (attributes requested: " +
123 (attribute_tags.empty() ? "all" :
124 std::to_string(attribute_tags.size())) + ")");
125
126 // Call the handler to retrieve attributes
127 auto result = handler_(sop_class_uid, sop_instance_uid, attribute_tags);
128 if (result.is_err()) {
129 logger_->warn("N-GET handler failed for instance: " + sop_instance_uid +
130 " - " + result.error().message);
131 return send_n_get_response(
132 assoc, context_id, request.message_id(),
133 sop_class_uid, sop_instance_uid,
134 status_error_invalid_object_instance);
135 }
136
137 // Update statistics
138 ++gets_processed_;
139
140 // Send success response with dataset
141 auto dataset = std::move(result.value());
142 return send_n_get_response(
143 assoc, context_id, request.message_id(),
144 sop_class_uid, sop_instance_uid,
145 status_success, &dataset);
146}

168network::Result<std::monostate> n_get_scp::send_n_get_response(
169 network::association& assoc,
170 uint8_t context_id,
171 uint16_t message_id,
172 const std::string& sop_class_uid,
173 const std::string& sop_instance_uid,
174 network::dimse::status_code status,
175 core::dicom_dataset* dataset) {
176
177 using namespace network::dimse;
178
179 // Create N-GET response using factory function
180 auto response = make_n_get_rsp(
181 message_id, sop_class_uid, sop_instance_uid, status);
182
183 // Attach dataset if provided (successful responses include attributes)
184 if (dataset != nullptr) {
185 response.set_dataset(std::move(*dataset));
186 }
187
188 // Send the response
189 return assoc.send_dimse(context_id, response);
190}
