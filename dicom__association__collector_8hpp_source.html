221inline std::vector<association_metric> dicom_association_collector::collect() {
222 std::vector<association_metric> metrics;
223
224 if (!initialized_) {
225 return metrics;
226 }
227
228 const auto& counters = pacs_metrics::global_metrics().associations();
229
230 // Active associations (gauge)
231 metrics.push_back(create_metric(
232 "dicom_associations_active",
233 static_cast<double>(counters.current_active.load(std::memory_order_relaxed)),
234 "gauge"));
235
236 // Peak active associations (gauge)
237 metrics.push_back(create_metric(
238 "dicom_associations_peak_active",
239 static_cast<double>(counters.peak_active.load(std::memory_order_relaxed)),
240 "gauge"));
241
242 // Total established (counter)
243 metrics.push_back(create_metric(
244 "dicom_associations_total",
245 static_cast<double>(counters.total_established.load(std::memory_order_relaxed)),
246 "counter"));
247
248 // Total rejected (counter)
249 metrics.push_back(create_metric(
250 "dicom_associations_rejected_total",
251 static_cast<double>(counters.total_rejected.load(std::memory_order_relaxed)),
252 "counter"));
253
254 // Total aborted (counter)
255 metrics.push_back(create_metric(
256 "dicom_associations_aborted_total",
257 static_cast<double>(counters.total_aborted.load(std::memory_order_relaxed)),
258 "counter"));
259
260 // Success rate (gauge) - calculated metric
261 const auto total = counters.total_established.load(std::memory_order_relaxed);
262 const auto rejected = counters.total_rejected.load(std::memory_order_relaxed);
263 const auto attempted = total + rejected;
264 const double success_rate = (attempted > 0)
265 ? static_cast<double>(total) / static_cast<double>(attempted)
266 : 1.0;
267
268 metrics.push_back(create_metric(
269 "dicom_associations_success_rate",
270 success_rate,
271 "gauge"));
272
273 // Update statistics
274 {
275 std::lock_guard<std::mutex> lock(stats_mutex_);
276 ++collection_count_;
277 }
278
279 return metrics;
280}

301inline std::unordered_map<std::string, double>
302dicom_association_collector::get_statistics() const {
303 std::lock_guard<std::mutex> lock(stats_mutex_);
304
305 std::unordered_map<std::string, double> stats;
306 stats["collection_count"] = static_cast<double>(collection_count_);
307
308 if (initialized_) {
309 const auto uptime = std::chrono::duration_cast<std::chrono::seconds>(
310 std::chrono::steady_clock::now() - init_time_);
311 stats["uptime_seconds"] = static_cast<double>(uptime.count());
312 }
313
314 return stats;
315}
