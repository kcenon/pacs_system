96void dicom_server::register_service(services::scp_service_ptr service) {
97 if (!service) {
98 return;
99 }
100
101 std::lock_guard<std::mutex> lock(services_mutex_);
102
103 // Register SOP Class mappings
104 for (const auto& sop_class : service->supported_sop_classes()) {
105 sop_class_to_service_[sop_class] = service.get();
106 }
107
108 services_.push_back(std::move(service));
109}

112 std::lock_guard<std::mutex> lock(services_mutex_);
113
114 std::vector<std::string> sop_classes;
115 sop_classes.reserve(sop_class_to_service_.size());
116
117 for (const auto& [uid, _] : sop_class_to_service_) {
118 sop_classes.push_back(uid);

128Result<std::monostate> dicom_server::start() {
129 if (running_.exchange(true)) {
130 return error_info("Server already running");
131 }
132
133 // Validate configuration
134 if (config_.ae_title.empty()) {
135 running_ = false;
136 return error_info("AE Title cannot be empty");
137 }
138
139 if (config_.ae_title.length() > AE_TITLE_LENGTH) {
140 running_ = false;
141 return error_info("AE Title exceeds 16 characters");
142 }
143
144 if (config_.port == 0) {
145 running_ = false;
146 return error_info("Invalid port number");
147 }
148
149 // Check that at least one service is registered
150 {
151 std::lock_guard<std::mutex> lock(services_mutex_);
152 if (services_.empty()) {
153 running_ = false;
154 return error_info("No services registered");
155 }
156 }
157
158 // Reset statistics
159 {
160 std::lock_guard<std::mutex> lock(stats_mutex_);
161 stats_.start_time = clock::now();
162 stats_.last_activity = stats_.start_time;
163 stats_.total_associations = 0;
164 stats_.active_associations = 0;
165 stats_.rejected_associations = 0;
166 stats_.messages_processed = 0;
167 stats_.bytes_received = 0;
168 stats_.bytes_sent = 0;
169 }
170
171 // Create and start accept worker
172 accept_worker_ = std::make_unique<detail::accept_worker>(
173 config_.port,
174 // Connection callback (currently unused - placeholder for future TCP integration)
175 [](uint64_t /*session_id*/) {
176 // Future: handle new TCP connection
177 },
178 // Maintenance callback for periodic idle timeout checks
179 [this]() {
180 check_idle_timeouts();
181 }
182 );
183 accept_worker_->set_wake_interval(std::chrono::milliseconds(100));
184
185 auto start_result = accept_worker_->start();
186 if (start_result.is_err()) {
187 running_ = false;
188 return error_info("Failed to start accept worker: " +
189 start_result.error().message);
190 }
191
192 // Register in global registry
193 {
194 std::lock_guard<std::mutex> lock(registry_mutex_);
195 server_registry_[config_.port] = this;
196 }
197
198 return std::monostate{};
199}

201void dicom_server::stop(duration timeout) {
202 if (!running_.exchange(false)) {
203 return; // Already stopped
204 }
205
206 // Unregister from global registry
207 {
208 std::lock_guard<std::mutex> lock(registry_mutex_);
209 server_registry_.erase(config_.port);
210 }
211
212 // Stop accept worker (handles graceful shutdown via on_stop_requested hook)
213 if (accept_worker_) {
214 accept_worker_->stop();
215 accept_worker_.reset();
216 }
217
218 // =========================================================================
219 // Phase 1: Request graceful cancellation via cancellation tokens
220 // =========================================================================
221 // This triggers cooperative cancellation in all message loops.
222 // The message_loop will detect cancellation and attempt graceful release.
223 {
224 std::lock_guard<std::mutex> lock(associations_mutex_);
225 for (auto& [id, info] : associations_) {
226 info->cancel_token.cancel();
227 }
228 }
229
230 // =========================================================================
231 // Phase 2: Wait for graceful shutdown with timeout
232 // =========================================================================
233 // Give associations time to complete their graceful release.
234 // The message_loop will call release() and remove itself from the map.
235 auto deadline = clock::now() + timeout;
236 {
237 std::unique_lock<std::mutex> lock(associations_mutex_);
238 while (!associations_.empty() && clock::now() < deadline) {
239 // Release lock while waiting to allow message_loop to remove associations
240 lock.unlock();
241 std::this_thread::sleep_for(std::chrono::milliseconds{100});
242 lock.lock();
243 }
244 }
245
246 // =========================================================================
247 // Phase 3: Force abort remaining associations
248 // =========================================================================
249 // If any associations are still active after timeout, force abort them.
250 // This ensures we don't hang indefinitely on unresponsive connections.
251 {
252 std::lock_guard<std::mutex> lock(associations_mutex_);
253 for (auto& [id, info] : associations_) {
254 if (info->assoc.is_established()) {
255 info->assoc.abort(
256 static_cast<uint8_t>(abort_source::service_provider),
257 static_cast<uint8_t>(abort_reason::not_specified));
258 }
259 // Mark as no longer processing
260 // Note: Already holding associations_mutex_ so no atomic needed
261 info->processing = false;
262 }
263 associations_.clear();
264 }
265
266 // Notify shutdown waiters
267 {
268 std::lock_guard<std::mutex> lock(shutdown_mutex_);
269 shutdown_cv_.notify_all();
270 }
271}

273void dicom_server::wait_for_shutdown() {

275 shutdown_cv_.wait(lock, [this]() { return !running_; });

325void dicom_server::handle_association(uint64_t session_id, association assoc) {
326 // Check max associations limit
327 {
328 std::lock_guard<std::mutex> lock(associations_mutex_);
329 if (config_.max_associations > 0 &&
330 associations_.size() >= config_.max_associations) {
331 // Reject due to resource limit
332 assoc.abort(
333 static_cast<uint8_t>(abort_source::service_provider),
334 static_cast<uint8_t>(abort_reason::not_specified));
335
336 std::lock_guard<std::mutex> stats_lock(stats_mutex_);
337 stats_.rejected_associations++;
338 return;
339 }
340 }
341
342 // Create association info
343 association_info info;
344 info.id = session_id;
345 info.assoc = std::move(assoc);
346 info.connected_at = clock::now();
347 info.last_activity = info.connected_at;
348
349 // Invoke established callback
350 {
351 std::lock_guard<std::mutex> lock(callback_mutex_);
352 if (on_established_cb_) {
353 on_established_cb_(info.assoc);
354 }
355 }
356
357 // Publish association established event
358 kcenon::common::get_event_bus().publish(
359 kcenon::pacs::events::association_established_event{
360 std::string(info.assoc.calling_ae()),
361 std::string(info.assoc.called_ae()),
362 info.remote_address,
363 config_.port,
364 info.assoc.max_pdu_size()
365 }
366 );
367
368 // Add to active associations
369 add_association(std::move(info));
370
371 // Update statistics
372 {
373 std::lock_guard<std::mutex> lock(stats_mutex_);
374 stats_.total_associations++;
375 stats_.last_activity = clock::now();
376 }
377}

379void dicom_server::message_loop(association_info& info) {
380 // Check both running flag and cancellation token for cooperative shutdown
381 while (running_ && !info.cancel_token.is_cancelled() && info.assoc.is_established()) {
382 // Receive DIMSE message with timeout
383 // Use shorter timeout to check cancellation more frequently
384 auto receive_timeout = (std::min)(
385 std::chrono::duration_cast<association::duration>(config_.idle_timeout),
386 association::duration{1000} // Check cancellation at least every second
387 );
388
389 auto result = info.assoc.receive_dimse(receive_timeout);
390
391 // Check for cancellation after potentially blocking receive
392 if (info.cancel_token.is_cancelled()) {
393 break;
394 }
395
396 if (result.is_err()) {
397 // Check if it was a timeout vs. actual error
398 if (info.assoc.is_established()) {
399 // Still established, might be timeout - check idle
400 auto now = clock::now();
401 auto idle_duration = std::chrono::duration_cast<std::chrono::seconds>(
402 now - info.last_activity);
403
404 if (config_.idle_timeout.count() > 0 &&
405 idle_duration >= config_.idle_timeout) {
406 // Idle timeout - abort association
407 info.assoc.abort(
408 static_cast<uint8_t>(abort_source::service_provider),
409 static_cast<uint8_t>(abort_reason::not_specified));
410 }
411 }
412 break;
413 }
414
415 // Update activity timestamp
416 info.last_activity = clock::now();
417 touch_association(info.id);
418
419 // Dispatch message to service
420 auto& [context_id, msg] = result.value();
421 auto dispatch_result = dispatch_to_service(info.assoc, context_id, msg);
422
423 if (dispatch_result.is_err()) {
424 report_error(dispatch_result.error().message);
425 }
426
427 // Update statistics
428 {
429 std::lock_guard<std::mutex> lock(stats_mutex_);
430 stats_.messages_processed++;
431 stats_.last_activity = clock::now();
432 }
433 }
434
435 // If cancelled, attempt graceful release before forced abort
436 if (info.cancel_token.is_cancelled() && info.assoc.is_established()) {
437 // Try graceful release first - ignore result as we'll abort anyway if it fails
438 // The Phase 3 shutdown will handle any remaining established associations
439 (void)info.assoc.release();
440 }
441
442 // Calculate association duration
443 auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(
444 clock::now() - info.connected_at);
445
446 // Determine if aborted
447 bool is_aborted = info.assoc.state() == association_state::aborted;
448
449 // Association ended - invoke released callback
450 {
451 std::lock_guard<std::mutex> lock(callback_mutex_);
452 if (on_released_cb_) {
453 on_released_cb_(info.assoc);
454 }
455 }
456
457 // Publish association event (released or aborted)
458 if (is_aborted) {
459 kcenon::common::get_event_bus().publish(
460 kcenon::pacs::events::association_aborted_event{
461 std::string(info.assoc.calling_ae()),
462 std::string(info.assoc.called_ae()),
463 "Association aborted"
464 }
465 );
466 } else {
467 kcenon::common::get_event_bus().publish(
468 kcenon::pacs::events::association_released_event{
469 std::string(info.assoc.calling_ae()),
470 std::string(info.assoc.called_ae()),
471 duration,
472 0 // operations_count not tracked yet
473 }
474 );
475 }
476
477 // Remove from active associations
478 remove_association(info.id);
479}

481Result<std::monostate> dicom_server::dispatch_to_service(
482 association& assoc,
483 uint8_t context_id,
484 const dimse::dimse_message& msg) {
485
486 // Get the SOP Class from the message
487 std::string sop_class_uid = msg.affected_sop_class_uid();
488
489 if (sop_class_uid.empty()) {
490 return error_info("Cannot determine SOP Class UID from message");
491 }
492
493 // Find the appropriate service
494 auto* service = find_service(sop_class_uid);
495 if (!service) {
496 return error_info("No service registered for SOP Class: " + sop_class_uid);
497 }
498
499 // Dispatch to service
500 return service->handle_message(assoc, context_id, msg);
501}

558void dicom_server::check_idle_timeouts() {
559 if (config_.idle_timeout.count() == 0) {
560 return; // No timeout configured
561 }
562
563 auto now = clock::now();
564 std::vector<uint64_t> timed_out;
565
566 {
567 std::lock_guard<std::mutex> lock(associations_mutex_);
568 for (auto& [id, info] : associations_) {
569 auto idle_duration = std::chrono::duration_cast<std::chrono::seconds>(
570 now - info->last_activity);
571
572 if (idle_duration >= config_.idle_timeout) {
573 timed_out.push_back(id);
574 }
575 }
576
577 // Abort timed-out associations
578 for (auto id : timed_out) {
579 auto it = associations_.find(id);
580 if (it != associations_.end()) {
581 it->second->assoc.abort(
582 static_cast<uint8_t>(abort_source::service_provider),
583 static_cast<uint8_t>(abort_reason::not_specified));
584 }
585 }
586 }

606Result<associate_ac> dicom_server::simulate_association_request(const associate_rq& rq, association* client_peer) {
607 // Validate called AE title
608 if (!validate_called_ae(rq.called_ae_title)) {
609 return error_info("Called AE Title not recognized");
610 }
611
612 // Validate calling AE title
613 if (!validate_calling_ae(rq.calling_ae_title)) {
614 return error_info("Calling AE Title not recognized");
615 }
616
617 // Create SCP configuration for negotiation
618 scp_config negotiation_config;
619 negotiation_config.ae_title = config_.ae_title;
620 negotiation_config.max_pdu_length = config_.max_pdu_size;
621 negotiation_config.implementation_class_uid = config_.implementation_class_uid;
622 negotiation_config.implementation_version_name = config_.implementation_version_name;
623 negotiation_config.supported_abstract_syntaxes = supported_sop_classes();
624 // Accept compressed and uncompressed transfer syntaxes
625 negotiation_config.supported_transfer_syntaxes = {
626 "1.2.840.10008.1.2.4.201", // HTJ2K Lossless Only
627 "1.2.840.10008.1.2.4.202", // HTJ2K RPCL Lossless Only
628 "1.2.840.10008.1.2.4.203", // HTJ2K (Lossless or Lossy)
629 "1.2.840.10008.1.2.4.90", // JPEG 2000 Lossless
630 "1.2.840.10008.1.2.4.91", // JPEG 2000 Lossy
631 "1.2.840.10008.1.2.4.80", // JPEG-LS Lossless
632 "1.2.840.10008.1.2.4.81", // JPEG-LS Near-Lossless
633 "1.2.840.10008.1.2.4.70", // JPEG Lossless
634 "1.2.840.10008.1.2.4.50", // JPEG Baseline
635 "1.2.840.10008.1.2.5", // RLE Lossless
636 "1.2.840.10008.1.2.1", // Explicit VR LE
637 "1.2.840.10008.1.2" // Implicit VR LE
638 };
639
640 // Accept association
641 auto assoc = association::accept(rq, negotiation_config);
642
643 if (!assoc.is_established()) {
644 return error_info("Association rejected: no acceptable presentation contexts");
645 }
646
647 // Build AC PDU before moving association
648 auto ac = assoc.build_associate_ac();
649
650 // Register association
651 uint64_t id = next_association_id();
652 handle_association(id, std::move(assoc));
653
654 // Link peers and start message processing via thread pool
655 {
656 std::lock_guard<std::mutex> lock(associations_mutex_);
657 auto it = associations_.find(id);
658 if (it != associations_.end()) {
659 // Link peers
660 it->second->assoc.set_peer(client_peer);
661 client_peer->set_peer(&it->second->assoc);
662
663 // Submit message loop to thread_pool_adapter instead of creating dedicated thread.
664 // We need to capture the raw pointer to info because unique_ptr is in the map.
665 // The map might rehash but the unique_ptr value (the pointer to association_info) is stable.
666 auto* info_ptr = it->second.get();
667
668 // Mark as processing before submitting to pool
669 // Note: Already holding associations_mutex_ so no atomic needed
670 info_ptr->processing = true;
671
672 // Submit with high priority - association message handling is important
673 thread_pool_->submit_fire_and_forget(
674 [this, info_ptr]() {
675 message_loop(*info_ptr);
676 }
677 );
678 }
679 }
680
681 return ac;
682}
