56bool hevc_codec::can_encode(const image_params& params) const noexcept {
57 if (params.width == 0 || params.height == 0) {
58 return false;
59 }
60
61 if (params.samples_per_pixel != 1 && params.samples_per_pixel != 3) {
62 return false;
63 }
64
65 if (main10_) {
66 // Main 10 Profile supports up to 10-bit
67 if (params.bits_stored < 1 || params.bits_stored > 10) {
68 return false;
69 }
70 } else {
71 // Main Profile supports up to 8-bit
72 if (params.bits_stored < 1 || params.bits_stored > 8) {
73 return false;
74 }
75 }
76
77 return true;
78}
