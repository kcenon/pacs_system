103struct cache_stats {
104 std::atomic<std::uint64_t> hits{0};
105 std::atomic<std::uint64_t> misses{0};
106 std::atomic<std::uint64_t> insertions{0};
107 std::atomic<std::uint64_t> evictions{0};
108 std::atomic<std::uint64_t> expirations{0};
109 std::atomic<std::size_t> current_size{0};
110
115 [[nodiscard]] double hit_rate() const noexcept {
116 const auto total_hits = hits.load(std::memory_order_relaxed);
117 const auto total_misses = misses.load(std::memory_order_relaxed);
118 const auto total = total_hits + total_misses;
119 if (total == 0) {
120 return 0.0;
121 }
122 return (static_cast<double>(total_hits) / static_cast<double>(total)) * 100.0;
123 }
124
129 [[nodiscard]] std::uint64_t total_accesses() const noexcept {
130 return hits.load(std::memory_order_relaxed) +
131 misses.load(std::memory_order_relaxed);
132 }
133
137 void reset() noexcept {
138 hits.store(0, std::memory_order_relaxed);
139 misses.store(0, std::memory_order_relaxed);
140 insertions.store(0, std::memory_order_relaxed);
141 evictions.store(0, std::memory_order_relaxed);
142 expirations.store(0, std::memory_order_relaxed);
143 // Note: current_size is not reset as it reflects actual cache state
144 }
145};

278 [[nodiscard]] std::optional<Value> get(const Key& key) {
279 std::unique_lock lock(mutex_);
280
281 auto it = cache_map_.find(key);
282 if (it == cache_map_.end()) {
283 // Cache miss
284 stats_.misses.fetch_add(1, std::memory_order_relaxed);
285 return std::nullopt;
286 }
287
288 auto& entry = it->second;
289
290 // Check TTL expiration
291 if (is_expired(entry->expiry_time)) {
292 // Entry expired - remove it
293 stats_.expirations.fetch_add(1, std::memory_order_relaxed);
294 stats_.misses.fetch_add(1, std::memory_order_relaxed);
295 remove_entry(it);
296 return std::nullopt;
297 }
298
299 // Cache hit - move to front (most recently used)
300 stats_.hits.fetch_add(1, std::memory_order_relaxed);
301 lru_list_.splice(lru_list_.begin(), lru_list_, entry);
302
303 return entry->value;
304 }
305
315 void put(const Key& key, const Value& value) {
316 std::unique_lock lock(mutex_);
317
318 auto it = cache_map_.find(key);
319 if (it != cache_map_.end()) {
320 // Update existing entry
321 auto& entry = it->second;
322 entry->value = value;
323 entry->expiry_time = calculate_expiry();
324 lru_list_.splice(lru_list_.begin(), lru_list_, entry);
325 return;
326 }
327
328 // Evict if at capacity
329 while (cache_map_.size() >= max_size_ && !lru_list_.empty()) {
330 evict_oldest();
331 }
332
333 // Insert new entry at front
334 lru_list_.emplace_front(cache_entry{key, value, calculate_expiry()});
335 cache_map_[key] = lru_list_.begin();
336
337 stats_.insertions.fetch_add(1, std::memory_order_relaxed);
338 stats_.current_size.store(cache_map_.size(), std::memory_order_relaxed);
339 }
340
347 void put(const Key& key, Value&& value) {
348 std::unique_lock lock(mutex_);
349
350 auto it = cache_map_.find(key);
351 if (it != cache_map_.end()) {
352 // Update existing entry
353 auto& entry = it->second;
354 entry->value = std::move(value);
355 entry->expiry_time = calculate_expiry();
356 lru_list_.splice(lru_list_.begin(), lru_list_, entry);
357 return;
358 }
359
360 // Evict if at capacity
361 while (cache_map_.size() >= max_size_ && !lru_list_.empty()) {
362 evict_oldest();
363 }
364
365 // Insert new entry at front
366 lru_list_.emplace_front(cache_entry{key, std::move(value), calculate_expiry()});

368
369 stats_.insertions.fetch_add(1, std::memory_order_relaxed);
370 stats_.current_size.store(cache_map_.size(), std::memory_order_relaxed);
371 }

382 [[nodiscard]] bool contains(const Key& key) const {

393 bool invalidate(const Key& key) {
394 std::unique_lock lock(mutex_);
395
396 auto it = cache_map_.find(key);
397 if (it == cache_map_.end()) {
398 return false;
399 }
400
401 remove_entry(it);
402 return true;
403 }

430 template <typename Predicate>
431 size_type invalidate_if(Predicate pred) {
432 std::unique_lock lock(mutex_);
433
434 size_type removed = 0;
435 auto it = lru_list_.begin();
436
437 while (it != lru_list_.end()) {
438 if (pred(it->key, it->value)) {
439 cache_map_.erase(it->key);
440 it = lru_list_.erase(it);
441 ++removed;
442 } else {
443 ++it;
444 }
445 }
446
447 stats_.current_size.store(cache_map_.size(), std::memory_order_relaxed);
448 return removed;
449 }
450
454 void clear() {
455 std::unique_lock lock(mutex_);
456
457 lru_list_.clear();
458 cache_map_.clear();
459 stats_.current_size.store(0, std::memory_order_relaxed);
460 }
461
470 size_type purge_expired() {
471 std::unique_lock lock(mutex_);
472
473 size_type removed = 0;
474 const auto now = clock_type::now();
475
476 auto it = lru_list_.begin();
477 while (it != lru_list_.end()) {
478 if (is_expired(it->expiry_time, now)) {
479 cache_map_.erase(it->key);
480 it = lru_list_.erase(it);
481 ++removed;
482 stats_.expirations.fetch_add(1, std::memory_order_relaxed);
483 } else {
484 ++it;
485 }
486 }
487
488 stats_.current_size.store(cache_map_.size(), std::memory_order_relaxed);
489 return removed;
490 }

594 [[nodiscard]] time_point calculate_expiry() const {
595 return clock_type::now() + config_.ttl;
596 }
597
598 [[nodiscard]] bool is_expired(time_point expiry) const {
599 return clock_type::now() > expiry;
600 }
601
602 [[nodiscard]] bool is_expired(time_point expiry, time_point now) const {
603 return now > expiry;
604 }
605
606 void evict_oldest() {
607 // Evict from the back (least recently used)
608 if (lru_list_.empty()) {
609 return;
610 }
611
612 auto& oldest = lru_list_.back();
613 cache_map_.erase(oldest.key);
614 lru_list_.pop_back();
615
616 stats_.evictions.fetch_add(1, std::memory_order_relaxed);
617 stats_.current_size.store(cache_map_.size(), std::memory_order_relaxed);
618 }
619
620 void remove_entry(typename map_type::iterator map_it) {
621 lru_list_.erase(map_it->second);
622 cache_map_.erase(map_it);
623 stats_.current_size.store(cache_map_.size(), std::memory_order_relaxed);
624 }
