52[[nodiscard]] charls::interleave_mode get_interleave_mode(const image_params& params) {
53 if (params.samples_per_pixel == 1) {
54 return charls::interleave_mode::none;
55 }
56 // For color images, use sample interleaved mode (RGBRGB...)
57 return (params.planar_configuration == 0)
58 ? charls::interleave_mode::sample
59 : charls::interleave_mode::line;
60}
61
65[[nodiscard]] charls::color_transformation get_color_transform(const image_params& params) {
66 if (params.samples_per_pixel == 1) {
67 return charls::color_transformation::none;
68 }
69 // For RGB images, no color transformation is applied
70 // (JPEG-LS can optionally use HP color transforms, but DICOM typically doesn't)
71 return charls::color_transformation::none;
72}

81class jpeg_ls_codec::impl {
82public:
83 explicit impl(bool lossless, int near_value)
84 : lossless_(lossless)
85 , near_value_(0) {
86 // Handle sentinel value: auto-determine NEAR based on mode
87 if (near_value == kAutoNearValue) {
88 near_value_ = lossless_ ? kLosslessNearValue : kDefaultNearLosslessValue;
89 }
90 // Lossless mode always forces NEAR=0 (DICOM compliance)
91 else if (lossless_) {
92 near_value_ = kLosslessNearValue;
93 }
94 // Explicit NEAR=0 forces lossless mode
95 else if (near_value <= 0) {
96 lossless_ = true;
97 near_value_ = kLosslessNearValue;
98 }
99 // Near-lossless with explicit NEAR value
100 else {
101 near_value_ = std::clamp(near_value, 1, kMaxNearValue);
102 }
103 }
104

106 return lossless_;
107 }

110 return near_value_;

113 [[nodiscard]] codec_result encode(std::span<const uint8_t> pixel_data,
114 const image_params& params,
115 const compression_options& options) const {
116#ifndef PACS_WITH_JPEGLS_CODEC
117 (void)pixel_data;
118 (void)params;
119 (void)options;
120 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error,
121 "JPEG-LS codec not available: CharLS library not found at build time");
122#else
123 // Validate input
124 if (pixel_data.empty()) {
125 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Empty pixel data");
126 }
127
128 if (params.width == 0 || params.height == 0) {
129 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Invalid image dimensions");
130 }
131
132 // Determine effective NEAR value
133 int effective_near = near_value_;
134 bool use_lossless = lossless_ || options.lossless;
135
136 if (use_lossless) {
137 effective_near = 0;
138 } else if (!lossless_ && options.quality > 0 && options.quality <= 100) {
139 // Map quality (1-100) to NEAR parameter
140 // quality 100 -> NEAR 0 (lossless)
141 // quality 1 -> NEAR ~10 (more compression)
142 effective_near = static_cast<int>((100 - options.quality) * 10 / 100);
143 effective_near = std::clamp(effective_near, 0, kMaxNearValue);
144 }
145
146 try {
147 // Create encoder
148 charls::jpegls_encoder encoder;
149
150 // Set frame info
151 charls::frame_info frame_info{};
152 frame_info.width = params.width;
153 frame_info.height = params.height;
154 frame_info.bits_per_sample = params.bits_stored;
155 frame_info.component_count = params.samples_per_pixel;
156
157 encoder.frame_info(frame_info);
158
159 // Set interleave mode
160 encoder.interleave_mode(get_interleave_mode(params));
161
162 // Set NEAR parameter (0 = lossless)
163 encoder.near_lossless(effective_near);
164
165 // Set color transformation
166 encoder.color_transformation(get_color_transform(params));
167
168 // Calculate destination buffer size
169 // For high-entropy data, JPEG-LS output may exceed input size.
170 // Use the maximum of: CharLS estimate, input size + 20%, or input + 1KB overhead.
171 size_t estimated_size = encoder.estimated_destination_size();
172 size_t safe_size = (std::max)({
173 estimated_size,
174 pixel_data.size() + pixel_data.size() / 5, // input + 20%
175 pixel_data.size() + 1024 // input + 1KB header overhead
176 });
177 std::vector<uint8_t> destination(safe_size);
178
179 // Encode
180 encoder.destination(destination);
181 size_t bytes_written = encoder.encode(pixel_data);
182
183 // Resize to actual size
184 destination.resize(bytes_written);
185
186 return kcenon::pacs::ok<compression_result>(compression_result{std::move(destination), params});
187
188 } catch (const charls::jpegls_error& e) {
189 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error,
190 std::string("JPEG-LS encoding failed: ") + e.what());
191 } catch (const std::exception& e) {
192 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error,
193 std::string("JPEG-LS encoding failed: ") + e.what());
194 }
195#endif // PACS_WITH_JPEGLS_CODEC
196 }

198 [[nodiscard]] codec_result decode(std::span<const uint8_t> compressed_data,
199 const image_params& params) const {
200#ifndef PACS_WITH_JPEGLS_CODEC
201 (void)compressed_data;
202 (void)params;
203 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error,
204 "JPEG-LS codec not available: CharLS library not found at build time");
205#else
206 if (compressed_data.empty()) {
207 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Empty compressed data");
208 }
209
210 try {
211 // Create decoder
212 charls::jpegls_decoder decoder;
213 decoder.source(compressed_data);
214
215 // Read header to get frame info
216 decoder.read_header();
217
218 // Get frame info
219 const charls::frame_info& frame_info = decoder.frame_info();
220
221 // Build output parameters
222 image_params output_params = params;
223 output_params.width = static_cast<uint16_t>(frame_info.width);
224 output_params.height = static_cast<uint16_t>(frame_info.height);
225 output_params.bits_stored = static_cast<uint16_t>(frame_info.bits_per_sample);
226 output_params.bits_allocated = (frame_info.bits_per_sample <= 8) ? 8 : 16;
227 output_params.high_bit = output_params.bits_stored - 1;
228 output_params.samples_per_pixel = static_cast<uint16_t>(frame_info.component_count);
229
230 // Determine photometric interpretation
231 if (frame_info.component_count == 1) {
232 output_params.photometric = photometric_interpretation::monochrome2;
233 } else if (frame_info.component_count == 3) {
234 output_params.photometric = photometric_interpretation::rgb;
235 }
236
237 // Validate dimensions if provided
238 if (params.width > 0 && params.width != output_params.width) {
239 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Image width mismatch: expected " +
240 std::to_string(params.width) + ", got " +
241 std::to_string(output_params.width));
242 }
243 if (params.height > 0 && params.height != output_params.height) {
244 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error, "Image height mismatch: expected " +
245 std::to_string(params.height) + ", got " +
246 std::to_string(output_params.height));
247 }
248
249 // Allocate destination buffer
250 size_t destination_size = decoder.destination_size();
251 std::vector<uint8_t> destination(destination_size);
252
253 // Decode
254 decoder.decode(destination);
255
256 // Set output as interleaved
257 output_params.planar_configuration = 0;
258
259 return kcenon::pacs::ok<compression_result>(compression_result{std::move(destination), output_params});
260
261 } catch (const charls::jpegls_error& e) {
262 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error,
263 std::string("JPEG-LS decoding failed: ") + e.what());
264 } catch (const std::exception& e) {
265 return kcenon::pacs::pacs_error<compression_result>(kcenon::pacs::error_codes::decompression_error,
266 std::string("JPEG-LS decoding failed: ") + e.what());
267 }
268#endif // PACS_WITH_JPEGLS_CODEC
269 }

271private:
272 bool lossless_;
273 int near_value_;
274};

298bool jpeg_ls_codec::can_encode(const image_params& params) const noexcept {
299 // JPEG-LS supports 2-16 bit precision per component
300 if (params.bits_stored < 2 || params.bits_stored > 16) {
301 return false;
302 }
303
304 // bits_allocated must be 8 or 16
305 if (params.bits_allocated != 8 && params.bits_allocated != 16) {
306 return false;
307 }
308
309 // Support grayscale (1) and color (3) images
310 if (params.samples_per_pixel != 1 && params.samples_per_pixel != 3) {
311 return false;
312 }
313
314 // Require valid dimensions
315 if (params.width == 0 || params.height == 0) {
316 return false;
317 }
318
319 // Maximum dimension check (JPEG-LS limit)
320 if (params.width > 65535 || params.height > 65535) {
321 return false;
322 }
323
324 return true;
325}

327bool jpeg_ls_codec::can_decode(const image_params& params) const noexcept {
328 // For decoding, we're more lenient as actual parameters come from the bitstream
329 // Just validate samples_per_pixel if specified
330 if (params.samples_per_pixel != 0 &&
331 params.samples_per_pixel != 1 &&
332 params.samples_per_pixel != 3) {
333 return false;
334 }
335 return true;
336}
