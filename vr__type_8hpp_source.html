157[[nodiscard]] constexpr std::optional<vr_type> from_string(std::string_view str) noexcept {
158 if (str.size() != 2) {
159 return std::nullopt;
160 }
161
162 // Convert two ASCII chars to uint16_t (big-endian: first char is high byte)
163 const auto code = static_cast<uint16_t>(
164 (static_cast<uint16_t>(static_cast<unsigned char>(str[0])) << 8) |
165 static_cast<uint16_t>(static_cast<unsigned char>(str[1]))
166 );
167
168 // Validate by checking if the code corresponds to a known VR
169 switch (static_cast<vr_type>(code)) {
170 case vr_type::AE: case vr_type::AS: case vr_type::AT:
171 case vr_type::CS: case vr_type::DA: case vr_type::DS:
172 case vr_type::DT: case vr_type::FD: case vr_type::FL:
173 case vr_type::IS: case vr_type::LO: case vr_type::LT:
174 case vr_type::OB: case vr_type::OD: case vr_type::OF:
175 case vr_type::OL: case vr_type::OV: case vr_type::OW:
176 case vr_type::PN: case vr_type::SH: case vr_type::SL:
177 case vr_type::SQ: case vr_type::SS: case vr_type::ST:
178 case vr_type::SV: case vr_type::TM: case vr_type::UC:
179 case vr_type::UI: case vr_type::UL: case vr_type::UN:
180 case vr_type::UR: case vr_type::US: case vr_type::UT:
181 case vr_type::UV:
182 return static_cast<vr_type>(code);
183 default:
184 return std::nullopt;
185 }
186}

285[[nodiscard]] constexpr std::size_t fixed_length(vr_type vr) noexcept {
286 switch (vr) {
287 case vr_type::AT: return 4; // Attribute Tag
288 case vr_type::FL: return 4; // Float
289 case vr_type::FD: return 8; // Double
290 case vr_type::SL: return 4; // Signed Long
291 case vr_type::SS: return 2; // Signed Short
292 case vr_type::SV: return 8; // Signed 64-bit
293 case vr_type::UL: return 4; // Unsigned Long
294 case vr_type::US: return 2; // Unsigned Short
295 case vr_type::UV: return 8; // Unsigned 64-bit
296 default: return 0; // Variable length
297 }
298}

317[[nodiscard]] constexpr char padding_char(vr_type vr) noexcept {
318 if (vr == vr_type::UI) {
319 return '\0'; // UI uses null padding
320 }
321 if (is_string_vr(vr)) {
322 return ' '; // Other string VRs use space padding
323 }
324 return '\0'; // Binary VRs use null padding if needed
325}
