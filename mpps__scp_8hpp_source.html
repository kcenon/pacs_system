85[[nodiscard]] inline auto to_string(mpps_status status) -> std::string_view {
86 switch (status) {
87 case mpps_status::in_progress:
88 return "IN PROGRESS";
89 case mpps_status::completed:
90 return "COMPLETED";
91 case mpps_status::discontinued:
92 return "DISCONTINUED";
93 default:
94 return "IN PROGRESS";
95 }
96}
97
104[[nodiscard]] inline auto parse_mpps_status(std::string_view str)
105 -> std::optional<mpps_status> {
106 if (str == "IN PROGRESS") {
107 return mpps_status::in_progress;
108 }
109 if (str == "COMPLETED") {
110 return mpps_status::completed;
111 }
112 if (str == "DISCONTINUED") {
113 return mpps_status::discontinued;
114 }
115 return std::nullopt;
116}
