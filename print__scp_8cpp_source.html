87network::Result<std::monostate> print_scp::handle_message(
88 network::association& assoc,
89 uint8_t context_id,
90 const network::dimse::dimse_message& request) {
91
92 using namespace network::dimse;
93
94 switch (request.command()) {
95 case command_field::n_create_rq:
96 return handle_n_create(assoc, context_id, request);
97 case command_field::n_set_rq:
98 return handle_n_set(assoc, context_id, request);
99 case command_field::n_get_rq:
100 return handle_n_get(assoc, context_id, request);
101 case command_field::n_action_rq:
102 return handle_n_action(assoc, context_id, request);
103 case command_field::n_delete_rq:
104 return handle_n_delete(assoc, context_id, request);
105 default:
106 return kcenon::pacs::pacs_void_error(
107 kcenon::pacs::error_codes::print_unexpected_command,
108 "Unexpected command for Print SCP: " +
109 std::string(to_string(request.command())));
110 }
111}

154 network::association& assoc,

171 request.message_id(), sop_class_uid,
172 "", status_refused_sop_class_not_supported);
173}

175// =============================================================================
176// N-SET Handler
177// =============================================================================
178
179network::Result<std::monostate> print_scp::handle_n_set(
180 network::association& assoc,
181 uint8_t context_id,
182 const network::dimse::dimse_message& request) {
183
184 using namespace network::dimse;
185
186 auto sop_class_uid = request.affected_sop_class_uid();
187 if (sop_class_uid.empty()) {
188 sop_class_uid = request.command_set().get_string(
189 tag_requested_sop_class_uid);
190 }
191
192 auto sop_instance_uid = request.command_set().get_string(
193 tag_requested_sop_instance_uid);
194 if (sop_instance_uid.empty()) {
195 sop_instance_uid = request.affected_sop_instance_uid();
196 }
197
198 if (sop_instance_uid.empty()) {
199 return send_response(
200 assoc, context_id, command_field::n_set_rsp,
201 request.message_id(), sop_class_uid,
202 "", status_error_missing_attribute);
203 }
204
205 // Handle Image Box N-SET (grayscale or color)
206 if (sop_class_uid == basic_grayscale_image_box_sop_class_uid ||
207 sop_class_uid == basic_color_image_box_sop_class_uid) {
208
209 std::lock_guard<std::mutex> lock(mutex_);
210
211 auto it = image_boxes_.find(sop_instance_uid);
212 if (it == image_boxes_.end()) {
213 return send_response(
214 assoc, context_id, command_field::n_set_rsp,
215 request.message_id(), sop_class_uid,
216 sop_instance_uid, status_error_invalid_object_instance);
217 }
218
219 if (request.has_dataset()) {
220 it->second.data = request.dataset().value().get();
221 it->second.has_pixel_data = true;
222 }
223
224 ++images_set_;
225
226 return send_response(
227 assoc, context_id, command_field::n_set_rsp,
228 request.message_id(), sop_class_uid,
229 sop_instance_uid, status_success);
230 }
231
232 // Handle Film Session N-SET
233 if (sop_class_uid == basic_film_session_sop_class_uid) {
234 std::lock_guard<std::mutex> lock(mutex_);
235
236 auto it = sessions_.find(sop_instance_uid);
237 if (it == sessions_.end()) {
238 return send_response(
239 assoc, context_id, command_field::n_set_rsp,
240 request.message_id(), sop_class_uid,
241 sop_instance_uid, status_error_invalid_object_instance);
242 }
243
244 if (request.has_dataset()) {
245 const auto& ds = request.dataset().value().get();
246 it->second.data = ds;
247
248 if (ds.contains(print_tags::number_of_copies)) {
249 auto copies_str = ds.get_string(print_tags::number_of_copies);
250 if (!copies_str.empty()) {
251 it->second.number_of_copies =
252 static_cast<uint32_t>(std::stoul(copies_str));
253 }
254 }
255 if (ds.contains(print_tags::print_priority)) {
256 it->second.print_priority = ds.get_string(print_tags::print_priority);
257 }
258 if (ds.contains(print_tags::medium_type)) {
259 it->second.medium_type = ds.get_string(print_tags::medium_type);
260 }
261 }
262
263 return send_response(
264 assoc, context_id, command_field::n_set_rsp,
265 request.message_id(), sop_class_uid,
266 sop_instance_uid, status_success);
267 }
268
269 // Handle Film Box N-SET
270 if (sop_class_uid == basic_film_box_sop_class_uid) {
271 std::lock_guard<std::mutex> lock(mutex_);
272
273 auto it = film_boxes_.find(sop_instance_uid);
274 if (it == film_boxes_.end()) {
275 return send_response(
276 assoc, context_id, command_field::n_set_rsp,
277 request.message_id(), sop_class_uid,
278 sop_instance_uid, status_error_invalid_object_instance);
279 }
280
281 if (request.has_dataset()) {
282 const auto& ds = request.dataset().value().get();
283 it->second.data = ds;
284
285 if (ds.contains(print_tags::image_display_format)) {
286 it->second.image_display_format =
287 ds.get_string(print_tags::image_display_format);
288 }
289 if (ds.contains(print_tags::film_orientation)) {
290 it->second.film_orientation = ds.get_string(print_tags::film_orientation);
291 }
292 if (ds.contains(print_tags::film_size_id)) {
293 it->second.film_size_id = ds.get_string(print_tags::film_size_id);
294 }
295 }
296
297 return send_response(
298 assoc, context_id, command_field::n_set_rsp,
299 request.message_id(), sop_class_uid,
300 sop_instance_uid, status_success);
301 }
302
303 return send_response(
304 assoc, context_id, command_field::n_set_rsp,
305 request.message_id(), sop_class_uid,
306 sop_instance_uid, status_refused_sop_class_not_supported);
307}

309// =============================================================================
310// N-GET Handler (Printer Status)
311// =============================================================================
312
313network::Result<std::monostate> print_scp::handle_n_get(
314 network::association& assoc,
315 uint8_t context_id,
316 const network::dimse::dimse_message& request) {
317
318 using namespace network::dimse;
319
320 auto sop_class_uid = request.affected_sop_class_uid();
321 if (sop_class_uid.empty()) {
322 sop_class_uid = request.command_set().get_string(
323 tag_requested_sop_class_uid);
324 }
325
326 if (sop_class_uid != printer_sop_class_uid) {
327 return send_response(
328 assoc, context_id, command_field::n_get_rsp,
329 request.message_id(), sop_class_uid,
330 "", status_refused_sop_class_not_supported);
331 }
332
333 ++printer_queries_;
334
335 // Build printer status response
336 core::dicom_dataset response_ds;
337
338 using namespace encoding;
339
340 if (printer_status_handler_) {
341 auto result = printer_status_handler_();
342 if (result.is_ok()) {
343 auto& [status, info_ds] = result.value();
344 response_ds = std::move(info_ds);
345 response_ds.set_string(print_tags::printer_status_tag, vr_type::CS,
346 std::string(to_string(status)));
347 } else {
348 response_ds.set_string(print_tags::printer_status_tag, vr_type::CS, "NORMAL");
349 response_ds.set_string(print_tags::printer_status_info, vr_type::ST, "");
350 }
351 } else {
352 // Default: report printer as normal
353 response_ds.set_string(print_tags::printer_status_tag, vr_type::CS, "NORMAL");
354 response_ds.set_string(print_tags::printer_status_info, vr_type::ST, "");
355 response_ds.set_string(print_tags::printer_name, vr_type::LO, "PACS_PRINTER");
356 }
357
358 // Build response message
359 dimse_message response{command_field::n_get_rsp, 0};
360 response.set_message_id_responded_to(request.message_id());
361 response.set_affected_sop_class_uid(printer_sop_class_uid);
362 response.set_status(status_success);
363
364 auto sop_instance_uid = request.command_set().get_string(
365 tag_requested_sop_instance_uid);
366 if (sop_instance_uid.empty()) {
367 sop_instance_uid = request.affected_sop_instance_uid();
368 }
369 if (!sop_instance_uid.empty()) {
370 response.set_affected_sop_instance_uid(sop_instance_uid);
371 }
372
373 response.set_dataset(std::move(response_ds));
374 return assoc.send_dimse(context_id, response);
375}

377// =============================================================================
378// N-ACTION Handler (Print Film Box)
379// =============================================================================
380
381network::Result<std::monostate> print_scp::handle_n_action(
382 network::association& assoc,
383 uint8_t context_id,
384 const network::dimse::dimse_message& request) {
385
386 using namespace network::dimse;
387
388 auto sop_class_uid = request.affected_sop_class_uid();
389
390 // N-ACTION is used on Film Session or Film Box to initiate printing
391 if (sop_class_uid != basic_film_session_sop_class_uid &&
392 sop_class_uid != basic_film_box_sop_class_uid) {
393 return send_response(
394 assoc, context_id, command_field::n_action_rsp,
395 request.message_id(), sop_class_uid,
396 "", status_refused_sop_class_not_supported);
397 }
398
399 auto sop_instance_uid = request.affected_sop_instance_uid();
400 if (sop_instance_uid.empty()) {
401 return send_response(
402 assoc, context_id, command_field::n_action_rsp,
403 request.message_id(), sop_class_uid,
404 "", status_error_missing_attribute);
405 }
406
407 // Verify the object exists
408 {
409 std::lock_guard<std::mutex> lock(mutex_);
410
411 if (sop_class_uid == basic_film_box_sop_class_uid) {
412 if (film_boxes_.find(sop_instance_uid) == film_boxes_.end()) {
413 return send_response(
414 assoc, context_id, command_field::n_action_rsp,
415 request.message_id(), sop_class_uid,
416 sop_instance_uid, status_error_invalid_object_instance);
417 }
418 } else {
419 if (sessions_.find(sop_instance_uid) == sessions_.end()) {
420 return send_response(
421 assoc, context_id, command_field::n_action_rsp,
422 request.message_id(), sop_class_uid,
423 sop_instance_uid, status_error_invalid_object_instance);
424 }
425 }
426 }
427
428 // Call print handler
429 if (print_handler_) {
430 auto result = print_handler_(sop_instance_uid);
431 if (result.is_err()) {
432 return send_response(
433 assoc, context_id, command_field::n_action_rsp,
434 request.message_id(), sop_class_uid,
435 sop_instance_uid, status_error_unable_to_process);
436 }
437 }
438
439 ++prints_executed_;
440
441 // Build action response with action type ID
442 dimse_message response{command_field::n_action_rsp, 0};
443 response.set_message_id_responded_to(request.message_id());
444 response.set_affected_sop_class_uid(sop_class_uid);
445 response.set_affected_sop_instance_uid(sop_instance_uid);
446 response.set_status(status_success);
447
448 auto action_type = request.action_type_id();
449 if (action_type.has_value()) {
450 response.set_action_type_id(action_type.value());
451 }
452
453 return assoc.send_dimse(context_id, response);
454}

456// =============================================================================
457// N-DELETE Handler
458// =============================================================================
459
460network::Result<std::monostate> print_scp::handle_n_delete(
461 network::association& assoc,
462 uint8_t context_id,
463 const network::dimse::dimse_message& request) {
464
465 using namespace network::dimse;
466
467 auto sop_class_uid = request.affected_sop_class_uid();
468 auto sop_instance_uid = request.affected_sop_instance_uid();
469
470 if (sop_instance_uid.empty()) {
471 return send_response(
472 assoc, context_id, command_field::n_delete_rsp,
473 request.message_id(), sop_class_uid,
474 "", status_error_missing_attribute);
475 }
476
477 std::lock_guard<std::mutex> lock(mutex_);
478
479 if (sop_class_uid == basic_film_session_sop_class_uid) {
480 auto it = sessions_.find(sop_instance_uid);
481 if (it == sessions_.end()) {
482 return send_response(
483 assoc, context_id, command_field::n_delete_rsp,
484 request.message_id(), sop_class_uid,
485 sop_instance_uid, status_error_invalid_object_instance);
486 }
487
488 // Delete associated film boxes and their image boxes
489 for (const auto& fb_uid : it->second.film_box_uids) {
490 auto fb_it = film_boxes_.find(fb_uid);
491 if (fb_it != film_boxes_.end()) {
492 for (const auto& ib_uid : fb_it->second.image_box_uids) {
493 image_boxes_.erase(ib_uid);
494 }
495 film_boxes_.erase(fb_it);
496 }
497 }
498 sessions_.erase(it);
499
500 } else if (sop_class_uid == basic_film_box_sop_class_uid) {
501 auto it = film_boxes_.find(sop_instance_uid);
502 if (it == film_boxes_.end()) {
503 return send_response(
504 assoc, context_id, command_field::n_delete_rsp,
505 request.message_id(), sop_class_uid,
506 sop_instance_uid, status_error_invalid_object_instance);
507 }
508
509 // Delete associated image boxes
510 for (const auto& ib_uid : it->second.image_box_uids) {
511 image_boxes_.erase(ib_uid);
512 }
513
514 // Remove from parent session's film box list
515 for (auto& [_, session] : sessions_) {
516 auto& fb_uids = session.film_box_uids;
517 fb_uids.erase(
518 std::remove(fb_uids.begin(), fb_uids.end(), sop_instance_uid),
519 fb_uids.end());
520 }
521 film_boxes_.erase(it);
522
523 } else {
524 return send_response(
525 assoc, context_id, command_field::n_delete_rsp,
526 request.message_id(), sop_class_uid,
527 sop_instance_uid, status_refused_sop_class_not_supported);
528 }
529
530 return send_response(
531 assoc, context_id, command_field::n_delete_rsp,
532 request.message_id(), sop_class_uid,
533 sop_instance_uid, status_success);
534}

537// Film Session Creation
538// =============================================================================
539
540network::Result<std::monostate> print_scp::create_film_session(
541 network::association& assoc,
542 uint8_t context_id,
543 const network::dimse::dimse_message& request) {
544
545 using namespace network::dimse;
546
547 auto sop_instance_uid = request.affected_sop_instance_uid();
548 if (sop_instance_uid.empty()) {
549 sop_instance_uid = generate_uid();
550 }
551
552 film_session session;
553 session.sop_instance_uid = sop_instance_uid;
554
555 if (request.has_dataset()) {
556 const auto& ds = request.dataset().value().get();
557 session.data = ds;
558
559 if (ds.contains(print_tags::number_of_copies)) {
560 auto copies_str = ds.get_string(print_tags::number_of_copies);
561 if (!copies_str.empty()) {
562 session.number_of_copies =
563 static_cast<uint32_t>(std::stoul(copies_str));
564 }
565 }
566 if (ds.contains(print_tags::print_priority)) {
567 session.print_priority = ds.get_string(print_tags::print_priority);
568 }
569 if (ds.contains(print_tags::medium_type)) {
570 session.medium_type = ds.get_string(print_tags::medium_type);
571 }
572 if (ds.contains(print_tags::film_destination)) {
573 session.film_destination = ds.get_string(print_tags::film_destination);
574 }
575 }
576
577 // Call session handler
578 if (session_handler_) {
579 auto result = session_handler_(session);
580 if (result.is_err()) {
581 return send_response(
582 assoc, context_id, command_field::n_create_rsp,
583 request.message_id(), basic_film_session_sop_class_uid,
584 sop_instance_uid, status_error_unable_to_process);
585 }
586 }
587
588 {
589 std::lock_guard<std::mutex> lock(mutex_);
590 sessions_.emplace(sop_instance_uid, std::move(session));
591 }
592
593 ++sessions_created_;
594
595 return send_response(
596 assoc, context_id, command_field::n_create_rsp,
597 request.message_id(), basic_film_session_sop_class_uid,
598 sop_instance_uid, status_success);
599}

601// =============================================================================
602// Film Box Creation
603// =============================================================================
604
605network::Result<std::monostate> print_scp::create_film_box(
606 network::association& assoc,
607 uint8_t context_id,
608 const network::dimse::dimse_message& request) {
609
610 using namespace network::dimse;
611
612 auto sop_instance_uid = request.affected_sop_instance_uid();
613 if (sop_instance_uid.empty()) {
614 sop_instance_uid = generate_uid();
615 }
616
617 film_box box;
618 box.sop_instance_uid = sop_instance_uid;
619
620 if (request.has_dataset()) {
621 const auto& ds = request.dataset().value().get();
622 box.data = ds;
623
624 if (ds.contains(print_tags::image_display_format)) {
625 box.image_display_format = ds.get_string(print_tags::image_display_format);
626 }
627 if (ds.contains(print_tags::film_orientation)) {
628 box.film_orientation = ds.get_string(print_tags::film_orientation);
629 }
630 if (ds.contains(print_tags::film_size_id)) {
631 box.film_size_id = ds.get_string(print_tags::film_size_id);
632 }
633 }
634
635 // Parse image display format to determine image box count
636 // Format: "STANDARD\C,R" where C=columns, R=rows
637 uint16_t num_image_boxes = 1;
638 auto format = box.image_display_format;
639 auto backslash_pos = format.find('\\');
640 if (backslash_pos != std::string::npos) {
641 auto dims = format.substr(backslash_pos + 1);
642 auto comma_pos = dims.find(',');
643 if (comma_pos != std::string::npos) {
644 auto cols = static_cast<uint16_t>(
645 std::stoul(dims.substr(0, comma_pos)));
646 auto rows = static_cast<uint16_t>(
647 std::stoul(dims.substr(comma_pos + 1)));
648 num_image_boxes = static_cast<uint16_t>(cols * rows);
649 }
650 }
651
652 // Create image boxes for this film box
653 std::lock_guard<std::mutex> lock(mutex_);
654
655 // Link to parent film session if found
656 for (auto& [_, session] : sessions_) {
657 session.film_box_uids.push_back(sop_instance_uid);
658 box.film_session_uid = session.sop_instance_uid;
659 break; // Link to first (most recent) session
660 }
661
662 for (uint16_t i = 1; i <= num_image_boxes; ++i) {
663 image_box ib;
664 ib.sop_instance_uid = generate_uid();
665 ib.film_box_uid = sop_instance_uid;
666 ib.image_position = i;
667 box.image_box_uids.push_back(ib.sop_instance_uid);
668 image_boxes_.emplace(ib.sop_instance_uid, std::move(ib));
669 }
670
671 film_boxes_.emplace(sop_instance_uid, std::move(box));
672 ++film_boxes_created_;
673
674 // Build response with Referenced Image Box Sequence
675 dimse_message response{command_field::n_create_rsp, 0};
676 response.set_message_id_responded_to(request.message_id());
677 response.set_affected_sop_class_uid(basic_film_box_sop_class_uid);

679 response.set_status(status_success);
680
681 // Add referenced image box UIDs in response dataset
682 using namespace encoding;
683 core::dicom_dataset response_ds;
684 auto& fb = film_boxes_.at(sop_instance_uid);

693 seq.push_back(std::move(ref_item));
694 }
695 response.set_dataset(std::move(response_ds));
