157struct mpps_result {
159 std::string mpps_sop_instance_uid;
160
162 uint16_t status{0};
163
165 std::string error_comment;
166
168 std::chrono::milliseconds elapsed{0};
169
171 [[nodiscard]] bool is_success() const noexcept {
172 return status == 0x0000;
173 }
174
176 [[nodiscard]] bool is_warning() const noexcept {
177 return (status & 0xF000) == 0xB000;
178 }
179
181 [[nodiscard]] bool is_error() const noexcept {
182 return !is_success() && !is_warning();
183 }
184};
