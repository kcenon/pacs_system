41std::string get_decoded_string(
42 const core::dicom_dataset& ds,
43 core::dicom_tag tag,
44 std::string_view default_value) {
45
46 const auto* elem = ds.get(tag);
47 if (elem == nullptr) {
48 return std::string{default_value};
49 }
50
51 auto raw = elem->as_string();
52 if (!raw.is_ok()) {
53 return std::string{default_value};
54 }
55
56 // Look up Specific Character Set (0008,0005)
57 auto scs_value = ds.get_string(core::tags::specific_character_set);
58 auto scs = parse_specific_character_set(scs_value);
59
60 return decode_to_utf8(raw.value(), scs);
61}

68
69 // Look up Specific Character Set (0008,0005)

72
