86auto uid_mapping::get_or_create(std::string_view original_uid)
87 -> kcenon::common::Result<std::string> {
88 // First try read-only lookup
89 {
90 std::shared_lock lock(mutex_);
91 auto it = original_to_anon_.find(original_uid);
92 if (it != original_to_anon_.end()) {
93 return it->second;
94 }
95 }
96
97 // Need to create new mapping - upgrade to write lock
98 std::unique_lock lock(mutex_);
99
100 // Double-check after acquiring write lock
101 auto it = original_to_anon_.find(original_uid);
102 if (it != original_to_anon_.end()) {
103 return it->second;
104 }
105
106 // Generate new UID
107 auto new_uid = generate_uid();
108 std::string original_str{original_uid};
109
110 original_to_anon_[original_str] = new_uid;
111 anon_to_original_[new_uid] = original_str;
112
113 return new_uid;
114}

136auto uid_mapping::add_mapping(
137 std::string_view original_uid,
138 std::string_view anonymized_uid
139) -> kcenon::common::VoidResult {
140 std::unique_lock lock(mutex_);
141
142 auto existing = original_to_anon_.find(original_uid);
143 if (existing != original_to_anon_.end()) {
144 if (existing->second != anonymized_uid) {
145 return kcenon::common::make_error<std::monostate>(
146 1,
147 "Original UID already mapped to different value",
148 "uid_mapping"
149 );
150 }
151 return kcenon::common::ok();
152 }
153
154 std::string original_str{original_uid};
155 std::string anon_str{anonymized_uid};
156
157 original_to_anon_[original_str] = anon_str;
158 anon_to_original_[anon_str] = original_str;
159
160 return kcenon::common::ok();
161}

184auto uid_mapping::remove(std::string_view original_uid) -> bool {
185 std::unique_lock lock(mutex_);
186
187 auto it = original_to_anon_.find(original_uid);
188 if (it == original_to_anon_.end()) {
189 return false;
190 }
191
192 auto anon_uid = it->second;
193 original_to_anon_.erase(it);
194 anon_to_original_.erase(anon_uid);
195
196 return true;
197}

258auto uid_mapping::generate_uid() const -> std::string {
259 // Generate UID based on timestamp and counter
260 auto now = std::chrono::system_clock::now();
261 auto timestamp = std::chrono::duration_cast<std::chrono::microseconds>(
262 now.time_since_epoch()
263 ).count();
264
265 auto counter = uid_counter_.fetch_add(1);
266
267 // Random component for additional uniqueness
268 std::random_device rd;
269 std::mt19937 gen(rd());
270 std::uniform_int_distribution<std::uint32_t> dist(0, 999999);
271 auto random_part = dist(gen);
272
273 std::ostringstream oss;
274 oss << uid_root_ << "." << timestamp << "." << counter << "." << random_part;
275
276 return oss.str();
277}
