62auto commitment_repository::record_request(
63 const std::string& transaction_uid,
64 const std::string& requesting_ae,
65 const std::vector<services::sop_reference>& references) -> VoidResult {
66 if (!db() || !db()->is_connected()) {
67 return VoidResult(kcenon::common::error_info{
68 -1, "Database not connected", "storage"});
69 }
70
71 return in_transaction([&]() -> VoidResult {
72 commitment_record record;
73 record.transaction_uid = transaction_uid;
74 record.requesting_ae = requesting_ae;
75 record.request_time = format_timestamp(
76 std::chrono::system_clock::now());
77 record.status = commitment_status::pending;
78 record.total_instances = static_cast<int>(references.size());
79
80 auto insert_result = insert(record);
81 if (insert_result.is_err()) {
82 return VoidResult(insert_result.error());
83 }
84
85 return insert_references(transaction_uid, references);
86 });
87}

89auto commitment_repository::update_result(
90 const std::string& transaction_uid,
91 const services::commitment_result& result) -> VoidResult {
92 if (!db() || !db()->is_connected()) {
93 return VoidResult(kcenon::common::error_info{
94 -1, "Database not connected", "storage"});
95 }
96
97 return in_transaction([&]() -> VoidResult {
98 // Determine overall status
99 commitment_status new_status;
100 if (result.failed_references.empty() &&
101 !result.success_references.empty()) {
102 new_status = commitment_status::success;
103 } else if (result.success_references.empty() &&
104 !result.failed_references.empty()) {
105 new_status = commitment_status::failed;
106 } else if (!result.success_references.empty() &&
107 !result.failed_references.empty()) {
108 new_status = commitment_status::partial;
109 } else {
110 new_status = commitment_status::failed;
111 }
112
113 // Update main record
114 auto now_str = format_timestamp(std::chrono::system_clock::now());
115 std::string sql =
116 "UPDATE storage_commitment SET status = '" +
117 std::string(to_string(new_status)) +
118 "', completion_time = '" + now_str +
119 "', success_count = " +
120 std::to_string(result.success_references.size()) +
121 ", failure_count = " +
122 std::to_string(result.failed_references.size()) +
123 " WHERE transaction_uid = '" + transaction_uid + "'";
124
125 auto exec_result = storage_session().execute(sql);
126 if (exec_result.is_err()) {
127 return VoidResult(exec_result.error());
128 }
129
130 // Update successful references
131 for (const auto& ref : result.success_references) {
132 std::string ref_sql =
133 "UPDATE commitment_references SET status = 'success'"
134 " WHERE transaction_uid = '" + transaction_uid +
135 "' AND sop_instance_uid = '" + ref.sop_instance_uid + "'";
136 auto ref_result = storage_session().execute(ref_sql);
137 if (ref_result.is_err()) {
138 return VoidResult(ref_result.error());
139 }
140 }
141
142 // Update failed references
143 for (const auto& [ref, reason] : result.failed_references) {
144 std::string ref_sql =
145 "UPDATE commitment_references SET status = 'failed'"
146 ", failure_reason = " +
147 std::to_string(static_cast<uint16_t>(reason)) +
148 " WHERE transaction_uid = '" + transaction_uid +
149 "' AND sop_instance_uid = '" + ref.sop_instance_uid + "'";
150 auto ref_result = storage_session().execute(ref_sql);
151 if (ref_result.is_err()) {
152 return VoidResult(ref_result.error());
153 }
154 }
155
156 return ok();
157 });
158}

176auto commitment_repository::get_references(
177 const std::string& transaction_uid)
178 -> Result<std::vector<commitment_reference_record>> {
179 if (!db() || !db()->is_connected()) {
180 return Result<std::vector<commitment_reference_record>>(
181 kcenon::common::error_info{-1, "Database not connected", "storage"});
182 }
183
184 auto builder = query_builder();
185 builder.select({"transaction_uid", "sop_class_uid", "sop_instance_uid",
186 "status", "failure_reason"})
187 .from("commitment_references")
188 .where("transaction_uid", "=", transaction_uid);
189
190 auto result = storage_session().select(builder.build());
191 if (result.is_err()) {
192 return Result<std::vector<commitment_reference_record>>(result.error());
193 }
194
195 std::vector<commitment_reference_record> records;
196 records.reserve(result.value().size());
197 for (const auto& row : result.value()) {
198 commitment_reference_record ref;
199 ref.transaction_uid = row.at("transaction_uid");
200 ref.sop_class_uid = row.at("sop_class_uid");
201 ref.sop_instance_uid = row.at("sop_instance_uid");
202 ref.status = row.at("status");
203
204 auto reason_it = row.find("failure_reason");
205 if (reason_it != row.end() && !reason_it->second.empty()) {
206 ref.failure_reason = std::stoi(reason_it->second);
207 }
208
209 records.push_back(std::move(ref));
210 }
211
212 return Result<std::vector<commitment_reference_record>>::ok(
213 std::move(records));
214}

223auto commitment_repository::cleanup_old_transactions(
224 std::chrono::hours max_age) -> Result<size_t> {
225 if (!db() || !db()->is_connected()) {
226 return Result<size_t>(kcenon::common::error_info{
227 -1, "Database not connected", "storage"});
228 }
229
230 auto cutoff = std::chrono::system_clock::now() - max_age;
231 auto cutoff_str = format_timestamp(cutoff);
232
233 // Delete references first (FK constraint)
234 std::string ref_sql =
235 "DELETE FROM commitment_references WHERE transaction_uid IN "
236 "(SELECT transaction_uid FROM storage_commitment "
237 "WHERE status IN ('success', 'failed', 'partial') "
238 "AND completion_time < '" + cutoff_str + "')";
239
240 auto ref_result = storage_session().execute(ref_sql);
241 if (ref_result.is_err()) {
242 return Result<size_t>(ref_result.error());
243 }
244
245 // Count records to be deleted
246 std::string count_sql =
247 "SELECT count(*) as cnt FROM storage_commitment "
248 "WHERE status IN ('success', 'failed', 'partial') "
249 "AND completion_time < '" + cutoff_str + "'";
250
251 auto count_result = storage_session().select(count_sql);
252 size_t deleted_count = 0;
253 if (count_result.is_ok() && !count_result.value().empty()) {
254 deleted_count = std::stoull(count_result.value()[0].at("cnt"));
255 }
256
257 // Delete main records (CASCADE handles references)
258 std::string sql =
259 "DELETE FROM storage_commitment "
260 "WHERE status IN ('success', 'failed', 'partial') "
261 "AND completion_time < '" + cutoff_str + "'";
262
263 auto result = storage_session().execute(sql);
264 if (result.is_err()) {
265 return Result<size_t>(result.error());
266 }
267
268 return kcenon::common::ok(deleted_count);
269}

275auto commitment_repository::map_row_to_entity(const database_row& row) const
276 -> commitment_record {
277 commitment_record record;
278 record.transaction_uid = row.at("transaction_uid");
279 record.requesting_ae = row.at("requesting_ae");
280 record.request_time = row.at("request_time");
281
282 auto completion_it = row.find("completion_time");
283 if (completion_it != row.end()) {
284 record.completion_time = completion_it->second;
285 }
286
287 record.status = commitment_status_from_string(row.at("status"));
288
289 auto total_it = row.find("total_instances");
290 if (total_it != row.end() && !total_it->second.empty()) {
291 record.total_instances = std::stoi(total_it->second);
292 }
293
294 auto success_it = row.find("success_count");
295 if (success_it != row.end() && !success_it->second.empty()) {
296 record.success_count = std::stoi(success_it->second);
297 }
298
299 auto failure_it = row.find("failure_count");
300 if (failure_it != row.end() && !failure_it->second.empty()) {
301 record.failure_count = std::stoi(failure_it->second);
302 }
303
304 return record;
305}

338auto commitment_repository::format_timestamp(
339 std::chrono::system_clock::time_point tp) const -> std::string {
340 auto time_t = std::chrono::system_clock::to_time_t(tp);
341 std::tm tm{};
342#ifdef _WIN32
343 gmtime_s(&tm, &time_t);
344#else
345 gmtime_r(&time_t, &tm);
346#endif
347 std::ostringstream oss;
348 oss << std::put_time(&tm, "%Y-%m-%d %H:%M:%S");
349 return oss.str();
350}
