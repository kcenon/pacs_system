102auto study_lock_manager::lock(
103 const std::string& study_uid,
104 lock_type type,
105 const std::string& reason,
106 const std::string& holder,
107 std::chrono::seconds timeout) -> kcenon::common::Result<lock_token> {
108 const auto resolved_holder = resolve_holder(holder);
109 const auto expiry = calculate_expiry(timeout);
110 const auto now = std::chrono::system_clock::now();
111
112 std::unique_lock lock{mutex_};
113
114 // Check if lock can be acquired
115 auto it = locks_.find(study_uid);
116 if (it != locks_.end()) {
117 auto& existing = it->second;
118
119 // Check if existing lock is expired
120 if (existing.info.is_expired()) {
121 // Clean up expired lock
122 auto token_it = token_to_study_.find(existing.info.token_id);
123 if (token_it != token_to_study_.end()) {
124 token_to_study_.erase(token_it);
125 }
126 locks_.erase(it);
127 it = locks_.end();
128
129 // Notify expiration
130 if (on_lock_expired_) {
131 lock.unlock();
132 on_lock_expired_(study_uid, existing.info);
133 lock.lock();
134 }
135 }
136 }
137
138 // Re-check after potential cleanup
139 it = locks_.find(study_uid);
140 if (it != locks_.end()) {
141 auto& existing = it->second;
142
143 // Handle shared lock upgrade/coexistence
144 if (type == lock_type::shared &&
145 existing.info.type == lock_type::shared) {
146 // Check max shared locks
147 if (existing.shared_holders.size() >= config_.max_shared_locks) {
148 return kcenon::common::Result<lock_token>::err(
149 kcenon::common::error_info{
150 lock_error::max_shared_exceeded,
151 "Maximum shared locks exceeded",
152 "study_lock_manager"});
153 }
154
155 // Add holder to shared list
156 existing.shared_holders.push_back(resolved_holder);
157 existing.info.shared_count = existing.shared_holders.size();
158
159 // Generate new token for this shared holder
160 const auto token_id = generate_token_id();
161 token_to_study_[token_id] = study_uid;
162
163 lock_token token;
164 token.token_id = token_id;
165 token.study_uid = study_uid;
166 token.type = lock_type::shared;
167 token.acquired_at = now;
168 token.expires_at = expiry;
169
170 record_acquisition(type);
171
172 if (on_lock_acquired_) {
173 lock_info info = existing.info;
174 info.token_id = token_id;
175 info.holder = resolved_holder;
176 lock.unlock();
177 on_lock_acquired_(study_uid, info);
178 }
179
180 return kcenon::common::Result<lock_token>::ok(token);
181 }
182
183 // Cannot acquire lock - already locked
184 {
185 std::lock_guard stats_lock{stats_mutex_};
186 ++stats_.contention_count;
187 }
188
189 return kcenon::common::Result<lock_token>::err(
190 kcenon::common::error_info{
191 lock_error::already_locked,
192 "Study is already locked by: " + existing.info.holder,
193 "study_lock_manager",
194 "Lock type: " + to_string(existing.info.type)});
195 }
196
197 // Create new lock
198 const auto token_id = generate_token_id();
199
200 lock_entry entry;
201 entry.info.study_uid = study_uid;
202 entry.info.type = type;
203 entry.info.reason = reason;
204 entry.info.holder = resolved_holder;
205 entry.info.token_id = token_id;
206 entry.info.acquired_at = now;
207 entry.info.expires_at = expiry;
208 entry.info.shared_count = (type == lock_type::shared) ? 1 : 0;
209
210 if (type == lock_type::shared) {
211 entry.shared_holders.push_back(resolved_holder);
212 }
213
214 locks_[study_uid] = std::move(entry);
215 token_to_study_[token_id] = study_uid;
216
217 lock_token token;
218 token.token_id = token_id;
219 token.study_uid = study_uid;
220 token.type = type;
221 token.acquired_at = now;
222 token.expires_at = expiry;
223
224 record_acquisition(type);
225
226 if (on_lock_acquired_) {
227 lock_info info = locks_[study_uid].info;
228 lock.unlock();
229 on_lock_acquired_(study_uid, info);
230 }
231
232 return kcenon::common::Result<lock_token>::ok(token);
233}

250auto study_lock_manager::unlock(const lock_token& token)
251 -> kcenon::common::Result<std::monostate> {
252 std::unique_lock lock{mutex_};
253
254 // Find study by token
255 auto token_it = token_to_study_.find(token.token_id);
256 if (token_it == token_to_study_.end()) {
257 return kcenon::common::Result<std::monostate>::err(
258 kcenon::common::error_info{
259 lock_error::invalid_token,
260 "Invalid or expired token",
261 "study_lock_manager"});
262 }
263
264 const auto study_uid = token_it->second;
265 auto lock_it = locks_.find(study_uid);
266 if (lock_it == locks_.end()) {
267 // Token exists but lock doesn't - clean up
268 token_to_study_.erase(token_it);
269 return kcenon::common::Result<std::monostate>::err(
270 kcenon::common::error_info{
271 lock_error::not_found,
272 "Lock not found",
273 "study_lock_manager"});
274 }
275
276 auto& entry = lock_it->second;
277 const auto duration = entry.info.duration();
278 const auto type = entry.info.type;
279 lock_info released_info = entry.info;
280
281 // Handle shared lock release
282 if (type == lock_type::shared && entry.shared_holders.size() > 1) {
283 // Just remove this holder from shared list
284 auto holder_it = std::find(
285 entry.shared_holders.begin(),
286 entry.shared_holders.end(),
287 token.token_id); // Using token_id as holder identifier for shared
288
289 if (holder_it != entry.shared_holders.end()) {
290 entry.shared_holders.erase(holder_it);
291 }
292 token_to_study_.erase(token_it);
293 entry.info.shared_count = entry.shared_holders.size();
294
295 record_release(type, duration);
296
297 if (on_lock_released_) {
298 lock.unlock();
299 on_lock_released_(study_uid, released_info);
300 }
301
302 return kcenon::common::Result<std::monostate>::ok(std::monostate{});
303 }
304
305 // Remove lock entirely
306 token_to_study_.erase(token_it);
307 locks_.erase(lock_it);
308
309 record_release(type, duration);
310
311 if (on_lock_released_) {
312 lock.unlock();
313 on_lock_released_(study_uid, released_info);
314 }
315
316 return kcenon::common::Result<std::monostate>::ok(std::monostate{});
317}

319auto study_lock_manager::unlock(
320 const std::string& study_uid,
321 const std::string& holder)
322 -> kcenon::common::Result<std::monostate> {
323 const auto resolved_holder = resolve_holder(holder);
324
325 std::unique_lock lock{mutex_};
326
327 auto lock_it = locks_.find(study_uid);
328 if (lock_it == locks_.end()) {
329 return kcenon::common::Result<std::monostate>::err(
330 kcenon::common::error_info{
331 lock_error::not_found,
332 "Lock not found for study",
333 "study_lock_manager"});
334 }
335
336 auto& entry = lock_it->second;
337
338 // Verify holder matches
339 if (entry.info.holder != resolved_holder) {
340 // Check shared holders
341 bool found_shared = false;
342 if (entry.info.type == lock_type::shared) {
343 auto holder_it = std::find(
344 entry.shared_holders.begin(),
345 entry.shared_holders.end(),
346 resolved_holder);
347 if (holder_it != entry.shared_holders.end()) {
348 found_shared = true;
349 entry.shared_holders.erase(holder_it);
350 entry.info.shared_count = entry.shared_holders.size();
351 }
352 }
353
354 if (!found_shared) {
355 return kcenon::common::Result<std::monostate>::err(
356 kcenon::common::error_info{
357 lock_error::permission_denied,
358 "Lock held by different holder: " + entry.info.holder,
359 "study_lock_manager"});
360 }
361
362 if (!entry.shared_holders.empty()) {
363 // Other shared holders remain
364 record_release(entry.info.type, entry.info.duration());
365 return kcenon::common::Result<std::monostate>::ok(std::monostate{});
366 }
367 }
368
369 const auto duration = entry.info.duration();
370 const auto type = entry.info.type;
371 lock_info released_info = entry.info;
372
373 // Remove token mapping
374 auto token_it = token_to_study_.find(entry.info.token_id);
375 if (token_it != token_to_study_.end()) {
376 token_to_study_.erase(token_it);
377 }
378
379 locks_.erase(lock_it);
380
381 record_release(type, duration);
382
383 if (on_lock_released_) {
384 lock.unlock();
385 on_lock_released_(study_uid, released_info);
386 }
387
388 return kcenon::common::Result<std::monostate>::ok(std::monostate{});
389}

391auto study_lock_manager::force_unlock(
392 const std::string& study_uid,
393 [[maybe_unused]] const std::string& admin_reason)
394 -> kcenon::common::Result<std::monostate> {
395 // Note: admin_reason can be used for audit logging in future
396 if (!config_.allow_force_unlock) {
397 return kcenon::common::Result<std::monostate>::err(
398 kcenon::common::error_info{
399 lock_error::permission_denied,
400 "Force unlock is not allowed",
401 "study_lock_manager"});
402 }
403
404 std::unique_lock lock{mutex_};
405
406 auto lock_it = locks_.find(study_uid);
407 if (lock_it == locks_.end()) {
408 return kcenon::common::Result<std::monostate>::err(
409 kcenon::common::error_info{
410 lock_error::not_found,
411 "Lock not found for study",
412 "study_lock_manager"});
413 }
414
415 const auto duration = lock_it->second.info.duration();
416 const auto type = lock_it->second.info.type;
417 lock_info released_info = lock_it->second.info;
418
419 // Remove token mapping
420 auto token_it = token_to_study_.find(lock_it->second.info.token_id);
421 if (token_it != token_to_study_.end()) {
422 token_to_study_.erase(token_it);
423 }
424
425 // Remove all shared holder tokens
426 for (const auto& shared_holder : lock_it->second.shared_holders) {
427 // Shared holders may have separate tokens - clean up if needed
428 (void)shared_holder;
429 }
430
431 locks_.erase(lock_it);
432
433 {
434 std::lock_guard stats_lock{stats_mutex_};
435 ++stats_.force_unlock_count;
436 }
437 record_release(type, duration);
438
439 if (on_lock_released_) {
440 lock.unlock();
441 on_lock_released_(study_uid, released_info);
442 }
443
444 return kcenon::common::Result<std::monostate>::ok(std::monostate{});
445}

447auto study_lock_manager::unlock_all_by_holder(const std::string& holder)
448 -> std::size_t {
449 const auto resolved_holder = resolve_holder(holder);
450 std::size_t count = 0;
451
452 std::unique_lock lock{mutex_};
453
454 std::vector<std::string> to_remove;
455 for (auto& [study_uid, entry] : locks_) {
456 if (entry.info.holder == resolved_holder) {
457 to_remove.push_back(study_uid);
458 } else if (entry.info.type == lock_type::shared) {
459 // Check shared holders
460 auto it = std::find(
461 entry.shared_holders.begin(),
462 entry.shared_holders.end(),
463 resolved_holder);
464 if (it != entry.shared_holders.end()) {
465 entry.shared_holders.erase(it);
466 entry.info.shared_count = entry.shared_holders.size();
467 ++count;
468
469 if (entry.shared_holders.empty()) {
470 to_remove.push_back(study_uid);
471 }
472 }
473 }
474 }
475
476 for (const auto& study_uid : to_remove) {
477 auto it = locks_.find(study_uid);
478 if (it != locks_.end()) {
479 // Remove token mapping
480 auto token_it = token_to_study_.find(it->second.info.token_id);
481 if (token_it != token_to_study_.end()) {
482 token_to_study_.erase(token_it);
483 }
484
485 record_release(it->second.info.type, it->second.info.duration());
486 locks_.erase(it);
487 ++count;
488 }
489 }
490
491 return count;
492}

589auto study_lock_manager::refresh_lock(
590 const lock_token& token,
591 std::chrono::seconds extension)
592 -> kcenon::common::Result<lock_token> {
593 std::unique_lock lock{mutex_};
594
595 auto token_it = token_to_study_.find(token.token_id);
596 if (token_it == token_to_study_.end()) {
597 return kcenon::common::Result<lock_token>::err(
598 kcenon::common::error_info{
599 lock_error::invalid_token,
600 "Invalid or expired token",
601 "study_lock_manager"});
602 }
603
604 auto lock_it = locks_.find(token_it->second);
605 if (lock_it == locks_.end()) {
606 return kcenon::common::Result<lock_token>::err(
607 kcenon::common::error_info{
608 lock_error::not_found,
609 "Lock not found",
610 "study_lock_manager"});
611 }
612
613 if (lock_it->second.info.is_expired()) {
614 return kcenon::common::Result<lock_token>::err(
615 kcenon::common::error_info{
616 lock_error::expired,
617 "Lock has expired",
618 "study_lock_manager"});
619 }
620
621 // Calculate new expiry
622 auto new_extension = extension.count() > 0 ? extension : config_.default_timeout;
623 if (new_extension.count() > 0) {
624 lock_it->second.info.expires_at =
625 std::chrono::system_clock::now() + new_extension;
626 }
627
628 lock_token updated_token = token;
629 updated_token.expires_at = lock_it->second.info.expires_at;
630
631 return kcenon::common::Result<lock_token>::ok(updated_token);
632}

657

669 resolved_holder);
670 if (it != entry.shared_holders.end()) {
671 result.push_back(entry.info);
672 }
673 }
674 }
675
676 return result;

708// =============================================================================
709// Maintenance
710// =============================================================================
711
712auto study_lock_manager::cleanup_expired_locks() -> std::size_t {
713 std::unique_lock lock{mutex_};
714
715 std::vector<std::string> expired_studies;
716
717 for (const auto& [study_uid, entry] : locks_) {
718 if (entry.info.is_expired()) {
719 expired_studies.push_back(study_uid);
720 }
721 }
722
723 for (const auto& study_uid : expired_studies) {
724 auto it = locks_.find(study_uid);
725 if (it != locks_.end()) {
726 lock_info expired_info = it->second.info;
727
728 // Remove token mapping
729 auto token_it = token_to_study_.find(it->second.info.token_id);
730 if (token_it != token_to_study_.end()) {
731 token_to_study_.erase(token_it);
732 }
733
734 locks_.erase(it);
735
736 if (on_lock_expired_) {
737 lock.unlock();
738 on_lock_expired_(study_uid, expired_info);
739 lock.lock();
740 }
741 }
742 }
743
744 return expired_studies.size();
745}

747auto study_lock_manager::get_stats() const -> lock_manager_stats {
748 std::lock_guard stats_lock{stats_mutex_};
749 std::shared_lock lock{mutex_};
750
751 lock_manager_stats current_stats = stats_;
752 current_stats.active_locks = 0;
753 current_stats.exclusive_locks = 0;
754 current_stats.shared_locks = 0;
755 current_stats.migration_locks = 0;
756
757 for (const auto& [study_uid, entry] : locks_) {
758 if (!entry.info.is_expired()) {
759 ++current_stats.active_locks;
760 switch (entry.info.type) {
761 case lock_type::exclusive:
762 ++current_stats.exclusive_locks;
763 break;
764 case lock_type::shared:
765 ++current_stats.shared_locks;

838auto study_lock_manager::calculate_expiry(std::chrono::seconds timeout) const
839 -> std::optional<std::chrono::system_clock::time_point> {
840 auto effective_timeout = timeout.count() > 0 ? timeout : config_.default_timeout;
841 if (effective_timeout.count() <= 0) {

843 }
844
845 return std::chrono::system_clock::now() + effective_timeout;

848auto study_lock_manager::can_acquire_lock(
849 const std::string& study_uid,
850 lock_type type) const -> bool {
851 auto it = locks_.find(study_uid);
852 if (it == locks_.end()) {
853 return true;
854 }
855
856 if (it->second.info.is_expired()) {
857 return true;
858 }
859
860 // Shared locks can coexist with other shared locks
861 if (type == lock_type::shared &&
862 it->second.info.type == lock_type::shared) {
863 return it->second.shared_holders.size() < config_.max_shared_locks;
864 }
865
866 return false;
867}

875void study_lock_manager::record_release(
876 lock_type type,
877 std::chrono::milliseconds duration) {
878 std::lock_guard stats_lock{stats_mutex_};
879 ++stats_.total_releases;
880
881 // Update average duration
882 if (stats_.total_releases > 0) {
883 auto total_duration =
884 stats_.avg_lock_duration.count() * (stats_.total_releases - 1) +
885 duration.count();
886 stats_.avg_lock_duration =
887 std::chrono::milliseconds{total_duration / stats_.total_releases};
888 }
889
890 // Update max duration
891 if (duration > stats_.max_lock_duration) {
892 stats_.max_lock_duration = duration;
893 }
894
895 (void)type; // Type-specific stats are calculated in get_stats()
896}
