233inline bool dicom_storage_collector::initialize(
234 const std::unordered_map<std::string, std::string>& config) {
235 // Extract AE title from config if provided
236 if (auto it = config.find("ae_title"); it != config.end()) {
237 ae_title_ = it->second;
238 }
239
240 // Check for pool metrics configuration
241 if (auto it = config.find("collect_pool_metrics"); it != config.end()) {
242 collect_pool_metrics_ = (it->second == "true" || it->second == "1");
243 }
244
245 init_time_ = std::chrono::steady_clock::now();
246 prev_collection_time_ = init_time_;
247 initialized_ = true;
248 return true;
249}

251inline std::vector<storage_metric> dicom_storage_collector::collect() {
252 std::vector<storage_metric> metrics;
253
254 if (!initialized_) {
255 return metrics;
256 }
257
258 // Collect transfer metrics
259 collect_transfer_metrics(metrics);
260
261 // Collect pool metrics if enabled
262 if (collect_pool_metrics_) {
263 collect_pool_metrics(metrics);
264 }
265
266 // Update statistics
267 {
268 std::lock_guard<std::mutex> lock(stats_mutex_);
269 ++collection_count_;
270 }
271
272 return metrics;
273}

275inline void dicom_storage_collector::collect_transfer_metrics(
276 std::vector<storage_metric>& metrics) {
277
278 const auto& transfer = pacs_metrics::global_metrics().transfer();
279 const auto now = std::chrono::steady_clock::now();
280
281 // Current values
282 const auto bytes_sent = transfer.bytes_sent.load(std::memory_order_relaxed);
283 const auto bytes_received = transfer.bytes_received.load(std::memory_order_relaxed);
284 const auto images_stored = transfer.images_stored.load(std::memory_order_relaxed);
285 const auto images_retrieved = transfer.images_retrieved.load(std::memory_order_relaxed);
286
287 // Total bytes sent (counter)
288 metrics.push_back(create_metric(
289 "dicom_bytes_sent_total",
290 static_cast<double>(bytes_sent),
291 "counter",
292 "bytes"));
293
294 // Total bytes received (counter)
295 metrics.push_back(create_metric(
296 "dicom_bytes_received_total",
297 static_cast<double>(bytes_received),
298 "counter",
299 "bytes"));
300
301 // Images stored (counter)
302 metrics.push_back(create_metric(
303 "dicom_images_stored_total",
304 static_cast<double>(images_stored),
305 "counter",
306 "count"));
307
308 // Images retrieved (counter)
309 metrics.push_back(create_metric(
310 "dicom_images_retrieved_total",
311 static_cast<double>(images_retrieved),
312 "counter",
313 "count"));
314
315 // Calculate throughput rates
316 const auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
317 now - prev_collection_time_);
318
319 if (elapsed.count() > 0) {
320 // Bytes per second sent (gauge)
321 const auto sent_delta = bytes_sent - prev_bytes_sent_;
322 const double send_rate =
323 static_cast<double>(sent_delta) * 1000.0 / static_cast<double>(elapsed.count());
324 metrics.push_back(create_metric(
325 "dicom_bytes_sent_rate",
326 send_rate,
327 "gauge",
328 "bytes_per_second"));
329
330 // Bytes per second received (gauge)
331 const auto received_delta = bytes_received - prev_bytes_received_;
332 const double receive_rate =
333 static_cast<double>(received_delta) * 1000.0 / static_cast<double>(elapsed.count());
334 metrics.push_back(create_metric(
335 "dicom_bytes_received_rate",
336 receive_rate,
337 "gauge",
338 "bytes_per_second"));
339 }
340
341 // Update previous values for next rate calculation
342 prev_bytes_sent_ = bytes_sent;
343 prev_bytes_received_ = bytes_received;
344 prev_collection_time_ = now;
345}

347inline void dicom_storage_collector::collect_pool_metrics(
348 std::vector<storage_metric>& metrics) {
349
350 const auto& pacs = pacs_metrics::global_metrics();
351
352 // Element pool metrics
353 const auto& element_pool = pacs.element_pool();
354 metrics.push_back(create_metric(
355 "dicom_element_pool_acquisitions_total",
356 static_cast<double>(element_pool.total_acquisitions.load(std::memory_order_relaxed)),
357 "counter",
358 "count"));
359 metrics.push_back(create_metric(
360 "dicom_element_pool_hit_ratio",
361 element_pool.hit_ratio(),
362 "gauge",
363 "ratio"));
364 metrics.push_back(create_metric(
365 "dicom_element_pool_size",
366 static_cast<double>(element_pool.current_pool_size.load(std::memory_order_relaxed)),
367 "gauge",
368 "count"));
369
370 // Dataset pool metrics
371 const auto& dataset_pool = pacs.dataset_pool();
372 metrics.push_back(create_metric(
373 "dicom_dataset_pool_acquisitions_total",
374 static_cast<double>(dataset_pool.total_acquisitions.load(std::memory_order_relaxed)),
375 "counter",
376 "count"));
377 metrics.push_back(create_metric(
378 "dicom_dataset_pool_hit_ratio",
379 dataset_pool.hit_ratio(),
380 "gauge",
381 "ratio"));
382 metrics.push_back(create_metric(
383 "dicom_dataset_pool_size",
384 static_cast<double>(dataset_pool.current_pool_size.load(std::memory_order_relaxed)),
385 "gauge",
386 "count"));
387
388 // PDU buffer pool metrics
389 const auto& pdu_pool = pacs.pdu_buffer_pool();
390 metrics.push_back(create_metric(
391 "dicom_pdu_buffer_pool_acquisitions_total",
392 static_cast<double>(pdu_pool.total_acquisitions.load(std::memory_order_relaxed)),
393 "counter",
394 "count"));
395 metrics.push_back(create_metric(
396 "dicom_pdu_buffer_pool_hit_ratio",
397 pdu_pool.hit_ratio(),
398 "gauge",
399 "ratio"));
400 metrics.push_back(create_metric(
401 "dicom_pdu_buffer_pool_size",
402 static_cast<double>(pdu_pool.current_pool_size.load(std::memory_order_relaxed)),
403 "gauge",
404 "count"));
405}

411inline std::vector<std::string> dicom_storage_collector::get_metric_types() const {
412 std::vector<std::string> types = {
413 // Transfer metrics
414 "dicom_bytes_sent_total",
415 "dicom_bytes_received_total",
416 "dicom_images_stored_total",
417 "dicom_images_retrieved_total",
418 "dicom_bytes_sent_rate",
419 "dicom_bytes_received_rate"
420 };
421
422 if (collect_pool_metrics_) {
423 // Element pool

425 types.push_back("dicom_element_pool_hit_ratio");
426 types.push_back("dicom_element_pool_size");
427 // Dataset pool
428 types.push_back("dicom_dataset_pool_acquisitions_total");
429 types.push_back("dicom_dataset_pool_hit_ratio");
430 types.push_back("dicom_dataset_pool_size");
431 // PDU buffer pool
432 types.push_back("dicom_pdu_buffer_pool_acquisitions_total");
433 types.push_back("dicom_pdu_buffer_pool_hit_ratio");
434 types.push_back("dicom_pdu_buffer_pool_size");
435 }
436
437 return types;

444inline std::unordered_map<std::string, double>
445dicom_storage_collector::get_statistics() const {
446 std::lock_guard<std::mutex> lock(stats_mutex_);
447
448 std::unordered_map<std::string, double> stats;
449 stats["collection_count"] = static_cast<double>(collection_count_);
450
451 if (initialized_) {
452 const auto uptime = std::chrono::duration_cast<std::chrono::seconds>(
453 std::chrono::steady_clock::now() - init_time_);
454 stats["uptime_seconds"] = static_cast<double>(uptime.count());
455 }
456
457 return stats;
458}
