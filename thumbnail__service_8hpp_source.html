258 struct cache_key_hash {
259 size_t operator()(const cache_key& k) const {
260 size_t h = std::hash<std::string>{}(k.uid);
261 h ^= std::hash<uint16_t>{}(k.size) << 1;
262 h ^= std::hash<std::string>{}(k.format) << 2;
263 h ^= std::hash<int>{}(k.quality) << 3;
264 h ^= std::hash<uint32_t>{}(k.frame) << 4;
265 return h;
266 }
267 };
