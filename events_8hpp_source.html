191enum class query_level {
192 patient,
193 study,
194 series,
195 image
196};
197
201[[nodiscard]] inline auto query_level_to_string(query_level level) -> std::string {
202 switch (level) {
203 case query_level::patient: return "PATIENT";
204 case query_level::study: return "STUDY";
205 case query_level::series: return "SERIES";
206 case query_level::image: return "IMAGE";
207 default: return "UNKNOWN";
208 }
209}
