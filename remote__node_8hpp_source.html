80 error,

89[[nodiscard]] constexpr const char* to_string(node_status status) noexcept {
90 switch (status) {
91 case node_status::unknown: return "unknown";
92 case node_status::online: return "online";
93 case node_status::offline: return "offline";
94 case node_status::error: return "error";
95 case node_status::verifying: return "verifying";
96 default: return "unknown";
97 }
98}
99
105[[nodiscard]] inline node_status node_status_from_string(std::string_view str) noexcept {
106 if (str == "online") return node_status::online;
107 if (str == "offline") return node_status::offline;
108 if (str == "error") return node_status::error;
109 if (str == "verifying") return node_status::verifying;
110 return node_status::unknown;
111}

189

192 // =========================================================================

201 // =========================================================================
202
203 int64_t pk{0};

205 std::chrono::system_clock::time_point updated_at;
206
207 // =========================================================================
208 // Convenience Methods
209 // =========================================================================
210
214 [[nodiscard]] bool supports_query_retrieve() const noexcept {
215 return supports_find && (supports_move || supports_get);
216 }
217
221 [[nodiscard]] bool is_online() const noexcept {
222 return status == node_status::online;
223 }
224
228 [[nodiscard]] bool has_tls() const noexcept {
229 return tls.has_value() && tls->is_configured();
230 }

236 return host + ":" + std::to_string(port);
237 }
