85std::string generate_uuid() {
86 static std::random_device rd;
87 static std::mt19937 gen(rd());
88 static std::uniform_int_distribution<> dis(0, 15);
89 static const char *hex = "0123456789abcdef";
90
91 std::string uuid = "xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx";
92 for (char &c : uuid) {
93 if (c == 'x') {
94 c = hex[dis(gen)];
95 } else if (c == 'y') {
96 c = hex[(dis(gen) & 0x3) | 0x8];
97 }
98 }
99 return uuid;
100}

105std::string format_timestamp(
106 const std::chrono::system_clock::time_point &tp) {
107 auto time = std::chrono::system_clock::to_time_t(tp);
108 std::ostringstream oss;
109 oss << std::put_time(std::gmtime(&time), "%Y-%m-%dT%H:%M:%SZ");

176std::string parse_json_string(const std::string &json, const std::string &key) {
177 std::string search = "\"" + key + "\":\"";
178 auto pos = json.find(search);
179 if (pos == std::string::npos) {
180 return "";
181 }
182 pos += search.length();
183 auto end = json.find("\"", pos);
184 if (end == std::string::npos) {
185 return "";
186 }
187 return json.substr(pos, end - pos);
188}

193std::string parse_json_object(const std::string &json, const std::string &key) {
194 std::string search = "\"" + key + "\":";
195 auto pos = json.find(search);
196 if (pos == std::string::npos) {
197 return "{}";
198 }
199 pos += search.length();
200 while (pos < json.size() && (json[pos] == ' ' || json[pos] == '\t')) {
201 ++pos;
202 }
203 if (pos >= json.size() || json[pos] != '{') {
204 return "{}";
205 }
206
207 int depth = 0;
208 size_t start = pos;
209 for (; pos < json.size(); ++pos) {
210 if (json[pos] == '{') {
211 ++depth;
212 } else if (json[pos] == '}') {
213 --depth;
214 if (depth == 0) {
215 return json.substr(start, pos - start + 1);
216 }
217 }
218 }
219 return "{}";
220}

225std::string build_state_json(const std::string &body) {
226 // Extract individual state components and build combined JSON
227 std::string layout = parse_json_object(body, "layout");
228 std::string viewports = "[]";
229 std::string active_viewport = "";
230
231 // Parse viewports array
232 std::string viewports_search = "\"viewports\":";
233 auto viewports_pos = body.find(viewports_search);
234 if (viewports_pos != std::string::npos) {
235 viewports_pos += viewports_search.length();
236 while (viewports_pos < body.size() &&
237 (body[viewports_pos] == ' ' || body[viewports_pos] == '\t')) {
238 ++viewports_pos;
239 }
240 if (viewports_pos < body.size() && body[viewports_pos] == '[') {
241 int depth = 0;
242 size_t start = viewports_pos;
243 for (; viewports_pos < body.size(); ++viewports_pos) {
244 if (body[viewports_pos] == '[') {
245 ++depth;
246 } else if (body[viewports_pos] == ']') {
247 --depth;
248 if (depth == 0) {
249 viewports = body.substr(start, viewports_pos - start + 1);
250 break;
251 }
252 }
253 }
254 }
255 }
256
257 active_viewport = parse_json_string(body, "active_viewport");
258
259 std::ostringstream oss;
260 oss << R"({"layout":)" << layout
261 << R"(,"viewports":)" << viewports
262 << R"(,"active_viewport":")" << json_escape(active_viewport) << R"("})";
263 return oss.str();
264}

270 std::shared_ptr<rest_server_context> ctx) {
271 // POST /api/v1/viewer-states - Create viewer state
272 CROW_ROUTE(app, "/api/v1/viewer-states")
273 .methods(crow::HTTPMethod::POST)([ctx](const crow::request &req) {
274 crow::response res;
275 res.add_header("Content-Type", "application/json");
276 add_cors_headers(res, *ctx);
277
278 if (!ctx->database) {
279 res.code = 503;
280 res.body =
281 make_error_json("DATABASE_UNAVAILABLE", "Database not configured");
282 return res;
283 }
284
285 std::string body = req.body;
286 if (body.empty()) {
287 res.code = 400;
288 res.body = make_error_json("INVALID_REQUEST", "Request body is empty");
289 return res;
290 }
291
292 storage::viewer_state_record state;
293 state.state_id = generate_uuid();
294 state.study_uid = parse_json_string(body, "study_uid");
295 state.user_id = parse_json_string(body, "user_id");
296 state.state_json = build_state_json(body);
297 state.created_at = std::chrono::system_clock::now();
298 state.updated_at = state.created_at;
299
300 if (state.study_uid.empty()) {
301 res.code = 400;
302 res.body = make_error_json("MISSING_FIELD", "study_uid is required");
303 return res;
304 }
305
306 #ifdef PACS_WITH_DATABASE_SYSTEM
307 storage::viewer_state_record_repository repo(ctx->database->db_adapter());
308 storage::recent_study_repository recent_repo(ctx->database->db_adapter());
309#else
310 storage::viewer_state_repository repo(ctx->database->native_handle());
311#endif
312 auto save_result =
313#ifdef PACS_WITH_DATABASE_SYSTEM
314 repo.save(state);
315#else
316 repo.save_state(state);
317#endif
318 if (!save_result.is_ok()) {
319 res.code = 500;
320 res.body =
321 make_error_json("SAVE_ERROR", save_result.error().message);
322 return res;
323 }
324
325 // Also record study access if user_id is provided
326 if (!state.user_id.empty()) {
327 #ifdef PACS_WITH_DATABASE_SYSTEM
328 (void)recent_repo.record_access(state.user_id, state.study_uid);
329 #else
330 (void)repo.record_study_access(state.user_id, state.study_uid);
331 #endif
332 }
333
334 res.code = 201;
335 std::ostringstream oss;
336 oss << R"({"state_id":")" << json_escape(state.state_id)
337 << R"(","created_at":")" << format_timestamp(state.created_at)
338 << R"("})";
339 res.body = oss.str();
340 return res;
341 });
