96void pipeline_adapter::register_session(uint64_t session_id,
97 session_context context) {
98 std::lock_guard<std::mutex> lock(sessions_mutex_);
99 context.session_id = session_id;
100 context.created_at = std::chrono::steady_clock::now();
101 context.last_activity = context.created_at;
102 sessions_[session_id] = std::move(context);
103
104 // Update metrics
105 coordinator_->get_metrics().increment_active_associations();
106
107 // Notify session event
108 session_event_callback callback;
109 {
110 std::lock_guard<std::mutex> cb_lock(callbacks_mutex_);
111 callback = session_event_callback_;
112 }
113 if (callback) {
114 callback(session_id, "session_registered");
115 }
116}
117
118void pipeline_adapter::unregister_session(uint64_t session_id) {
119 std::lock_guard<std::mutex> lock(sessions_mutex_);
120 if (sessions_.erase(session_id) > 0) {
121 coordinator_->get_metrics().decrement_active_associations();
122
123 // Notify session event
124 session_event_callback callback;
125 {
126 std::lock_guard<std::mutex> cb_lock(callbacks_mutex_);
127 callback = session_event_callback_;
128 }
129 if (callback) {
130 callback(session_id, "session_unregistered");
131 }
132 }
133}

154auto pipeline_adapter::on_data_received(uint64_t session_id,
155 std::vector<uint8_t> data) -> VoidResult {
156 if (!is_running()) {
157 return pacs_void_error(error_codes::not_initialized,
158 "Pipeline adapter is not running");
159 }
160
161 // Update session activity
162 {
163 std::lock_guard<std::mutex> lock(sessions_mutex_);
164 auto it = sessions_.find(session_id);
165 if (it != sessions_.end()) {
166 it->second.last_activity = std::chrono::steady_clock::now();
167 }
168 }
169
170 // Get send callback for use in jobs
171 send_callback send_fn;
172 {
173 std::lock_guard<std::mutex> lock(callbacks_mutex_);
174 send_fn = send_callback_;
175 }
176
177 // Get association callback
178 association_callback assoc_cb;
179 {
180 std::lock_guard<std::mutex> lock(callbacks_mutex_);
181 assoc_cb = association_callback_;
182 }
183
184 // Create receive job
185 auto job = std::make_unique<receive_network_io_job>(
186 session_id,
187 std::move(data),
188 nullptr, // on_data callback not needed
189 nullptr // on_error callback not needed
190 );
191
192 return coordinator_->submit_to_stage(
193 pipeline_stage::network_receive,
194 std::move(job)
195 );
196}

301void pipeline_adapter::on_job_completed(const job_context& ctx, bool success) {
302 // Record end-to-end metrics if this is the final stage
303 if (ctx.stage == pipeline_stage::network_send) {
304 auto now = std::chrono::steady_clock::now();
305 auto enqueue_time = std::chrono::steady_clock::time_point(
306 std::chrono::nanoseconds(ctx.enqueue_time_ns)
307 );
308 auto latency_ns = std::chrono::duration_cast<std::chrono::nanoseconds>(
309 now - enqueue_time
310 ).count();
311
312 coordinator_->get_metrics().record_operation_completion(
313 ctx.category,
314 static_cast<uint64_t>(latency_ns),
315 success
316 );
317 }
318}

320void pipeline_adapter::on_backpressure(pipeline_stage stage, size_t queue_depth) {
321 // Log backpressure event
322 // In a full implementation, this could trigger adaptive throttling
323 session_event_callback callback;
324 {
325 std::lock_guard<std::mutex> lock(callbacks_mutex_);
326 callback = session_event_callback_;
327 }
328
329 if (callback) {
330 std::string event = "backpressure_" + std::string(get_stage_name(stage)) +
331 "_depth_" + std::to_string(queue_depth);
332 callback(0, event); // session_id 0 for system events
333 }
334}
