54void dicom_dictionary::initialize_standard_tags() {
55 const auto tags = get_standard_tags();
56
57 tag_map_.reserve(tags.size());
58 keyword_map_.reserve(tags.size());
59
60 for (const auto& info : tags) {
61 tag_map_.emplace(info.tag, info);
62 if (!info.keyword.empty()) {
63 keyword_map_.emplace(info.keyword, info.tag);
64 }
65 }
66
67 standard_count_ = tags.size();
68}

70auto dicom_dictionary::find(dicom_tag tag) const -> std::optional<tag_info> {
71 std::shared_lock lock(mutex_);
72
73 const auto it = tag_map_.find(tag);

80auto dicom_dictionary::find_by_keyword(std::string_view keyword) const
81 -> std::optional<tag_info> {
82 std::shared_lock lock(mutex_);
83
84 const auto it = keyword_map_.find(keyword);
85 if (it != keyword_map_.end()) {
86 const auto tag_it = tag_map_.find(it->second);

120auto dicom_dictionary::register_private_tag(const tag_info& info) -> bool {
121 // Only allow private tags
122 if (!info.tag.is_private()) {
123 return false;
124 }
125
126 std::unique_lock lock(mutex_);
127
128 // Check if already exists
129 if (tag_map_.contains(info.tag)) {
130 return false;
131 }
132
133 tag_map_.emplace(info.tag, info);
134 if (!info.keyword.empty()) {
135 keyword_map_.emplace(info.keyword, info.tag);
136 }
137
138 return true;
139}

155auto dicom_dictionary::get_tags_in_group(uint16_t group) const
156 -> std::vector<tag_info> {
157 std::shared_lock lock(mutex_);
158
159 std::vector<tag_info> result;
160 for (const auto& [tag, info] : tag_map_) {
161 if (tag.group() == group) {
162 result.push_back(info);
163 }
164 }
165
166 std::sort(result.begin(), result.end(),
167 [](const auto& a, const auto& b) {
168 return a.tag < b.tag;
169 });
170
171 return result;
172}
