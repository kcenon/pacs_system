446 template <typename Func>
447 [[nodiscard]] auto transaction(Func&& func) -> VoidResult {
448 auto begin_result = begin_transaction();
449 if (begin_result.is_err()) {
450 return begin_result;
451 }
452
453 try {
454 auto result = std::forward<Func>(func)();
455 if (result.is_err()) {
456 (void)rollback();
457 return result;
458 }
459
460 return commit();
461 } catch (const std::exception& e) {
462 (void)rollback();
463 return VoidResult(kcenon::common::error_info{
464 -1, std::string("Transaction failed: ") + e.what(), "storage"});
465 }
466 }
