217 [[nodiscard]] auto should_migrate(const tier_policy& policy,
218 storage_tier target_tier) const -> bool {
219 // Can only migrate to a "colder" tier
220 if (target_tier <= current_tier) {
221 return false;
222 }
223
224 // Check size threshold
225 if (size_bytes < policy.min_migration_size) {
226 return false;
227 }
228
229 auto time_inactive = time_since_access();
230
231 // Check if instance has been inactive long enough
232 if (current_tier == storage_tier::hot &&
233 target_tier == storage_tier::warm) {
234 return time_inactive >= policy.hot_to_warm;
235 }
236
237 if (current_tier == storage_tier::warm &&
238 target_tier == storage_tier::cold) {
239 return time_inactive >= policy.warm_to_cold;
240 }
241
242 // Hot directly to cold (must meet both thresholds)
243 if (current_tier == storage_tier::hot &&
244 target_tier == storage_tier::cold) {
245 return time_inactive >= (policy.hot_to_warm + policy.warm_to_cold);
246 }
247
248 return false;
249 }
