131[[nodiscard]] inline std::string_view
132to_string(commitment_failure_reason reason) noexcept {
133 switch (reason) {
134 case commitment_failure_reason::processing_failure:
135 return "Processing failure";
136 case commitment_failure_reason::no_such_object_instance:
137 return "No such object instance";
138 case commitment_failure_reason::resource_limitation:
139 return "Resource limitation";
140 case commitment_failure_reason::referenced_sop_class_not_supported:
141 return "Referenced SOP Class not supported";
142 case commitment_failure_reason::class_instance_conflict:
143 return "Class/Instance conflict";
144 case commitment_failure_reason::duplicate_transaction_uid:
145 return "Duplicate Transaction UID";
146 }
147 return "Unknown failure reason";
148}

174 [[nodiscard]] bool all_successful() const noexcept {
175 return failed_references.empty() && !success_references.empty();
176 }
177
179 [[nodiscard]] std::size_t total_instances() const noexcept {
180 return success_references.size() + failed_references.size();
181 }
