287inline bool dicom_metrics_collector::do_initialize(const config_map& config) {
288 // Parse collection configuration
289 if (auto it = config.find("collect_associations"); it != config.end()) {
290 collect_associations_ = (it->second == "true" || it->second == "1");
291 }
292 if (auto it = config.find("collect_transfers"); it != config.end()) {
293 collect_transfers_ = (it->second == "true" || it->second == "1");
294 }
295 if (auto it = config.find("collect_storage"); it != config.end()) {
296 collect_storage_ = (it->second == "true" || it->second == "1");
297 }
298 if (auto it = config.find("collect_queries"); it != config.end()) {
299 collect_queries_ = (it->second == "true" || it->second == "1");
300 }
301 if (auto it = config.find("collect_pools"); it != config.end()) {
302 collect_pools_ = (it->second == "true" || it->second == "1");
303 }
304
305 return true;
306}

308inline std::vector<dicom_metric> dicom_metrics_collector::do_collect() {
309 std::vector<dicom_metric> metrics;
310 metrics.reserve(64); // Pre-allocate for typical metric count
311
312 if (collect_associations_) {
313 collect_association_metrics(metrics);
314 }
315 if (collect_transfers_) {
316 collect_transfer_metrics(metrics);
317 }
318 if (collect_storage_) {
319 collect_storage_metrics(metrics);
320 }
321 if (collect_queries_) {
322 collect_query_metrics(metrics);
323 }
324 if (collect_pools_) {
325 collect_pool_metrics(metrics);
326 }
327
328 return metrics;
329}

335inline std::vector<std::string> dicom_metrics_collector::do_get_metric_types() const {
336 std::vector<std::string> types;
337
338 // Association metrics
339 types.push_back("dicom_associations_active");
340 types.push_back("dicom_associations_peak");
341 types.push_back("dicom_associations_total");
342 types.push_back("dicom_associations_rejected_total");
343 types.push_back("dicom_associations_aborted_total");
344
345 // Transfer metrics
346 types.push_back("dicom_images_sent_total");
347 types.push_back("dicom_images_received_total");
348 types.push_back("dicom_bytes_sent_total");
349 types.push_back("dicom_bytes_received_total");
350 types.push_back("dicom_transfer_rate_mbps");
351
352 // DIMSE operation metrics
353 static constexpr std::array<const char*, 11> ops = {
354 "c_echo", "c_store", "c_find", "c_move", "c_get",
355 "n_create", "n_set", "n_get", "n_action", "n_event", "n_delete"
356 };
357 for (const auto* op : ops) {
358 types.push_back(std::string("dicom_") + op + "_total");
359 types.push_back(std::string("dicom_") + op + "_success_total");
360 types.push_back(std::string("dicom_") + op + "_failure_total");
361 types.push_back(std::string("dicom_") + op + "_duration_seconds");
362 }
363
364 // Pool metrics
365 types.push_back("dicom_element_pool_hits");
366 types.push_back("dicom_element_pool_misses");
367 types.push_back("dicom_dataset_pool_hits");
368 types.push_back("dicom_dataset_pool_misses");
369
370 return types;
371}

373inline void dicom_metrics_collector::do_add_statistics(stats_map& stats) const {
374 std::lock_guard<std::mutex> lock(snapshot_mutex_);
375
376 stats["active_associations"] =
377 static_cast<double>(last_snapshot_.active_associations);
378 stats["total_images_processed"] =

384inline dicom_metrics_snapshot dicom_metrics_collector::get_snapshot() const {
385 dicom_metrics_snapshot snapshot;
386 snapshot.timestamp = std::chrono::system_clock::now();
387
388 const auto& pacs = pacs_metrics::global_metrics();
389 const auto& assoc = pacs.associations();
390 const auto& transfer = pacs.transfer();
391
392 // Association metrics
393 snapshot.total_associations =
394 assoc.total_established.load(std::memory_order_relaxed);
395 snapshot.active_associations =
396 assoc.current_active.load(std::memory_order_relaxed);
397 snapshot.failed_associations =
398 assoc.total_rejected.load(std::memory_order_relaxed) +
399 assoc.total_aborted.load(std::memory_order_relaxed);
400 snapshot.peak_active_associations =
401 assoc.peak_active.load(std::memory_order_relaxed);
402
403 // Transfer metrics
404 snapshot.images_sent = 0; // Not tracked separately
405 snapshot.images_received =
406 transfer.images_stored.load(std::memory_order_relaxed);
407 snapshot.bytes_sent =
408 transfer.bytes_sent.load(std::memory_order_relaxed);
409 snapshot.bytes_received =
410 transfer.bytes_received.load(std::memory_order_relaxed);
411
412 // Storage metrics (from C-STORE counter)
413 const auto& store = pacs.get_counter(dimse_operation::c_store);
414 snapshot.store_operations = store.total_count();
415 snapshot.successful_stores =
416 store.success_count.load(std::memory_order_relaxed);
417 snapshot.failed_stores =
418 store.failure_count.load(std::memory_order_relaxed);
419 snapshot.avg_store_latency_ms =
420 static_cast<double>(store.average_duration_us()) / 1000.0;
421
422 // Query metrics (from C-FIND counter)
423 const auto& query = pacs.get_counter(dimse_operation::c_find);
424 snapshot.query_operations = query.total_count();
425 snapshot.successful_queries =
426 query.success_count.load(std::memory_order_relaxed);
427 snapshot.failed_queries =
428 query.failure_count.load(std::memory_order_relaxed);
429 snapshot.avg_query_latency_ms =
430 static_cast<double>(query.average_duration_us()) / 1000.0;
431
432 return snapshot;
433}

460inline void dicom_metrics_collector::collect_association_metrics(
461 std::vector<dicom_metric>& metrics) {
462
463 const auto& assoc = pacs_metrics::global_metrics().associations();
464
465 // Active associations (gauge)
466 metrics.push_back(create_base_metric(
467 "dicom_associations_active",
468 static_cast<double>(assoc.current_active.load(std::memory_order_relaxed)),
469 "gauge"));
470
471 // Peak active associations (gauge)
472 metrics.push_back(create_base_metric(
473 "dicom_associations_peak",
474 static_cast<double>(assoc.peak_active.load(std::memory_order_relaxed)),
475 "gauge"));
476
477 // Total established (counter)
478 metrics.push_back(create_base_metric(
479 "dicom_associations_total",
480 static_cast<double>(assoc.total_established.load(std::memory_order_relaxed)),
481 "counter"));
482
483 // Rejected (counter)
484 metrics.push_back(create_base_metric(
485 "dicom_associations_rejected_total",
486 static_cast<double>(assoc.total_rejected.load(std::memory_order_relaxed)),
487 "counter"));
488
489 // Aborted (counter)
490 metrics.push_back(create_base_metric(
491 "dicom_associations_aborted_total",
492 static_cast<double>(assoc.total_aborted.load(std::memory_order_relaxed)),
493 "counter"));
494
495 // Success rate (gauge)
496 const auto total = assoc.total_established.load(std::memory_order_relaxed);
497 const auto rejected = assoc.total_rejected.load(std::memory_order_relaxed);
498 const auto attempted = total + rejected;
499 const double success_rate = (attempted > 0)
500 ? static_cast<double>(total) / static_cast<double>(attempted)
501 : 1.0;
502
503 metrics.push_back(create_base_metric(
504 "dicom_associations_success_rate",
505 success_rate,
506 "gauge"));
507}

654inline void dicom_metrics_collector::collect_dimse_operation_metrics(
655 std::vector<dicom_metric>& metrics,
656 const std::string& op_name,
657 const operation_counter& counter) {
658
659 std::unordered_map<std::string, std::string> op_tags{{"operation", op_name}};
660
661 // Total requests (counter)
662 metrics.push_back(create_base_metric(
663 "dicom_" + op_name + "_total",
664 static_cast<double>(counter.total_count()),
665 "counter",
666 op_tags));
667
668 // Successful requests (counter)
669 metrics.push_back(create_base_metric(
670 "dicom_" + op_name + "_success_total",
671 static_cast<double>(counter.success_count.load(std::memory_order_relaxed)),
672 "counter",
673 op_tags));
674
675 // Failed requests (counter)
676 metrics.push_back(create_base_metric(
677 "dicom_" + op_name + "_failure_total",
678 static_cast<double>(counter.failure_count.load(std::memory_order_relaxed)),
679 "counter",
680 op_tags));
681
682 // Duration metrics (only if there have been requests)
683 if (counter.total_count() > 0) {
684 // Average duration in seconds (gauge)
685 metrics.push_back(create_base_metric(
686 "dicom_" + op_name + "_duration_seconds_avg",
687 static_cast<double>(counter.average_duration_us()) / 1'000'000.0,
688 "gauge",
689 op_tags));
690
691 // Total duration in seconds (counter) - for rate calculations
692 metrics.push_back(create_base_metric(
693 "dicom_" + op_name + "_duration_seconds_sum",
694 static_cast<double>(counter.total_duration_us.load(std::memory_order_relaxed)) / 1'000'000.0,
695 "counter",
696 op_tags));
697
698 // Min duration (gauge)
699 const auto min_us = counter.min_duration_us.load(std::memory_order_relaxed);
700 if (min_us != UINT64_MAX) {
701 metrics.push_back(create_base_metric(
702 "dicom_" + op_name + "_duration_seconds_min",
703 static_cast<double>(min_us) / 1'000'000.0,
704 "gauge",
705 op_tags));
706 }
707
708 // Max duration (gauge)
709 const auto max_us = counter.max_duration_us.load(std::memory_order_relaxed);
710 if (max_us > 0) {
711 metrics.push_back(create_base_metric(
712 "dicom_" + op_name + "_duration_seconds_max",
713 static_cast<double>(max_us) / 1'000'000.0,
714 "gauge",
715 op_tags));
716 }
717 }
