70[[nodiscard]] inline auto to_string(mpps_status status) -> std::string {
71 switch (status) {
72 case mpps_status::in_progress:
73 return "IN PROGRESS";
74 case mpps_status::completed:
75 return "COMPLETED";
76 case mpps_status::discontinued:
77 return "DISCONTINUED";
78 default:
79 return "IN PROGRESS";
80 }
81}
82
89[[nodiscard]] inline auto parse_mpps_status(std::string_view str)
90 -> std::optional<mpps_status> {
91 if (str == "IN PROGRESS") {
92 return mpps_status::in_progress;
93 }
94 if (str == "COMPLETED") {
95 return mpps_status::completed;
96 }
97 if (str == "DISCONTINUED") {
98 return mpps_status::discontinued;
99 }
100 return std::nullopt;
101}
