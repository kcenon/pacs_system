102struct retrieve_progress {
103 uint16_t remaining{0};
104 uint16_t completed{0};
105 uint16_t failed{0};
106 uint16_t warning{0};
107 std::chrono::steady_clock::time_point start_time;
108
113 [[nodiscard]] uint16_t total() const noexcept {
114 return remaining + completed + failed + warning;
115 }
116
121 [[nodiscard]] float percent() const noexcept {
122 uint16_t t = total();
123 if (t == 0) return 0.0f;
124 return (static_cast<float>(completed + failed + warning) / t) * 100.0f;
125 }
126
131 [[nodiscard]] std::chrono::milliseconds elapsed() const noexcept {
132 return std::chrono::duration_cast<std::chrono::milliseconds>(
133 std::chrono::steady_clock::now() - start_time);
134 }
135};

137// =============================================================================
138// Retrieve Result Structure
139// =============================================================================
140
146struct retrieve_result {
148 uint16_t completed{0};
149
151 uint16_t failed{0};
152
154 uint16_t warning{0};
155
157 uint16_t final_status{0};
158
160 std::chrono::milliseconds elapsed{0};
161
163 std::vector<core::dicom_dataset> received_instances;
164
169 [[nodiscard]] bool is_success() const noexcept {
170 return final_status == 0x0000 && failed == 0;
171 }
172
177 [[nodiscard]] bool is_cancelled() const noexcept {
178 return final_status == 0xFE00;
179 }
180
185 [[nodiscard]] bool has_failures() const noexcept {
186 return failed > 0;
187 }
188
193 [[nodiscard]] bool has_warnings() const noexcept {
194 return warning > 0;
195 }
196};
