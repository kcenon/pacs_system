119[[nodiscard]] inline kcenon::common::VoidResult register_services(
120 kcenon::common::di::IServiceContainer& container,
121 const registration_config& config = {}) {
122
123 using namespace kcenon::common::di;
124
125 const auto lifetime = config.use_singletons
126 ? service_lifetime::singleton
127 : service_lifetime::transient;
128
129 // -------------------------------------------------------------------------
130 // Register IDicomStorage
131 // -------------------------------------------------------------------------
132 {
133 auto storage_path = config.storage_path;
134 if (storage_path.empty()) {
135 storage_path = std::filesystem::temp_directory_path() / "pacs_storage";
136 }
137
138 auto result = container.register_factory<IDicomStorage>(
139 [path = storage_path](IServiceContainer&) -> std::shared_ptr<IDicomStorage> {
140 storage::file_storage_config storage_config;
141 storage_config.root_path = path;
142 storage_config.create_directories = true;
143 return std::make_shared<storage::file_storage>(storage_config);
144 },
145 lifetime
146 );
147
148 if (result.is_err()) {
149 return result;
150 }
151 }
152
153 // -------------------------------------------------------------------------
154 // Register IDicomNetwork
155 // -------------------------------------------------------------------------
156 if (config.enable_network) {
157 auto result = container.register_factory<IDicomNetwork>(
158 [](IServiceContainer&) -> std::shared_ptr<IDicomNetwork> {
159 return std::make_shared<DicomNetworkService>();
160 },
161 lifetime
162 );
163
164 if (result.is_err()) {
165 return result;
166 }
167 }
168
169 // -------------------------------------------------------------------------
170 // Register ILogger
171 // -------------------------------------------------------------------------
172 if (config.enable_logger) {
173 auto result = container.register_factory<ILogger>(
174 [](IServiceContainer&) -> std::shared_ptr<ILogger> {
175 return std::make_shared<LoggerService>();
176 },
177 lifetime
178 );
179
180 if (result.is_err()) {
181 return result;
182 }
183 }
184
185 return kcenon::common::VoidResult::ok({});
186}
