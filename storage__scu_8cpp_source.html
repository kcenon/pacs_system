65[[nodiscard]] bool is_dicom_file(const std::filesystem::path& path) {
66 const auto ext = path.extension().string();
67 if (ext.empty()) {
68 // Files without extension might still be DICOM
69 return true;
70 }
71 return std::find(dicom_extensions.begin(), dicom_extensions.end(), ext)
72 != dicom_extensions.end();
73}

99network::Result<store_result> storage_scu::store_impl(
100 network::association& assoc,
101 const core::dicom_dataset& dataset,
102 uint16_t message_id) {
103
104 using namespace network::dimse;
105
106 // Extract SOP Class UID from dataset
107 const auto sop_class_uid = dataset.get_string(tag_sop_class_uid);
108 if (sop_class_uid.empty()) {
109 return kcenon::pacs::pacs_error<store_result>(
110 kcenon::pacs::error_codes::store_missing_sop_class_uid,
111 "Missing SOP Class UID in dataset");
112 }
113
114 // Extract SOP Instance UID from dataset
115 const auto sop_instance_uid = dataset.get_string(tag_sop_instance_uid);
116 if (sop_instance_uid.empty()) {
117 return kcenon::pacs::pacs_error<store_result>(
118 kcenon::pacs::error_codes::store_missing_sop_instance_uid,
119 "Missing SOP Instance UID in dataset");
120 }
121
122 // Verify association is established
123 if (!assoc.is_established()) {
124 return kcenon::pacs::pacs_error<store_result>(
125 kcenon::pacs::error_codes::association_not_established,
126 "Association not established");
127 }
128
129 // Get accepted presentation context for this SOP class
130 auto context_id = assoc.accepted_context_id(sop_class_uid);
131 if (!context_id) {
132 return kcenon::pacs::pacs_error<store_result>(
133 kcenon::pacs::error_codes::store_no_accepted_context,
134 "No accepted presentation context for SOP Class: " + sop_class_uid);
135 }
136
137 // Build C-STORE-RQ message
138 auto request = make_c_store_rq(
139 message_id,
140 sop_class_uid,
141 sop_instance_uid,
142 config_.default_priority
143 );
144
145 // Attach the dataset
146 request.set_dataset(dataset);
147
148 // Send the request
149 auto send_result = assoc.send_dimse(*context_id, request);
150 if (send_result.is_err()) {
151 failures_.fetch_add(1, std::memory_order_relaxed);
152 return send_result.error();
153 }
154
155 // Receive the response
156 auto recv_result = assoc.receive_dimse(config_.response_timeout);
157 if (recv_result.is_err()) {
158 failures_.fetch_add(1, std::memory_order_relaxed);
159 return recv_result.error();
160 }
161
162 const auto& [recv_context_id, response] = recv_result.value();
163
164 // Verify it's a C-STORE response
165 if (response.command() != command_field::c_store_rsp) {
166 failures_.fetch_add(1, std::memory_order_relaxed);
167 return kcenon::pacs::pacs_error<store_result>(
168 kcenon::pacs::error_codes::store_unexpected_command,
169 "Expected C-STORE-RSP but received " +
170 std::string(to_string(response.command())));
171 }
172
173 // Build result from response
174 store_result result;
175 result.sop_instance_uid = sop_instance_uid;
176 result.status = static_cast<uint16_t>(response.status());
177
178 // Extract error comment if present
179 if (response.command_set().contains(tag_error_comment)) {
180 result.error_comment = response.command_set().get_string(tag_error_comment);
181 }
182
183 // Update statistics
184 if (result.is_success() || result.is_warning()) {
185 images_sent_.fetch_add(1, std::memory_order_relaxed);
186 // Estimate dataset size
187 bytes_sent_.fetch_add(
188 dataset.size() * sizeof(uint32_t),
189 std::memory_order_relaxed
190 );
191 } else {
192 failures_.fetch_add(1, std::memory_order_relaxed);
193 }
194
195 return result;
196}

202std::vector<store_result> storage_scu::store_batch(
203 network::association& assoc,
204 const std::vector<core::dicom_dataset>& datasets,
205 store_progress_callback progress_callback) {
206
207 std::vector<store_result> results;
208 results.reserve(datasets.size());
209
210 const size_t total = datasets.size();
211 size_t completed = 0;
212
213 for (const auto& dataset : datasets) {
214 auto result = store(assoc, dataset);
215
216 if (result.is_ok()) {
217 results.push_back(std::move(result.value()));
218 } else {
219 // Create a failure result
220 store_result failure_result;
221 failure_result.sop_instance_uid = dataset.get_string(tag_sop_instance_uid);
222 failure_result.status = static_cast<uint16_t>(storage_status::cannot_understand);
223 failure_result.error_comment = result.error().message;
224 results.push_back(std::move(failure_result));
225
226 // Stop on error if configured
227 if (!config_.continue_on_error) {
228 break;
229 }
230 }
231
232 ++completed;
233
234 // Report progress
235 if (progress_callback) {
236 progress_callback(completed, total);
237 }
238 }
239
240 return results;
241}

247network::Result<store_result> storage_scu::store_file(
248 network::association& assoc,
249 const std::filesystem::path& file_path) {
250
251 // Check if file exists
252 if (!std::filesystem::exists(file_path)) {
253 return kcenon::pacs::pacs_error<store_result>(
254 kcenon::pacs::error_codes::file_not_found_service,
255 "File not found: " + file_path.string());
256 }
257
258 // Check if it's a regular file
259 if (!std::filesystem::is_regular_file(file_path)) {
260 return kcenon::pacs::pacs_error<store_result>(
261 kcenon::pacs::error_codes::not_a_regular_file,
262 "Not a regular file: " + file_path.string());
263 }
264
265 // Parse the DICOM file
266 auto file_result = core::dicom_file::open(file_path);
267 if (file_result.is_err()) {
268 const auto error_msg = "Failed to parse DICOM file: " +
269 file_path.string() + ": " +
270 file_result.error().message;
271 logger_->error(error_msg);
272 return kcenon::pacs::pacs_error<store_result>(
273 kcenon::pacs::error_codes::file_parse_failed, error_msg);
274 }
275
276 // Get the dataset from the parsed file
277 const auto& dataset = file_result.value().dataset();
278
279 // Store the dataset using the existing store method
280 return store(assoc, dataset);
281}

337std::vector<std::filesystem::path> storage_scu::collect_dicom_files(
338 const std::filesystem::path& directory,
339 bool recursive) const {
340
341 std::vector<std::filesystem::path> files;
342
343 if (!std::filesystem::exists(directory) ||
344 !std::filesystem::is_directory(directory)) {
345 return files;
346 }
347
348 if (recursive) {
349 for (const auto& entry :
350 std::filesystem::recursive_directory_iterator(directory)) {
351 if (entry.is_regular_file() && is_dicom_file(entry.path())) {
352 files.push_back(entry.path());
353 }
354 }
355 } else {
356 for (const auto& entry :
357 std::filesystem::directory_iterator(directory)) {
358 if (entry.is_regular_file() && is_dicom_file(entry.path())) {
359 files.push_back(entry.path());
360 }
361 }
362 }
363
364 // Sort files for deterministic order
365 std::sort(files.begin(), files.end());
366
367 return files;
368}

396uint16_t storage_scu::next_message_id() noexcept {
397 uint16_t id = message_id_counter_.fetch_add(1, std::memory_order_relaxed);
398 // Wrap around at 0xFFFF, skip 0 (reserved)
399 if (id == 0) {
400 id = message_id_counter_.fetch_add(1, std::memory_order_relaxed);
401 }
402 return id;
