94template <typename T>
95class Result {
96public:
97 Result(T value) : data_(std::move(value)), has_value_(true) {}
98 Result(const error_info& err) : error_(err), has_value_(false) {}
99
100 [[nodiscard]] bool is_ok() const noexcept { return has_value_; }
101 [[nodiscard]] bool is_err() const noexcept { return !has_value_; }
102 [[nodiscard]] T& value() & { return data_; }
103 [[nodiscard]] const T& value() const& { return data_; }
104 [[nodiscard]] T&& value() && { return std::move(data_); }
105 [[nodiscard]] const error_info& error() const { return error_; }
106
107private:
108 T data_{};
109 error_info error_;
110 bool has_value_;
111};
