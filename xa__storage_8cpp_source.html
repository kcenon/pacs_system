141std::vector<std::string> get_xa_storage_sop_classes(bool include_3d) {
142 std::vector<std::string> result;
143 result.reserve(include_3d ? 5 : 3);
144
145 for (const auto& info : xa_sop_classes) {
146 if (!info.is_3d || include_3d) {
147 result.emplace_back(info.uid);
148 }
149 }
150
151 return result;
152}
153
154const xa_sop_class_info*
155get_xa_sop_class_info(std::string_view uid) noexcept {
156 auto it = std::find_if(
157 xa_sop_classes.begin(),
158 xa_sop_classes.end(),
159 [uid](const auto& info) { return info.uid == uid; }
160 );
161
162 if (it != xa_sop_classes.end()) {
163 return &(*it);
164 }
165 return nullptr;
166}

201bool xa_positioner_angles::is_valid() const noexcept {
202 // Typical clinical range: primary angle -90 to +90 (LAO/RAO)
203 // Secondary angle: -45 to +45 (Cranial/Caudal)
204 return primary_angle >= -180.0 && primary_angle <= 180.0 &&
205 secondary_angle >= -90.0 && secondary_angle <= 90.0;
206}

212double xa_calibration_data::magnification_factor() const noexcept {
213 if (distance_source_to_patient <= 0.0 ||
214 distance_source_to_detector <= 0.0) {
215 return 0.0;
216 }
217 return distance_source_to_detector / distance_source_to_patient;
218}
219
220double xa_calibration_data::isocenter_pixel_spacing() const noexcept {
221 double mag = magnification_factor();
222 if (mag <= 0.0 || imager_pixel_spacing[0] <= 0.0) {
223 return 0.0;
224 }
225 // Pixel spacing at isocenter = detector pixel spacing / magnification
226 return imager_pixel_spacing[0] / mag;
227}
228
229bool xa_calibration_data::is_valid() const noexcept {
230 return imager_pixel_spacing[0] > 0.0 &&
231 imager_pixel_spacing[1] > 0.0 &&
232 distance_source_to_detector > 0.0 &&
233 distance_source_to_patient > 0.0 &&
234 distance_source_to_detector > distance_source_to_patient;
235}
