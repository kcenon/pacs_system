213struct health_status {

220
222 database_status database;

231 storage_metrics metrics;
232
234 version_info version;
235
237 std::optional<std::string> message;
238
247 void update_level() noexcept {
248 if (!database.connected) {
249 level = health_level::unhealthy;
250 return;
251 }
252
253 if (!storage.writable || !storage.readable) {
254 level = health_level::unhealthy;
255 return;
256 }
257
258 // Check for degraded conditions
259 if (storage.usage_percent() > 90.0) {
260 level = health_level::degraded;
261 return;
262 }
263
264 if (associations.active_associations >=
265 associations.max_associations * 0.9) {
266 level = health_level::degraded;
267 return;
268 }
269
270 level = health_level::healthy;
271 }
272
277 [[nodiscard]] bool is_healthy() const noexcept {
278 return level == health_level::healthy;
279 }
280
285 [[nodiscard]] bool is_operational() const noexcept {
286 return level != health_level::unhealthy;
287 }
288};
