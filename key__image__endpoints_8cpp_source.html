81std::string generate_uuid() {
82 static std::random_device rd;
83 static std::mt19937 gen(rd());
84 static std::uniform_int_distribution<> dis(0, 15);
85 static const char *hex = "0123456789abcdef";
86
87 std::string uuid = "xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx";
88 for (char &c : uuid) {
89 if (c == 'x') {
90 c = hex[dis(gen)];
91 } else if (c == 'y') {
92 c = hex[(dis(gen) & 0x3) | 0x8];
93 }
94 }
95 return uuid;
96}

101std::string format_timestamp(
102 const std::chrono::system_clock::time_point &tp) {
103 auto time = std::chrono::system_clock::to_time_t(tp);
104 std::ostringstream oss;
105 oss << std::put_time(std::gmtime(&time), "%Y-%m-%dT%H:%M:%SZ");
106 return oss.str();
107}

112std::string key_image_to_json(const storage::key_image_record &ki) {
113 std::ostringstream oss;
114 oss << R"({"key_image_id":")" << json_escape(ki.key_image_id)
115 << R"(","study_uid":")" << json_escape(ki.study_uid)
116 << R"(","sop_instance_uid":")" << json_escape(ki.sop_instance_uid)
117 << R"(","frame_number":)";
118 if (ki.frame_number.has_value()) {
119 oss << ki.frame_number.value();
120 } else {
121 oss << "null";
122 }
123 oss << R"(,"user_id":")" << json_escape(ki.user_id)
124 << R"(","reason":")" << json_escape(ki.reason)
125 << R"(","document_title":")" << json_escape(ki.document_title)
126 << R"(","created_at":")" << format_timestamp(ki.created_at) << R"("})";
127 return oss.str();
128}

150std::string parse_json_string(const std::string &json, const std::string &key) {
151 std::string search = "\"" + key + "\":\"";
152 auto pos = json.find(search);
153 if (pos == std::string::npos) {
154 return "";
155 }
156 pos += search.length();
157 auto end = json.find("\"", pos);
158 if (end == std::string::npos) {
159 return "";
160 }
161 return json.substr(pos, end - pos);
162}

167std::optional<int> parse_json_int(const std::string &json,
168 const std::string &key) {
169 std::string search = "\"" + key + "\":";
170 auto pos = json.find(search);
171 if (pos == std::string::npos) {
172 return std::nullopt;
173 }
174 pos += search.length();
175 while (pos < json.size() && (json[pos] == ' ' || json[pos] == '\t')) {
176 ++pos;
177 }
178 if (pos >= json.size()) {
179 return std::nullopt;
180 }
181 if (json.substr(pos, 4) == "null") {
182 return std::nullopt;
183 }
184 try {
185 return std::stoi(json.substr(pos));
186 } catch (...) {
187 return std::nullopt;
188 }
189}

194void register_key_image_endpoints_impl(crow::SimpleApp &app,
195 std::shared_ptr<rest_server_context> ctx) {
196 // POST /api/v1/studies/<studyUid>/key-images - Create key image
197 CROW_ROUTE(app, "/api/v1/studies/<string>/key-images")
198 .methods(crow::HTTPMethod::POST)(
199 [ctx](const crow::request &req, const std::string &study_uid) {
200 crow::response res;
201 res.add_header("Content-Type", "application/json");
202 add_cors_headers(res, *ctx);
203
204 if (!ctx->database) {
205 res.code = 503;
206 res.body =
207 make_error_json("DATABASE_UNAVAILABLE", "Database not configured");
208 return res;
209 }
210
211 std::string body = req.body;
212 if (body.empty()) {
213 res.code = 400;
214 res.body = make_error_json("INVALID_REQUEST", "Request body is empty");
215 return res;
216 }
217
218 storage::key_image_record ki;
219 ki.key_image_id = generate_uuid();
220 ki.study_uid = study_uid;
221 ki.sop_instance_uid = parse_json_string(body, "sop_instance_uid");
222 ki.frame_number = parse_json_int(body, "frame_number");
223 ki.user_id = parse_json_string(body, "user_id");
224 ki.reason = parse_json_string(body, "reason");
225 ki.document_title = parse_json_string(body, "document_title");
226 ki.created_at = std::chrono::system_clock::now();
227
228 if (ki.sop_instance_uid.empty()) {
229 res.code = 400;
230 res.body = make_error_json("MISSING_FIELD", "sop_instance_uid is required");
231 return res;
232 }
233
234#ifdef PACS_WITH_DATABASE_SYSTEM
235 storage::key_image_repository repo(ctx->database->db_adapter());
236#else
237 storage::key_image_repository repo(ctx->database->native_handle());
238#endif
239 auto save_result = repo.save(ki);
240 if (!save_result.is_ok()) {
241 res.code = 500;
242 res.body =
243 make_error_json("SAVE_ERROR", save_result.error().message);
244 return res;
245 }
246
247 res.code = 201;
248 std::ostringstream oss;
249 oss << R"({"key_image_id":")" << json_escape(ki.key_image_id)
250 << R"(","created_at":")" << format_timestamp(ki.created_at)
251 << R"("})";
252 res.body = oss.str();
253 return res;
254 });

304
305#ifdef PACS_WITH_DATABASE_SYSTEM
306 storage::key_image_repository repo(ctx->database->db_adapter());
307 auto exists_result = repo.exists(key_image_id);
308 if (!exists_result.is_ok()) {
309 res.code = 500;
310 res.add_header("Content-Type", "application/json");
311 res.body = make_error_json("QUERY_ERROR", exists_result.error().message);
312 return res;
313 }
314 if (!exists_result.value()) {
315 res.code = 404;
316 res.add_header("Content-Type", "application/json");
317 res.body = make_error_json("NOT_FOUND", "Key image not found");
318 return res;
319 }
320#else
321 storage::key_image_repository repo(ctx->database->native_handle());
322 if (!repo.exists(key_image_id)) {
323 res.code = 404;
324 res.add_header("Content-Type", "application/json");
325 res.body = make_error_json("NOT_FOUND", "Key image not found");
326 return res;
327 }
328#endif
329
330 auto remove_result = repo.remove(key_image_id);
331 if (!remove_result.is_ok()) {
332 res.code = 500;
333 res.add_header("Content-Type", "application/json");
334 res.body =
335 make_error_json("DELETE_ERROR", remove_result.error().message);
336 return res;
337 }
338
339 res.code = 204;
340 return res;
341 });

344 CROW_ROUTE(app, "/api/v1/studies/<string>/key-images/export-sr")

351 res.code = 503;

357

368#else

370 auto key_images = repo.find_by_study(study_uid);

372 if (key_images.empty()) {
373 res.code = 404;
374 res.add_header("Content-Type", "application/json");
375 res.body = make_error_json("NO_KEY_IMAGES",
376 "No key images found for study");
377 return res;
378 }

380 // Build Key Object Selection Document as JSON
381 // Note: Full DICOM SR encoding requires dcmtk/dicom_dataset integration
382 // This endpoint returns a structured JSON that can be converted to DICOM SR
383 std::ostringstream oss;
384 oss << R"({"document_type":"Key Object Selection",)";
385 oss << R"("study_uid":")" << json_escape(study_uid) << R"(",)";
386 oss << R"("document_title":"Key Images",)";
387 oss << R"("referenced_instances":[)";
388
389 for (size_t i = 0; i < key_images.size(); ++i) {
390 if (i > 0) {
391 oss << ",";
392 }
393 const auto &ki = key_images[i];
394 oss << R"({"sop_instance_uid":")" << json_escape(ki.sop_instance_uid) << R"(",)";
395 oss << R"("frame_number":)";
396 if (ki.frame_number.has_value()) {
397 oss << ki.frame_number.value();
398 } else {
399 oss << "null";
400 }
401 oss << R"(,"reason":")" << json_escape(ki.reason) << R"("})";
402 }
403
404 oss << R"(],"created_at":")" << format_timestamp(std::chrono::system_clock::now()) << R"("})";
405
406 res.code = 200;
407 res.add_header("Content-Type", "application/json");
408 res.body = oss.str();
409 return res;
410 });
