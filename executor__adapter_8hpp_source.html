90class lambda_job : public kcenon::common::interfaces::IJob {
91public:
92 using job_function = std::function<kcenon::common::VoidResult()>;
93
101 explicit lambda_job(
102 job_function func,
103 std::string name = "lambda_job",
104 int priority = 0)
105 : func_(std::move(func))
106 , name_(std::move(name))
107 , priority_(priority) {}
108
116 template <typename F>
117 requires std::is_void_v<std::invoke_result_t<F>>
118 explicit lambda_job(F&& func, std::string name = "lambda_job", int priority = 0)
119 : func_([f = std::forward<F>(func)]() -> kcenon::common::VoidResult {
120 f();
121 return kcenon::common::VoidResult(std::monostate{});
122 })
123 , name_(std::move(name))
124 , priority_(priority) {}
125
126 kcenon::common::VoidResult execute() override {
127 if (!func_) {
128 return kcenon::common::VoidResult(
129 kcenon::common::error_info{-1, "No function provided", "executor"});
130 }
131 return func_();
132 }
133
134 [[nodiscard]] std::string get_name() const override { return name_; }
135 [[nodiscard]] int get_priority() const override { return priority_; }
136
137private:
138 job_function func_;
139 std::string name_;
140 int priority_;
141};
