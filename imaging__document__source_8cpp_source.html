88std::string current_datetime() {
89 auto now = std::chrono::system_clock::now();
90 auto time_t_now = std::chrono::system_clock::to_time_t(now);
91 std::tm tm_now{};
92#if defined(_WIN32)
93 gmtime_s(&tm_now, &time_t_now);
94#else
95 gmtime_r(&time_t_now, &tm_now);
96#endif
97 char buf[16];
98 std::strftime(buf, sizeof(buf), "%Y%m%d%H%M%S", &tm_now);
99 return buf;
100}

103 return current_datetime().substr(0, 8);

120kos_creation_result imaging_document_source::create_kos_document(
121 const std::string& study_instance_uid,
122 const std::vector<kos_instance_reference>& references,
123 const std::optional<core::dicom_dataset>& patient_demographics) const {
124
125 kos_creation_result result;
126
127 if (references.empty()) {
128 result.error_message = "No instance references provided";
129 return result;
130 }
131
132 core::dicom_dataset kos;
133
134 // --- SOP Common Module ---
135 kos.set_string(tags::sop_class_uid, vr_type::UI,
136 std::string(sop_classes::key_object_selection_document_storage_uid));
137 result.kos_instance_uid = generate_uid();
138 kos.set_string(tags::sop_instance_uid, vr_type::UI, result.kos_instance_uid);
139
140 // --- Patient Module (Type 2) ---
141 if (patient_demographics) {
142 auto copy_if_present = [&](dicom_tag tag, vr_type vr) {
143 if (patient_demographics->contains(tag)) {
144 kos.set_string(tag, vr, patient_demographics->get_string(tag));
145 } else {
146 kos.set_string(tag, vr, "");
147 }
148 };
149 copy_if_present(tags::patient_name, vr_type::PN);
150 copy_if_present(tags::patient_id, vr_type::LO);
151 copy_if_present(tags::patient_birth_date, vr_type::DA);
152 copy_if_present(tags::patient_sex, vr_type::CS);
153 } else {
154 kos.set_string(tags::patient_name, vr_type::PN, "");
155 kos.set_string(tags::patient_id, vr_type::LO, "");
156 kos.set_string(tags::patient_birth_date, vr_type::DA, "");
157 kos.set_string(tags::patient_sex, vr_type::CS, "");
158 }
159
160 // --- General Study Module ---
161 kos.set_string(tags::study_instance_uid, vr_type::UI, study_instance_uid);
162 kos.set_string(tags::study_date, vr_type::DA, current_date());
163 kos.set_string(tags::study_time, vr_type::TM, current_time());
164 kos.set_string(tags::referring_physician_name, vr_type::PN, "");
165 kos.set_string(tags::study_id, vr_type::SH, "");
166 kos.set_string(tags::accession_number, vr_type::SH, "");
167
168 // --- General Series Module ---
169 kos.set_string(tags::modality, vr_type::CS, "KO");
170 kos.set_string(tags::series_instance_uid, vr_type::UI, generate_uid());
171 kos.set_string(tags::series_number, vr_type::IS, "1");
172
173 // --- General Equipment Module ---
174 kos.set_string(dicom_tag{0x0008, 0x0070}, vr_type::LO, ""); // Manufacturer
175
176 // --- SR Document General Module ---
177 kos.set_string(tags::instance_number, vr_type::IS, "1");
178 kos.set_string(tags::content_date, vr_type::DA, current_date());
179 kos.set_string(tags::content_time, vr_type::TM, current_time());
180 kos.set_string(kos_tags::completion_flag, vr_type::CS, "COMPLETE");
181 kos.set_string(kos_tags::verification_flag, vr_type::CS, "UNVERIFIED");
182
183 // --- SR Document Content Module ---
184 // Value type: CONTAINER (root)
185 kos.set_string(kos_tags::value_type, vr_type::CS, "CONTAINER");
186
187 // Concept Name Code Sequence: Key Object Selection (113000, DCM)
188 dicom_dataset concept_name;
189 concept_name.set_string(kos_tags::code_value, vr_type::SH, "113000");
190 concept_name.set_string(kos_tags::coding_scheme_designator, vr_type::SH, "DCM");
191 concept_name.set_string(kos_tags::code_meaning, vr_type::LO, "Of Interest");
192 insert_sequence(kos, kos_tags::concept_name_code_sequence, {concept_name});
193
194 // Content Template Sequence: TID 2010
195 dicom_dataset template_item;
196 template_item.set_string(kos_tags::template_identifier, vr_type::CS, "2010");
197 template_item.set_string(kos_tags::mapping_resource, vr_type::CS, "DCMR");
198 insert_sequence(kos, kos_tags::content_template_sequence, {template_item});
199
200 // --- Current Requested Procedure Evidence Sequence ---
201 build_evidence_sequence(kos, references);
202
203 result.success = true;
204 result.reference_count = references.size();
205 result.kos_dataset = std::move(kos);
206
207 return result;
208}

210xds_document_entry imaging_document_source::build_document_entry(
211 const core::dicom_dataset& kos_dataset) const {
212
213 xds_document_entry entry;
214
215 entry.entry_uuid = "urn:uuid:" + generate_uid();
216 entry.unique_id = kos_dataset.get_string(tags::sop_instance_uid);
217 entry.patient_id = kos_dataset.get_string(tags::patient_id);
218 entry.source_patient_id = entry.patient_id;
219
220 entry.class_code = "IMG";
221 entry.class_code_scheme = "1.3.6.1.4.1.19376.3.840.1.1.4";
222 entry.class_code_display = "Imaging Procedure";
223
224 entry.type_code = "KOS";
225 entry.type_code_scheme = "1.2.840.10008.2.16.4";
226 entry.type_code_display = "Key Object Selection";
227
228 entry.format_code = std::string(sop_classes::key_object_selection_document_storage_uid);
229 entry.format_code_scheme = "1.2.840.10008.2.6.1";
230
231 entry.creation_time = current_datetime();
232 entry.service_start_time = entry.creation_time;
233 entry.service_stop_time = entry.creation_time;
234
235 entry.facility_type_code = config_.facility_type_code;
236 entry.facility_type_code_scheme = config_.facility_type_code_scheme;
237 entry.practice_setting_code = config_.practice_setting_code;
238 entry.practice_setting_code_scheme = config_.practice_setting_code_scheme;
239
240 // Extract author from patient demographics if available
241 auto ref_physician = kos_dataset.get_string(tags::referring_physician_name);
242 if (!ref_physician.empty()) {
243 entry.author_person = ref_physician;
244 }
245
246 entry.title = "Key Object Selection Document";
247
248 return entry;
249}

298std::string imaging_document_source::generate_uid() const {
299 static constexpr const char* uid_root = "1.2.826.0.1.3680043.2.1545.1";
300 static std::mt19937_64 gen{std::random_device{}()};
301 static std::uniform_int_distribution<uint64_t> dist;
302
303 auto now = std::chrono::system_clock::now();
304 auto timestamp = std::chrono::duration_cast<std::chrono::milliseconds>(
305 now.time_since_epoch()).count();
306
307 return std::string(uid_root) + "." + std::to_string(timestamp) +
308 "." + std::to_string(dist(gen) % 100000);
309}

311void imaging_document_source::build_evidence_sequence(
312 core::dicom_dataset& kos_dataset,
313 const std::vector<kos_instance_reference>& references) const {
314
315 // Group references by study -> series -> instances
316 struct series_group {
317 std::string series_uid;
318 std::vector<std::pair<std::string, std::string>> instances; // sop_class, sop_instance
319 };
320
321 std::unordered_map<std::string, std::vector<series_group>> study_map;
322
323 for (const auto& ref : references) {
324 auto& series_list = study_map[ref.study_instance_uid];
325
326 auto it = std::find_if(series_list.begin(), series_list.end(),
327 [&](const series_group& sg) {
328 return sg.series_uid == ref.series_instance_uid;
329 });
330
331 if (it != series_list.end()) {
332 it->instances.emplace_back(ref.sop_class_uid, ref.sop_instance_uid);
333 } else {
334 series_group sg;
335 sg.series_uid = ref.series_instance_uid;
336 sg.instances.emplace_back(ref.sop_class_uid, ref.sop_instance_uid);
337 series_list.push_back(std::move(sg));
338 }
339 }
340
341 // Build the sequence structure
342 std::vector<dicom_dataset> study_items;
343 for (const auto& [study_uid, series_list] : study_map) {
344 dicom_dataset study_item;
345 study_item.set_string(tags::study_instance_uid, vr_type::UI, study_uid);
346
347 std::vector<dicom_dataset> series_items;
348 for (const auto& sg : series_list) {
349 dicom_dataset series_item;
350 series_item.set_string(tags::series_instance_uid, vr_type::UI, sg.series_uid);
351
352 std::vector<dicom_dataset> sop_items;
353 for (const auto& [sop_class, sop_instance] : sg.instances) {
354 dicom_dataset sop_item;
355 sop_item.set_string(kos_tags::referenced_sop_class_uid, vr_type::UI, sop_class);
356 sop_item.set_string(kos_tags::referenced_sop_instance_uid, vr_type::UI, sop_instance);
357 sop_items.push_back(std::move(sop_item));
358 }
359 insert_sequence(series_item, kos_tags::referenced_sop_sequence, std::move(sop_items));
360 series_items.push_back(std::move(series_item));
361 }
362 insert_sequence(study_item, kos_tags::referenced_series_sequence, std::move(series_items));
363 study_items.push_back(std::move(study_item));
364 }
365
366 insert_sequence(kos_dataset,
367 kos_tags::current_requested_procedure_evidence_sequence,
368 std::move(study_items));
369}
