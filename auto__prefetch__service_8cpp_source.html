104void auto_prefetch_service::start() {
105 if (enabled_.exchange(true)) {
106 return; // Already enabled
107 }
108
109 stop_requested_.store(false);
110 next_cycle_time_ = std::chrono::steady_clock::now();
111
112 worker_thread_ = std::thread([this]() {
113 run_loop();
114 });
115
116 integration::logger_adapter::info(
117 "Auto prefetch service started interval_seconds={} max_concurrent={}",
118 config_.prefetch_interval.count(),
119 config_.max_concurrent_prefetches);
120}

122void auto_prefetch_service::disable(bool wait_for_completion) {
123 stop(wait_for_completion);
124}
125
126void auto_prefetch_service::stop(bool wait_for_completion) {
127 if (!enabled_.exchange(false)) {
128 return; // Already disabled
129 }
130
131 stop_requested_.store(true);
132
133 // Wake up the worker thread
134 cv_.notify_all();
135
136 if (wait_for_completion && worker_thread_.joinable()) {
137 worker_thread_.join();
138 } else if (worker_thread_.joinable()) {
139 worker_thread_.detach();
140 }
141
142 integration::logger_adapter::info("Auto prefetch service stopped");
143}

195void auto_prefetch_service::on_worklist_query(
196 const std::vector<storage::worklist_item>& worklist_items) {
197
198 for (const auto& item : worklist_items) {
199 if (item.patient_id.empty()) {
200 continue;
201 }
202
203 prefetch_request request;
204 request.patient_id = item.patient_id;
205 request.patient_name = item.patient_name;
206 request.scheduled_modality = item.modality;
207 request.scheduled_study_uid = item.study_uid;
208 request.request_time = std::chrono::system_clock::now();
209
210 queue_request(request);
211 }
212
213 // Wake up worker if there are new requests
214 cv_.notify_one();
215
216 integration::logger_adapter::debug(
217 "Queued prefetch requests from worklist worklist_items={} queue_size={}",
218 worklist_items.size(),
219 pending_requests());
220}

306void auto_prefetch_service::run_loop() {
307 integration::logger_adapter::debug("Prefetch service worker thread started");
308
309 while (!stop_requested_.load()) {
310 std::unique_lock<std::mutex> lock(mutex_);
311
312 // Wait until next cycle time or until woken up
313 auto wait_until = next_cycle_time_;
314 cv_.wait_until(lock, wait_until, [this]() {
315 return stop_requested_.load() ||
316 std::chrono::steady_clock::now() >= next_cycle_time_ ||
317 !request_queue_.empty();
318 });
319
320 if (stop_requested_.load()) {
321 break;
322 }
323
324 // Check if we should run a cycle
325 auto now = std::chrono::steady_clock::now();
326 bool should_run_cycle = (now >= next_cycle_time_) ||
327 (!request_queue_.empty());
328
329 if (should_run_cycle) {
330 lock.unlock();
331
332 cycle_in_progress_.store(true);
333 auto result = execute_cycle();
334 cycle_in_progress_.store(false);
335
336 lock.lock();
337
338 // Update last result
339 last_result_ = result;
340 update_stats(result);
341 ++cycles_count_;
342
343 // Schedule next cycle
344 next_cycle_time_ = std::chrono::steady_clock::now() +
345 config_.prefetch_interval;
346
347 // Invoke callback
348 if (config_.on_cycle_complete) {
349 lock.unlock();
350 config_.on_cycle_complete(result);
351 lock.lock();
352 }
353 }
354 }
355
356 integration::logger_adapter::debug("Prefetch service worker thread stopped");
357}

359auto auto_prefetch_service::execute_cycle() -> prefetch_result {
360 prefetch_result cycle_result;
361 cycle_result.timestamp = std::chrono::system_clock::now();
362 auto cycle_start = std::chrono::steady_clock::now();
363
364 // Process all pending requests
365 while (auto request = dequeue_request()) {
366 if (stop_requested_.load()) {
367 break;
368 }
369
370 auto result = process_request(*request);
371 cycle_result += result;
372 }
373
374 cycle_result.duration = std::chrono::duration_cast<std::chrono::milliseconds>(
375 std::chrono::steady_clock::now() - cycle_start);
376
377 integration::logger_adapter::info(
378 "Prefetch cycle completed patients={} studies_prefetched={} studies_failed={} duration_ms={}",
379 cycle_result.patients_processed,
380 cycle_result.studies_prefetched,
381 cycle_result.studies_failed,
382 cycle_result.duration.count());
383
384 // Record metrics
385 integration::monitoring_adapter::record_histogram(
386 "prefetch_cycle_duration_ms",
387 static_cast<double>(cycle_result.duration.count()));
388 integration::monitoring_adapter::increment_counter(
389 "prefetch_studies_total",
390 static_cast<int64_t>(cycle_result.studies_prefetched));
391 integration::monitoring_adapter::increment_counter(
392 "prefetch_failures_total",
393 static_cast<int64_t>(cycle_result.studies_failed));
394
395 return cycle_result;
396}

398auto auto_prefetch_service::process_request(const prefetch_request& request)
399 -> prefetch_result {
400
401 prefetch_result result;
402 result.patients_processed = 1;
403 result.timestamp = std::chrono::system_clock::now();
404 auto start_time = std::chrono::steady_clock::now();
405
406 integration::logger_adapter::debug(
407 "Processing prefetch request patient_id={} scheduled_modality={}",
408 request.patient_id,
409 request.scheduled_modality);
410
411 // Query each remote PACS for prior studies
412 for (const auto& pacs : config_.remote_pacs) {
413 if (!pacs.is_valid()) {
414 continue;
415 }
416
417 // Query for prior studies
418 auto prior_studies = query_prior_studies(
419 pacs,
420 request.patient_id,
421 config_.criteria.lookback_period);
422
423 // Filter based on criteria
424 auto filtered_studies = filter_studies(prior_studies, request);
425
426 // Prefetch each study
427 for (const auto& study : filtered_studies) {
428 // Skip if already present locally
429 if (study_exists_locally(study.study_instance_uid)) {
430 ++result.studies_already_present;
431 continue;
432 }
433
434 // Skip the scheduled study itself
435 if (study.study_instance_uid == request.scheduled_study_uid) {
436 continue;
437 }
438
439 // Attempt prefetch
440 bool success = prefetch_study(pacs, study);
441
442 if (success) {
443 ++result.studies_prefetched;
444 result.series_prefetched += study.number_of_series;
445 result.instances_prefetched += study.number_of_instances;
446
447 if (config_.on_prefetch_complete) {
448 config_.on_prefetch_complete(
449 request.patient_id, study, true, "");
450 }
451 } else {
452 ++result.studies_failed;
453
454 if (config_.on_prefetch_error) {
455 config_.on_prefetch_error(
456 request.patient_id,
457 study.study_instance_uid,
458 "Failed to prefetch study");
459 }
460 }
461
462 // Check rate limiting
463 if (config_.rate_limit_per_minute > 0) {
464 auto elapsed = std::chrono::steady_clock::now() - start_time;
465 auto elapsed_minutes =
466 std::chrono::duration_cast<std::chrono::minutes>(elapsed)
467 .count();
468 if (elapsed_minutes < 1) {
469 auto prefetched_this_minute =
470 result.studies_prefetched + result.studies_failed;
471 if (prefetched_this_minute >= config_.rate_limit_per_minute) {
472 // Wait until the next minute
473 std::this_thread::sleep_for(
474 std::chrono::seconds(60) - elapsed);
475 }
476 }
477 }
478 }
479 }
480
481 result.duration = std::chrono::duration_cast<std::chrono::milliseconds>(
482 std::chrono::steady_clock::now() - start_time);
483
484 return result;
485}

487auto auto_prefetch_service::query_prior_studies(
488 const remote_pacs_config& pacs_config,
489 const std::string& patient_id,
490 std::chrono::days lookback) -> std::vector<prior_study_info> {
491
492 std::vector<prior_study_info> results;
493
494 // Calculate date range
495 auto now = std::chrono::system_clock::now();
496 auto from_time = now - lookback;
497
498 // Format dates as YYYYMMDD
499 auto format_date = [](std::chrono::system_clock::time_point tp) {
500 auto time_t = std::chrono::system_clock::to_time_t(tp);
501 std::tm tm = *std::localtime(&time_t);
502 char buffer[9];
503 std::strftime(buffer, sizeof(buffer), "%Y%m%d", &tm);
504 return std::string(buffer);
505 };
506
507 std::string from_date = format_date(from_time);
508 std::string to_date = format_date(now);
509
510 integration::logger_adapter::debug(
511 "Querying prior studies remote_pacs={} patient_id={} from_date={} to_date={}",
512 pacs_config.ae_title,
513 patient_id,
514 from_date,
515 to_date);
516
517 try {
518 // Configure association for C-FIND
519 network::association_config assoc_config;
520 assoc_config.calling_ae_title = pacs_config.local_ae_title;
521 assoc_config.called_ae_title = pacs_config.ae_title;
522 assoc_config.proposed_contexts.push_back({
523 1,
524 std::string(services::study_root_find_sop_class_uid),
525 {"1.2.840.10008.1.2.1", "1.2.840.10008.1.2"}
526 });
527
528 // Connect to remote PACS
529 auto connect_result = network::association::connect(
530 pacs_config.host,
531 pacs_config.port,
532 assoc_config,
533 std::chrono::duration_cast<network::association::duration>(
534 pacs_config.connection_timeout));
535
536 if (connect_result.is_err()) {
537 integration::logger_adapter::error(
538 "Failed to connect to remote PACS for C-FIND remote_pacs={} error={}",
539 pacs_config.ae_title,
540 connect_result.error().message);
541 return results;
542 }
543
544 auto assoc = std::move(connect_result.value());
545
546 // Build query keys
547 services::study_query_keys keys;
548 keys.patient_id = patient_id;
549 keys.study_date = from_date + "-" + to_date;
550
551 // Execute C-FIND query
552 services::query_scu_config query_config;
553 query_config.model = services::query_model::study_root;
554 query_config.level = services::query_level::study;
555 query_config.timeout = std::chrono::duration_cast<std::chrono::milliseconds>(
556 pacs_config.association_timeout);
557
558 services::query_scu scu(query_config);
559 auto find_result = scu.find_studies(assoc, keys);
560
561 if (find_result.is_ok() && find_result.value().is_success()) {
562 for (const auto& dataset : find_result.value().matches) {
563 prior_study_info info;
564 info.study_instance_uid =
565 dataset.get_string(core::tags::study_instance_uid);
566 info.patient_id = patient_id;
567 info.patient_name =
568 dataset.get_string(core::tags::patient_name);
569 info.study_date =
570 dataset.get_string(core::tags::study_date);
571 info.study_description =
572 dataset.get_string(core::tags::study_description);
573
574 // Parse modalities in study (backslash-separated)
575 auto modalities_str =
576 dataset.get_string(core::tags::modalities_in_study);
577 if (!modalities_str.empty()) {
578 std::istringstream ss(modalities_str);
579 std::string mod;
580 while (std::getline(ss, mod, '\\')) {
581 if (!mod.empty()) {
582 info.modalities.insert(mod);
583 }
584 }
585 }
586
587 // Parse numeric fields
588 auto num_series_str = dataset.get_string(
589 core::tags::number_of_study_related_series);
590 if (!num_series_str.empty()) {
591 try {
592 info.number_of_series = std::stoull(num_series_str);
593 } catch (...) {}
594 }
595
596 auto num_instances_str = dataset.get_string(
597 core::tags::number_of_study_related_instances);
598 if (!num_instances_str.empty()) {
599 try {
600 info.number_of_instances = std::stoull(num_instances_str);
601 } catch (...) {}
602 }
603
604 results.push_back(std::move(info));
605 }
606
607 integration::logger_adapter::debug(
608 "C-FIND query returned matches={} remote_pacs={} patient_id={}",
609 results.size(),
610 pacs_config.ae_title,
611 patient_id);
612 } else if (find_result.is_err()) {
613 integration::logger_adapter::error(
614 "C-FIND query failed remote_pacs={} error={}",
615 pacs_config.ae_title,
616 find_result.error().message);
617 }
618
619 // Release association
620 (void)assoc.release();
621 } catch (const std::exception& e) {
622 integration::logger_adapter::error(
623 "Exception querying prior studies remote_pacs={} error={}",
624 pacs_config.ae_title,
625 e.what());
626 }
627
628 return results;
629}

631auto auto_prefetch_service::filter_studies(
632 const std::vector<prior_study_info>& studies,
633 const prefetch_request& request) -> std::vector<prior_study_info> {
634
635 std::vector<prior_study_info> filtered;
636
637 for (const auto& study : studies) {
638 // Apply modality filters
639 bool modality_match = true;
640
641 // Check include list (if not empty, study must match)
642 if (!config_.criteria.include_modalities.empty()) {
643 modality_match = false;
644 for (const auto& mod : study.modalities) {
645 if (config_.criteria.include_modalities.count(mod) > 0) {
646 modality_match = true;
647 break;
648 }
649 }
650 }
651
652 // Check exclude list
653 if (modality_match && !config_.criteria.exclude_modalities.empty()) {
654 for (const auto& mod : study.modalities) {
655 if (config_.criteria.exclude_modalities.count(mod) > 0) {
656 modality_match = false;
657 break;
658 }
659 }
660 }
661
662 if (!modality_match) {
663 continue;
664 }
665
666 // Apply body part filter
667 if (!config_.criteria.include_body_parts.empty()) {
668 if (config_.criteria.include_body_parts.count(
669 study.body_part_examined) == 0) {
670 continue;
671 }
672 }
673
674 filtered.push_back(study);
675 }
676
677 // Sort by preference if enabled
678 if (config_.criteria.prefer_same_modality ||
679 config_.criteria.prefer_same_body_part) {
680 std::ranges::sort(filtered, [&](const auto& a, const auto& b) {
681 int score_a = 0;
682 int score_b = 0;
683
684 if (config_.criteria.prefer_same_modality) {
685 if (a.modalities.count(request.scheduled_modality) > 0) {
686 score_a += 10;
687 }
688 if (b.modalities.count(request.scheduled_modality) > 0) {
689 score_b += 10;
690 }
691 }
692
693 if (config_.criteria.prefer_same_body_part) {
694 if (a.body_part_examined == request.scheduled_body_part) {
695 score_a += 5;
696 }
697 if (b.body_part_examined == request.scheduled_body_part) {
698 score_b += 5;
699 }
700 }
701
702 // Higher score first, then by date (newer first)
703 if (score_a != score_b) {
704 return score_a > score_b;
705 }
706 return a.study_date > b.study_date;
707 });
708 }
709
710 // Limit results
711 if (config_.criteria.max_studies_per_patient > 0 &&
712 filtered.size() > config_.criteria.max_studies_per_patient) {
713 filtered.resize(config_.criteria.max_studies_per_patient);
714 }
715
716 return filtered;
717}

726auto auto_prefetch_service::prefetch_study(
727 const remote_pacs_config& pacs_config,
728 const prior_study_info& study) -> bool {
729
730 integration::logger_adapter::debug(
731 "Prefetching study study_uid={} patient_id={} remote_pacs={}",
732 study.study_instance_uid,
733 study.patient_id,
734 pacs_config.ae_title);
735
736 try {
737 // Configure association for C-MOVE
738 network::association_config assoc_config;
739 assoc_config.calling_ae_title = pacs_config.local_ae_title;
740 assoc_config.called_ae_title = pacs_config.ae_title;
741 assoc_config.proposed_contexts.push_back({
742 1,
743 std::string(services::study_root_move_sop_class_uid),
744 {"1.2.840.10008.1.2.1", "1.2.840.10008.1.2"}
745 });
746
747 // Connect to remote PACS
748 auto connect_result = network::association::connect(
749 pacs_config.host,
750 pacs_config.port,
751 assoc_config,
752 std::chrono::duration_cast<network::association::duration>(
753 pacs_config.connection_timeout));
754
755 if (connect_result.is_err()) {
756 integration::logger_adapter::error(
757 "Failed to connect to remote PACS for C-MOVE remote_pacs={} error={}",
758 pacs_config.ae_title,
759 connect_result.error().message);
760 return false;
761 }
762
763 auto assoc = std::move(connect_result.value());
764
765 // Configure retrieve SCU with our local AE as move destination
766 services::retrieve_scu_config retrieve_config;
767 retrieve_config.mode = services::retrieve_mode::c_move;
768 retrieve_config.model = services::query_model::study_root;
769 retrieve_config.level = services::query_level::study;
770 retrieve_config.move_destination = pacs_config.local_ae_title;
771 retrieve_config.timeout = std::chrono::duration_cast<std::chrono::milliseconds>(
772 pacs_config.association_timeout);
773
774 services::retrieve_scu scu(retrieve_config);
775 auto move_result = scu.retrieve_study(
776 assoc, study.study_instance_uid);
777
778 // Release association
779 (void)assoc.release();
780
781 if (move_result.is_ok() && move_result.value().is_success()) {
782 integration::logger_adapter::info(
783 "Successfully prefetched study study_uid={} completed={} remote_pacs={}",
784 study.study_instance_uid,
785 move_result.value().completed,
786 pacs_config.ae_title);
787 return true;
788 }
789
790 if (move_result.is_err()) {
791 integration::logger_adapter::error(
792 "C-MOVE failed study_uid={} remote_pacs={} error={}",
793 study.study_instance_uid,
794 pacs_config.ae_title,
795 move_result.error().message);
796 } else if (move_result.value().has_failures()) {
797 integration::logger_adapter::error(
798 "C-MOVE completed with failures study_uid={} completed={} failed={} remote_pacs={}",
799 study.study_instance_uid,
800 move_result.value().completed,
801 move_result.value().failed,
802 pacs_config.ae_title);
803 }
804
805 return false;
806 } catch (const std::exception& e) {
807 integration::logger_adapter::error(
808 "Exception during prefetch study_uid={} remote_pacs={} error={}",
809 study.study_instance_uid,
810 pacs_config.ae_title,
811 e.what());
812 return false;
813 }
814}

820void auto_prefetch_service::queue_request(const prefetch_request& request) {
821 std::lock_guard<std::mutex> lock(queue_mutex_);
822
823 // Deduplicate by patient ID
824 if (queued_patients_.count(request.patient_id) > 0) {
825 return;
826 }
827
828 request_queue_.push(request);
829 queued_patients_.insert(request.patient_id);
830}
831
832auto auto_prefetch_service::dequeue_request()
833 -> std::optional<prefetch_request> {
834 std::lock_guard<std::mutex> lock(queue_mutex_);
835
836 if (request_queue_.empty()) {
837 return std::nullopt;
838 }
839
840 auto request = std::move(request_queue_.front());
841 request_queue_.pop();
842 queued_patients_.erase(request.patient_id);
843
844 return request;
845}
