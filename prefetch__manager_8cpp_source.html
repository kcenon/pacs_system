73std::string generate_uuid() {
74 static thread_local std::random_device rd;
75 static thread_local std::mt19937_64 gen(rd());
76 static thread_local std::uniform_int_distribution<uint64_t> dis;
77
78 uint64_t ab = dis(gen);
79 uint64_t cd = dis(gen);
80
81 // Set version (4) and variant (8, 9, A, or B)
82 ab = (ab & 0xFFFFFFFFFFFF0FFFULL) | 0x0000000000004000ULL;
83 cd = (cd & 0x3FFFFFFFFFFFFFFFULL) | 0x8000000000000000ULL;
84
85 std::ostringstream oss;
86 oss << std::hex << std::setfill('0');
87 oss << std::setw(8) << (ab >> 32);
88 oss << '-';
89 oss << std::setw(4) << ((ab >> 16) & 0xFFFF);
90 oss << '-';
91 oss << std::setw(4) << (ab & 0xFFFF);
92 oss << '-';
93 oss << std::setw(4) << (cd >> 48);
94 oss << '-';
95 oss << std::setw(12) << (cd & 0xFFFFFFFFFFFFULL);
96
97 return oss.str();
98}

103bool matches_filter(const std::string& value, const std::string& filter) {
104 if (filter.empty()) return true;
105 if (value.empty()) return false;
106
107 size_t start = 0;
108 while (start < filter.size()) {
109 auto end = filter.find(',', start);
110 if (end == std::string::npos) {
111 end = filter.size();
112 }
113
114 auto pattern = filter.substr(start, end - start);
115 // Trim whitespace
116 while (!pattern.empty() && pattern.front() == ' ') pattern.erase(0, 1);
117 while (!pattern.empty() && pattern.back() == ' ') pattern.pop_back();
118
119 if (pattern == value) {
120 return true;
121 }
122
123 start = end + 1;
124 }
125
126 return false;
127}

185 void load_rules_from_repo() {
186 #ifdef PACS_WITH_DATABASE_SYSTEM

188 auto result = repositories.rules->find_enabled();

255 std::vector<prefetch_rule> get_matching_rules(
256 prefetch_trigger trigger,
257 const std::string& modality,
258 const std::string& body_part,
259 const std::string& station_ae) {
260 std::vector<prefetch_rule> matches;
261
262 std::shared_lock lock(rules_mutex);
263 for (const auto& rule : rules_cache) {
264 if (!rule.enabled) continue;
265 if (rule.trigger != trigger) continue;
266
267 if (!matches_filter(modality, rule.modality_filter)) continue;
268 if (!matches_filter(body_part, rule.body_part_filter)) continue;
269 if (!matches_filter(station_ae, rule.station_ae_filter)) continue;
270
271 matches.push_back(rule);
272 }
273
274 return matches;
275 }

351 void scheduler_loop() {
352 while (scheduler_running.load()) {
353 std::unique_lock lock(scheduler_mutex);
354 scheduler_cv.wait_for(lock, std::chrono::minutes(1), [this] {
355 return !scheduler_running.load();
356 });
357
358 if (!scheduler_running.load()) break;
359
360 // Check scheduled rules
361 check_scheduled_rules();
362 }
363 }
364
365 void check_scheduled_rules() {
366 std::shared_lock lock(rules_mutex);
367 for (const auto& rule : rules_cache) {
368 if (!rule.enabled) continue;
369 if (rule.trigger != prefetch_trigger::scheduled_exam) continue;
370 if (rule.schedule_cron.empty()) continue;
371
372 // Simple cron check - in production would use proper cron parsing
373 // For now, just check if rule should run
374 // This is a placeholder for actual cron implementation
375 }
376 }

498kcenon::pacs::VoidResult prefetch_manager::add_rule(const prefetch_rule& rule) {
499 prefetch_rule new_rule = rule;
500 if (new_rule.rule_id.empty()) {
501 new_rule.rule_id = generate_uuid();
502 }
503
504 // Save to repository
505 #ifdef PACS_WITH_DATABASE_SYSTEM
506 if (impl_->repositories.rules) {
507 auto result = impl_->repositories.rules->save(new_rule);
508 if (result.is_err()) {
509 return result.error();
510 }
511 } else if (impl_->compatibility_repo) {
512 auto result = impl_->compatibility_repo->save_rule(new_rule);
513 if (result.is_err()) {
514 return result;
515 }
516 }
517 #else
518 if (impl_->compatibility_repo) {
519 auto result = impl_->compatibility_repo->save_rule(new_rule);
520 if (result.is_err()) {
521 return result;
522 }
523 }
524 #endif
525
526 // Update cache
527 {
528 std::unique_lock lock(impl_->rules_mutex);
529 impl_->rules_cache.push_back(new_rule);
530 }
531
532 impl_->logger->info_fmt("Added prefetch rule: {} ({})", new_rule.name, new_rule.rule_id);
533 return kcenon::common::ok();
534}

536kcenon::pacs::VoidResult prefetch_manager::update_rule(const prefetch_rule& rule) {
537 if (rule.rule_id.empty()) {
538 return kcenon::pacs::pacs_void_error(-1, "Rule ID is required for update");
539 }
540
541 // Save to repository
542 #ifdef PACS_WITH_DATABASE_SYSTEM
543 if (impl_->repositories.rules) {
544 auto result = impl_->repositories.rules->save(rule);
545 if (result.is_err()) {
546 return result.error();
547 }
548 } else if (impl_->compatibility_repo) {
549 auto result = impl_->compatibility_repo->save_rule(rule);
550 if (result.is_err()) {
551 return result;
552 }
553 }
554 #else
555 if (impl_->compatibility_repo) {
556 auto result = impl_->compatibility_repo->save_rule(rule);
557 if (result.is_err()) {
558 return result;
559 }
560 }
561 #endif
562
563 // Update cache
564 {
565 std::unique_lock lock(impl_->rules_mutex);
566 for (auto& cached_rule : impl_->rules_cache) {
567 if (cached_rule.rule_id == rule.rule_id) {
568 cached_rule = rule;
569 break;
570 }
571 }
572 }
573
574 impl_->logger->info_fmt("Updated prefetch rule: {}", rule.rule_id);
575 return kcenon::common::ok();
576}

578kcenon::pacs::VoidResult prefetch_manager::remove_rule(std::string_view rule_id) {
579 // Remove from repository
580 #ifdef PACS_WITH_DATABASE_SYSTEM
581 if (impl_->repositories.rules) {
582 auto result = impl_->repositories.rules->remove(std::string(rule_id));
583 if (result.is_err()) {
584 return result;
585 }
586 } else if (impl_->compatibility_repo) {
587 auto result = impl_->compatibility_repo->remove_rule(rule_id);
588 if (result.is_err()) {
589 return result;
590 }
591 }
592 #else
593 if (impl_->compatibility_repo) {
594 auto result = impl_->compatibility_repo->remove_rule(rule_id);
595 if (result.is_err()) {
596 return result;
597 }
598 }
599 #endif
600
601 // Remove from cache
602 {
603 std::unique_lock lock(impl_->rules_mutex);
604 impl_->rules_cache.erase(
605 std::remove_if(impl_->rules_cache.begin(), impl_->rules_cache.end(),
606 [&rule_id](const prefetch_rule& r) { return r.rule_id == rule_id; }),
607 impl_->rules_cache.end());
608 }
609
610 impl_->logger->info_fmt("Removed prefetch rule: {}", std::string(rule_id));
611 return kcenon::common::ok();
612}

633void prefetch_manager::process_worklist(
634 const std::vector<core::dicom_dataset>& worklist_items) {
635 for (const auto& item : worklist_items) {
636 // Extract relevant fields from worklist item
637 auto patient_id = get_dataset_value(item, core::tags::patient_id);
638 auto modality_val = get_dataset_value(item, core::tags::modality);
639 auto body_part = get_dataset_value(item, body_part_examined_tag);
640 auto station_ae = get_dataset_value(item, core::tags::scheduled_station_ae_title);
641
642 if (patient_id.empty()) {
643 continue;
644 }
645
646 // Find matching rules
647 auto rules = impl_->get_matching_rules(
648 prefetch_trigger::worklist_match, modality_val, body_part, station_ae);
649
650 for (const auto& rule : rules) {
651 // Trigger prefetch for this patient
652 auto result = prefetch_priors(patient_id, modality_val, body_part);
653
654 // Update rule statistics
655 impl_->increment_rule_stats(rule.rule_id, result.studies_prefetched);
656
657 impl_->logger->debug_fmt("Worklist prefetch for patient {}: {} studies prefetched",
658 patient_id, result.studies_prefetched);
659 }
660 }
661}

671// Prior Study Prefetch
672// =============================================================================
673
674prefetch_result prefetch_manager::prefetch_priors(
675 std::string_view patient_id,
676 std::string_view current_modality,
677 std::optional<std::string_view> body_part) {
678 auto start_time = std::chrono::steady_clock::now();
679 prefetch_result result;
680 result.patient_id = std::string(patient_id);
681
682 // Find matching prior study rules
683 auto rules = impl_->get_matching_rules(
684 prefetch_trigger::prior_studies,
685 std::string(current_modality),
686 body_part ? std::string(*body_part) : "",
687 "");
688
689 if (rules.empty()) {
690 impl_->logger->debug_fmt("No prior study rules match for patient {} modality {}",
691 std::string(patient_id), std::string(current_modality));
692 return result;
693 }
694
695 // Use the first matching rule (or could combine multiple)
696 const auto& rule = rules.front();
697
698 // Query prior studies from each source node
699 for (const auto& source_node_id : rule.source_node_ids) {
700 auto node = impl_->node_manager->get_node(source_node_id);
701 if (!node) {
702 impl_->logger->warn_fmt("Source node {} not found", source_node_id);
703 continue;
704 }
705
706 // Query for prior studies
707 // In production, this would use query_scu to search for studies
708 // For now, create prefetch jobs based on rule configuration
709
710 // Record that we attempted prefetch
711 impl_->logger->info_fmt("Prefetching priors for patient {} from node {}",
712 std::string(patient_id), source_node_id);
713 }
714
715 auto end_time = std::chrono::steady_clock::now();
716 result.elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
717 end_time - start_time);
718
719 return result;
720}

733 });

739
740std::string prefetch_manager::prefetch_study(
741 std::string_view source_node_id,
742 std::string_view study_uid) {
743 // Check deduplication
744 if (impl_->config.deduplicate_requests) {
745 if (impl_->is_study_pending(std::string(study_uid))) {

769 job_id,
