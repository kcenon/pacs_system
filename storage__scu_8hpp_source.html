68struct store_result {
70 std::string sop_instance_uid;
71
73 uint16_t status{0};
74
76 std::string error_comment;
77
79 [[nodiscard]] bool is_success() const noexcept {
80 return status == 0x0000;
81 }
82
84 [[nodiscard]] bool is_warning() const noexcept {
85 return (status & 0xF000) == 0xB000;
86 }
87
89 [[nodiscard]] bool is_error() const noexcept {
90 return !is_success() && !is_warning();
91 }
92};
