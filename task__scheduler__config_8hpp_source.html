99 [[nodiscard]] static auto every_minutes(int n) -> cron_schedule {
100 cron_schedule s;
101 s.minute = "*/" + std::to_string(n);
102 return s;
103 }
104
109 [[nodiscard]] static auto every_hours(int n) -> cron_schedule {
110 cron_schedule s;
111 s.minute = "0";
112 s.hour = "*/" + std::to_string(n);
113 return s;
114 }
115
121 [[nodiscard]] static auto daily_at(int hour, int minute = 0) -> cron_schedule {
122 cron_schedule s;
123 s.minute = std::to_string(minute);
124 s.hour = std::to_string(hour);
125 return s;
126 }
127
134 [[nodiscard]] static auto weekly_on(int day_of_week, int hour = 0, int minute = 0)
135 -> cron_schedule {
136 cron_schedule s;
137 s.minute = std::to_string(minute);
138 s.hour = std::to_string(hour);
139 s.day_of_week = std::to_string(day_of_week);
140 return s;
141 }
