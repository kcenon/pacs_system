126[[nodiscard]] inline std::string json_escape(std::string_view s) {
127 std::string result;
128 result.reserve(s.size() + 10);
129 for (char c : s) {
130 switch (c) {
131 case '"':
132 result += "\\\"";
133 break;
134 case '\\':
135 result += "\\\\";
136 break;
137 case '\b':
138 result += "\\b";
139 break;
140 case '\f':
141 result += "\\f";
142 break;
143 case '\n':
144 result += "\\n";
145 break;
146 case '\r':
147 result += "\\r";
148 break;
149 case '\t':
150 result += "\\t";
151 break;
152 default:
153 result += c;
154 break;
155 }
156 }
157 return result;
158}
