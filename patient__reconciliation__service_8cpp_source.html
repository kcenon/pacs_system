65reconciliation_result patient_reconciliation_service::update_demographics(
66 const demographics_update_request& request) {
67
68 reconciliation_result result;
69 result.type = reconciliation_type::demographics_update;
70
71 if (request.target_patient_id.empty()) {
72 result.error_message = "Target patient ID is required";
73 return result;
74 }
75
76 size_t updated = 0;
77 std::set<std::string> affected_studies;
78
79 for (auto& instance : instances_) {
80 auto pid = instance.get_string(tags::patient_id);
81 if (pid == request.target_patient_id) {
82 apply_demographics(instance, request.updated_demographics);
83 updated++;
84
85 auto study_uid = instance.get_string(tags::study_instance_uid);
86 if (!study_uid.empty()) {
87 affected_studies.insert(study_uid);
88 }
89 }
90 }
91
92 if (updated == 0) {
93 result.error_message = "No instances found for patient ID: " +
94 request.target_patient_id;
95 return result;
96 }
97
98 result.success = true;
99 result.instances_updated = updated;
100 result.studies_affected = affected_studies.size();
101
102 // Create audit record
103 reconciliation_audit_record audit;
104 audit.record_id = generate_record_id();
105 audit.type = reconciliation_type::demographics_update;
106 audit.primary_patient_id = request.target_patient_id;
107 audit.operator_name = request.operator_name.value_or("SYSTEM");
108 audit.instances_updated = updated;
109 audit.timestamp = std::chrono::system_clock::now();
110 audit.success = true;
111 audit_records_.push_back(std::move(audit));
112
113 return result;
114}

116reconciliation_result patient_reconciliation_service::merge_patients(
117 const patient_merge_request& request) {
118
119 reconciliation_result result;
120 result.type = reconciliation_type::patient_merge;
121
122 if (request.source_patient_id.empty()) {
123 result.error_message = "Source patient ID is required";
124 return result;
125 }
126
127 if (request.target_patient_id.empty()) {
128 result.error_message = "Target patient ID is required";
129 return result;
130 }
131
132 if (request.source_patient_id == request.target_patient_id) {
133 result.error_message = "Source and target patient IDs must be different";
134 return result;
135 }
136
137 size_t updated = 0;
138 std::set<std::string> affected_studies;
139
140 for (auto& instance : instances_) {
141 auto pid = instance.get_string(tags::patient_id);
142 if (pid == request.source_patient_id) {
143 // Reassign to target patient
144 instance.set_string(tags::patient_id, vr_type::LO,
145 request.target_patient_id);
146
147 // Apply target demographics if provided
148 if (request.target_demographics) {
149 apply_demographics(instance, request.target_demographics.value());
150 }
151
152 updated++;
153
154 auto study_uid = instance.get_string(tags::study_instance_uid);
155 if (!study_uid.empty()) {
156 affected_studies.insert(study_uid);
157 }
158 }
159 }
160
161 if (updated == 0) {
162 result.error_message = "No instances found for source patient ID: " +
163 request.source_patient_id;
164 return result;
165 }
166
167 result.success = true;
168 result.instances_updated = updated;
169 result.studies_affected = affected_studies.size();
170
171 // Create audit record
172 reconciliation_audit_record audit;
173 audit.record_id = generate_record_id();
174 audit.type = reconciliation_type::patient_merge;
175 audit.primary_patient_id = request.target_patient_id;
176 audit.secondary_patient_id = request.source_patient_id;
177 audit.operator_name = request.operator_name.value_or("SYSTEM");
178 audit.instances_updated = updated;
179 audit.timestamp = std::chrono::system_clock::now();
180 audit.success = true;
181 audit_records_.push_back(std::move(audit));
182
183 return result;
184}

220 const std::string& patient_id) const {
221
222 std::vector<reconciliation_audit_record> results;
223 for (const auto& record : audit_records_) {
224 if (record.primary_patient_id == patient_id ||
225 (record.secondary_patient_id &&
226 record.secondary_patient_id.value() == patient_id)) {
227 results.push_back(record);
228 }
229 }
230 return results;

237void patient_reconciliation_service::apply_demographics(
238 core::dicom_dataset& dataset,
239 const patient_demographics& demographics) const {
240
241 if (demographics.patient_name) {
242 dataset.set_string(tags::patient_name, vr_type::PN,
243 demographics.patient_name.value());
244 }
245 if (demographics.patient_id) {
246 dataset.set_string(tags::patient_id, vr_type::LO,
247 demographics.patient_id.value());
248 }
249 if (demographics.patient_birth_date) {
250 dataset.set_string(tags::patient_birth_date, vr_type::DA,
251 demographics.patient_birth_date.value());
252 }
253 if (demographics.patient_sex) {
254 dataset.set_string(tags::patient_sex, vr_type::CS,
255 demographics.patient_sex.value());
256 }
257 if (demographics.issuer_of_patient_id) {
258 dataset.set_string(tags::issuer_of_patient_id, vr_type::LO,
259 demographics.issuer_of_patient_id.value());
260 }
261}

263std::string patient_reconciliation_service::generate_record_id() const {
264 static std::mt19937_64 gen{std::random_device{}()};
265 static std::uniform_int_distribution<uint64_t> dist;
266
267 auto now = std::chrono::system_clock::now();
268 auto timestamp = std::chrono::duration_cast<std::chrono::milliseconds>(
269 now.time_since_epoch()).count();
270
271 return "PIR-" + std::to_string(timestamp) + "-" +
272 std::to_string(dist(gen) % 100000);
273}
